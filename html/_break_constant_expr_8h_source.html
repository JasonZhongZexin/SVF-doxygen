78  inline void UnifyFunctionExit(Module& module)
79  {
80  for (Module::const_iterator iter = module.begin(), eiter = module.end();
81  iter != eiter; ++iter)
82  {
83  const Function& fun = *iter;
84  if(fun.isDeclaration())
85  continue;
86  getUnifyExit(fun)->runOnFunction(const_cast<Function&>(fun));
87  }
88  }
90  inline UnifyFunctionExitNodes* getUnifyExit(const Function& fn)
91  {
92  assert(!fn.isDeclaration() && "external function does not have DF");
93  return &getAnalysis<UnifyFunctionExitNodes>(const_cast<Function&>(fn));
94  }
