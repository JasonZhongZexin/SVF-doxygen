38 static bool hasEdge(const CHNode *src, const CHNode *dst,
39  CHEdge::CHEDGETYPE et)
40 {
41  for (CHEdge::CHEdgeSetTy::const_iterator it = src->getOutEdges().begin(),
42  eit = src->getOutEdges().end(); it != eit; ++it)
43  {
44  CHNode *node = (*it)->getDstNode();
45  CHEdge::CHEDGETYPE edgeType = (*it)->getEdgeType();
46  if (node == dst && edgeType == et)
47  return true;
48  }
49  return false;
50 }

52 static bool checkArgTypes(CallSite cs, const Function *fn) {
53 
54  // here we skip the first argument (i.e., this pointer)
55  u32_t arg_size = (fn->arg_size() > cs.arg_size()) ? cs.arg_size(): fn->arg_size();
56  if(arg_size > 1){
57  for (unsigned i = 1; i < arg_size; i++) {
58  auto cs_arg = cs.getArgOperand(i);
59  auto fn_arg = fn->getArg(i);
60  if (cs_arg->getType() != fn_arg->getType()) {
61  return false;
62  }
63  }
64  }
65 
66  return true;
67 }

69 void CHGraph::addEdge(const string className, const string baseClassName,
70  CHEdge::CHEDGETYPE edgeType)
71 {
72  CHNode *srcNode = getNode(className);
73  CHNode *dstNode = getNode(baseClassName);
74  assert(srcNode && dstNode && "node not found?");
75 
76  if (!hasEdge(srcNode, dstNode, edgeType))
77  {
78  CHEdge *edge = new CHEdge(srcNode, dstNode, edgeType);
79  srcNode->addOutgoingEdge(edge);
80  dstNode->addIncomingEdge(edge);
81  }
82 }

93  * Get virtual functions for callsite "cs" based on vtbls (calculated
94  * based on pointsto set)
95  */
96 void CHGraph::getVFnsFromVtbls(CallSite cs, const VTableSet &vtbls, VFunSet &virtualFunctions)
97 {
98 
100  size_t idx = cppUtil::getVCallIdx(cs);
102  string funName = cppUtil::getFunNameOfVCallSite(cs);
103  for (const GlobalValue *vt : vtbls)
104  {
105  const CHNode *child = getNode(cppUtil::getClassNameFromVtblObj(vt));
106  if (child == nullptr)
107  continue;
108  CHNode::FuncVector vfns;
109  child->getVirtualFunctions(idx, vfns);
110  for (CHNode::FuncVector::const_iterator fit = vfns.begin(),
111  feit = vfns.end(); fit != feit; ++fit)
112  {
113  const SVFFunction* callee = *fit;
114  if (cs.arg_size() == callee->arg_size() ||
115  (cs.getFunctionType()->isVarArg() && callee->isVarArg()))
116  {
117 
118  // if argument types do not match
119  // skip this one
120  if (!checkArgTypes(cs, callee->getLLVMFun()))
121  continue;
122 
123  cppUtil::DemangledName dname = cppUtil::demangle(callee->getName().str());
124  string calleeName = dname.funcName;
125 
126  /*
127  * The compiler will add some special suffix (e.g.,
128  * "[abi:cxx11]") to the end of some virtual function:
129  * In dealII
130  * function: FE_Q<3>::get_name
131  * will be mangled as: _ZNK4FE_QILi3EE8get_nameB5cxx11Ev
132  * after demangling: FE_Q<3>::get_name[abi:cxx11]
133  * The special suffix ("[abi:cxx11]") needs to be removed
134  */
135  const std::string suffix("[abi:cxx11]");
136  size_t suffix_pos = calleeName.rfind(suffix);
137  if (suffix_pos != string::npos)
138  calleeName.erase(suffix_pos, suffix.size());
139 
140  /*
141  * if we can't get the function name of a virtual callsite, all virtual
142  * functions calculated by idx will be valid
143  */
144  if (funName.size() == 0)
145  {
146  virtualFunctions.insert(callee);
147  }
148  else if (funName[0] == '~')
149  {
150  /*
151  * if the virtual callsite is calling a destructor, then all
152  * destructors in the ch will be valid
153  * class A { virtual ~A(){} };
154  * class B: public A { virtual ~B(){} };
155  * int main() {
156  * A *a = new B;
157  * delete a; /// the function name of this virtual callsite is ~A()
158  * }
159  */
160  if (calleeName[0] == '~')
161  {
162  virtualFunctions.insert(callee);
163  }
164  }
165  else
166  {
167  /*
168  * for other virtual function calls, the function name of the callsite
169  * and the function name of the target callee should match exactly
170  */
171  if (funName.compare(calleeName) == 0)
172  {
173  virtualFunctions.insert(callee);
174  }
175  }
176  }
177  }
178  }
179 }

192 void CHGraph::printCH()
193 {
194  for (CHGraph::const_iterator it = this->begin(), eit = this->end();
195  it != eit; ++it)
196  {
197  const CHNode *node = it->second;
198  outs() << "class: " << node->getName() << "\n";
199  for (CHEdge::CHEdgeSetTy::const_iterator it = node->OutEdgeBegin();
200  it != node->OutEdgeEnd(); ++it)
201  {
202  if ((*it)->getEdgeType() == CHEdge::INHERITANCE)
203  outs() << (*it)->getDstNode()->getName() << " --inheritance--> "
204  << (*it)->getSrcNode()->getName() << "\n";
205  else
206  outs() << (*it)->getSrcNode()->getName() << " --instance--> "
207  << (*it)->getDstNode()->getName() << "\n";
208  }
209  }
210  outs() << '\n';
211 }
