358  void markCxtStmtFlag(const CxtStmt& tgr, const CxtStmt& src)
360  {
361  const CxtLockSet& srclockset = getCxtLockfromCxtStmt(src);
362  if(hasCxtLockfromCxtStmt(tgr)== false)
363  {
364  for(CxtLockSet::const_iterator it = srclockset.begin(), eit = srclockset.end(); it!=eit; ++it)
365  {
366  addCxtStmtToSpan(tgr,*it);
367  }
368  pushToCTSWorkList(tgr);
369  }
370  else
371  {
372  if(intersect(getCxtLockfromCxtStmt(tgr),srclockset))
373  pushToCTSWorkList(tgr);
374  }
375  }
376  bool intersect(CxtLockSet& tgrlockset, const CxtLockSet& srclockset)
377  {
378  CxtLockSet toBeDeleted;
379  for(CxtLockSet::const_iterator it = tgrlockset.begin(), eit = tgrlockset.end(); it!=eit; ++it)
380  {
381  if(srclockset.find(*it)==srclockset.end())
382  toBeDeleted.insert(*it);
383  }
384  for(CxtLockSet::const_iterator it = toBeDeleted.begin(), eit = toBeDeleted.end(); it!=eit; ++it)
385  {
386  tgrlockset.erase(*it);
387  }
388  if(!toBeDeleted.empty())
389  return true;
390  return false;
391  }
