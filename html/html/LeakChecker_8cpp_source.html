41 void LeakChecker::initSrcs()
42 {
43 
44  PAG* pag = getPAG();
45  ICFG* icfg = pag->getICFG();
46  for(PAG::CSToRetMap::iterator it = pag->getCallSiteRets().begin(),
47  eit = pag->getCallSiteRets().end(); it!=eit; ++it)
48  {
49  const RetBlockNode* cs = it->first;
52  if(isPtrInDeadFunction(cs->getCallSite()))
53  continue;
54 
55  PTACallGraph::FunctionSet callees;
56  getCallgraph()->getCallees(cs->getCallBlockNode(),callees);
57  for(PTACallGraph::FunctionSet::const_iterator cit = callees.begin(), ecit = callees.end(); cit!=ecit; cit++)
58  {
59  const SVFFunction* fun = *cit;
60  if (isSourceLikeFun(fun))
61  {
62  CSWorkList worklist;
63  SVFGNodeBS visited;
64  worklist.push(it->first->getCallBlockNode());
65  while (!worklist.empty())
66  {
67  const CallBlockNode* cs = worklist.pop();
68  const RetBlockNode* retBlockNode = icfg->getRetBlockNode(cs->getCallSite());
69  const PAGNode* pagNode = pag->getCallSiteRet(retBlockNode);
70  const SVFGNode* node = getSVFG()->getDefSVFGNode(pagNode);
71  if (visited.test(node->getId()) == 0)
72  visited.set(node->getId());
73  else
74  continue;
75 
76  CallSiteSet csSet;
77  // if this node is in an allocation wrapper, find all its call nodes
78  if (isInAWrapper(node, csSet))
79  {
80  for (CallSiteSet::iterator it = csSet.begin(), eit =
81  csSet.end(); it != eit; ++it)
82  {
83  worklist.push(*it);
84  }
85  }
86  // otherwise, this is the source we are interested
87  else
88  {
89  // exclude sources in dead functions
90  if (isPtrInDeadFunction(cs->getCallSite()) == false)
91  {
92  addToSources(node);
93  addSrcToCSID(node, cs);
94  }
95  }
96  }
97  }
98  }
99  }
100 
101 }

106 void LeakChecker::initSnks()
107 {
108 
109  PAG* pag = getPAG();
110 
111  for(PAG::CSToArgsListMap::iterator it = pag->getCallSiteArgsMap().begin(),
112  eit = pag->getCallSiteArgsMap().end(); it!=eit; ++it)
113  {
114 
115  PTACallGraph::FunctionSet callees;
116  getCallgraph()->getCallees(it->first,callees);
117  for(PTACallGraph::FunctionSet::const_iterator cit = callees.begin(), ecit = callees.end(); cit!=ecit; cit++)
118  {
119  const SVFFunction* fun = *cit;
120  if (isSinkLikeFun(fun)) {
121  PAG::PAGNodeList &arglist = it->second;
122  assert(!arglist.empty() && "no actual parameter at deallocation site?");
124  for (PAG::PAGNodeList::const_iterator ait = arglist.begin(),
125  aeit = arglist.end(); ait != aeit; ++ait) {
126  const PAGNode *pagNode = *ait;
127  if (pagNode->isPointer()) {
128  const SVFGNode *snk = getSVFG()->getActualParmVFGNode(pagNode, it->first);
129  addToSinks(snk);
130  }
131  }
132  }
133  }
134  }

152 void LeakChecker::reportBug(ProgSlice* slice)
153 {
154 
155  if(isAllPathReachable() == false && isSomePathReachable() == false)
156  {
157  reportNeverFree(slice->getSource());
158  }
159  else if (isAllPathReachable() == false && isSomePathReachable() == true)
160  {
161  reportPartialLeak(slice->getSource());
162  SVFUtil::errs() << "\t\t conditional free path: \n" << slice->evalFinalCond() << "\n";
163  slice->annotatePaths();
164  }
165 
166  if(Options::ValidateTests)
167  testsValidation(slice);
168 }

207  success = true;
208  }
209  else if(fun->getName() == "CLKMALLOC")
210  {
211  if(isAllPathReachable() == false && isSomePathReachable() == false)
212  success = true;
213  }
214  else if(fun->getName() == "NFRLEAKFP" || fun->getName() == "PLKLEAKFP"
215  || fun->getName() == "LEAKFN")
216  {
217  return;
218  }
219  else
220  {
221  writeWrnMsg("\t can not validate, check function not found, please put it at the right place!!");

226 
227  if (success)
