78 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallSite cs)
79 {
80  const PointerType *refType = nullptr;
81  // Case 1: heap object held by *argument, we should get its element type.
82  if (SVFUtil::isHeapAllocExtCallViaArg(cs))
83  {
84  int argPos = SVFUtil::getHeapAllocHoldingArgPosition(cs);
85  const Value* arg = cs.getArgument(argPos);
86  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
87  refType = SVFUtil::dyn_cast<PointerType>(getPtrElementType(argType));
88  }
89  // Case 2: heap/static object held by return value.
90  else
91  {
92  assert((SVFUtil::isStaticExtCall(cs) || SVFUtil::isHeapAllocExtCallViaRet(cs))
93  && "Must be heap alloc via ret, or static allocation site");
94  refType = SVFUtil::dyn_cast<PointerType>(cs.getType());
95  }
96  assert(refType && "Allocated object must be held by a pointer-typed value.");
97  return refType;
98 }
