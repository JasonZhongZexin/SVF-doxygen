213  inline const SVFFunction* getSVFFunction(std::string name)
214  {
215  Function* fun = nullptr;
216 
217  for (u32_t i = 0; i < llvmModuleSet->getModuleNum(); ++i)
218  {
219  Module* mod = llvmModuleSet->getModule(i);
220  fun = mod->getFunction(name);
221  if (fun)
222  {
223  return llvmModuleSet->getSVFFunction(fun);
224  }
225  }
226  return nullptr;
227  }

236  const Function* getDefinition(const Function* fun) const
237  {
238  assert(fun->isDeclaration() && "not a function declaration?");
239  FunDeclToDefMapTy::const_iterator it = FunDeclToDefMap.find(fun);

243 
244  bool hasDeclaration(const Function* fun) const
245  {
246  if(fun->isDeclaration() && !hasDefinition(fun))
247  return false;
248 
249  const Function* funDef = fun;
250  if(fun->isDeclaration() && hasDefinition(fun))
251  funDef = getDefinition(fun);
252 
253  FunDefToDeclsMapTy::const_iterator it = FunDefToDeclsMap.find(funDef);
254  return it != FunDefToDeclsMap.end();
255  }
256 
257  const FunctionSetType& getDeclaration(const Function* fun) const
258  {
259  const Function* funDef = fun;
260  if(fun->isDeclaration() && hasDefinition(fun))
261  funDef = getDefinition(fun);
262 
263  FunDefToDeclsMapTy::const_iterator it = FunDefToDeclsMap.find(funDef);
264  assert(it != FunDefToDeclsMap.end() && "does not have a function definition (body)?");
