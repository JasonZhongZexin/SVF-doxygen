56 {

58 
59  for (const auto& func : *M)

82 void PathCondAllocator::allocateForBB(const BasicBlock & bb)
83 {
84 
85  u32_t succ_number = getBBSuccessorNum(&bb);
86 
87  // if successor number greater than 1, allocate new decision variable for successors
88  if(succ_number > 1)
89  {
90 
91  //allocate log2(num_succ) decision variables
92  double num = log(succ_number)/log(2);
93  u32_t bit_num = (u32_t)ceil(num);
94  u32_t succ_index = 0;
95  std::vector<Condition*> condVec;
96  for(u32_t i = 0 ; i < bit_num; i++)
97  {
98  condVec.push_back(newCond(bb.getTerminator()));
99  }
100 
101  // iterate each successor
102  for (succ_const_iterator succ_it = succ_begin(&bb);
103  succ_it != succ_end(&bb);
104  succ_it++, succ_index++)
105  {
106 
107  const BasicBlock* succ = *succ_it;
108 
109  Condition* path_cond = getTrueCond();
110 
112 
113  // for each successor decide its bit representation
114  // decide whether each bit of succ_index is 1 or 0, if (three successor) succ_index is 000 then use C1^C2^C3
115  // if 001 use C1^C2^negC3
116  for(u32_t j = 0 ; j < bit_num; j++)
117  {
118  //test each bit of this successor's index (binary representation)
119  u32_t tool = 0x01 << j;
120  if(tool & succ_index)
121  {
122  path_cond = condAnd(path_cond, (condNeg(condVec.at(j))));
123  }
124  else
125  {
126  path_cond = condAnd(path_cond, condVec.at(j));
127  }
128  }
129  setBranchCond(&bb,succ,path_cond);
130  }
131 
132  }

181 PathCondAllocator::Condition* PathCondAllocator::evaluateTestNullLikeExpr(const BranchStmt* branchStmt, const BasicBlock *succ)
182 {
183 
184  const BasicBlock* succ1 = branchStmt->getSuccessor(0)->getBB();
185 
186  if(isTestNullExpr(branchStmt->getCondition()->getValue()))
187  {
188  // succ is then branch
189  if(succ1 == succ)
190  return getFalseCond();
191  // succ is else branch
192  else
193  return getTrueCond();
194  }
195  if(isTestNotNullExpr(branchStmt->getCondition()->getValue()))
196  {
197  // succ is then branch
198  if(succ1 == succ)
199  return getTrueCond();
200  // succ is else branch
201  else
202  return getFalseCond();
203  }
204 
205  return nullptr;

211 PathCondAllocator::Condition* PathCondAllocator::evaluateProgExit(const BranchStmt* branchStmt, const BasicBlock *succ)
212 {
213  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
214  const BasicBlock* succ2 = branchStmt->getSuccessor(1)->getBB();
215 
216  bool branch1 = isBBCallsProgExit(succ1);
217  bool branch2 = isBBCallsProgExit(succ2);
218 
220  if(branch1 && !branch2)
221  {
222  // succ is then branch
223  if(succ1 == succ)
224  return getFalseCond();
225  // succ is else branch
226  else
227  return getTrueCond();
228  }
230  else if(!branch1 && branch2)
231  {
232  // succ is else branch
233  if(succ2 == succ)
234  return getFalseCond();
235  // succ is then branch
236  else
237  return getTrueCond();
238  }
239  // two branches both call program exit
240  else if(branch1 && branch2)
241  {
242  return getFalseCond();
243  }
245  else
246  return nullptr;
247

255 PathCondAllocator::Condition* PathCondAllocator::evaluateLoopExitBranch(const BasicBlock * bb, const BasicBlock *dst)
256 {
257  const Function* fun = bb->getParent();
258  assert(fun==dst->getParent() && "two basic blocks should be in the same function");
259 
260  const LoopInfo* loopInfo = getLoopInfo(fun);
261  if(loopInfo->isLoopHeader(const_cast<BasicBlock*>(bb)))
262  {
263  const Loop *loop = loopInfo->getLoopFor(bb);
264  SmallBBVector exitbbs;
265  Set<BasicBlock*> filteredbbs;
266  loop->getExitBlocks(exitbbs);
268  while(!exitbbs.empty())
269  {
270  BasicBlock* eb = exitbbs.pop_back_val();
271  if(!isBBCallsProgExit(eb))
272  filteredbbs.insert(eb);
273  }
274 
276  bool allPDT = true;
277  PostDominatorTree* pdt = getPostDT(fun);
278  for(const auto& filteredbb : filteredbbs)
279  {
280  if(!pdt->dominates(dst,filteredbb))
281  allPDT =false;
282  }
283 
284  if(allPDT)
285  return getTrueCond();
286  }
287  return nullptr;
288 }

295 PathCondAllocator::Condition* PathCondAllocator::evaluateBranchCond(const BasicBlock * bb, const BasicBlock *succ)
296 {
297  if(getBBSuccessorNum(bb) == 1)
298  {
299  assert(bb->getTerminator()->getSuccessor(0) == succ && "not the unique successor?");
300  return getTrueCond();
301  }
302 
303  if (ICFGNode *icfgNode = getICFG()->getICFGNode(bb->getTerminator()))
304  {
305  for (const auto &svfStmt: icfgNode->getSVFStmts())
306  {
307  if (const BranchStmt *branchStmt = SVFUtil::dyn_cast<BranchStmt>(svfStmt))
308  {
309  if(branchStmt->getNumSuccessors() == 2)
310  {
311  const BasicBlock* succ1 = branchStmt->getSuccessor(0)->getBB();
312  const BasicBlock* succ2 = branchStmt->getSuccessor(1)->getBB();
313  assert((succ1 == succ || succ2 == succ) && "not a successor??");
314 
315  Condition* evalLoopExit = evaluateLoopExitBranch(bb,succ);
316  if(evalLoopExit)
317  return evalLoopExit;
318 
319  Condition* evalProgExit = evaluateProgExit(branchStmt,succ);
320  if(evalProgExit)
321  return evalProgExit;
322 
323  Condition* evalTestNullLike = evaluateTestNullLikeExpr(branchStmt,succ);
324  if(evalTestNullLike)
325  return evalTestNullLike;
326  break;
327  }
328  }
329  }
330  }
331 
332  return getBranchCond(bb, succ);
333 }

498 PathCondAllocator::Condition* PathCondAllocator::ComputeIntraVFGGuard(const BasicBlock* srcBB, const BasicBlock* dstBB)
499 {
500 
501  assert(srcBB->getParent() == dstBB->getParent() && "two basic blocks are not in the same function??");
502 
503  PostDominatorTree* postDT = getPostDT(srcBB->getParent());
504  if(postDT->dominates(dstBB,srcBB))
505  return getTrueCond();
506 
507  CFWorkList worklist;
508  worklist.push(srcBB);
509  setCFCond(srcBB,getTrueCond());
510 
511  while(!worklist.empty())
512  {
513  const BasicBlock* bb = worklist.pop();
514  Condition* cond = getCFCond(bb);
515 
518  if(Condition* loopExitCond = evaluateLoopExitBranch(bb,dstBB))
519  return condAnd(cond, loopExitCond);
520 
521 
522  for (succ_const_iterator succ_it = succ_begin(bb);
523  succ_it != succ_end(bb); succ_it++)
524  {
525  const BasicBlock* succ = *succ_it;
530  Condition* brCond;
531  if(postDT->dominates(succ,bb))
532  brCond = getTrueCond();
533  else
534  brCond = getEvalBrCond(bb, succ);
535 
536  DBOUT(DSaber, outs() << " bb (" << bb->getName().str() <<
537  ") --> " << "succ_bb (" << succ->getName().str() << ") condition: " << brCond << "\n");
538  Condition* succPathCond = condAnd(cond, brCond);
539  if(setCFCond(succ, condOr(getCFCond(succ), succPathCond)))
540  worklist.push(succ);
541  }
542  }
543 
544  DBOUT(DSaber, outs() << " src_bb (" << srcBB->getName().str() <<
545  ") --> " << "dst_bb (" << dstBB->getName().str() << ") condition: " << getCFCond(dstBB) << "\n");
546 
547  return getCFCond(dstBB);
548 }
