45 class ExtAPI
46 {
47 public:
48  //External Function types
49  //Assume a call in the form LHS= F(arg0, arg1, arg2, arg3).
50  enum extf_t
51  {
52  EFT_NOOP= 0, //no effect on pointers
53  EFT_ALLOC, //returns a ptr to a newly allocated object
54  EFT_REALLOC, //like L_A0 if arg0 is a non-null ptr, else ALLOC
55  EFT_FREE, //free memory arg0 and all pointers passing into free function
56  EFT_NOSTRUCT_ALLOC, //like ALLOC but only allocates non-struct data
57  EFT_STAT, //retval points to an unknown static var X
58  EFT_STAT2, //ret -> X -> Y (X, Y - external static vars)
59  EFT_L_A0, //copies arg0, arg1, or arg2 into LHS
60  EFT_L_A1,
61  EFT_L_A2,
62  EFT_L_A8,
63  EFT_L_A0__A0R_A1, //stores arg1 into *arg0 and returns arg0 (currently only for memset)
64  EFT_L_A0__A0R_A1R, //copies the data that arg1 points to into the location
65  // arg0 points to; note that several fields may be
66  // copied at once if both point to structs.
67  // Returns arg0.
68  EFT_A1R_A0R, //copies *arg0 into *arg1, with non-ptr return
69  EFT_A3R_A1R_NS, //copies *arg1 into *arg3 (non-struct copy only)
70  EFT_A1R_A0, //stores arg0 into *arg1
71  EFT_A2R_A1, //stores arg1 into *arg2
72  EFT_A4R_A1, //stores arg1 into *arg4
73  EFT_L_A0__A2R_A0, //stores arg0 into *arg2 and returns it
74  EFT_L_A0__A1_A0, //store arg1 into arg0's base and returns arg0
75  EFT_A0R_NEW, //stores a pointer to an allocated object in *arg0
76  EFT_A1R_NEW, //as above, into *arg1, etc.
77  EFT_A2R_NEW,
78  EFT_A4R_NEW,
79  EFT_A11R_NEW,
80  EFT_STD_RB_TREE_INSERT_AND_REBALANCE, // Some complex effects
81  EFT_STD_RB_TREE_INCREMENT, // Some complex effects
82  EFT_STD_LIST_HOOK, // Some complex effects
83 
84  CPP_EFT_A0R_A1, //stores arg1 into *arg0
85  CPP_EFT_A0R_A1R, //copies *arg1 into *arg0
86  CPP_EFT_A1R, //load arg1
87  EFT_CXA_BEGIN_CATCH, //__cxa_begin_catch
88  CPP_EFT_DYNAMIC_CAST, // dynamic_cast
89  EFT_OTHER //not found in the list
90  };
91 private:
92 
93  //Each Function name is mapped to its extf_t
94  // (hash_map and map are much slower).
95  llvm::StringMap<extf_t> info;
96  //A cache of is_ext results for all SVFFunction*'s (hash_map is fastest).
97  Map<const SVFFunction*, bool> isext_cache;
98 
99  void init(); //fill in the map (see ExtAPI.cpp)
100 
101  ExtAPI()
102  {
103  init();
104  isext_cache.clear();
105  }

122  //Return the extf_t of (F).
123  extf_t get_type(const SVFFunction* F) const
124  {
125  assert(F);
126  std::string funName = F->getName().str();
127  if(F->isIntrinsic())
128  {
129  funName = "llvm." + F->getName().split('.').second.split('.').first.str();
130  }
131  llvm::StringMap<extf_t>::const_iterator it= info.find(funName);
132  if(it == info.end() || !F->isDeclaration())
133  return EFT_OTHER;
134  else
135  return it->second;
136  }
137 
138  //Does (F) have a static var X (unavailable to us) that its return points to?
139  bool has_static(const SVFFunction* F) const
140  {
141  extf_t t= get_type(F);
142  return t==EFT_STAT || t==EFT_STAT2;
143  }
144  //Assuming hasStatic(F), does (F) have a second static Y where X -> Y?
145  bool has_static2(const SVFFunction* F) const
146  {
147  return get_type(F) == EFT_STAT2;
148  }
149  //Does (F) allocate a new object and return it?
150  bool is_alloc(const SVFFunction* F) const
151  {
152  extf_t t= get_type(F);
153  return t==EFT_ALLOC || t==EFT_NOSTRUCT_ALLOC;
154  }
155  //Does (F) allocate a new object and assign it to one of its arguments?
156  bool is_arg_alloc(const SVFFunction* F) const
157  {
158  extf_t t= get_type(F);
159  return t>=EFT_A0R_NEW && t<=EFT_A11R_NEW;
160  }
161  //Get the position of argument which holds the new object
162  int get_alloc_arg_pos(const SVFFunction* F) const
163  {
164  extf_t t= get_type(F);
165  switch(t)
166  {
167  case EFT_A0R_NEW:
168  return 0;
169  case EFT_A1R_NEW:
170  return 1;
171  case EFT_A2R_NEW:
172  return 2;
173  case EFT_A4R_NEW:
174  return 4;
175  case EFT_A11R_NEW:
176  return 11;
177  default:
178  assert(false && "Not an alloc call via argument.");
179  return -1;
180  }
181  }
182  //Does (F) allocate only non-struct objects?
183  bool no_struct_alloc(const SVFFunction* F) const
184  {
185  return get_type(F) == EFT_NOSTRUCT_ALLOC;
186  }
187  //Does (F) not free/release any memory?
188  bool is_dealloc(const SVFFunction* F) const
189  {
190  extf_t t= get_type(F);
191  return t == EFT_FREE;
192  }
193  //Does (F) not do anything with the known pointers?
194  bool is_noop(const SVFFunction* F) const
195  {
196  extf_t t= get_type(F);
197  return t == EFT_NOOP || t == EFT_FREE;
198  }
199  //Does (F) reallocate a new object?
200  bool is_realloc(const SVFFunction* F) const
201  {
202  extf_t t= get_type(F);
203  return t==EFT_REALLOC;
204  }
205  //Should (F) be considered "external" (either not defined in the program
206  // or a user-defined version of a known alloc or no-op)?
207  bool is_ext(const SVFFunction* F)
208  {
209  assert(F);
210  //Check the cache first; everything below is slower.
211  Map<const SVFFunction*, bool>::iterator i_iec= isext_cache.find(F);
212  if(i_iec != isext_cache.end())
213  return i_iec->second;
214 
215  bool res;
216  if(F->isDeclaration() || F->isIntrinsic())
217  {
218  res= 1;
219  }
220  else
221  {
222  extf_t t= get_type(F);
223  res= t==EFT_ALLOC || t==EFT_REALLOC || t==EFT_NOSTRUCT_ALLOC
224  || t==EFT_NOOP || t==EFT_FREE;
225  }
226  isext_cache[F]= res;
227  return res;
228  }
229 };
