219  inline Condition* getVFCond(const SVFGNode* node) const
220  {
221  SVFGNodeToCondMap::const_iterator it = svfgNodeToCondMap.find(node);
222  if(it==svfgNodeToCondMap.end())
223  {
224  return getFalseCond();
225  }
226  return it->second;
227  }
228  inline bool setVFCond(const SVFGNode* node, Condition* cond)
229  {
230  SVFGNodeToCondMap::iterator it = svfgNodeToCondMap.find(node);
231  // until a fixed-point is reached (condition is not changed)
232  if(it!=svfgNodeToCondMap.end() && isEquivalentBranchCond(it->second, cond))
233  return false;
234 
235  svfgNodeToCondMap[node] = cond;
236  return true;
237  }
