62 void CFLBase::buildCFLGraph()
63 {
64  // Start building CFLGraph
65  double start = stat->getClk(true);
66 
67  AliasCFLGraphBuilder cflGraphBuilder = AliasCFLGraphBuilder();
68  if (Options::CFLGraph.empty()) // built from svfir
69  {
70  PointerAnalysis::initialize();
71  ConstraintGraph *consCG = new ConstraintGraph(svfir);
72  if (Options::PEGTransfer)
73  graph = cflGraphBuilder.buildBiPEGgraph(consCG, grammarBase->getStartKind(), grammarBase, svfir);
74  else
75  graph = cflGraphBuilder.buildBigraph(consCG, grammarBase->getStartKind(), grammarBase);
76  delete consCG;
77  }
78  else
79  graph = cflGraphBuilder.buildFromDot(Options::CFLGraph, grammarBase);
80 
81  // Check CFL Graph and Grammar are accordance with grammar
82  CFLGramGraphChecker cflChecker = CFLGramGraphChecker();
83  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
84 
85  // Get time of build graph
86  double end = stat->getClk(true);
87  timeOfBuildCFLGraph += (end - start) / TIMEINTERVAL;
88 }
