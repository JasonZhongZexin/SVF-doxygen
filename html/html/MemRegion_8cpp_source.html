172 void MRGenerator::collectModRefForLoadStore()
173 {
174 
175  SVFModule* svfModule = pta->getModule();
176  for (SVFModule::const_iterator fi = svfModule->begin(), efi = svfModule->end(); fi != efi;
177  ++fi)
178  {
179  const SVFFunction& fun = **fi;
180 
182  if (Options::IgnoreDeadFun && isDeadFunction(fun.getLLVMFun()))
183  continue;
184 
185  for (Function::const_iterator iter = fun.getLLVMFun()->begin(), eiter = fun.getLLVMFun()->end();
186  iter != eiter; ++iter)
187  {
188  const BasicBlock& bb = *iter;
189  for (BasicBlock::const_iterator bit = bb.begin(), ebit = bb.end();
190  bit != ebit; ++bit)
191  {
192  const Instruction& inst = *bit;
193  PAGEdgeList& pagEdgeList = getPAGEdgesFromInst(&inst);
194  for (PAGEdgeList::iterator bit = pagEdgeList.begin(), ebit =
195  pagEdgeList.end(); bit != ebit; ++bit)
196  {
197  const PAGEdge* inst = *bit;
198  pagEdgeToFunMap[inst] = &fun;
199  if (const StorePE *st = SVFUtil::dyn_cast<StorePE>(inst))
200  {
201  NodeBS cpts(pta->getPts(st->getDstID()).toNodeBS());
202  // TODO: change this assertion check later when we have conditional points-to set
203  if (cpts.empty())
204  continue;
205  assert(!cpts.empty() && "null pointer!!");
206  addCPtsToStore(cpts, st, &fun);
207  }
208 
209  else if (const LoadPE *ld = SVFUtil::dyn_cast<LoadPE>(inst))
210  {
211  NodeBS cpts(pta->getPts(ld->getSrcID()).toNodeBS());
212  // TODO: change this assertion check later when we have conditional points-to set
213  if (cpts.empty())
214  continue;
215  assert(!cpts.empty() && "null pointer!!");
216  addCPtsToLoad(cpts, ld, &fun);
217  }
218  }
219  }
220  }
221  }

284 void MRGenerator::sortPointsTo(const NodeBS& cpts)
285 {
286 
287  if(cptsToRepCPtsMap.find(cpts)!=cptsToRepCPtsMap.end())
288  return;
289 
290  PointsToList subSetList;
291  NodeBS repCPts = cpts;
292  for(PtsToRepPtsSetMap::iterator it = cptsToRepCPtsMap.begin(),
293  eit = cptsToRepCPtsMap.end(); it!=eit; ++it)
294  {
295  NodeBS& existCPts = it->second;
296  if(cpts.contains(existCPts))
297  {
298  subSetList.insert(it->first);
299  }
300  else if(existCPts.contains(cpts))
301  {
302  repCPts = existCPts;
303  }
304  }
305 
306  for(PointsToList::iterator it = subSetList.begin(), eit = subSetList.end(); it!=eit; ++it)
307  {
308  cptsToRepCPtsMap[*it] = cpts;
309  }

311  cptsToRepCPtsMap[cpts] = repCPts;

480 {
482  NodeBS& argsPts = csToCallSiteArgsPtsMap[cs];
483  PAG* pag = pta->getPAG();
484  CallBlockNode* callBlockNode = pag->getICFG()->getCallBlockNode(cs->getCallSite());
485  RetBlockNode* retBlockNode = pag->getICFG()->getRetBlockNode(cs->getCallSite());
486 
487  WorkList worklist;
488  if (pag->hasCallSiteArgsMap(callBlockNode))
489  {
490  const PAG::PAGNodeList& args = pta->getPAG()->getCallSiteArgsList(callBlockNode);
491  for(PAG::PAGNodeList::const_iterator itA = args.begin(), ieA = args.end(); itA!=ieA; ++itA)
492  {
493  const PAGNode* node = *itA;
494  if(node->isPointer())
495  worklist.push(node->getId());
496  }
497  }
498 
499  while(!worklist.empty())
500  {
501  NodeID nodeId = worklist.pop();
502  const NodeBS& tmp = pta->getPts(nodeId).toNodeBS();
503  for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
504  argsPts |= CollectPtsChain(*it);
505  }
506 
508  NodeBS& retPts = csToCallSiteRetPtsMap[cs];
509 
510  if (pta->getPAG()->callsiteHasRet(retBlockNode))
511  {
512  const PAGNode* node = pta->getPAG()->getCallSiteRet(retBlockNode);
513  if(node->isPointer())
514  {
515  const NodeBS& tmp = pta->getPts(node->getId()).toNodeBS();
516  for(NodeBS::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)

603 bool MRGenerator::handleCallsiteModRef(NodeBS& mod, NodeBS& ref, const CallBlockNode* cs, const SVFFunction* callee)
604 {
606  if(isHeapAllocExtCall(cs->getCallSite()))
607  {
608  PAGEdgeList& pagEdgeList = getPAGEdgesFromInst(cs->getCallSite());
609  for (PAGEdgeList::const_iterator bit = pagEdgeList.begin(),
610  ebit = pagEdgeList.end(); bit != ebit; ++bit)
611  {
612  const PAGEdge* edge = *bit;
613  if (const AddrPE* addr = SVFUtil::dyn_cast<AddrPE>(edge))
614  mod.set(addr->getSrcID());
615  }
616  }
618  else
619  {
620  mod = getModSideEffectOfFunction(callee);
621  ref = getRefSideEffectOfFunction(callee);
622  }
623  // add ref set
624  bool refchanged = addRefSideEffectOfCallSite(cs, ref);
625  // add mod set
626  bool modchanged = addModSideEffectOfCallSite(cs, mod);
627 
628  return refchanged || modchanged;

635 void MRGenerator::modRefAnalysis(PTACallGraphNode* callGraphNode, WorkList& worklist)
636 {
637 
639  for(PTACallGraphNode::iterator it = callGraphNode->InEdgeBegin(), eit = callGraphNode->InEdgeEnd();
640  it!=eit; ++it)
641  {
642  PTACallGraphEdge* edge = *it;
643 
645  for(PTACallGraphEdge::CallInstSet::iterator cit = edge->getDirectCalls().begin(),
646  ecit = edge->getDirectCalls().end(); cit!=ecit; ++cit)
647  {

653  }
655  for(PTACallGraphEdge::CallInstSet::iterator cit = edge->getIndirectCalls().begin(),
656  ecit = edge->getIndirectCalls().end(); cit!=ecit; ++cit)

738 ModRefInfo MRGenerator::getModRefInfo(const CallBlockNode* cs, const Value* V)
739 {
740  bool ref = false;
741  bool mod = false;
742 
743  if (pta->getPAG()->hasValueNode(V))
744  {
745  const NodeBS pts(pta->getPts(pta->getPAG()->getValueNode(V)).toNodeBS());
746  const NodeBS csRef = getRefInfoForCall(cs);
747  const NodeBS csMod = getModInfoForCall(cs);
748  NodeBS ptsExpanded, csRefExpanded, csModExpanded;
749  pta->expandFIObjs(pts, ptsExpanded);
750  pta->expandFIObjs(csRef, csRefExpanded);
751  pta->expandFIObjs(csMod, csModExpanded);
752 
753  if (csRefExpanded.intersects(ptsExpanded))
754  ref = true;
755  if (csModExpanded.intersects(ptsExpanded))
756  mod = true;
757  }
758 
759  if (mod && ref)
760  return ModRefInfo::ModRef;
761  else if (ref)
762  return ModRefInfo::Ref;
