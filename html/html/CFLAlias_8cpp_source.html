42 void CFLAlias::onTheFlyCallGraphSolve(const CallSiteToFunPtrMap& callsites, CallEdgeMap& newEdges)
43 {
44  for(CallSiteToFunPtrMap::const_iterator iter = callsites.begin(), eiter = callsites.end(); iter!=eiter; ++iter)
45  {
46  const CallICFGNode* cs = iter->first;
47 
48  if (isVirtualCallSite(SVFUtil::getLLVMCallSite(cs->getCallSite())))
49  {
50  const Value *vtbl = getVCallVtblPtr(SVFUtil::getLLVMCallSite(cs->getCallSite()));
51  assert(pag->hasValueNode(vtbl));
52  NodeID vtblId = pag->getValueNode(vtbl);
53  resolveCPPIndCalls(cs, getPts(vtblId), newEdges);
54  }
55  else
56  resolveIndCalls(iter->first,getPts(iter->second),newEdges);
57  }
58 }

64 void CFLAlias::connectCaller2CalleeParams(CallSite cs, const SVFFunction* F)
65 {
66  assert(F);
67 
68  DBOUT(DAndersen, outs() << "connect parameters from indirect callsite " << SVFUtil::value2String(cs.getInstruction()) << " to callee " << *F << "\n");
69 
70  CallICFGNode* callBlockNode = svfir->getICFG()->getCallICFGNode(cs.getInstruction());
71  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
72 
73  if(SVFUtil::isHeapAllocExtFunViaRet(F) && svfir->callsiteHasRet(retBlockNode))
74  {
75  heapAllocatorViaIndCall(cs);
76  }
77 
78  if (svfir->funHasRet(F) && svfir->callsiteHasRet(retBlockNode))
79  {
80  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
81  const PAGNode* fun_return = svfir->getFunRet(F);
82  if (cs_return->isPointer() && fun_return->isPointer())
83  {
84  NodeID dstrec = cs_return->getId();
85  NodeID srcret = fun_return->getId();
86  addCopyEdge(srcret, dstrec);
87  }
88  else
89  {
90  DBOUT(DAndersen, outs() << "not a pointer ignored\n");
91  }
92  }
93 
94  if (svfir->hasCallSiteArgsMap(callBlockNode) && svfir->hasFunArgsList(F))
95  {
96 
97  // connect actual and formal param
98  const SVFIR::SVFVarList& csArgList = svfir->getCallSiteArgsList(callBlockNode);
99  const SVFIR::SVFVarList& funArgList = svfir->getFunArgsList(F);
100  //Go through the fixed parameters.
101  DBOUT(DPAGBuild, outs() << " args:");
102  SVFIR::SVFVarList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
103  SVFIR::SVFVarList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
104  for (; funArgIt != funArgEit; ++csArgIt, ++funArgIt)
105  {
106  //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
107  if (csArgIt == csArgEit)
108  {
109  DBOUT(DAndersen, outs() << " !! not enough args\n");
110  break;
111  }
112  const PAGNode *cs_arg = *csArgIt ;
113  const PAGNode *fun_arg = *funArgIt;
114 
115  if (cs_arg->isPointer() && fun_arg->isPointer())
116  {
117  DBOUT(DAndersen, outs() << "process actual parm " << cs_arg->toString() << " \n");
118  NodeID srcAA = cs_arg->getId();
119  NodeID dstFA = fun_arg->getId();
120  addCopyEdge(srcAA, dstFA);
121  }
122  }
123 
124  //Any remaining actual args must be varargs.
125  if (F->isVarArg())
126  {
127  NodeID vaF = svfir->getVarargNode(F);
128  DBOUT(DPAGBuild, outs() << "\n varargs:");
129  for (; csArgIt != csArgEit; ++csArgIt)
130  {
131  const PAGNode *cs_arg = *csArgIt;
132  if (cs_arg->isPointer())
133  {
134  NodeID vnAA = cs_arg->getId();
135  addCopyEdge(vnAA,vaF);
136  }
137  }
138  }
139  if(csArgIt != csArgEit)
140  {
141  writeWrnMsg("too many args to non-vararg func.");
142  writeWrnMsg("(" + getSourceLoc(cs.getInstruction()) + ")");
143  }
144  }
145 }

147 void CFLAlias::heapAllocatorViaIndCall(CallSite cs)
148 {
149  assert(SVFUtil::getCallee(cs) == nullptr && "not an indirect callsite?");
150  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
151  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
152  NodeID srcret;
153  CallSite2DummyValPN::const_iterator it = callsite2DummyValPN.find(cs);
154  if(it != callsite2DummyValPN.end())
155  {
156  srcret = it->second;
157  }
158  else
159  {
160  NodeID valNode = svfir->addDummyValNode();
161  NodeID objNode = svfir->addDummyObjNode(cs.getType());
162  callsite2DummyValPN.insert(std::make_pair(cs,valNode));
163  graph->addCFLNode(valNode, new CFLNode(valNode));
164  graph->addCFLNode(objNode, new CFLNode(objNode));
165  srcret = valNode;
166  }
167 
168  NodeID dstrec = cs_return->getId();
169  addCopyEdge(srcret, dstrec);
170 }

198 void CFLAlias::analyze()
199 {
200  GrammarBuilder grammarBuilder = GrammarBuilder(Options::GrammarFilename);
201  CFGNormalizer normalizer = CFGNormalizer();
202  AliasCFLGraphBuilder cflGraphBuilder = AliasCFLGraphBuilder();
203  CFLGramGraphChecker cflChecker = CFLGramGraphChecker();
204  if (Options::GraphIsFromDot == false)
205  {
206  PointerAnalysis::initialize();
207  GrammarBase *grammarBase = grammarBuilder.build();
208  ConstraintGraph *consCG = new ConstraintGraph(svfir);
209  graph = cflGraphBuilder.buildBigraph(consCG, grammarBase->getStartKind(), grammarBase);
210  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
211  grammar = normalizer.normalize(grammarBase);
212  cflChecker.check(grammar, &cflGraphBuilder, graph);
213  delete consCG;
214  delete grammarBase;
215  }
216  else
217  {
218  GrammarBase *grammarBase = grammarBuilder.build();
219  graph = cflGraphBuilder.buildFromDot(Options::InputFilename, grammarBase);
220  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
221  grammar = normalizer.normalize(grammarBase);
222  cflChecker.check(grammar, &cflGraphBuilder, graph);
223  delete grammarBase;
224  }
225  solver = new CFLSolver(graph, grammar);
226  solver->solve();
227  while (updateCallGraph(svfir->getIndirectCallsites()))
228  solver->solve();
229  if(Options::PrintCFL == true)
230  {
231  std::string svfirName = Options::InputFilename.c_str();
232  svfir->dump(svfirName.append("_IR"));
233  std::string grammarName = Options::InputFilename.c_str();
234  grammar->dump(grammarName.append("_Grammar"));
235  std::string CFLGraphFileName = Options::InputFilename.c_str();
236  graph->dump(CFLGraphFileName.append("_CFL"));
237  }
238  if (Options::GraphIsFromDot == false)
239  {
240  PointerAnalysis::finalize();
241  }
242 }
