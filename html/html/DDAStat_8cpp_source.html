83 void DDAStat::performStatPerQuery(NodeID ptr)
84 {
85 
86  u32_t NumOfDPM = 0;
87  u32_t NumOfLoc = 0;
88  u32_t maxNumOfDPMPerLoc = 0;
89  u32_t cptsSize = 0;
90  PointsTo pts;
91  if(flowDDA)
92  {
93  for(FlowDDA::LocToDPMVecMap::const_iterator it = flowDDA->getLocToDPMVecMap().begin(),
94  eit = flowDDA->getLocToDPMVecMap().end(); it!=eit; ++it)
95  {
96  NumOfLoc++;
97  u32_t num = it->second.size();
98  NumOfDPM += num;
99  if(num > maxNumOfDPMPerLoc)
100  maxNumOfDPMPerLoc = num;
101  }
102  cptsSize = flowDDA->getPts(ptr).count();
103  pts = flowDDA->getPts(ptr);
104  }
105  else if(contextDDA)
106  {
107  for(ContextDDA::LocToDPMVecMap::const_iterator it = contextDDA->getLocToDPMVecMap().begin(),
108  eit = contextDDA->getLocToDPMVecMap().end(); it!=eit; ++it)
109  {
110  NumOfLoc++;
111  u32_t num = it->second.size();
112  NumOfDPM += num;
113  if(num > maxNumOfDPMPerLoc)
114  maxNumOfDPMPerLoc = num;
115  }
116  ContextCond cxt;
117  CxtVar var(cxt,ptr);
118  cptsSize = contextDDA->getPts(var).count();
119  pts = contextDDA->getBVPointsTo(contextDDA->getPts(var));
120  }
121  u32_t ptsSize = pts.count();
122 
123  double avgDPMAtLoc = NumOfLoc!=0 ? (double)NumOfDPM/NumOfLoc : 0;
124  _AvgNumOfDPMAtSVFGNode += avgDPMAtLoc;
125  if(maxNumOfDPMPerLoc > _MaxNumOfDPMAtSVFGNode)
126  _MaxNumOfDPMAtSVFGNode = maxNumOfDPMPerLoc;
127 
128  _TotalCPtsSize += cptsSize;
129  if (_MaxCPtsSize < cptsSize)
130  _MaxCPtsSize = cptsSize;
131 
132  _TotalPtsSize += ptsSize;
133  if(_MaxPtsSize < ptsSize)
134  _MaxPtsSize = ptsSize;
135 
136  if(cptsSize == 0)
137  _NumOfNullPtr++;
138 
139  if(getPTA()->containBlackHoleNode(pts))
140  {
141  _NumOfConstantPtr++;
142  }
143  if(getPTA()->containConstantNode(pts))
144  {
145  _NumOfBlackholePtr++;
146  }
147 
148  _TotalNumOfQuery++;
149  _TotalNumOfDPM += _NumOfDPM;
150  _TotalNumOfStrongUpdates += _NumOfStrongUpdates;
151  _TotalNumOfMustAliases += _NumOfMustAliases;
152  _TotalNumOfInfeasiblePath += _NumOfInfeasiblePath;
153 
154  _TotalNumOfStep += _NumOfStep;
155  _TotalNumOfStepInCycle += _NumOfStepInCycle;
156 
157  timeStatMap.clear();
158  NumPerQueryStatMap.clear();
159 
160  timeStatMap["TimePerQuery"] = _AnaTimePerQuery/TIMEINTERVAL;
161  timeStatMap["CyleTimePerQuery"] = _AnaTimeCyclePerQuery/TIMEINTERVAL;
162 
163  NumPerQueryStatMap["CPtsSize"] = cptsSize;
164  NumPerQueryStatMap["PtsSize"] = ptsSize;
165  NumPerQueryStatMap["NumOfStep"] = _NumOfStep;
166  NumPerQueryStatMap["NumOfStepInCycle"] = _NumOfStepInCycle;
167  NumPerQueryStatMap["NumOfDPM"] = _NumOfDPM;
168  NumPerQueryStatMap["NumOfSU"] = _NumOfStrongUpdates;
169  NumPerQueryStatMap["IndEdgeResolved"] = getPTA()->getNumOfResolvedIndCallEdge() - _NumOfIndCallEdgeSolved;
170  NumPerQueryStatMap["AvgDPMAtLoc"] = avgDPMAtLoc;
171  NumPerQueryStatMap["MaxDPMAtLoc"] = maxNumOfDPMPerLoc;
172  NumPerQueryStatMap["MaxPathPerQuery"] = VFPathCond::maximumPath;
173  NumPerQueryStatMap["MaxCxtPerQuery"] = ContextCond::maximumCxt;
174  NumPerQueryStatMap["NumOfMustAA"] = _NumOfMustAliases;
175  NumPerQueryStatMap["NumOfInfePath"] = _NumOfInfeasiblePath;
176 
178  _NumOfStep = 0;
179  _NumOfStepInCycle = 0;
180  _NumOfDPM = 0;
181  _NumOfStrongUpdates = 0;
182  _NumOfMustAliases = 0;
183  _NumOfInfeasiblePath = 0;
184  _AnaTimeCyclePerQuery = 0;
185  _NumOfIndCallEdgeSolved = getPTA()->getNumOfResolvedIndCallEdge();
186 }

196 void DDAStat::performStat()
197 {
198 
199  generalNumMap.clear();
200  PTNumStatMap.clear();
201  timeStatMap.clear();
202 
203  callgraphStat();
204 
205  getNumOfOOBQuery();
206 
207  for (PAG::const_iterator nodeIt = PAG::getPAG()->begin(), nodeEit = PAG::getPAG()->end(); nodeIt != nodeEit; nodeIt++)
208  {
209  PAGNode* pagNode = nodeIt->second;
210  if(SVFUtil::isa<ObjPN>(pagNode))
211  {
212  if(getPTA()->isLocalVarInRecursiveFun(nodeIt->first))
213  {
214  localVarInRecursion.set(nodeIt->first);
215  }
216  }
217  }
218 
219  timeStatMap["TotalQueryTime"] = _TotalTimeOfQueries/TIMEINTERVAL;
220  timeStatMap["AvgTimePerQuery"] = (_TotalTimeOfQueries/TIMEINTERVAL)/_TotalNumOfQuery;
221  timeStatMap["TotalBKCondTime"] = (_TotalTimeOfBKCondition/TIMEINTERVAL);
222 
223  PTNumStatMap["NumOfQuery"] = _TotalNumOfQuery;
224  PTNumStatMap["NumOfOOBQuery"] = _TotalNumOfOutOfBudgetQuery;
225  PTNumStatMap["NumOfDPM"] = _TotalNumOfDPM;
226  PTNumStatMap["NumOfSU"] = _TotalNumOfStrongUpdates;
227  PTNumStatMap["NumOfStoreSU"] = _StrongUpdateStores.count();
228  PTNumStatMap["NumOfStep"] = _TotalNumOfStep;
229  PTNumStatMap["NumOfStepInCycle"] = _TotalNumOfStepInCycle;
230  timeStatMap["AvgDPMAtLoc"] = (double)_AvgNumOfDPMAtSVFGNode/_TotalNumOfQuery;
231  PTNumStatMap["MaxDPMAtLoc"] = _MaxNumOfDPMAtSVFGNode;
232  PTNumStatMap["MaxPathPerQuery"] = VFPathCond::maximumPath;
233  PTNumStatMap["MaxCxtPerQuery"] = ContextCond::maximumCxt;
234  PTNumStatMap["MaxCPtsSize"] = _MaxCPtsSize;
235  PTNumStatMap["MaxPtsSize"] = _MaxPtsSize;
236  timeStatMap["AvgCPtsSize"] = (double)_TotalCPtsSize/_TotalNumOfQuery;
237  timeStatMap["AvgPtsSize"] = (double)_TotalPtsSize/_TotalNumOfQuery;
238  PTNumStatMap["IndEdgeSolved"] = getPTA()->getNumOfResolvedIndCallEdge();
239  PTNumStatMap["NumOfNullPtr"] = _NumOfNullPtr;
240  PTNumStatMap["PointsToConstPtr"] = _NumOfConstantPtr;
241  PTNumStatMap["PointsToBlkPtr"] = _NumOfBlackholePtr;
242  PTNumStatMap["NumOfMustAA"] = _TotalNumOfMustAliases;
243  PTNumStatMap["NumOfInfePath"] = _TotalNumOfInfeasiblePath;
244  PTNumStatMap["NumOfStore"] = PAG::getPAG()->getPTAEdgeSet(PAGEdge::Store).size();
245  PTNumStatMap["MemoryUsageVmrss"] = _vmrssUsageAfter - _vmrssUsageBefore;
246  PTNumStatMap["MemoryUsageVmsize"] = _vmsizeUsageAfter - _vmsizeUsageBefore;
247 
248  printStat();
249 }
