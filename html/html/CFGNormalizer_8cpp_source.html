42 
43  Usage Notation
44  definition =
45  concatenation ,
46  termination ;
47  alternation |
48  optional [ ... ]

51  terminal string " ... "
52  terminal string ' ... '
53  comment (* ... *)
54  special sequence ? ... ?
55  exception -
56 
57  Output: Backus-Naur form(BNF)

60 
61  Possible Step:
62

65  Convert every option [ E ] to a fresh non-terminal X and add

67  (We can convert X = A [ E ] B. to X = A E B | A B.)
68  Convert every group ( E ) to a fresh non-terminal X and add
69  X = E.
70  We can even do away with alternatives by having several productions with the same non-terminal.
71  X = E | E'. becomes X = E. X = E'.

74 
75  And no differentiation between terminal and non-terminal

82 CFLGrammar* CFGNormalizer::normalize(GrammarBase *generalGrammar)
83 {
84  /*
85  ebnf_sign_replace('*');
86  ebnf_sign_replace('?');
87  */
88  CFLGrammar *grammar = new CFLGrammar();
89  grammar->startSymbol = generalGrammar->startSymbol;
90  grammar->terminals = generalGrammar->terminals;
91  grammar->nonterminals = generalGrammar->nonterminals;
92  grammar->totalSymbol = generalGrammar->totalSymbol;
93  ebnf_bin(generalGrammar, grammar);
94 
95  for(auto symProdsPair: generalGrammar->rawProductions)
96  {
97  for(auto prod: symProdsPair.second)
98  {
99  Production tempP = prod;
100  tempP.insert(tempP.begin(), symProdsPair.first);
101  if (prod.size() == 1)
102  {
103  if ((std::find(tempP.begin(), tempP.end(), grammar->str2Sym("epsilon")) != tempP.end()))
104  {
105  if (std::find(grammar->epsilonProds.begin(), grammar->epsilonProds.end(), tempP) == grammar->epsilonProds.end())
106  grammar->epsilonProds.insert(tempP);
107  }
108  grammar->singleRHS2Prods[tempP[1]].insert(tempP);
109  }
110  if (prod.size() == 2)
111  {
112  grammar->firstRHS2Prods[tempP[1]].insert(tempP);
113  grammar->secondRHS2Prods[tempP[2]].insert(tempP);
114  }
115  }
116  }
117 
118  return grammar;
119 }

121 void CFGNormalizer::ebnf_bin(GrammarBase* generalGrammar, CFLGrammar *grammar)
122 {
123  Map<Symbol, Productions> new_grammar = {};
124  std::string tempStr = "";
125 
126  for(auto head : generalGrammar->rawProductions)
127  {
128  for(auto rule: head.second)
129  {
130 
131  Production long_run = rule;
132  long_run.erase(long_run.begin());
133  auto it = generalGrammar->rawProductions[head.first].find(rule);
134  generalGrammar->rawProductions[head.first].erase(it);
135  generalGrammar->rawProductions[head.first].insert(long_run);
136  }
137  }
138 
139  for(auto head : generalGrammar->rawProductions)
140  {
141  for(auto rule: head.second)
142  {
143  if (rule.size() < 3) continue;
144 
145  Production long_run = rule;
146  Symbol first = long_run[0];
147  long_run.erase(long_run.begin());
148  auto it = generalGrammar->rawProductions[head.first].find(rule);
149  generalGrammar->rawProductions[head.first].erase(it);
150  rule = {first};
151  generalGrammar->rawProductions[head.first].insert(rule);
152 
153  Symbol X = check_head(new_grammar, long_run);
154  if (int(X) == -1)
155  {
156  X = check_head(generalGrammar->rawProductions, long_run);
157  }
158  if (int(X) != -1)
159  {
160  it = generalGrammar->rawProductions[head.first].find(rule);
161  generalGrammar->rawProductions[head.first].erase(it);
162  rule.push_back(X);
163  generalGrammar->rawProductions[head.first].insert(rule);
164  }
165  else
166  {
167  tempStr = "X";
168  std::ostringstream ss;
169  ss << grammar->num_generator();
170  tempStr.append(ss.str());
171  Symbol tempSym = grammar->insertNonTerminalSymbol(tempStr);
172  it = generalGrammar->rawProductions[head.first].find(rule);
173  generalGrammar->rawProductions[head.first].erase(it);
174  rule.push_back(tempSym);
175  generalGrammar->rawProductions[head.first].insert(rule);
176  X = tempSym;
177  }
178  new_grammar[X] = {};
179  Production temp_p = long_run;
180  Symbol RHX;
181  if (long_run.size() ==2)
182  {
183  new_grammar[X].insert(temp_p);
184  long_run.clear();
185  }
186  else
187  {
188  new_grammar[X].insert(long_run);
189  RHX = X;
190  }
191  while (long_run.size() > 2)
192  {
193  first = long_run[0];
194  Production prev_rule = long_run;
195  long_run.erase(long_run.begin());
196 
197  X = RHX;
198  temp_p = long_run;
199 
200  RHX = check_head(new_grammar, long_run);
201  if (int(RHX) == -1)
202  {
203  RHX = check_head(generalGrammar->rawProductions, long_run);
204  }
205  if(int(RHX) == -1)
206  {
207  tempStr = "X";
208  std::ostringstream ss;
209  ss << grammar->num_generator();
210  tempStr.append(ss.str());
211  Symbol tempSym = grammar->insertNonTerminalSymbol(tempStr);
212  auto it = new_grammar[X].find(prev_rule);
213  new_grammar[X].erase(it);
214  new_grammar[X].insert({first, tempSym});
215  new_grammar[tempSym].insert(long_run);
216  RHX = tempSym;
217  }
218  }
219 
220 
221  for (auto new_head : new_grammar)
222  {
223  for (auto prod : new_head.second)
224  {
225  auto it = generalGrammar->rawProductions[new_head.first].find(prod);
226  if (it == generalGrammar->rawProductions[new_head.first].end())
227  {
228  generalGrammar->rawProductions[new_head.first].insert(prod);
229  }
230  }
231  }
232 
233  }
234  }

251 void CFGNormalizer::ebnf_sign_replace(char sign, GrammarBase* generalGrammar, CFLGrammar *grammar)
252 {
253  SVF::Map<std::string, std::string> new_rule_checker;
254  std::string X = "X";
255  for (auto ebnfHead : generalGrammar->rawProductions)
256  {
257  for (auto ebnfProd : ebnfHead.second)
258  {
259  size_t i = 1;
260  while (i < ebnfProd.size())
261  {
262  int repetition_start = -1;
263  if (grammar->sym2Str(ebnfProd[i]) == std::string(1, sign))
264  {
265  if (i == 1)
266  {
267  // abort("EBNF Form is not correct");
268  std::cout << "wr";
269  }
270  else if (grammar->sym2Str(ebnfProd[i - 1]) != std::string(1, ')'))
271  {
272  repetition_start = i - 1;
273  }
274  else
275  {
276  repetition_start = ebnf_bracket_match(ebnfProd, i, grammar);
277  }
278  std::string repetition = "";
279  for (size_t j = repetition_start; j < i; j++)
280  {
281  repetition.append(grammar->sym2Str(ebnfProd[j]));
282  repetition.append(" ");
283  }
284  repetition.append(grammar->sym2Str(ebnfProd[i]));
285  if (new_rule_checker.find(repetition) != new_rule_checker.end())
286  {
287 
288  ebnfProd.erase(ebnfProd.begin() + repetition_start, ebnfProd.begin() + i + 1);
289  ebnfProd.insert(ebnfProd.begin() + repetition_start, grammar->str2Sym(new_rule_checker[repetition]));
290  }
291  else
292  {
293  X = "X";
294  std::ostringstream ss;
295  ss << grammar->num_generator();
296  X.append(ss.str());
297  Symbol tempSym = grammar->insertNonTerminalSymbol(X);
298  ebnfProd.erase(ebnfProd.begin() + repetition_start, ebnfProd.begin() + i + 1);
299  ebnfProd.insert(ebnfProd.begin() + repetition_start, tempSym);
300  new_rule_checker[repetition] = X;
301  }
302  i = repetition_start;
303  }
304  i++;
305  }
306  }
307  }
308  for(auto rep: new_rule_checker)
309  {
310  Production temp_list = {};
311  std::string new_nonterminal = rep.second;
312  if (sign == '*')
313  {
314  temp_list.push_back(grammar->str2Sym(new_nonterminal));
315  }
316  Production temp_p = strTrans(rep.second, grammar);
317  temp_list.insert(temp_list.end(), temp_p.begin(),temp_p.end());
318  temp_list.insert(temp_list.begin(), grammar->str2Sym("epsilon"));
319  generalGrammar->rawProductions[grammar->str2Sym(new_nonterminal)].insert(temp_list);
320  }
321 
322 }

324 CFGNormalizer::Production CFGNormalizer::strTrans(std::string LHS, CFLGrammar *grammar)
325 {
326  Production prod = {};
327  std::smatch matches;
328  std::regex LHSReg("\\s*(.*)");
329  std::string delimiter;
330  size_t pos;
331  std::string word;
332  std::regex_search(LHS, matches, LHSReg);
333  LHS = matches.str(1);
334  delimiter = " ";
335  while ((pos = LHS.find(delimiter)) != std::string::npos)
336  {
337  word = LHS.substr(0, pos);
338  LHS.erase(0, pos + delimiter.length());
339  prod.push_back(grammar->str2Sym(word));
340  }
341  prod.push_back(grammar->str2Sym(LHS));
342  return prod;
343 }
