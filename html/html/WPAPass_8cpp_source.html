98 void WPAPass::runPointerAnalysis(SVFModule* svfModule, u32_t kind)
99 {
101  PAGBuilder builder;
102  PAG* pag = builder.build(svfModule);
104  switch (kind)
105  {
106  case PointerAnalysis::Andersen_WPA:
107  _pta = new Andersen(pag);
108  break;
109  case PointerAnalysis::AndersenLCD_WPA:
110  _pta = new AndersenLCD(pag);
111  break;
112  case PointerAnalysis::AndersenHCD_WPA:
113  _pta = new AndersenHCD(pag);
114  break;
115  case PointerAnalysis::AndersenHLCD_WPA:
116  _pta = new AndersenHLCD(pag);
117  break;
118  case PointerAnalysis::AndersenSCD_WPA:
119  _pta = new AndersenSCD(pag);
120  break;
121  case PointerAnalysis::AndersenSFR_WPA:
122  _pta = new AndersenSFR(pag);
123  break;
124  case PointerAnalysis::AndersenWaveDiff_WPA:
125  _pta = new AndersenWaveDiff(pag);
126  break;
127  case PointerAnalysis::AndersenWaveDiffWithType_WPA:
128  _pta = new AndersenWaveDiffWithType(pag);
129  break;
130  case PointerAnalysis::Steensgaard_WPA:
131  _pta = new Steensgaard(pag);
132  break;
133  case PointerAnalysis::FSSPARSE_WPA:
134  _pta = new FlowSensitive(pag);
135  break;
136  case PointerAnalysis::FSTBHC_WPA:
137  _pta = new FlowSensitiveTBHC(pag);
138  break;
139  case PointerAnalysis::VFS_WPA:
140  _pta = new VersionedFlowSensitive(pag);
141  break;
142  case PointerAnalysis::TypeCPP_WPA:
143  _pta = new TypeAnalysis(pag);
144  break;
145  default:
146  assert(false && "This pointer analysis has not been implemented yet.\n");
147  return;
148  }
149 
150  ptaVector.push_back(_pta);
151  _pta->analyze();
152  if (Options::AnderSVFG)
153  {
154  SVFGBuilder memSSA(true);
155  assert(SVFUtil::isa<AndersenBase>(_pta) && "supports only andersen/steensgaard for pre-computed SVFG");
156  SVFG *svfg;
157  if (Options::WPAOPTSVFG)
158  {
159  svfg = memSSA.buildFullSVFG((BVDataPTAImpl*)_pta);
160  } else
161  {
162  svfg = memSSA.buildFullSVFGWithoutOPT((BVDataPTAImpl*)_pta);
163  }
164 
166  if (Options::PASelected.isSet(PointerAnalysis::AndersenWaveDiff_WPA))
167  _svfg = svfg;
168  }
169 
170  if (Options::PrintAliases)
171  PrintAliasPairs(_pta);
172 }

174 void WPAPass::PrintAliasPairs(PointerAnalysis* pta)
175 {
176  PAG* pag = pta->getPAG();
177  for (PAG::iterator lit = pag->begin(), elit = pag->end(); lit != elit; ++lit)
178  {
179  PAGNode* node1 = lit->second;
180  PAGNode* node2 = node1;
181  for (PAG::iterator rit = lit, erit = pag->end(); rit != erit; ++rit)
182  {
183  node2 = rit->second;
184  if(node1==node2)
185  continue;
186  const Function* fun1 = node1->getFunction();
187  const Function* fun2 = node2->getFunction();
188  AliasResult result = pta->alias(node1->getId(), node2->getId());
189  SVFUtil::outs() << (result == AliasResult::NoAlias ? "NoAlias" : "MayAlias")
190  << " var" << node1->getId() << "[" << node1->getValueName()
191  << "@" << (fun1==nullptr?"":fun1->getName()) << "] --"
192  << " var" << node2->getId() << "[" << node2->getValueName()
193  << "@" << (fun2==nullptr?"":fun2->getName()) << "]\n";
194  }
195  }
196 }

202 AliasResult WPAPass::alias(const Value* V1, const Value* V2)
203 {
204 
205  AliasResult result = llvm::MayAlias;
206 
207  PAG* pag = _pta->getPAG();
208 
214  if (pag->hasValueNode(V1) && pag->hasValueNode(V2))
215  {
217  if (Options::AliasRule.getBits() == 0 || Options::AliasRule.isSet(Veto))
218  {
220  result = llvm::MayAlias;
221 
222  for (PTAVector::const_iterator it = ptaVector.begin(), eit = ptaVector.end();
223  it != eit; ++it)
224  {
225  if ((*it)->alias(V1, V2) == llvm::NoAlias)
226  result = llvm::NoAlias;
227  }
228  }
229  else if (Options::AliasRule.isSet(Conservative))
230  {
232  result = llvm::NoAlias;
233 
234  for (PTAVector::const_iterator it = ptaVector.begin(), eit = ptaVector.end();
235  it != eit; ++it)
236  {
237  if ((*it)->alias(V1, V2) == llvm::MayAlias)
238  result = llvm::MayAlias;
239  }
240  }
241  }
242 
243  return result;
244 }
