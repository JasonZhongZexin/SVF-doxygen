38 PTACallGraph* CallGraphBuilder::buildCallGraph(SVFModule* svfModule)
39 {
41  for (SVFModule::llvm_iterator F = svfModule->llvmFunBegin(), E = svfModule->llvmFunEnd(); F != E; ++F)
42  {
43  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(*F);
44  callgraph->addCallGraphNode(fun);
45  }
46 
48  for (SVFModule::llvm_iterator F = svfModule->llvmFunBegin(), E = svfModule->llvmFunEnd(); F != E; ++F)
49  {
50  Function *fun = *F;
51  for (inst_iterator I = inst_begin(*fun), J = inst_end(*fun); I != J; ++I)
52  {
53  const Instruction *inst = &*I;
54  if (SVFUtil::isNonInstricCallSite(inst))
55  {
56  if(const SVFFunction* callee = getCallee(inst))
57  {
58  const CallICFGNode* callBlockNode = icfg->getCallICFGNode(inst);
59  const SVFFunction* caller = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
60  callgraph->addDirectCallGraphEdge(callBlockNode,caller,callee);
61  }
62  }
63  }
64  }
65 
66  return callgraph;
67 }

69 PTACallGraph* ThreadCallGraphBuilder::buildThreadCallGraph(SVFModule* svfModule)
70 {
71 
72  buildCallGraph(svfModule);
73 
74  ThreadCallGraph* cg = dyn_cast<ThreadCallGraph>(callgraph);
75  assert(cg && "not a thread callgraph?");
76 
77  ThreadAPI* tdAPI = ThreadAPI::getThreadAPI();
78  for (SVFModule::llvm_const_iterator fi = svfModule->llvmFunBegin(), efi = svfModule->llvmFunEnd(); fi != efi; ++fi)
79  {
80  const Function *fun = *fi;
81  for (const_inst_iterator II = inst_begin(*fun), E = inst_end(*fun); II != E; ++II)
82  {
83  const Instruction *inst = &*II;
84  if (tdAPI->isTDFork(inst))
85  {
86  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
87  cg->addForksite(cs);
88  const Function* forkee = SVFUtil::dyn_cast<Function>(tdAPI->getForkedFun(inst));
89  if (forkee)
90  {
91  cg->addDirectForkEdge(cs);
92  }
93  // indirect call to the start routine function
94  else
95  {
96  cg->addThreadForkEdgeSetMap(cs,nullptr);
97  }
98  }
99  else if (tdAPI->isHareParFor(inst))
100  {
101  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
102  cg->addParForSite(cs);
103  const Function* taskFunc = SVFUtil::dyn_cast<Function>(tdAPI->getTaskFuncAtHareParForSite(inst));
104  if (taskFunc)
105  {
106  cg->addDirectParForEdge(cs);
107  }
108  // indirect call to the start routine function
109  else
110  {
111  cg->addHareParForEdgeSetMap(cs,nullptr);
112  }
113  }
114  }
115  }
116  // record join sites
117  for (SVFModule::llvm_const_iterator fi = svfModule->llvmFunBegin(), efi = svfModule->llvmFunEnd(); fi != efi; ++fi)
118  {
119  const Function *fun = *fi;
120  for (const_inst_iterator II = inst_begin(*fun), E = inst_end(*fun); II != E; ++II)
121  {
122  const Instruction *inst = &*II;
123  if (tdAPI->isTDJoin(inst))
124  {
125  const CallICFGNode* cs = icfg->getCallICFGNode(inst);
126  cg->addJoinsite(cs);
127  }
128  }
129  }
130 
131  return cg;
132 }
