42 /*
43  svf.main() is used to model the real entry point of a C++ program, which
44  initializes all global C++ objects and then call main().
45 
46  LLVM may generate two global arrays @llvm.global_ctors and @llvm.global_dtors
47  that contain constructor and destructor functions for global variables. They
48  are not called explicitly, so we have to add them in the svf.main function.
49  The order to call these constructor and desctructor functions are also
50  specified in the global arrays.
51 
52  Related part in LLVM language reference:
53  https://llvm.org/docs/LangRef.html#the-llvm-global-ctors-global-variable
54 
55  For example, given a "int main(int argc, char * argv[])", the corresponding
56  svf.main will be generated as follows:
57  define void @svf.main(i32, i8**, i8**) {
58  entry:
59  call void @ctor1()
60  call void @ctor2()
61  %3 = call i32 @main(i32 %0, i8** %1)
62  call void @dtor1()
63  call void @dtor2()
64  ret void
65  }
66 */
67 
68 #define SVF_MAIN_FUNC_NAME "svf.main"
69 #define SVF_GLOBAL_CTORS "llvm.global_ctors"
70 #define SVF_GLOBAL_DTORS "llvm.global_dtors"

128 void LLVMModuleSet::build()
129 {
130  buildFunToFunMap();
131  initialize();
132  buildGlobalDefToRepMap();
133  if(preProcessed==false)
134  prePassSchedule();
135  const SVFModule::FunctionSetType& functions = svfModule->getFunctionSet();
136  for (SVFModule::FunctionSetType::const_iterator func_iter = functions.begin(); func_iter != functions.end(); func_iter++)
137  {
138  const SVFFunction* func = *func_iter;
139  const bool isUncalledFunction = LLVMUtil::isUncalledFunction(func->getLLVMFun());
140  const bool isNotRetFunction = LLVMUtil::functionDoesNotRet(func->getLLVMFun());
141  SVFFunction* svffun = const_cast<SVFFunction*>(func);
142  svffun->setIsNotRet(isNotRetFunction);
143  svffun->setIsUncalledFunction(isUncalledFunction);
144  if (SVFUtil::isExtCall(func) == false)
145  {
146  std::vector<const BasicBlock*> reachableBBs;
147  LLVMUtil::getFunReachableBBs(func, reachableBBs);
148  BasicBlock *exitBB = const_cast<BasicBlock*>(LLVMUtil::getFunExitBB(func));
149  svffun->setReachableBBs(reachableBBs);
150  svffun->setExitBB(exitBB);
151 
152  //process and stored dt & df
153  DominatorTree dt;
154  DominanceFrontier df;
155  dt.recalculate(*svffun->getLLVMFun());
156  df.analyze(dt);
157  LoopInfo loopInfo = LoopInfo(dt);
158  PostDominatorTree pdt = PostDominatorTree(*(func->getLLVMFun()));
159  Map<const BasicBlock*,Set<const BasicBlock*>> & dfBBsMap = svffun->getDomFrontierMap();
160  for (DominanceFrontierBase::const_iterator dfIter = df.begin(), eDfIter = df.end(); dfIter != eDfIter; dfIter++)
161  {
162  const BasicBlock* keyBB = dfIter->first;
163  const std::set<BasicBlock *>& domSet = dfIter->second;
164  Set<const BasicBlock*>& valueBasicBlocks = dfBBsMap[keyBB];
165  for (const BasicBlock* bbValue:domSet)
166  {
167  valueBasicBlocks.insert(bbValue);
168  }
169  }
170  for (Function::const_iterator bit = svffun->getLLVMFun()->begin(), ebit = svffun->getLLVMFun()->end(); bit != ebit; ++bit)
171  {
172  const BasicBlock *bb = &*bit;
173  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
174  {
175  DomTreeNode::iterator DI = dtNode->begin();
176  if (DI != dtNode->end())
177  {
178  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end(); DI != DE; ++DI)
179  {
180  svffun->getDomTreeMap()[bb].insert((*DI)->getBlock());
181  }
182  }
183  else
184  {
185  svffun->getDomTreeMap()[bb] = Set<const BasicBlock *>();
186  }
187  }
188 
189  if(DomTreeNode * pdtNode = pdt.getNode(const_cast<BasicBlock*>(bb)))
190  {
191  DomTreeNode::iterator DI = pdtNode->begin();
192  if (DI != pdtNode->end())
193  {
194  for (DomTreeNode::iterator DI = pdtNode->begin(), DE = pdtNode->end(); DI != DE; ++DI)
195  {
196  svffun->getPostDomTreeMap()[bb].insert((*DI)->getBlock());
197  }
198  }
199  else
200  {
201  svffun->getPostDomTreeMap()[bb] = Set<const BasicBlock *>();
202  }
203  }
204  if (const Loop *loop = loopInfo.getLoopFor(bb))
205  {
206  for (BasicBlock* loopBlock:loop->getBlocks())
207  {
208  svffun->addToBB2LoopMap(bb,loopBlock);
209  }
210  }
211  }
212  }
213  }

220 {
222  std::unique_ptr<BreakConstantGEPs> p1 = std::make_unique<BreakConstantGEPs>();
223  for (u32_t i = 0; i < LLVMModuleSet::getLLVMModuleSet()->getModuleNum(); ++i)
224  {
225  Module *module = LLVMModuleSet::getLLVMModuleSet()->getModule(i);
226  p1->runOnModule(*module);
227  }
228 
230  std::unique_ptr<UnifyFunctionExitNodes> p2 =
231  std::make_unique<UnifyFunctionExitNodes>();
232  for (u32_t i = 0; i < LLVMModuleSet::getLLVMModuleSet()->getModuleNum(); ++i)
233  {
234  Module *module = LLVMModuleSet::getLLVMModuleSet()->getModule(i);
235  for (auto F = module->begin(), E = module->end(); F != E; ++F)
236  {
237  Function &fun = *F;
238  if (fun.isDeclaration())
239  continue;
240  p2->runOnFunction(fun);
241  }

246 void LLVMModuleSet::loadModules(const std::vector<std::string> &moduleNameVec)
247 {
248 
249  // We read SVFIR from LLVM IR
250  if(Options::Graphtxt.getValue().empty())
251  {
252  if(moduleNameVec.empty())
253  {
254  SVFUtil::outs() << "no LLVM bc file is found!\n";
255  exit(0);
256  }
257  //assert(!moduleNameVec.empty() && "no LLVM bc file is found!");
258  }
259  // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
260  else
261  SVFModule::setPagFromTXT(Options::Graphtxt.getValue());
262 
263  //
264  // To avoid the following type bugs (t1 != t3) when parsing multiple modules,
265  // We should use only one LLVMContext object for multiple modules in the same thread.
266  // No such problem if only one module is processed by SVF.
267  // ------------------------------------------------------------------
268  // LLVMContext ctxa,ctxb;
269  // IntegerType * t1 = IntegerType::get(ctxa,32);
270  // IntegerType * t2 = IntegerType::get(ctxa,32);
271  // assert(t1 == t2);
272  // IntegerType * t3 = IntegerType::get(ctxb,32);
273  // IntegerType * t4 = IntegerType::get(ctxb,32);
274  // assert(t3 == t4);
275  // assert(t1 != t3);
276  // ------------------------------------------------------------------
277  //
278  cxts = std::make_unique<LLVMContext>();
279 
280  for (const std::string& moduleName : moduleNameVec)
281  {
282  SMDiagnostic Err;
283  std::unique_ptr<Module> mod = parseIRFile(moduleName, Err, *cxts);
284  if (mod == nullptr)
285  {
286  SVFUtil::errs() << "load module: " << moduleName << "failed!!\n\n";
287  Err.print("SVFModuleLoader", llvm::errs());
288  abort();
289  }
290  modules.emplace_back(*mod);
291  owned_modules.emplace_back(std::move(mod));
292  }

329 std::vector<const Function *> LLVMModuleSet::getLLVMGlobalFunctions(
330  const GlobalVariable *global)
331 {
332  // This function is used to extract constructor and destructor functions
333  // sorted by their priority from @llvm.global_ctors or @llvm.global_dtors.
334  // For example, given following @llvm.global_ctors, the returning sorted
335  // function list should be [ctor3, ctor1, ctor2].
336  // ------------------------------------------------------------------
337  // ; Each struct in the array is {priority, function, associated data}
338  //
339  // @llvm.global_ctors = appending global [2 x { i32, void ()*, i8* }]
340  // [{ i32, void ()*, i8* } { i32 1234, void ()* @ctor1.cpp, i8* null },
341  // { i32, void ()*, i8* } { i32 2345, void ()* @ctor2.cpp, i8* null },
342  // { i32, void ()*, i8* } { i32 345, void ()* @ctor3.cpp, i8* null }]
343  // ------------------------------------------------------------------
344  // TODO: According to LLVM language reference, if the third field is
345  // non-null, and points to a global variable or function, the initializer
346  // function will only run if the associated data from the current module is
347  // not discarded. However the associated data is currently ignored.
348 
349 
350  // This class is used for the priority queue that sorts the functions by
351  // their priority. Each object of this class stands for an item in the
352  // function array.
353  class LLVMGlobalFunction
354  {
355  public:
356  u32_t priority;
357  const Function *func;
358  LLVMGlobalFunction() {};
359  LLVMGlobalFunction(u32_t _priority, const Function *_func)
360  : priority(_priority), func(_func) {};
361  bool operator>(const LLVMGlobalFunction &other) const
362  {
363  if (priority != other.priority)
364  {
365  return priority > other.priority;
366  }
367  else
368  {
369  return func > other.func;
370  }
371  }
372  };
373 
374  std::priority_queue<LLVMGlobalFunction, std::vector<LLVMGlobalFunction>,
375  greater<LLVMGlobalFunction>>
376  queue;
377  std::vector<const Function *> result;
378 
379  // The @llvm.global_ctors/dtors global variable is an array of struct. Each
380  // struct has three fields: {i32 priority, void ()* @ctor/dtor, i8* @data}.
381  // First get the array here.
382  if(const ConstantArray *globalFuncArray =
383  SVFUtil::dyn_cast<ConstantArray>(global->getInitializer()))
384  {
385  // Get each struct in the array.
386  for (unsigned int i = 0; i < globalFuncArray->getNumOperands(); ++i)
387  {
388  if (
389  const ConstantStruct *globalFuncItem =
390  SVFUtil::dyn_cast<ConstantStruct>(
391  globalFuncArray->getOperand(i)))
392  {
393 
394  // Extract priority and function from the struct
395  const ConstantInt *priority = SVFUtil::dyn_cast<ConstantInt>(
396  globalFuncItem->getOperand(0));
397  const Function *func = SVFUtil::dyn_cast<Function>(
398  globalFuncItem->getOperand(1));
399 
400  if (priority && func)
401  {
402  queue.push(LLVMGlobalFunction(priority
403  ->getZExtValue(),
404  func));
405  }
406  }
407  }
408  }
409 
410  // Generate a sorted vector of functions from the priority queue.
411  while (!queue.empty())
412  {
413  result.push_back(queue.top().func);
414  queue.pop();
415  }
416  return result;
417 }

419 void LLVMModuleSet::addSVFMain()
420 {
421  std::vector<const Function *> ctor_funcs;
422  std::vector<const Function *> dtor_funcs;
423  Function * orgMain = 0;
424  Module* mainMod = nullptr;
425 
426  for (Module &mod : modules)
427  {
428  // Collect ctor and dtor functions
429  for (Module::global_iterator it = mod.global_begin(),
430  eit = mod.global_end();
431  it != eit; ++it)
432  {
433  const GlobalVariable *global = &*it;
434 
435  if (global->getName().equals(SVF_GLOBAL_CTORS) &&
436  global->hasInitializer())
437  {
438  ctor_funcs = getLLVMGlobalFunctions(global);
439  }
440  else if (global->getName().equals(SVF_GLOBAL_DTORS) &&
441  global->hasInitializer())
442  {
443  dtor_funcs = getLLVMGlobalFunctions(global);
444  }
445  }
446 
447  // Find main function
448  for (auto &func : mod)
449  {
450  if (func.getName().equals(SVF_MAIN_FUNC_NAME))
451  assert(false && SVF_MAIN_FUNC_NAME " already defined");
452  if(func.getName().equals("main"))
453  {
454  orgMain = &func;
455  mainMod = &mod;
456  }
457  }
458  }
459 
460  // Only create svf.main when the original main function is found, and also
461  // there are global contructor or destructor functions.
462  if (orgMain && getModuleNum() > 0 &&
463  (ctor_funcs.size() > 0 || dtor_funcs.size() > 0))
464  {
465  assert(mainMod && "Module with main function not found.");
466  Module & M = *mainMod;
467  // char **
468  Type * i8ptr2 = PointerType::getInt8PtrTy(M.getContext())->getPointerTo();
469  Type * i32 = IntegerType::getInt32Ty(M.getContext());
470  // define void @svf.main(i32, i8**, i8**)
471 #if (LLVM_VERSION_MAJOR >= 9)
472  FunctionCallee svfmainFn = M.getOrInsertFunction(
473  SVF_MAIN_FUNC_NAME,
474  Type::getVoidTy(M.getContext()),
475  i32,i8ptr2,i8ptr2
476  );
477  Function *svfmain = SVFUtil::dyn_cast<Function>(svfmainFn.getCallee());
478 #else
479  Function *svfmain = SVFUtil::dyn_cast<Function>(M.getOrInsertFunction(
480  SVF_MAIN_FUNC_NAME,
481  Type::getVoidTy(M.getContext()),
482  i32,i8ptr2,i8ptr2
483  ));
484 #endif
485  svfmain->setCallingConv(llvm::CallingConv::C);
486  BasicBlock* block = BasicBlock::Create(M.getContext(), "entry", svfmain);
487  IRBuilder Builder(block);
488  // emit "call void @ctor()". ctor_funcs is sorted so the functions are
489  // emitted in the order of priority
490  for(auto & ctor: ctor_funcs)
491  {
492  auto target = M.getOrInsertFunction(
493  ctor->getName(),
494  Type::getVoidTy(M.getContext())
495  );
496  Builder.CreateCall(target);
497  }
498  // main() should be called after all ctor functions and before dtor
499  // functions.
500  Function::arg_iterator arg_it = svfmain->arg_begin();
501  Value * args[] = {arg_it, arg_it + 1, arg_it + 2 };
502  size_t cnt = orgMain->arg_size();
503  assert(cnt <= 3 && "Too many arguments for main()");
504  Builder.CreateCall(orgMain, llvm::ArrayRef<Value*>(args,args + cnt));
505  // emit "call void @dtor()". dtor_funcs is sorted so the functions are
506  // emitted in the order of priority
507  for (auto &dtor : dtor_funcs)
508  {
509  auto target = M.getOrInsertFunction(
510  dtor->getName(),
511  Type::getVoidTy(M.getContext())
512  );
513  Builder.CreateCall(target);
514  }
515  // return;
516  Builder.CreateRetVoid();
517  }
518 }

521 void LLVMModuleSet::buildFunToFunMap()
522 {
523  Set<Function*> funDecls, funDefs;
524  OrderedSet<string> declNames, defNames, intersectNames;
525  typedef Map<string, Function*> NameToFunDefMapTy;
526  typedef Map<string, Set<Function*>> NameToFunDeclsMapTy;
527 
528  for (SVFModule::LLVMFunctionSetType::iterator it = svfModule->llvmFunBegin(),
529  eit = svfModule->llvmFunEnd(); it != eit; ++it)
530  {
531  Function *fun = *it;
532  if (fun->isDeclaration())
533  {
534  funDecls.insert(fun);
535  declNames.insert(fun->getName().str());
536  }
537  else
538  {
539  funDefs.insert(fun);
540  defNames.insert(fun->getName().str());
541  }
542  }
543  // Find the intersectNames
544  OrderedSet<string>::iterator declIter, defIter;
545  declIter = declNames.begin();
546  defIter = defNames.begin();
547  while (declIter != declNames.end() && defIter != defNames.end())
548  {
549  if (*declIter < *defIter)
550  {
551  declIter++;
552  }
553  else
554  {
555  if (!(*defIter < *declIter))
556  {
557  intersectNames.insert(*declIter);
558  declIter++;
559  }
560  defIter++;
561  }
562  }
563 
565  NameToFunDefMapTy nameToFunDefMap;
566  for (Set<Function*>::iterator it = funDefs.begin(),
567  eit = funDefs.end(); it != eit; ++it)
568  {
569  Function *fdef = *it;
570  string funName = fdef->getName().str();
571  if (intersectNames.find(funName) == intersectNames.end())
572  continue;
573  nameToFunDefMap[funName] = fdef;
574  }
575 
577  NameToFunDeclsMapTy nameToFunDeclsMap;
578  for (Set<Function*>::iterator it = funDecls.begin(),
579  eit = funDecls.end(); it != eit; ++it)
580  {
581  Function *fdecl = *it;
582  string funName = fdecl->getName().str();
583  if (intersectNames.find(funName) == intersectNames.end())
584  continue;
585  NameToFunDeclsMapTy::iterator mit = nameToFunDeclsMap.find(funName);
586  if (mit == nameToFunDeclsMap.end())
587  {
588  Set<Function*> decls;
589  decls.insert(fdecl);
590  nameToFunDeclsMap[funName] = decls;
591  }
592  else
593  {
594  Set<Function*> &decls = mit->second;
595  decls.insert(fdecl);
596  }
597  }
598 
600  for (Set<Function*>::iterator it = funDecls.begin(),
601  eit = funDecls.end(); it != eit; ++it)
602  {
603  const Function *fdecl = *it;
604  string funName = fdecl->getName().str();
605  if (intersectNames.find(funName) == intersectNames.end())
606  continue;
607  NameToFunDefMapTy::iterator mit = nameToFunDefMap.find(funName);
608  if (mit == nameToFunDefMap.end())
609  continue;
610  FunDeclToDefMap[fdecl] = mit->second;
611  }
612 
614  for (Set<Function*>::iterator it = funDefs.begin(),
615  eit = funDefs.end(); it != eit; ++it)
616  {
617  const Function *fdef = *it;
618  string funName = fdef->getName().str();
619  if (intersectNames.find(funName) == intersectNames.end())
620  continue;
621  NameToFunDeclsMapTy::iterator mit = nameToFunDeclsMap.find(funName);
622  if (mit == nameToFunDeclsMap.end())
623  continue;
624  std::vector<const Function*>& decls = FunDefToDeclsMap[fdef];
625  for (Set<Function*>::iterator sit = mit->second.begin(),
626  seit = mit->second.end(); sit != seit; ++sit)
627  {
628  decls.push_back(*sit);
629  }
630  }
631 }

633 void LLVMModuleSet::buildGlobalDefToRepMap()
634 {
635  typedef Map<string, Set<GlobalVariable*>> NameToGlobalsMapTy;
636  NameToGlobalsMapTy nameToGlobalsMap;
637  for (SVFModule::global_iterator it = svfModule->global_begin(),
638  eit = svfModule->global_end(); it != eit; ++it)
639  {
640  GlobalVariable *global = *it;
641  if (global->hasPrivateLinkage())
642  continue;
643  string name = global->getName().str();
644  NameToGlobalsMapTy::iterator mit = nameToGlobalsMap.find(name);
645  if (mit == nameToGlobalsMap.end())
646  {
647  Set<GlobalVariable*> globals;
648  globals.insert(global);
649  nameToGlobalsMap[name] = globals;
650  }
651  else
652  {
653  Set<GlobalVariable*> &globals = mit->second;
654  globals.insert(global);
655  }
656  }
657 
658  for (NameToGlobalsMapTy::iterator it = nameToGlobalsMap.begin(),
659  eit = nameToGlobalsMap.end(); it != eit; ++it)
660  {
661  Set<GlobalVariable*> &globals = it->second;
662  GlobalVariable *rep = *(globals.begin());
663  Set<GlobalVariable*>::iterator repit = globals.begin();
664  while (repit != globals.end())
665  {
666  GlobalVariable *cur = *repit;
667  if (cur->hasInitializer())
668  {
669  rep = cur;
670  break;
671  }
672  repit++;
673  }
