303 ICFGEdge* ICFG::getICFGEdge(const ICFGNode* src, const ICFGNode* dst, ICFGEdge::ICFGEdgeK kind)
304 {
305 
306  ICFGEdge * edge = nullptr;
307  Size_t counter = 0;
308  for (ICFGEdge::ICFGEdgeSetTy::iterator iter = src->OutEdgeBegin();
309  iter != src->OutEdgeEnd(); ++iter)
310  {
311  if ((*iter)->getDstID() == dst->getId() && (*iter)->getEdgeKind() == kind)
312  {
313  counter++;
314  edge = (*iter);
315  }
316  }
317  assert(counter <= 1 && "there's more than one edge between two ICFG nodes");
318  return edge;
319 
320 }

414 void ICFG::updateCallGraph(PTACallGraph* callgraph)
415 {
416  PTACallGraph::CallEdgeMap::const_iterator iter = callgraph->getIndCallMap().begin();
417  PTACallGraph::CallEdgeMap::const_iterator eiter = callgraph->getIndCallMap().end();
418  for (; iter != eiter; iter++)
419  {
420  const CallBlockNode* callBlock = iter->first;
421  const Instruction* cs = callBlock->getCallSite();
422  assert(callBlock->isIndirectCall() && "this is not an indirect call?");
423  const PTACallGraph::FunctionSet & functions = iter->second;
424  for (PTACallGraph::FunctionSet::const_iterator func_iter = functions.begin(); func_iter != functions.end(); func_iter++)
425  {
426  const SVFFunction* callee = *func_iter;
427  CallBlockNode* CallBlockNode = getCallBlockNode(cs);
428  if (!isExtCall(callee))
429  {
430  FunEntryBlockNode* calleeEntryNode = getFunEntryICFGNode(callee);
431  addCallEdge(CallBlockNode, calleeEntryNode, cs);
432  RetBlockNode* retBlockNode = getRetBlockNode(cs);
433  FunExitBlockNode* calleeExitNode = getFunExitICFGNode(callee);
434  addRetEdge(calleeExitNode, retBlockNode, cs);
435  }
436  }
437  }
