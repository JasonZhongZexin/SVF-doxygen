42 void ConstraintGraph::buildCG()
43 {
44 
45  // initialize nodes
46  for(SVFIR::iterator it = pag->begin(), eit = pag->end(); it!=eit; ++it)
47  {
48  addConstraintNode(new ConstraintNode(it->first), it->first);
49  }
50 
51  // initialize edges
52  PAGEdge::SVFStmtSetTy& addrs = getPAGEdgeSet(PAGEdge::Addr);
53  for (PAGEdge::SVFStmtSetTy::iterator iter = addrs.begin(), eiter =
54  addrs.end(); iter != eiter; ++iter)
55  {
56  const AddrStmt* edge = SVFUtil::cast<AddrStmt>(*iter);
57  addAddrCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
58  }
59 
60  PAGEdge::SVFStmtSetTy& copys = getPAGEdgeSet(PAGEdge::Copy);
61  for (PAGEdge::SVFStmtSetTy::iterator iter = copys.begin(), eiter =
62  copys.end(); iter != eiter; ++iter)
63  {
64  const CopyStmt* edge = SVFUtil::cast<CopyStmt>(*iter);
65  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
66  }
67 
68  PAGEdge::SVFStmtSetTy& phis = getPAGEdgeSet(PAGEdge::Phi);
69  for (PAGEdge::SVFStmtSetTy::iterator iter = phis.begin(), eiter =
70  phis.end(); iter != eiter; ++iter)
71  {
72  const PhiStmt* edge = SVFUtil::cast<PhiStmt>(*iter);
73  for(const auto opVar : edge->getOpndVars())
74  addCopyCGEdge(opVar->getId(),edge->getResID());
75  }
76

165 AddrCGEdge* ConstraintGraph::addAddrCGEdge(NodeID src, NodeID dst)
166 {
167  ConstraintNode* srcNode = getConstraintNode(src);
168  ConstraintNode* dstNode = getConstraintNode(dst);
169  if(hasEdge(srcNode,dstNode,ConstraintEdge::Addr))
170  return nullptr;
171  AddrCGEdge* edge = new AddrCGEdge(srcNode, dstNode, edgeIndex++);
172  bool added = AddrCGEdgeSet.insert(edge).second;
173  assert(added && "not added??");
174  srcNode->addOutgoingAddrEdge(edge);
175  dstNode->addIncomingAddrEdge(edge);

280 void ConstraintGraph::reTargetDstOfEdge(ConstraintEdge* edge, ConstraintNode* newDstNode)
281 {
282  NodeID newDstNodeID = newDstNode->getId();
283  NodeID srcId = edge->getSrcID();
284  if(LoadCGEdge* load = SVFUtil::dyn_cast<LoadCGEdge>(edge))
285  {
286  removeLoadEdge(load);
287  addLoadCGEdge(srcId,newDstNodeID);
288  }
289  else if(StoreCGEdge* store = SVFUtil::dyn_cast<StoreCGEdge>(edge))
290  {
291  removeStoreEdge(store);
292  addStoreCGEdge(srcId,newDstNodeID);
293  }
294  else if(CopyCGEdge* copy = SVFUtil::dyn_cast<CopyCGEdge>(edge))
295  {
296  removeDirectEdge(copy);
297  addCopyCGEdge(srcId,newDstNodeID);
298  }
299  else if(NormalGepCGEdge* gep = SVFUtil::dyn_cast<NormalGepCGEdge>(edge))
300  {
301  const LocationSet ls = gep->getLocationSet();
302  removeDirectEdge(gep);
303  addNormalGepCGEdge(srcId,newDstNodeID,ls);
304  }
305  else if(VariantGepCGEdge* gep = SVFUtil::dyn_cast<VariantGepCGEdge>(edge))
306  {
307  removeDirectEdge(gep);
308  addVariantGepCGEdge(srcId,newDstNodeID);
309  }
310  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
311  {
312  removeAddrEdge(addr);
313  }
314  else
315  assert(false && "no other edge type!!");
316 }

416 bool ConstraintGraph::moveInEdgesToRepNode(ConstraintNode* node, ConstraintNode* rep )
417 {
418  std::vector<ConstraintEdge*> sccEdges;
419  std::vector<ConstraintEdge*> nonSccEdges;
420  for (ConstraintNode::const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit;
421  ++it)
422  {
423  ConstraintEdge* subInEdge = *it;
424  if(sccRepNode(subInEdge->getSrcID()) != rep->getId())
425  nonSccEdges.push_back(subInEdge);
426  else
427  {
428  sccEdges.push_back(subInEdge);
429  }
430  }
431  // if this edge is outside scc, then re-target edge dst to rep
432  while(!nonSccEdges.empty())
433  {
434  ConstraintEdge* edge = nonSccEdges.back();
435  nonSccEdges.pop_back();
436  reTargetDstOfEdge(edge,rep);
437  }
438 
439  bool criticalGepInsideSCC = false;
440  // if this edge is inside scc, then remove this edge and two end nodes
441  while(!sccEdges.empty())
442  {
443  ConstraintEdge* edge = sccEdges.back();
444  sccEdges.pop_back();
446  if(SVFUtil::isa<CopyCGEdge>(edge))
447  removeDirectEdge(edge);
448  else if (SVFUtil::isa<GepCGEdge>(edge))
449  {
450  // If the GEP is critical (i.e. may have a non-zero offset),
451  // then it brings impact on field-sensitivity.
452  if (!isZeroOffsettedGepCGEdge(edge))
453  {
454  criticalGepInsideSCC = true;
455  }
456  removeDirectEdge(edge);
457  }
458  else if(SVFUtil::isa<LoadCGEdge>(edge) || SVFUtil::isa<StoreCGEdge>(edge))
459  reTargetDstOfEdge(edge,rep);
460  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
461  {
462  removeAddrEdge(addr);
463  }
464  else
465  assert(false && "no such edge");
466  }
467  return criticalGepInsideSCC;
468 }

541 void ConstraintGraph::print()
542 {
543 
544  outs() << "-----------------ConstraintGraph--------------------------------------\n";
545 
546  ConstraintEdge::ConstraintEdgeSetTy& addrs = this->getAddrCGEdges();
547  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = addrs.begin(),
548  eiter = addrs.end(); iter != eiter; ++iter)
549  {
550  outs() << (*iter)->getSrcID() << " -- Addr --> " << (*iter)->getDstID()
551  << "\n";
552  }
553 
554  ConstraintEdge::ConstraintEdgeSetTy& directs = this->getDirectCGEdges();
555  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = directs.begin(),
556  eiter = directs.end(); iter != eiter; ++iter)
557  {
558  if (CopyCGEdge* copy = SVFUtil::dyn_cast<CopyCGEdge>(*iter))
559  {
560  outs() << copy->getSrcID() << " -- Copy --> " << copy->getDstID()
561  << "\n";
562  }
563  else if (NormalGepCGEdge* ngep = SVFUtil::dyn_cast<NormalGepCGEdge>(*iter))
564  {
565  outs() << ngep->getSrcID() << " -- NormalGep (" << ngep->getFieldOffset()
566  << ") --> " << ngep->getDstID() << "\n";
567  }
568  else if (VariantGepCGEdge* vgep = SVFUtil::dyn_cast<VariantGepCGEdge>(*iter))
569  {
570  outs() << vgep->getSrcID() << " -- VarintGep --> "
571  << vgep->getDstID() << "\n";
572  }
573  else
574  assert(false && "wrong constraint edge kind!");
575  }
576 
577  ConstraintEdge::ConstraintEdgeSetTy& loads = this->getLoadCGEdges();
578  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = loads.begin(),
579  eiter = loads.end(); iter != eiter; ++iter)
580  {
581  outs() << (*iter)->getSrcID() << " -- Load --> " << (*iter)->getDstID()
582  << "\n";
583  }
584 
585  ConstraintEdge::ConstraintEdgeSetTy& stores = this->getStoreCGEdges();
586  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = stores.begin(),
587  eiter = stores.end(); iter != eiter; ++iter)
588  {
589  outs() << (*iter)->getSrcID() << " -- Store --> " << (*iter)->getDstID()
590  << "\n";
591  }
592 
593  outs()
594  << "--------------------------------------------------------------\n";
595
