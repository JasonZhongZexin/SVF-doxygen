262  {
263  NodeBS tds;
264  if(hasParentThread(tid) == false)
265  return tds;
266 
267  FIFOWorkList<NodeID> worklist;
268  worklist.push(getParentThread(tid));
269 
270  while(!worklist.empty())
271  {
272  NodeID t = worklist.pop();
273  if(tds.test_and_set(t))
274  {
275  if(hasParentThread(t))
276  worklist.push(getParentThread(t));
277  }
278  }
