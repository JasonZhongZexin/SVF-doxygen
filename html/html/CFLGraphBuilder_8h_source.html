61  template<class N, class E>
62  void build(GenericGraph<N,E>* graph, CFLGraph* cflGraph)
63  {
64  for(auto it = graph->begin(); it!= graph->end(); it++)
65  {
66  CFLNode* node = new CFLNode((*it).first);
67  cflGraph->addCFLNode((*it).first, node);
68  }
69  for(auto it = graph->begin(); it!= graph->end(); it++)
70  {
71  N* node = (*it).second;
72  for(E* edge : node->getOutEdges())
73  {
74  cflGraph->addCFLEdge(cflGraph->getGNode(edge->getSrcID()), cflGraph->getGNode(edge->getDstID()), edge->getEdgeKind());

80  template<class N, class E>
81  CFLGraph* buildBigraph(GenericGraph<N,E>* graph, Kind startKind, GrammarBase *grammar)
82  {
83  CFLGraph *cflGraph = new CFLGraph(startKind);
84  externMap = true;
85  for(auto pairV : grammar->getTerminals())
86  {
87  if(label2KindMap.find(pairV.first) == label2KindMap.end())
88  {
89  label2KindMap.insert(pairV);
90  }
91  if(kind2LabelMap.find(pairV.second) == kind2LabelMap.end())
92  {
93  kind2LabelMap.insert(make_pair(pairV.second, pairV.first));
94  }
95  }
96  for(auto pairV : grammar->getNonterminals())
97  {
98  if(label2KindMap.find(pairV.first) == label2KindMap.end())
99  {
100  label2KindMap.insert(pairV);
101  }
102  if(kind2LabelMap.find(pairV.second) == kind2LabelMap.end())
103  {
104  kind2LabelMap.insert(make_pair(pairV.second, pairV.first));
105  }
106  }
107  for(auto it = graph->begin(); it!= graph->end(); it++)
108  {
109  CFLNode* node = new CFLNode((*it).first);
110  cflGraph->addCFLNode((*it).first, node);
111  }
112  for(auto it = graph->begin(); it!= graph->end(); it++)
113  {
114  N* node = (*it).second;
115  for(E* edge : node->getOutEdges())
116  {
117  CFLGrammar::Kind edgeLabel = edge->getEdgeKind();
118  cflGraph->addCFLEdge(cflGraph->getGNode(edge->getSrcID()), cflGraph->getGNode(edge->getDstID()), edgeLabel);
119  std::string key = kind2LabelMap[edge->getEdgeKind()];
120  key.append("bar");
121  cflGraph->addCFLEdge(cflGraph->getGNode(edge->getDstID()), cflGraph->getGNode(edge->getSrcID()), label2KindMap[key]);
122  }
123  }
124  return cflGraph;
125  }
