175  {
176 
177  size_t h = exeState.getVarToVal().size() * 2;
178  SVF::Hash<SVF::u32_t> hf;
179  for (const auto &t: exeState.getVarToVal())
180  {
181  h ^= hf(t.first) + 0x9e3779b9 + (h << 6) + (h >> 2);
182  h ^= hf(t.second.id()) + 0x9e3779b9 + (h << 6) + (h >> 2);
183  }
184 
185  size_t h2 = exeState.getVarToVal().size() * 2;
186 
187  for (const auto &t: exeState.getLocToVal())
188  {
189  h2 ^= hf(t.first) + 0x9e3779b9 + (h2 << 6) + (h2 >> 2);
190  h2 ^= hf(t.second.id()) + 0x9e3779b9 + (h2 << 6) + (h2 >> 2);
191  }
192  SVF::Hash<std::pair<SVF::u32_t, SVF::u32_t>> pairH;
193 
194  return pairH(make_pair(pairH(make_pair(h, h2)), exeState.getPathConstraint().id()));
