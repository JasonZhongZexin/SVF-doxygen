41 CFLGraph * CFLGraphBuilder::buildFromDot(std::string fileName, GrammarBase *grammar)
42 {
43  CFLGraph *cflGraph = new CFLGraph();
44  cflGraph->setMap(grammar);
45  std::cout << "Building CFL Graph from dot file: " << fileName << "..\n";
46  std::string lineString;
47  std::ifstream inputFile(fileName);
48 
49  std::regex reg("Node(\\w+)\\s*->\\s*Node(\\w+)\\s*\\[.*label=(.*)\\]");
50 
51  std::cout << std::boolalpha;
52  u32_t lineNum = 0 ;
53  cflGraph->current = cflGraph->label2SymMap.size();
54 
55  while (getline(inputFile, lineString))
56  {
57  lineNum += 1;
58  std::smatch matches;
59  if (std::regex_search(lineString, matches, reg))
60  {
61  CFLNode *src, *dst;
62  if (cflGraph->hasGNode(std::stoul(matches.str(1), nullptr, 16))==false)
63  {
64  src = new CFLNode(std::stoul(matches.str(1), nullptr, 16));
65  cflGraph->addCFLNode(src->getId(), src);
66  }
67  else
68  {
69  src = cflGraph->getGNode(std::stoul(matches.str(1), nullptr, 16));
70  }
71  if (cflGraph->hasGNode(std::stoul(matches.str(2), nullptr, 16))==false)
72  {
73  dst = new CFLNode(std::stoul(matches.str(2), nullptr, 16));
74  cflGraph->addCFLNode(dst->getId(), dst);
75  }
76  else
77  {
78  dst = cflGraph->getGNode(std::stoul(matches.str(2), nullptr, 16));
79  }
80  if (cflGraph->externMap == false)
81  {
82  if (cflGraph->label2SymMap.find(matches.str(3)) != cflGraph->label2SymMap.end())
83  {
84  cflGraph->addCFLEdge(src, dst, cflGraph->label2SymMap[matches.str(3)]);
85  }
86  else
87  {
88  cflGraph->label2SymMap.insert({matches.str(3), cflGraph->current++});
89  cflGraph->addCFLEdge(src, dst, cflGraph->label2SymMap[matches.str(3)]);
90  }
91  }
92  else
93  {
94  if (cflGraph->label2SymMap.find(matches.str(3)) != cflGraph->label2SymMap.end())
95  {
96  cflGraph->addCFLEdge(src, dst, cflGraph->label2SymMap[matches.str(3)]);
97  }
98  else
99  {
100  if(Options::FlexSymMap == true)
101  {
102  cflGraph->label2SymMap.insert({matches.str(3), cflGraph->current++});
103  cflGraph->addCFLEdge(src, dst, cflGraph->label2SymMap[matches.str(3)]);
104  }
105  else
106  {
107  std::string msg = "In line " + std::to_string(lineNum) + " sym can not find in grammar, please correct the input dot or set --flexsymmap.";
108  SVFUtil::errMsg(msg);
109  std::cout << msg;
110  abort();
111  }
112  }
113  }
114  }
115  }
116  inputFile.close();
117  return cflGraph;
118 }
