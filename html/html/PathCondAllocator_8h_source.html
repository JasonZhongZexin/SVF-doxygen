253  inline bool setCFCond(const BasicBlock* bb, Condition* cond)
254  {
255  BBToCondMap::iterator it = bbToCondMap.find(bb);
256  // until a fixed-point is reached (condition is not changed)
257  if(it!=bbToCondMap.end() && isEquivalentBranchCond(it->second, cond))
258  return false;
259 
260  bbToCondMap[bb] = cond;
261  return true;
262  }

264  {
265  BBToCondMap::const_iterator it = bbToCondMap.find(bb);
