127  inline bool hasEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)
128  {
129  ConstraintEdge edge(src,dst,kind);
130  if(kind == ConstraintEdge::Copy ||
131  kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
132  return directEdgeSet.find(&edge) != directEdgeSet.end();
133  else if(kind == ConstraintEdge::Addr)
134  return AddrCGEdgeSet.find(&edge) != AddrCGEdgeSet.end();
135  else if(kind == ConstraintEdge::Store)
136  return StoreCGEdgeSet.find(&edge) != StoreCGEdgeSet.end();
137  else if(kind == ConstraintEdge::Load)
138  return LoadCGEdgeSet.find(&edge) != LoadCGEdgeSet.end();
139  else
140  assert(false && "no other kind!");
141  return false;
142  }
143 
145  inline ConstraintEdge* getEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)
146  {
147  ConstraintEdge edge(src,dst,kind);
148  if(kind == ConstraintEdge::Copy || kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
149  {
150  auto eit = directEdgeSet.find(&edge);
151  return *eit;
152  }
153  else if(kind == ConstraintEdge::Addr)
154  {
155  auto eit = AddrCGEdgeSet.find(&edge);
156  return *eit;
157  }
158  else if(kind == ConstraintEdge::Store)
159  {
160  auto eit = StoreCGEdgeSet.find(&edge);
161  return *eit;
162  }
163  else if(kind == ConstraintEdge::Load)
164  {
165  auto eit = LoadCGEdgeSet.find(&edge);
166  return *eit;
167  }
168  else{
169  assert(false && "no other kind!");
170  return NULL;
171  }
172  }
