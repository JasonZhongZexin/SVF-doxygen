261  inline bool setCFCond(const BasicBlock* bb, Condition* cond)
262  {
263  BBToCondMap::iterator it = bbToCondMap.find(bb);
264  // until a fixed-point is reached (condition is not changed)
265  if(it!=bbToCondMap.end() && isEquivalentBranchCond(it->second, cond))
266  return false;
267 
268  bbToCondMap[bb] = cond;
269  return true;
270  }
271  inline Condition* getCFCond(const BasicBlock* bb) const
272  {
273  BBToCondMap::const_iterator it = bbToCondMap.find(bb);
274  if(it==bbToCondMap.end())
275  {
276  return getFalseCond();
277  }
278  return it->second;
279  }
