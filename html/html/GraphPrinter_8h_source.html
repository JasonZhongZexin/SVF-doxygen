88  const GraphType &GT)
89  {
91  typedef llvm::GraphTraits<GraphType> GTraits;
92

98  // Print each node name and its edges
99  node_iterator I = GTraits::nodes_begin(GT);
100  node_iterator E = GTraits::nodes_end(GT);

103  NodeRef *Node = *I;
