374  inline bool mustAlias(const CVar& var1, const CVar& var2)
375  {
376  if(isSameVar(var1,var2))
377  return true;
378 
379  bool singleton = !(isHeapMemObj(var1.get_id()) || isLocalVarInRecursiveFun(var1.get_id()));
380  if(isCondCompatible(var1.get_cond(),var2.get_cond(),singleton) == false)
381  return false;
382 
383  const CPtSet& cpts1 = getPts(var1);
384  const CPtSet& cpts2 = getPts(var2);
385  return (contains(cpts1,cpts2) && contains(cpts2,cpts1));
386  }
387 
388  // Whether cpts1 contains all points-to targets of pts2
389  bool contains(const CPtSet& cpts1, const CPtSet& cpts2)
390  {
391  if (cpts1.empty() || cpts2.empty())
392  return false;
393 
394  for (typename CPtSet::const_iterator it2 = cpts2.begin(); it2 != cpts2.end(); ++it2)
395  {
396  bool hasObj = false;
397  for (typename CPtSet::const_iterator it1 = cpts1.begin(); it1 != cpts1.end(); ++it1)
398  {
399  if(isSameVar(*it1,*it2))
400  {
401  hasObj = true;
402  break;
403  }
404  }
405  if(hasObj == false)
406  return false;
407  }
408  return true;
409  }
410 
412  bool isSameVar(const CVar& var1, const CVar& var2) const
413  {
414  if(var1.get_id() != var2.get_id())
415  return false;
416 
418  bool singleton = !(isHeapMemObj(var1.get_id()) || isLocalVarInRecursiveFun(var1.get_id()));
419  return isCondCompatible(var1.get_cond(),var2.get_cond(),singleton);
420  }

424  virtual void NormalizePointsTo()
425  {
426  normalized = true;
427  if (hasPtsMap())
428  {
429  const typename MutPTDataTy::PtsMap& ptsMap = getPtsMap();
430  for(typename MutPTDataTy::PtsMap::const_iterator it = ptsMap.begin(), eit=ptsMap.end(); it!=eit; ++it)
431  {
432  for(typename CPtSet::const_iterator cit = it->second.begin(), ecit=it->second.end(); cit!=ecit; ++cit)
433  {
434  ptrToBVPtsMap[(it->first).get_id()].set(cit->get_id());
435  objToNSRevPtsMap[cit->get_id()].insert((it->first).get_id());
436  ptrToCPtsMap[(it->first).get_id()].set(*cit);
437  }
438  }
439  }
440  else
441  {
442  assert(false && "CondPTAImpl::NormalizePointsTo: could not normalize points-to sets");
443  }

510  virtual inline AliasResult alias(const CPtSet& pts1, const CPtSet& pts2)
511  {
512  CPtSet cpts1;
513  expandFIObjs(pts1,cpts1);
514  CPtSet cpts2;
515  expandFIObjs(pts2,cpts2);
516  if (containBlackHoleNode(cpts1) || containBlackHoleNode(cpts2))
517  return llvm::MayAlias;
518  else if(this->getAnalysisTy()==PathS_DDA && contains(cpts1,cpts2) && contains(cpts2,cpts1))
519  {
520  return llvm::MustAlias;
521  }
522  else if(overlap(cpts1,cpts2))
523  return llvm::MayAlias;
524  else
525  return llvm::NoAlias;
526  }

551  void dumpTopLevelPtsTo()
552  {
553  for (OrderedNodeSet::iterator nIter = this->getAllValidPtrs().begin(); nIter != this->getAllValidPtrs().end(); ++nIter)
554  {
555  const PAGNode* node = this->getPAG()->getPAGNode(*nIter);
556  if (this->getPAG()->isValidTopLevelPtr(node))
557  {
558  if (SVFUtil::isa<DummyObjPN>(node))
559  {
560  SVFUtil::outs() << "##<Blackhole or constant> id:" << node->getId();
561  }
562  else if (!SVFUtil::isa<DummyValPN>(node))
563  {
564  SVFUtil::outs() << "##<" << node->getValue()->getName() << "> ";
565  //SVFUtil::outs() << "Source Loc: " << SVFUtil::getSourceLoc(node->getValue());
566  }
567 
568  const PointsTo& pts = getPts(node->getId());
569  SVFUtil::outs() << "\nNodeID " << node->getId() << " ";
570  if (pts.empty())
571  {
572  SVFUtil::outs() << "\t\tPointsTo: {empty}\n\n";
573  }
574  else
575  {
576  SVFUtil::outs() << "\t\tPointsTo: { ";
577  for (PointsTo::iterator it = pts.begin(), eit = pts.end(); it != eit; ++it)
578  SVFUtil::outs() << *it << " ";
579  SVFUtil::outs() << "}\n\n";
580  }
581  }
582  }
583 
584  SVFUtil::outs().flush();
585  }
