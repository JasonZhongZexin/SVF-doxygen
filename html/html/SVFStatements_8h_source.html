138  static inline GEdgeFlag makeEdgeFlagWithAddionalOpnd(GEdgeKind k, const SVFVar* var)
139  {
140  Var2LabelMap::const_iterator iter = var2LabelMap.find(var);
141  u64_t label = (iter != var2LabelMap.end()) ?
142  iter->second : multiOpndLabelCounter++;
143  return (label << EdgeKindMaskBits) | k;
144  }

147  static inline GEdgeFlag makeEdgeFlagWithCallInst(GEdgeKind k, const ICFGNode* cs)
148  {
149  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(cs);
150  u64_t label = (iter != inst2LabelMap.end()) ?
151  iter->second : callEdgeLabelCounter++;
152  return (label << EdgeKindMaskBits) | k;
153  }
154 
157  static inline GEdgeFlag makeEdgeFlagWithStoreInst(GEdgeKind k, const ICFGNode* store)
158  {
159  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(store);
160  u64_t label = (iter != inst2LabelMap.end()) ?
161  iter->second : storeEdgeLabelCounter++;
162  return (label << EdgeKindMaskBits) | k;
163  }
