42 // Define a template that can be specialized by smart pointers to reflect the
43 // fact that they are automatically dereferenced, and are not involved with the
44 // template selection process... the default implementation is a noop.
45 //
46 template<typename From> struct simplify_type
47 {
48  using SimpleType = From; // The real type this represents...
49 
50  // An accessor to get the real value...
51  static SimpleType &getSimplifiedValue(From &Val)
52  {
53  return Val;
54  }
55 };
56 
57 template<typename From> struct simplify_type<const From>
58 {
59  using NonConstSimpleType = typename simplify_type<From>::SimpleType;
60  using SimpleType =
61  typename llvm::add_const_past_pointer<NonConstSimpleType>::type;
62  using RetType =
63  typename llvm::add_lvalue_reference_if_not_pointer<SimpleType>::type;
64 
65  static RetType getSimplifiedValue(const From& Val)
66  {
67  return simplify_type<From>::getSimplifiedValue(const_cast<From&>(Val));
68  }
69 };

71 // The core of the implementation of llvm::isa<X> is here; To and From should be
72 // the names of classes. This template can be specialized to customize the
73 // implementation of llvm::isa<> without rewriting it from scratch.
74 template <typename To, typename From, typename Enabler = void>
75 struct isa_impl
76 {
77  static inline bool doit(const From &Val)
78  {
79  return To::classof(&Val);
80  }

287 // cast<X> - Return the argument parameter cast to the specified type. This
288 // casting operator asserts that the type is correct, so it does not return null
289 // on failure. It does not allow a null argument (use cast_or_null for that).
290 // It is typically used like this:
291 //
292 // cast<Instruction>(myVal)->getParent()
293 //
294 template <class X, class Y>
295 inline typename std::enable_if<!is_simple_type<Y>::value,
296  typename cast_retty<X, const Y>::ret_type>::type
297  cast(const Y &Val)
298 {
299  assert(llvm::isa<X>(Val) && "cast<Ty>() argument of incompatible type!");
300  return cast_convert_val<
301  X, const Y, typename simplify_type<const Y>::SimpleType>::doit(Val);
302 }

304 template <class X, class Y>
305 inline typename cast_retty<X, Y>::ret_type cast(Y &Val)
306 {
307  assert(llvm::isa<X>(Val) && "cast<Ty>() argument of incompatible type!");
308  return cast_convert_val<X, Y,
309  typename simplify_type<Y>::SimpleType>::doit(Val);

320 template <class X, class Y>
321 inline typename cast_retty<X, std::unique_ptr<Y>>::ret_type
322  cast(std::unique_ptr<Y> &&Val)
323 {
324  assert(llvm::isa<X>(Val.get()) && "cast<Ty>() argument of incompatible type!");
325  using ret_type = typename cast_retty<X, std::unique_ptr<Y>>::ret_type;
326  return ret_type(
327  cast_convert_val<X, Y *, typename simplify_type<Y *>::SimpleType>::doit(
328  Val.release()));

331 // dyn_cast<X> - Return the argument parameter cast to the specified type. This
332 // casting operator returns null if the argument is of the wrong type, so it can
333 // be used to test for a type as well as cast if successful. This should be
334 // used in the context of an if statement like this:
335 //
336 // if (const Instruction *I = dyn_cast<Instruction>(myVal)) { ... }
337 //

343  dyn_cast(const Y &Val)
344 {
345  return llvm::isa<X>(Val) ? llvm::cast<X>(Val) : nullptr;
