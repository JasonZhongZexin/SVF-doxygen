331  void markCxtStmtFlag(const CxtStmt& tgr, const CxtStmt& src)
333  {
334  const CxtLockSet& srclockset = getCxtLockfromCxtStmt(src);
335  if(hasCxtLockfromCxtStmt(tgr)== false)
336  {
337  for(CxtLockSet::const_iterator it = srclockset.begin(), eit = srclockset.end(); it!=eit; ++it)
338  {
339  addCxtStmtToSpan(tgr,*it);
340  }
341  pushToCTSWorkList(tgr);
342  }
343  else
344  {
345  if(intersect(getCxtLockfromCxtStmt(tgr),srclockset))
346  pushToCTSWorkList(tgr);
347  }
348  }
349  bool intersect(CxtLockSet& tgrlockset, const CxtLockSet& srclockset)
350  {
351  CxtLockSet toBeDeleted;
352  for(CxtLockSet::const_iterator it = tgrlockset.begin(), eit = tgrlockset.end(); it!=eit; ++it)
353  {
354  if(srclockset.find(*it)==srclockset.end())
355  toBeDeleted.insert(*it);
356  }
357  for(CxtLockSet::const_iterator it = toBeDeleted.begin(), eit = toBeDeleted.end(); it!=eit; ++it)
358  {
359  tgrlockset.erase(*it);
360  }
361  if(!toBeDeleted.empty())
362  return true;
363  return false;
364  }
