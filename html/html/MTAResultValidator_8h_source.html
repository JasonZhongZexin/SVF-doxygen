272  void collectValidationTargets()
273  {
274  // Collect call sites of all RC_ACCESS function calls.
275  std::vector<const CallInst*> csInsts;
276  const Function *F;
277  for(auto it = M->llvmFunBegin(); it != M->llvmFunEnd(); it++){
278  const std::string fName = (*it)->getName().str();
279  if(fName.find(RC_ACCESS) != std::string::npos) {
280  F = (*it);
281  break;
282  }
283  }
284  if (!F) return;
285 
286  for (Value::const_use_iterator it = F->use_begin(), ie =
287  F->use_end(); it != ie; ++it)
288  {
289  const Use *u = &*it;
290  const Value *user = u->getUser();
291  const CallInst *csInst = SVFUtil::dyn_cast<CallInst>(user);
292  assert(csInst);
293  csInsts.push_back(csInst);
294  }
295  assert(csInsts.size() % 2 == 0 && "We should have RC_ACCESS called in pairs.");
296 
297  // Sort the validation sites according to their ids.
298  std::sort(csInsts.begin(), csInsts.end(), compare);
299 
300  // Generate access pairs.
301  for (int i = 0, e = csInsts.size(); i != e;)
302  {
303  const CallInst *CI1 = csInsts[i++];
304  const CallInst *CI2 = csInsts[i++];
305  const ConstantInt *C = SVFUtil::dyn_cast<ConstantInt>(CI1->getOperand(1));
306  assert(C);
307  const Instruction *I1 = getPreviousMemoryAccessInst(CI1);
308  const Instruction *I2 = getPreviousMemoryAccessInst(CI2);
309  assert(I1 && I2 && "RC_ACCESS should be placed immediately after the target memory access.");
310  RC_FLAG flags = C->getZExtValue();
311  accessPairs.push_back(AccessPair(I1, I2, flags));
312  }
313  }

316  void validateAll()
317  {
318  SVFUtil::outs() << SVFUtil::pasMsg(" --- Analysis Result Validation ---\n");
319 
320  // Iterate every memory access pair to perform the validation.
321  for (int i = 0, e = accessPairs.size(); i != e; ++i)
322  {
323  const AccessPair &ap = accessPairs[i];
324  const Instruction *I1 = ap.getInstruction1();
325  const Instruction *I2 = ap.getInstruction2();
326 
327  bool mhp = mayHappenInParallel(I1, I2);
328  bool alias = mayAccessAliases(I1, I2);
329  bool protect = protectedByCommonLocks(I1, I2);
330  bool racy = mayHaveDataRace(I1, I2);
331 
332  SVFUtil::outs() << "For the memory access pair at ("
333  << SVFUtil::getSourceLoc(I1) << ", "
334  << SVFUtil::getSourceLoc(I2) << ")\n";
335  if (selectedValidationScenarios & RC_ALIASES)
336  {
337  SVFUtil::outs() << "\t"
338  << getOutput("ALIASES", alias, ap.isFlaged(RC_ALIASES))
339  << "\n";
340  }
341  if (selectedValidationScenarios & RC_MHP)
342  {
343  SVFUtil::outs() << "\t"
344  << getOutput("MHP", mhp, ap.isFlaged(RC_MHP)) << "\n";
345  }
346  if (selectedValidationScenarios & RC_PROTECTED)
347  {
348  SVFUtil::outs() << "\t"
349  << getOutput("PROTECT", protect,
350  ap.isFlaged(RC_PROTECTED)) << "\n";
351  }
352  if (selectedValidationScenarios & RC_RACE)
353  {
354  SVFUtil::outs() << "\t"
355  << getOutput("RACE", racy, ap.isFlaged(RC_RACE))
356  << "\n";
357  }
358  }
359 
360  SVFUtil::outs() << "\n";
361  }

389  static bool compare(const CallInst *CI1, const CallInst *CI2)
390  {
391  const Value *V1 = CI1->getOperand(0);
392  const Value *V2 = CI2->getOperand(0);
393  const ConstantInt *C1 = SVFUtil::dyn_cast<ConstantInt>(V1);
394  const ConstantInt *C2 = SVFUtil::dyn_cast<ConstantInt>(V2);
395  assert(0 != C1 && 0 != C2);
396  return C1->getZExtValue() < C2->getZExtValue();
397  }

404  const Instruction *getPreviousMemoryAccessInst(
405  const Instruction *I)
406  {
407  I = I->getPrevNode();
408  while (I)
409  {
410  if (SVFUtil::isa<LoadInst>(I) || SVFUtil::isa<StoreInst>(I))
411  return I;
412 
413  if (const SVFFunction *callee = SVFUtil::getCallee(I))
414  {
415  if (ExtAPI::EFT_L_A0__A0R_A1R == ExtAPI::getExtAPI()->get_type(callee)
416  || callee->getName().find("llvm.memset") != StringRef::npos)
417  return I;
418  }
419  I = I->getPrevNode();
420  }
421  return nullptr;
422  }
