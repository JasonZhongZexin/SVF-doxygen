145  WtoNesting operator^(const WtoNesting &other) const
146  {
147  WtoNesting res;
148  for (auto this_it = begin(), other_it = other.begin();
149  this_it != end() && other_it != other.end();
150  ++this_it, ++other_it)
151  {
152  if (*this_it == *other_it)
153  {
154  res.add(*this_it);
155  }
156  else
157  {
158  break;
159  }
160  }
161  return res;
162  }
163 
164 private:
166  int compare(const WtoNesting &other) const
167  {
168  if (this == &other)
169  {
170  return 0; // equals
171  }
172 
173  auto this_it = begin();
174  auto other_it = other.begin();
175  while (this_it != end())
176  {
177  if (other_it == other.end())
178  {
179  return 1; // `this` is nested within `other`
180  }
181  else if (*this_it == *other_it)
182  {
183  ++this_it;
184  ++other_it;
185  }
186  else
187  {
188  return 2; // not comparable
189  }
190  }
191  if (other_it == other.end())
192  {
193  return 0; // equals
194  }
195  else
196  {
197  return -1; // `other` is nested within `this`
198  }
199  }

203  {

205  }

747  const WtoCycleT *component(const SVFBasicBlock *vertex)
748  {
749  WtoComponentRefList partition;
750  for (auto it = vertex->getSuccessors().begin(), et = vertex->getSuccessors().end(); it != et; ++it)
751  {

753  if (dfn(succ) == 0)

766  virtual Dfn visit(const SVFBasicBlock *vertex, WtoComponentRefList &partition)
767  {
768  Dfn head(0);
769  Dfn min(0);
770  bool loop;
771 
772  push(vertex);
773  _num += Dfn(1);
774  head = _num;
775  set_dfn(vertex, head);
776  loop = false;
777  for (auto it = vertex->getSuccessors().begin(), et = vertex->getSuccessors().end(); it != et; ++it)
778  {
779  const SVFBasicBlock *succ = *it;
780  Dfn succ_dfn = dfn(succ);
781  if (succ_dfn == Dfn(0))
782  {
783  min = visit(succ, partition);
784  }
785  else
786  {
787  min = succ_dfn;
788  }
789  if (min <= head)
790  {
791  head = min;
792  loop = true;
793  }
794  }
795  if (head == dfn(vertex))
796  {
797  set_dfn(vertex, UINT_MAX);
798  const SVFBasicBlock *element = pop();
799  if (loop)
800  {
801  while (element != vertex)
802  {
803  set_dfn(element, 0);
804  element = pop();
805  }
806  partition.push_front(component(vertex));
807  }
808  else
809  {
810  partition.push_front(newVertex(vertex));
811  }
812  }
813  return head;
814  }
