139  static inline GEdgeFlag makeEdgeFlagWithAddionalOpnd(GEdgeKind k, const SVFVar* var)
140  {
141  Var2LabelMap::const_iterator iter = var2LabelMap.find(var);
142  u64_t label = (iter != var2LabelMap.end()) ?
143  iter->second : multiOpndLabelCounter++;
144  return (label << EdgeKindMaskBits) | k;
145  }
146 
148  static inline GEdgeFlag makeEdgeFlagWithCallInst(GEdgeKind k, const ICFGNode* cs)
149  {
150  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(cs);
151  u64_t label = (iter != inst2LabelMap.end()) ?
152  iter->second : callEdgeLabelCounter++;
153  return (label << EdgeKindMaskBits) | k;
154  }
155 
158  static inline GEdgeFlag makeEdgeFlagWithStoreInst(GEdgeKind k, const ICFGNode* store)
159  {
160  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(store);
161  u64_t label = (iter != inst2LabelMap.end()) ?
162  iter->second : storeEdgeLabelCounter++;
163  return (label << EdgeKindMaskBits) | k;
164  }
