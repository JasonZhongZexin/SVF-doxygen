47 /*
48  * Update call graph using pointer analysis results
49  * (1) resolve function pointers for non-fork calls
50  * (2) resolve function pointers for fork sites
51  * (3) resolve function pointers for parallel_for sites
52  */
53 void ThreadCallGraph::updateCallGraph(PointerAnalysis* pta)
54 {
55 
56  PointerAnalysis::CallEdgeMap::const_iterator iter = pta->getIndCallMap().begin();
57  PointerAnalysis::CallEdgeMap::const_iterator eiter = pta->getIndCallMap().end();
58  for (; iter != eiter; iter++)
59  {
60  const CallICFGNode* cs = iter->first;
61  const PTACallGraph::FunctionSet &functions = iter->second;
62  for (PTACallGraph::FunctionSet::const_iterator func_iter =
63  functions.begin(); func_iter != functions.end(); func_iter++)
64  {
65  const SVFFunction* callee = *func_iter;
66  this->addIndirectCallGraphEdge(cs, cs->getCaller(), callee);
67  }
68  }
69 
70  // Fork sites
71  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
72  {
73  const Value* forkedval = tdAPI->getForkedFun((*it)->getCallSite());
74  if(SVFUtil::dyn_cast<Function>(forkedval)==nullptr)
75  {
76  SVFIR* pag = pta->getPAG();
77  const NodeBS targets = pta->getPts(pag->getValueNode(forkedval)).toNodeBS();
78  for (NodeBS::iterator ii = targets.begin(), ie = targets.end(); ii != ie; ii++)
79  {
80  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
81  {
82  const MemObj* obj = pag->getObject(objPN);
83  if(obj->isFunction())
84  {
85  const Function* callee = SVFUtil::cast<Function>(obj->getValue());
86  const SVFFunction* svfCallee = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(callee);
87  this->addIndirectForkEdge(*it, svfCallee);
88  }
89  }
90  }
91  }
92  }
93 
94  // parallel_for sites
95  for (CallSiteSet::const_iterator it = parForSitesBegin(), eit = parForSitesEnd(); it != eit; ++it)
96  {
97  const Value* forkedval = tdAPI->getTaskFuncAtHareParForSite((*it)->getCallSite());
98  if(SVFUtil::dyn_cast<Function>(forkedval)==nullptr)
99  {
100  SVFIR* pag = pta->getPAG();
101  const NodeBS targets = pta->getPts(pag->getValueNode(forkedval)).toNodeBS();
102  for (NodeBS::iterator ii = targets.begin(), ie = targets.end(); ii != ie; ii++)
103  {
104  if(ObjVar* objPN = SVFUtil::dyn_cast<ObjVar>(pag->getGNode(*ii)))
105  {
106  const MemObj* obj = pag->getObject(objPN);
107  if(obj->isFunction())
108  {
109  const Function* callee = SVFUtil::cast<Function>(obj->getValue());
110  const SVFFunction* svfCallee = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(callee);
111  this->addIndirectForkEdge(*it, svfCallee);
112  }
113  }
114  }
115  }
116  }
117 }

123 void ThreadCallGraph::updateJoinEdge(PointerAnalysis* pta)
124 {
125 
126  for (CallSiteSet::const_iterator it = joinsitesBegin(), eit = joinsitesEnd(); it != eit; ++it)
127  {
128  const Value* jointhread = tdAPI->getJoinedThread((*it)->getCallSite());
129  // find its corresponding fork sites first
130  CallSiteSet forkset;
131  for (CallSiteSet::const_iterator it = forksitesBegin(), eit = forksitesEnd(); it != eit; ++it)
132  {
133  const Value* forkthread = tdAPI->getForkedThread((*it)->getCallSite());
134  if (pta->alias(jointhread, forkthread))
135  {
136  forkset.insert(*it);
137  }
138  }
139  assert(!forkset.empty() && "Can't find a forksite for this join!!");
140  addDirectJoinEdge(*it,forkset);
141  }
142 }

147 void ThreadCallGraph::addDirectForkEdge(const CallICFGNode* cs)
148 {
149 
150  PTACallGraphNode* caller = getCallGraphNode(cs->getCaller());
151  const Function* forkee = SVFUtil::dyn_cast<Function>(tdAPI->getForkedFun(cs->getCallSite()));
152  assert(forkee && "callee does not exist");
153  PTACallGraphNode* callee = getCallGraphNode(getDefFunForMultipleModule(forkee));
154  CallSiteID csId = addCallSite(cs, callee->getFunction());
155 
156  if (!hasGraphEdge(caller, callee, PTACallGraphEdge::TDForkEdge, csId))
157  {
158  assert(cs->getCaller() == caller->getFunction() && "callee instruction not inside caller??");
159 
160  ThreadForkEdge* edge = new ThreadForkEdge(caller, callee, csId);
161  edge->addDirectCallSite(cs);
162 
163  addEdge(edge);
164  addThreadForkEdgeSetMap(cs, edge);
165  }
166 }

196 void ThreadCallGraph::addDirectJoinEdge(const CallICFGNode* cs,const CallSiteSet& forkset)
197 {
198 
199  PTACallGraphNode* joinFunNode = getCallGraphNode(cs->getCaller());
200 
201  for (CallSiteSet::const_iterator it = forkset.begin(), eit = forkset.end(); it != eit; ++it)
202  {
203 
204  const Function* threadRoutineFun = SVFUtil::dyn_cast<Function>(tdAPI->getForkedFun((*it)->getCallSite()));
205  assert(threadRoutineFun && "thread routine function does not exist");
206  const SVFFunction* svfRoutineFun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(threadRoutineFun);
207  PTACallGraphNode* threadRoutineFunNode = getCallGraphNode(svfRoutineFun);
208  CallSiteID csId = addCallSite(cs, svfRoutineFun);
209 
210  if (!hasThreadJoinEdge(cs,joinFunNode,threadRoutineFunNode, csId))
211  {
212  assert(cs->getCaller() == joinFunNode->getFunction() && "callee instruction not inside caller??");
213  ThreadJoinEdge* edge = new ThreadJoinEdge(joinFunNode,threadRoutineFunNode,csId);
214  edge->addDirectCallSite(cs);
215 
216  addThreadJoinEdgeSetMap(cs, edge);
217  }
218  }
