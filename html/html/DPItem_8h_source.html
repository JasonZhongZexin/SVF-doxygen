244  }
246  inline virtual bool pushContext(NodeID ctx)
247  {
248 
249  if(context.size() < maximumCxtLen)
250  {
251  context.push_back(ctx);
252 
253  if(context.size() > maximumCxt)
254  maximumCxt = context.size();
255  return true;
256  }
257  else
258  {
259  if(!context.empty())
260  {
261  setNonConcreteCxt();
262  context.erase(context.begin());
263  context.push_back(ctx);
264  }
265  return false;
266  }
267  }

271  {
273  if(context.empty())
274  return true;
276  else if(context.back() == ctx)
277  {
278  context.pop_back();
279  return true;
280  }
281  return false;
282  }
