56 OrderedNodeSet& FunptrDDAClient::collectCandidateQueries(PAG* p)
57 {
58  setPAG(p);
59  for(PAG::CallSiteToFunPtrMap::const_iterator it = pag->getIndirectCallsites().begin(),
60  eit = pag->getIndirectCallsites().end(); it!=eit; ++it)
61  {
62  if (cppUtil::isVirtualCallSite(SVFUtil::getLLVMCallSite(it->first->getCallSite())))
63  {
64  const Value *vtblPtr = cppUtil::getVCallVtblPtr(SVFUtil::getLLVMCallSite(it->first->getCallSite()));
65  assert(pag->hasValueNode(vtblPtr) && "not a vtable pointer?");
66  NodeID vtblId = pag->getValueNode(vtblPtr);
67  addCandidate(vtblId);
68  vtableToCallSiteMap[vtblId] = it->first;
69  }
70  else
71  {
72  addCandidate(it->second);
73  }
74  }
75  return candidateQueries;

154 OrderedNodeSet& AliasDDAClient::collectCandidateQueries(PAG* pag)
155 {
156  setPAG(pag);
157  PAGEdge::PAGEdgeSetTy& loads = pag->getEdgeSet(PAGEdge::Load);
158  for (PAGEdge::PAGEdgeSetTy::iterator iter = loads.begin(), eiter =
159  loads.end(); iter != eiter; ++iter)
160  {
161  PAGNode* loadsrc = (*iter)->getSrcNode();
162  loadSrcNodes.insert(loadsrc);
163  addCandidate(loadsrc->getId());
164  }
165 
166  PAGEdge::PAGEdgeSetTy& stores = pag->getEdgeSet(PAGEdge::Store);
167  for (PAGEdge::PAGEdgeSetTy::iterator iter = stores.begin(), eiter =
168  stores.end(); iter != eiter; ++iter)
169  {
170  PAGNode* storedst = (*iter)->getDstNode();
171  storeDstNodes.insert(storedst);
172  addCandidate(storedst->getId());
173  }
174  PAGEdge::PAGEdgeSetTy& geps = pag->getEdgeSet(PAGEdge::NormalGep);
175  for (PAGEdge::PAGEdgeSetTy::iterator iter = geps.begin(), eiter =
176  geps.end(); iter != eiter; ++iter)
177  {
178  PAGNode* gepsrc = (*iter)->getSrcNode();
179  gepSrcNodes.insert(gepsrc);
180  addCandidate(gepsrc->getId());
181  }
182  return candidateQueries;
183 }
