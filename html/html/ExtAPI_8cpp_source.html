42 // Get environment variables $SVF_DIR and "npm root" through popen() method
43 static std::string GetStdoutFromCommand(const std::string &command)
44 {
45  char buffer[128];
46  std::string result = "";
47  // Open pipe to file
48  FILE *pipe = popen(command.c_str(), "r");
49  if (!pipe)
50  {
51  return "popen failed!";
52  }
53  // read till end of process:
54  while (!feof(pipe))
55  {
56  // use buffer to read and add to result
57  if (fgets(buffer, 128, pipe) != NULL)
58  result += buffer;
59  }
60  pclose(pipe);
61  // remove "\n"
62  result.erase(remove(result.begin(), result.end(), '\n'), result.end());
63  return result;
64 }

90 static cJSON *parseJson(const std::string &path, off_t fileSize)
91 {
92  FILE *file = fopen(path.c_str(), "r");
93  if (!file)
94  {
95  return nullptr;
96  }
97 
98  // allocate memory size matched with file size
99  char *jsonStr = (char *)calloc(fileSize + 1, sizeof(char));
100 
101  // read json string from file
102  u32_t size = fread(jsonStr, sizeof(char), fileSize, file);
103  if (size == 0)
104  {
105  SVFUtil::errs() << SVFUtil::errMsg("\t Wrong ExtAPI.json path!! ") << "The current ExtAPI.json path is: " << path << "\n";
106  assert(false && "Read ExtAPI.json file fails!");
107  return nullptr;
108  }
109  fclose(file);
110 
111  // convert json string to json pointer variable
112  cJSON *root = cJSON_Parse(jsonStr);
113  if (!root)
114  {
115  free(jsonStr);
116  return nullptr;
117  }
118  free(jsonStr);
119  return root;
120 }

122 ExtAPI *ExtAPI::getExtAPI(const std::string &path)
123 {
124  if (extOp == nullptr)
125  {
126  extOp = new ExtAPI;
127  }
128  if (root == nullptr)
129  {
130  struct stat statbuf;
131 
132  // Four ways to get ExtAPI.json path
133  // 1. Explicit path provided
134  // 2. default path (get ExtAPI.json path from Util/config.h)
135  // 3. from $SVF_DIR
136  // 4. from "npm root"(If SVF is installed via npm)
137 
138  std::string jsonFilePath = path;
139  if (!jsonFilePath.empty() && !stat(jsonFilePath.c_str(), &statbuf))
140  {
141  root = parseJson(jsonFilePath, statbuf.st_size);
142  return extOp;
143  }
144 
145  jsonFilePath = PROJECT_PATH + std::string(EXTAPI_JSON_PATH);
146  if (!stat(jsonFilePath.c_str(), &statbuf))
147  {
148  root = parseJson(jsonFilePath, statbuf.st_size);
149  return extOp;
150  }
151 
152  jsonFilePath = getJsonFile("$SVF_DIR");
153  if (!stat(jsonFilePath.c_str(), &statbuf))
154  {
155  root = parseJson(jsonFilePath, statbuf.st_size);
156  return extOp;
157  }
158 
159  jsonFilePath = getJsonFile("npm root");
160  if (!stat(jsonFilePath.c_str(), &statbuf))
161  {
162  root = parseJson(jsonFilePath, statbuf.st_size);
163  return extOp;
164  }
165 
166  assert(false && "Open ExtAPI.json file fails!");
167  }
168  return extOp;

185 void ExtAPI::add_entry(const char* funName, extType type, bool overwrite_app_function)
186 {
187  assert(root);
188  assert(get_type(funName) == EFT_NULL);
189  auto entry = cJSON_CreateObject();
190 
191  // add the type field
192  auto typeString = cJSON_CreateString(extType_toString(type).c_str());
193  cJSON_AddItemToObject(entry, JSON_OPT_FUNCTIONTYPE, typeString);
194 
195  // add the 'overwrite_app_function' field
196  auto overwriteBool = cJSON_CreateNumber(overwrite_app_function ? 1 : 0);
197  cJSON_AddItemToObject(entry, JSON_OPT_OVERWRITE, overwriteBool);
198 
199  // we don't know where the `funName` comes from, so copy it just in case
200  cJSON_AddItemToObject(root, strdup(funName), entry);
201 }

383 // Should (F) be considered "external" (either not defined in the program
384 // or a user-defined version of a known alloc or no-op)?
385 bool ExtAPI::is_ext(const SVFFunction *F)
386 {
387  assert(F);
388  bool res;
389  if (F->isDeclaration() || F->isIntrinsic())
390  {
391  res = 1;
392  }
393  else
394  {
395  ExtAPI::extType t = get_type(F);
396  if (t != EFT_NULL)
397  {
398  u32_t overwrittenAppFunction = isOverwrittenAppFunction(F);
399  // overwrittenAppFunction = 1: Execute function specification in ExtAPI.json
400  // F is considered as external function
401  if (overwrittenAppFunction == 1)
402  res = 1;
403  // overwrittenAppFunction = 0: Execute user-defined functions
404  // F is not considered as external function
405  else
406  res = 0;
407  }
408  else
409  res = 0;
410  }
411  return res;
