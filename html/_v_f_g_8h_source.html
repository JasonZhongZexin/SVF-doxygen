265  inline bool hasBlackHoleConstObjAddrAsDef(const PAGNode* pagNode) const
266  {
267  if (hasDef(pagNode))
268  {
269  const VFGNode* defNode = getVFGNode(getDef(pagNode));
270  if (const AddrVFGNode* addr = SVFUtil::dyn_cast<AddrVFGNode>(defNode))
271  {
272  if (PAG::getPAG()->isBlkObjOrConstantObj(addr->getPAGEdge()->getSrcID()))
273  return true;
274  }
275  else if(const CopyVFGNode* copy = SVFUtil::dyn_cast<CopyVFGNode>(defNode))
276  {
277  if (PAG::getPAG()->isNullPtr(copy->getPAGEdge()->getSrcID()))
278  return true;
279  }
280  }
281  return false;
282  }

402  {
403  PAGNodeToDefMapTy::iterator it = PAGNodeToDefMap.find(pagNode);
404  if(it == PAGNodeToDefMap.end())

532  inline void addFormalParmVFGNode(const PAGNode* fparm, const SVFFunction* fun, CallPESet& callPEs)
533  {
534  FormalParmVFGNode* sNode = new FormalParmVFGNode(totalVFGNode++,fparm,fun);
535  addVFGNode(sNode, pag->getICFG()->getFunEntryBlockNode(fun));
536  for(CallPESet::const_iterator it = callPEs.begin(), eit=callPEs.end();
537  it!=eit; ++it)
538  sNode->addCallPE(*it);
539 
540  setDef(fparm,sNode);
541  PAGNodeToFormalParmMap[fparm] = sNode;
542  }
546  inline void addFormalRetVFGNode(const PAGNode* uniqueFunRet, const SVFFunction* fun, RetPESet& retPEs)
547  {
548  FormalRetVFGNode *sNode = new FormalRetVFGNode(totalVFGNode++, uniqueFunRet, fun);
549  addVFGNode(sNode, pag->getICFG()->getFunExitBlockNode(fun));
550  for (RetPESet::const_iterator it = retPEs.begin(), eit = retPEs.end(); it != eit; ++it)
551  sNode->addRetPE(*it);
552 
553  PAGNodeToFormalRetMap[uniqueFunRet] = sNode;
556  if (!pag->isPhiNode(uniqueFunRet))
557  setDef(uniqueFunRet, sNode);
558  }

568  inline void addIntraPHIVFGNode(const PAGNode* phiResNode, PAG::CopyPEList& oplist)
569  {
570  IntraPHIVFGNode* sNode = new IntraPHIVFGNode(totalVFGNode++,phiResNode);
571  u32_t pos = 0;
572  const PAGEdge* edge = NULL;
573  for(PAG::CopyPEList::const_iterator it = oplist.begin(), eit=oplist.end(); it!=eit; ++it,++pos)
574  {
575  edge = *it;
576  sNode->setOpVerAndBB(pos, edge->getSrcNode(), edge->getICFGNode());
577  }
578  assert(edge && "edge not found?");
579  addVFGNode(sNode,edge->getICFGNode());
580  setDef(phiResNode,sNode);
581  PAGNodeToIntraPHIVFGNodeMap[phiResNode] = sNode;
582  }
