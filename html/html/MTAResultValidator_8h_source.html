275  void collectValidationTargets()
276  {
277  // Collect call sites of all RC_ACCESS function calls.
278  std::vector<const CallBase*> csInsts;
279  const Function* F = nullptr;
280  for (Module &M : LLVMModuleSet::getLLVMModuleSet()->getLLVMModules())
281  {
282  for(auto it = M.begin(); it != M.end(); it++)
283  {
284  const std::string fName = (*it).getName().str();
285  if(fName.find(RC_ACCESS) != std::string::npos)
286  {
287  F = &(*it);
288  break;
289  }
290  }
291  }
292  if (!F) return;
293 
294  for (Value::const_use_iterator it = F->use_begin(), ie =
295  F->use_end(); it != ie; ++it)
296  {
297  const Use *u = &*it;
298  const Value *user = u->getUser();
299  if(LLVMUtil::isCallSite(user))
300  {
301  const CallBase* csInst = LLVMUtil::getLLVMCallSite(user);
302  csInsts.push_back(csInst);
303  }
304  }
305  assert(csInsts.size() % 2 == 0 && "We should have RC_ACCESS called in pairs.");
306 
307  // Sort the validation sites according to their ids.
308  std::sort(csInsts.begin(), csInsts.end(), compare);
309 
310  // Generate access pairs.
311  for (int i = 0, e = csInsts.size(); i != e;)
312  {
313  const CallBase* CI1 = csInsts[i++];
314  const CallBase* CI2 = csInsts[i++];
315  const ConstantInt* C = SVFUtil::dyn_cast<ConstantInt>(CI1->getArgOperand(1));
316  assert(C);
317  const Instruction* I1 = getPreviousMemoryAccessInst(CI1);
318  const Instruction* I2 = getPreviousMemoryAccessInst(CI2);
319  assert(I1 && I2 && "RC_ACCESS should be placed immediately after the target memory access.");
320  RC_FLAG flags = C->getZExtValue();
321  accessPairs.push_back(AccessPair(I1, I2, flags));
322  }
323  }

326  void validateAll()
327  {
328  SVFUtil::outs() << SVFUtil::pasMsg(" --- Analysis Result Validation ---\n");
329 
330  // Iterate every memory access pair to perform the validation.
331  for (int i = 0, e = accessPairs.size(); i != e; ++i)
332  {
333  const AccessPair &ap = accessPairs[i];
334  const Instruction* I1 = ap.getInstruction1();
335  const Instruction* I2 = ap.getInstruction2();
336 
337  bool mhp = mayHappenInParallel(I1, I2);
338  bool alias = mayAccessAliases(I1, I2);
339  bool protect = protectedByCommonLocks(I1, I2);
340  bool racy = mayHaveDataRace(I1, I2);
341 
342  SVFUtil::outs() << "For the memory access pair at ("
343  << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(I1)->getSourceLoc() << ", "
344  << LLVMModuleSet::getLLVMModuleSet()->getSVFValue(I2)->getSourceLoc() << ")\n";
345  if (selectedValidationScenarios & RC_ALIASES)
346  {
347  SVFUtil::outs() << "\t"
348  << getOutput("ALIASES", alias, ap.isFlaged(RC_ALIASES))
349  << "\n";
350  }
351  if (selectedValidationScenarios & RC_MHP)
352  {
353  SVFUtil::outs() << "\t"
354  << getOutput("MHP", mhp, ap.isFlaged(RC_MHP)) << "\n";
355  }
356  if (selectedValidationScenarios & RC_PROTECTED)
357  {
358  SVFUtil::outs() << "\t"
359  << getOutput("PROTECT", protect,
360  ap.isFlaged(RC_PROTECTED)) << "\n";
361  }
362  if (selectedValidationScenarios & RC_RACE)
363  {
364  SVFUtil::outs() << "\t"
365  << getOutput("RACE", racy, ap.isFlaged(RC_RACE))
366  << "\n";
367  }
368  }
369 
370  SVFUtil::outs() << "\n";
371  }

399  static bool compare(const CallBase* CI1, const CallBase* CI2)
400  {
401  const Value *V1 = CI1->getArgOperand(0);
402  const Value *V2 = CI2->getArgOperand(0);
403  const ConstantInt* C1 = SVFUtil::dyn_cast<ConstantInt>(V1);
404  const ConstantInt* C2 = SVFUtil::dyn_cast<ConstantInt>(V2);
405  assert(0 != C1 && 0 != C2);
406  return C1->getZExtValue() < C2->getZExtValue();
407  }

414  const Instruction* getPreviousMemoryAccessInst(
415  const Instruction* I)
416  {
417  I = I->getPrevNode();
418  while (I)
419  {
420  if (SVFUtil::isa<LoadInst>(I) || SVFUtil::isa<StoreInst>(I))
421  return I;
422 
423  const SVFInstruction* inst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(I);
424 
425  if (const SVFFunction *callee = SVFUtil::getCallee(inst))
426  {
427  if (callee->getName().find("llvm.memset") != std::string::npos)
428  return I;
429 
430  }
431  I = I->getPrevNode();
432  }
433  return nullptr;
434  }
