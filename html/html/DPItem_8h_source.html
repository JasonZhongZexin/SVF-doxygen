222  {
223  return concreteCxt;
224  }
226  inline void setNonConcreteCxt()
227  {
228  concreteCxt = false;
229  }
231  inline bool containCallStr(NodeID cxt) const
232  {
233  return std::find(context.begin(),context.end(),cxt) != context.end();
234  }
236  inline u32_t cxtSize() const
237  {
238  return context.size();
239  }
241  static inline void setMaxCxtLen(u32_t max)
242  {
243  maximumCxtLen = max;
244  }
246  static inline void setMaxPathLen(u32_t max)
247  {
248  maximumPathLen = max;
249  }
250  inline u32_t getMaxPathLen() const
251  {
252  return maximumPathLen;
253  }
255  inline virtual bool pushContext(NodeID ctx)
256  {
257 
258  if(context.size() < maximumCxtLen)
259  {
260  context.push_back(ctx);
261 
262  if(context.size() > maximumCxt)
263  maximumCxt = context.size();
264  return true;
265  }
266  else
267  {
268  if(!context.empty())
269  {
270  setNonConcreteCxt();
271  context.erase(context.begin());
272  context.push_back(ctx);
273  }
274  return false;
275  }
276  }
277 
279  inline virtual bool matchContext(NodeID ctx)
280  {
282  if(context.empty())
283  return true;
285  else if(context.back() == ctx)
286  {
287  context.pop_back();
288  return true;
289  }
290  return false;
291  }
292 
295  inline bool operator< (const ContextCond& rhs) const
296  {
297  return context < rhs.context;
298  }
300  inline NodeID operator[] (const u32_t index) const
301  {
302  assert(index < context.size());
303  return context[index];
304  }
306  inline ContextCond& operator= (const ContextCond& rhs)
307  {
308  if(*this!=rhs)
309  {
310  context = rhs.getContexts();
311  concreteCxt = rhs.isConcreteCxt();
312  }
313  return *this;
314  }
316  inline bool operator== (const ContextCond& rhs) const
317  {
318  return (context == rhs.getContexts());
319  }
321  inline bool operator!= (const ContextCond& rhs) const
322  {
323  return !(*this==rhs);
324  }
326  inline const_iterator begin() const
327  {
328  return context.begin();
329  }
331  inline const_iterator end() const
332  {
333  return context.end();
334  }
336  inline std::string toString() const
337  {
338  std::string str;
339  raw_string_ostream rawstr(str);
340  rawstr << "[:";
341  for(CallStrCxt::const_iterator it = context.begin(), eit = context.end(); it!=eit; ++it)
342  {
343  rawstr << *it << " ";
344  }
345  rawstr << " ]";
346  return rawstr.str();
347  }
348 protected:
349  CallStrCxt context;
350  static u32_t maximumCxtLen;
351  static u32_t maximumPathLen;
352  bool concreteCxt;
353 public:
