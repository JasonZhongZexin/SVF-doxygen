25 BVDataPTAImpl::BVDataPTAImpl(SVFIR* p, PointerAnalysis::PTATY type, bool alias_check) :
26  PointerAnalysis(p, type, alias_check), ptCache()
27 {
28  if (type == Andersen_BASE || type == Andersen_WPA || type == AndersenWaveDiff_WPA
29  || type == TypeCPP_WPA || type == FlowS_DDA
30  || type == AndersenSCD_WPA || type == AndersenSFR_WPA || type == CFLFICI_WPA )
31  {
32  // Only maintain reverse points-to when the analysis is field-sensitive, as objects turning
33  // field-insensitive is all it is used for.
34  bool maintainRevPts = Options::MaxFieldLimit != 0;
35  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(maintainRevPts);
36  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), maintainRevPts);
37  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
38  }
39  else if (type == Steensgaard_WPA)
40  {
41  // Steensgaard is only field-insensitive (for now?), so no reverse points-to.
42  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(false);
43  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), false);
44  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
45  }
46  else if (type == FSSPARSE_WPA)
47  {
48  if (Options::INCDFPTData)
49  {
50  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutIncDFPTDataTy(false);
51  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersIncDFPTDataTy(getPtCache(), false);
52  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
53  }
54  else
55  {
56  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDFPTDataTy(false);
57  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDFPTDataTy(getPtCache(), false);
58  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
59  }
60  }

63  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutVersionedPTDataTy(false);

65  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");

193 bool BVDataPTAImpl::readFromFile(const string& filename)
194 {
195  // If the module annotations are available, read from there instead
196  auto mainModule = SVF::LLVMModuleSet::getLLVMModuleSet()->getMainLLVMModule();
197  if (mainModule->getNamedMetadata("SVFIR-Annotated") != nullptr)
198  {
199  return readFromModule();
200  }
201 
202  outs() << "Loading pointer analysis results from '" << filename << "'...";
203 
204  ifstream F(filename.c_str());
205  if (!F.is_open())
206  {
207  outs() << " error opening file for reading!\n";
208  return false;
209  }
210 
211  // Read analysis results from file
212  PTDataTy *ptD = getPTDataTy();
213  string line;
214 
215  // Read points-to sets
216  string delimiter1 = " -> { ";
217  string delimiter2 = " }";
218  map<NodeID, string> nodePtsMap;
219  map<string, PointsTo> strPtsMap;
220 
221  while (F.good())
222  {
223  // Parse a single line in the form of "var -> { obj1 obj2 obj3 }"
224  getline(F, line);
225  size_t pos = line.find(delimiter1);
226  if (pos == string::npos) break;
227  if (line.back() != '}') break;
228 
229  // var
230  NodeID var = atoi(line.substr(0, pos).c_str());
231 
232  // objs
233  pos = pos + delimiter1.length();
234  size_t len = line.length() - pos - delimiter2.length();
235  string objs = line.substr(pos, len);
236  PointsTo dstPts;
237 
238  if (!objs.empty())
239  {
240  // map the variable ID to its unique string pointer set
241  nodePtsMap[var] = objs;
242  if (strPtsMap.count(objs)) continue;
243 
244  istringstream ss(objs);
245  NodeID obj;
246  while (ss.good())
247  {
248  ss >> obj;
249  dstPts.set(obj);
250  }
251  // map the string pointer set to the parsed PointsTo set
252  strPtsMap[objs] = dstPts;
253  }
254  }
255 
256  // map the variable ID to its pointer set
257  for (auto t: nodePtsMap)
258  ptD->unionPts(t.first, strPtsMap[t.second]);
259 
260  // Read SVFIR offset nodes
261  while (F.good())
262  {
263  if (line == "------") break;
264  // Parse a single line in the form of "ID baseNodeID offset"
265  istringstream ss(line);
266  NodeID id;
267  NodeID base;
268  size_t offset;
269  ss >> id >> base >> offset;
270 
271  NodeID n = pag->getGepObjVar(base, LocationSet(offset));
272  assert(id == n && "Error adding GepObjNode into SVFIR!");
273 
274  getline(F, line);
275  }
276 
277  // Read BaseNode insensitivity
278  while (F.good())
279  {
280  getline(F, line);
281  // Parse a single line in the form of "baseNodeID insensitive"
282  istringstream ss(line);
283  NodeID base;
284  bool insensitive;
285  ss >> base >> insensitive;
286 
287  if (insensitive)
288  setObjFieldInsensitive(base);
289  }
290 
291  // Update callgraph
292  updateCallGraph(pag->getIndirectCallsites());
293 
294  F.close();
295  outs() << "\n";
296 
297  return true;
298 }

404 void BVDataPTAImpl::normalizePointsTo()
405 {
406  SVFIR::MemObjToFieldsMap &memToFieldsMap = pag->getMemToFieldsMap();
407  SVFIR::NodeLocationSetMap &GepObjVarMap = pag->getGepObjNodeMap();
408 
409  // collect each gep node whose base node has been set as field-insensitive
410  NodeBS dropNodes;
411  for (auto t: memToFieldsMap)
412  {
413  NodeID base = t.first;
414  const MemObj* memObj = pag->getObject(base);
415  assert(memObj && "Invalid memobj in memToFieldsMap");
416  if (memObj->isFieldInsensitive())
417  {
418  for (NodeID id : t.second)
419  {
420  if (SVFUtil::isa<GepObjVar>(pag->getGNode(id)))
421  {
422  dropNodes.set(id);
423  }
424  else
425  assert(id == base && "Not a GepObj Node or a baseObj Node?");
426  }
427  }
428  }
429 
430  // remove the collected redundant gep nodes in each pointers's pts
431  for (SVFIR::iterator nIter = pag->begin(); nIter != pag->end(); ++nIter)
432  {
433  NodeID n = nIter->first;
434 
435  const PointsTo &tmpPts = getPts(n);
436  for (NodeID obj : tmpPts)
437  {
438  if (!dropNodes.test(obj))
439  continue;
440  NodeID baseObj = pag->getBaseObjVar(obj);
441  clearPts(n, obj);
442  addPts(n, baseObj);
443  }
444  }
445 
446  // clear GepObjVarMap and memToFieldsMap for redundant gepnodes
447  // and remove those nodes from pag
448  for (NodeID n: dropNodes)
449  {
450  NodeID base = pag->getBaseObjVar(n);
451  GepObjVar *gepNode = SVFUtil::dyn_cast<GepObjVar>(pag->getGNode(n));
452  const LocationSet ls = gepNode->getLocationSet();
453  GepObjVarMap.erase(std::make_pair(base, ls));
454  memToFieldsMap[base].reset(n);
455 
456  pag->removeGNode(gepNode);
457  }
458 }
