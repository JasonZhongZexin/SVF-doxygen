258  inline void RenameChiSet(const CHISet& chiSet, MRVector& memRegs)
259  {
260  for (CHISet::const_iterator cit = chiSet.begin(), ecit = chiSet.end();
261  cit != ecit; ++cit)
262  {
263  CHI* chi = (*cit);
264  chi->setOpVer(getTopStackVer(chi->getMR()));
265  chi->setResVer(newSSAName(chi->getMR(),chi));
266  memRegs.push_back(chi->getMR());
267  }
