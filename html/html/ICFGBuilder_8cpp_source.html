42 void ICFGBuilder::build(SVFModule* svfModule)
43 {
44  for (SVFModule::const_iterator iter = svfModule->begin(), eiter = svfModule->end(); iter != eiter; ++iter)
45  {
46  const SVFFunction *fun = *iter;
47  if (SVFUtil::isExtCall(fun))
48  continue;
49  WorkList worklist;
50  processFunEntry(fun,worklist);
51  processFunBody(worklist);
52  processFunExit(fun);
53  }
54  connectGlobalToProgEntry(svfModule);
55 }

60 void ICFGBuilder::processFunEntry(const SVFFunction* fun, WorkList& worklist)
61 {
62  FunEntryICFGNode* FunEntryICFGNode = icfg->getFunEntryICFGNode(fun);
63  const Instruction* entryInst = &((fun->getLLVMFun()->getEntryBlock()).front());
64  InstVec insts;
65  if (isIntrinsicInst(entryInst))
66  getNextInsts(entryInst, insts);
67  else
68  insts.push_back(entryInst);
69  for (InstVec::const_iterator nit = insts.begin(), enit = insts.end();
70  nit != enit; ++nit)
71  {
72  ICFGNode* instNode = getOrAddBlockICFGNode(*nit); //add interprocedure edge
73  icfg->addIntraEdge(FunEntryICFGNode, instNode);
74  worklist.push(*nit);
75  }
76 }

81 void ICFGBuilder::processFunBody(WorkList& worklist)
82 {
83  BBSet visited;
85  while (!worklist.empty())
86  {
87  const Instruction* inst = worklist.pop();
88  if (visited.find(inst) == visited.end())
89  {
90  visited.insert(inst);
91  ICFGNode* srcNode = getOrAddBlockICFGNode(inst);
92  if (isReturn(inst))
93  {
94  const Function* fun = inst->getFunction();
95  const SVFFunction* svfFun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
96  FunExitICFGNode* FunExitICFGNode = icfg->getFunExitICFGNode(svfFun);
97  icfg->addIntraEdge(srcNode, FunExitICFGNode);
98  }
99  InstVec nextInsts;
100  getNextInsts(inst, nextInsts);
101  u32_t branchID = 0;
102  for (InstVec::const_iterator nit = nextInsts.begin(), enit =
103  nextInsts.end(); nit != enit; ++nit)
104  {
105  const Instruction* succ = *nit;
106  ICFGNode* dstNode = getOrAddBlockICFGNode(succ);
107  if (isNonInstricCallSite(inst))
108  {
109  RetICFGNode* retICFGNode = getRetICFGNode(inst);
110  srcNode = retICFGNode;
111  }
112 
113 
114  if (const BranchInst* br = SVFUtil::dyn_cast<BranchInst>(inst))
115  {
116  assert(branchID <= 2 && "if/else has more than two branches?");
117  if(br->isConditional())
118  icfg->addConditionalIntraEdge(srcNode, dstNode, br->getCondition(), 1 - branchID);
119  else
120  icfg->addIntraEdge(srcNode, dstNode);
121  }
122  else if (const SwitchInst* si = SVFUtil::dyn_cast<SwitchInst>(inst))
123  {
125  const ConstantInt* condVal = const_cast<SwitchInst*>(si)->findCaseDest(const_cast<BasicBlock*>(succ->getParent()));
127  s32_t val = condVal ? condVal->getSExtValue() : -1;
128  icfg->addConditionalIntraEdge(srcNode, dstNode, si->getCondition(),val);
129  }
130  else
131  icfg->addIntraEdge(srcNode, dstNode);
132 
133  worklist.push(succ);
134  branchID++;
135  }
136  }
137  }
138 }

167 InterICFGNode* ICFGBuilder::getOrAddInterBlockICFGNode(const Instruction* inst)
168 {
169  assert(SVFUtil::isCallSite(inst) && "not a call instruction?");
170  assert(SVFUtil::isNonInstricCallSite(inst) && "associating an intrinsic debug instruction with an ICFGNode!");
171  CallICFGNode* callICFGNode = getCallICFGNode(inst);
172  addICFGInterEdges(inst, getCallee(inst)); //creating interprocedural edges
173  return callICFGNode;

179 void ICFGBuilder::addICFGInterEdges(const Instruction* cs, const SVFFunction* callee)
180 {
181  CallICFGNode* callICFGNode = getCallICFGNode(cs);
182  RetICFGNode* retBlockNode = getRetICFGNode(cs);
183 
185  if(callee)
186  {
188  if (isExtCall(callee))
189  {
190  icfg->addIntraEdge(callICFGNode, retBlockNode);
191  }
193  else
194  {
195  FunEntryICFGNode* calleeEntryNode = icfg->getFunEntryICFGNode(callee);
196  FunExitICFGNode* calleeExitNode = icfg->getFunExitICFGNode(callee);
197  icfg->addCallEdge(callICFGNode, calleeEntryNode, cs);
198  icfg->addRetEdge(calleeExitNode, retBlockNode, cs);
199  }
200  }
202  else
203  {
204  icfg->addIntraEdge(callICFGNode, retBlockNode);
205  }
206 }
