57 OrderedNodeSet& FunptrDDAClient::collectCandidateQueries(SVFIR* p)
58 {
59  setPAG(p);
60  for(SVFIR::CallSiteToFunPtrMap::const_iterator it = pag->getIndirectCallsites().begin(),
61  eit = pag->getIndirectCallsites().end(); it!=eit; ++it)
62  {
63  if (cppUtil::isVirtualCallSite(SVFUtil::getLLVMCallSite(it->first->getCallSite())))
64  {
65  const Value *vtblPtr = cppUtil::getVCallVtblPtr(SVFUtil::getLLVMCallSite(it->first->getCallSite()));
66  assert(pag->hasValueNode(vtblPtr) && "not a vtable pointer?");
67  NodeID vtblId = pag->getValueNode(vtblPtr);
68  addCandidate(vtblId);
69  vtableToCallSiteMap[vtblId] = it->first;
70  }
71  else
72  {
73  addCandidate(it->second);
74  }
75  }
76  return candidateQueries;
