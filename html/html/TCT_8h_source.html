312  const NodeBS getAncestorThread(NodeID tid) const
313  {
314  NodeBS tds;
315  if(hasParentThread(tid) == false)
316  return tds;
317 
318  FIFOWorkList<NodeID> worklist;
319  worklist.push(getParentThread(tid));
320 
321  while(!worklist.empty())
322  {
323  NodeID t = worklist.pop();
324  if(tds.test_and_set(t))
325  {
326  if(hasParentThread(t))
327  worklist.push(getParentThread(t));
328  }
329  }
330  return tds;
