371  inline bool mustAlias(const CVar& var1, const CVar& var2)
372  {
373  if(isSameVar(var1,var2))
374  return true;
375 
376  bool singleton = !(isHeapMemObj(var1.get_id()) || isLocalVarInRecursiveFun(var1.get_id()));
377  if(isCondCompatible(var1.get_cond(),var2.get_cond(),singleton) == false)
378  return false;
379 
380  const CPtSet& cpts1 = getPts(var1);
381  const CPtSet& cpts2 = getPts(var2);
382  return (contains(cpts1,cpts2) && contains(cpts2,cpts1));
383  }

385  // Whether cpts1 contains all points-to targets of pts2
386  bool contains(const CPtSet& cpts1, const CPtSet& cpts2)
387  {
388  if (cpts1.empty() || cpts2.empty())
389  return false;
390 
391  for (typename CPtSet::const_iterator it2 = cpts2.begin(); it2 != cpts2.end(); ++it2)
392  {
393  bool hasObj = false;
394  for (typename CPtSet::const_iterator it1 = cpts1.begin(); it1 != cpts1.end(); ++it1)
395  {
396  if(isSameVar(*it1,*it2))
397  {
398  hasObj = true;
399  break;
400  }
401  }
402  if(hasObj == false)
403  return false;
404  }
405  return true;
406  }
