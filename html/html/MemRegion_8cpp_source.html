148 void MRGenerator::collectModRefForLoadStore()
149 {
150 
151  SVFModule* svfModule = pta->getModule();
152  for (SVFModule::const_iterator fi = svfModule->begin(), efi = svfModule->end(); fi != efi;
153  ++fi)
154  {
155  const SVFFunction& fun = **fi;
156 
158  if (Options::IgnoreDeadFun && isDeadFunction(fun.getLLVMFun()))
159  continue;
160 
161  for (Function::const_iterator iter = fun.getLLVMFun()->begin(), eiter = fun.getLLVMFun()->end();
162  iter != eiter; ++iter)
163  {
164  const BasicBlock& bb = *iter;
165  for (BasicBlock::const_iterator bit = bb.begin(), ebit = bb.end();
166  bit != ebit; ++bit)
167  {
168  const Instruction& inst = *bit;
169  PAGEdgeList& pagEdgeList = getPAGEdgesFromInst(&inst);
170  for (PAGEdgeList::iterator bit = pagEdgeList.begin(), ebit =
171  pagEdgeList.end(); bit != ebit; ++bit)
172  {
173  const PAGEdge* inst = *bit;
174  pagEdgeToFunMap[inst] = &fun;
175  if (const StorePE *st = SVFUtil::dyn_cast<StorePE>(inst))
176  {
177  PointsTo cpts(pta->getPts(st->getDstID()));
178  // TODO: change this assertion check later when we have conditional points-to set
179  if (cpts.empty())
180  continue;
181  assert(!cpts.empty() && "null pointer!!");
182  addCPtsToStore(cpts, st, &fun);
183  }
184 
185  else if (const LoadPE *ld = SVFUtil::dyn_cast<LoadPE>(inst))
186  {
187  PointsTo cpts(pta->getPts(ld->getSrcID()));
188  // TODO: change this assertion check later when we have conditional points-to set
189  if (cpts.empty())
190  continue;
191  assert(!cpts.empty() && "null pointer!!");
192  addCPtsToLoad(cpts, ld, &fun);
193  }
194  }
195  }
196  }
197  }

260 void MRGenerator::sortPointsTo(const PointsTo& cpts)
261 {
262 
263  if(cptsToRepCPtsMap.find(cpts)!=cptsToRepCPtsMap.end())
264  return;
265 
266  PointsToList subSetList;
267  PointsTo repCPts = cpts;
268  for(PtsToRepPtsSetMap::iterator it = cptsToRepCPtsMap.begin(),
269  eit = cptsToRepCPtsMap.end(); it!=eit; ++it)
270  {
271  PointsTo& existCPts = it->second;

273  {

283  {

455 void MRGenerator::collectCallSitePts(const CallBlockNode* cs)
456 {
458  NodeBS& argsPts = csToCallSiteArgsPtsMap[cs];
459  PAG* pag = pta->getPAG();
460  CallBlockNode* callBlockNode = pag->getICFG()->getCallBlockNode(cs->getCallSite());
461  RetBlockNode* retBlockNode = pag->getICFG()->getRetBlockNode(cs->getCallSite());
462 
463  WorkList worklist;
464  if (pag->hasCallSiteArgsMap(callBlockNode))
465  {
466  const PAG::PAGNodeList& args = pta->getPAG()->getCallSiteArgsList(callBlockNode);
467  for(PAG::PAGNodeList::const_iterator itA = args.begin(), ieA = args.end(); itA!=ieA; ++itA)
468  {
469  const PAGNode* node = *itA;
470  if(node->isPointer())
471  worklist.push(node->getId());
472  }
473  }
474 
475  while(!worklist.empty())
476  {
477  NodeID nodeId = worklist.pop();
478  const PointsTo& tmp = pta->getPts(nodeId);
479  for(PointsTo::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
480  argsPts |= CollectPtsChain(*it);
481  }
482 
484  NodeBS& retPts = csToCallSiteRetPtsMap[cs];
485 
486  if (pta->getPAG()->callsiteHasRet(retBlockNode))
487  {
488  const PAGNode* node = pta->getPAG()->getCallSiteRet(retBlockNode);
489  if(node->isPointer())
490  {
491  const PointsTo& tmp = pta->getPts(node->getId());
492  for(PointsTo::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
493  retPts |= CollectPtsChain(*it);
494  }
495  }

579 bool MRGenerator::handleCallsiteModRef(NodeBS& mod, NodeBS& ref, const CallBlockNode* cs, const SVFFunction* callee)
580 {
582  if(isHeapAllocExtCall(cs->getCallSite()))
583  {
584  PAGEdgeList& pagEdgeList = getPAGEdgesFromInst(cs->getCallSite());
585  for (PAGEdgeList::const_iterator bit = pagEdgeList.begin(),
586  ebit = pagEdgeList.end(); bit != ebit; ++bit)
587  {
588  const PAGEdge* edge = *bit;
589  if (const AddrPE* addr = SVFUtil::dyn_cast<AddrPE>(edge))
590  mod.set(addr->getSrcID());
591  }
592  }
594  else
595  {
596  mod = getModSideEffectOfFunction(callee);
597  ref = getRefSideEffectOfFunction(callee);
598  }
599  // add ref set
600  bool refchanged = addRefSideEffectOfCallSite(cs, ref);
601  // add mod set
602  bool modchanged = addModSideEffectOfCallSite(cs, mod);
603 
604  return refchanged || modchanged;
605 }
