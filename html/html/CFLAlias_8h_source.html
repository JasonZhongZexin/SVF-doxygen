65  svfir->dump("SVFIR");
66  // Can be put in build, copy from general graph

68  graph->buildBigraph(consCG);
69  graph->dump("PAG");
70  delete consCG;
71  GrammarBase *generalGrammar = gReader->build(&ConstMap);
72  grammar = normalizer->normalize(generalGrammar);
73  graph->setMap(&grammar->terminals, &grammar->nonterminals);
74  }
75  else {
76  GrammarBase *generalGrammar = gReader->build();
77  grammar = normalizer->normalize(generalGrammar);
78  graph->setMap(&grammar->terminals, &grammar->nonterminals);
79  graph->buildFromDot(Options::InputFilename);

81  graph->startSymbol = grammar->startSymbol;
82  solver = new CFLSolver(graph, grammar);
83  solver->solve();
84  graph->dump("map");
