154 void BDDExprManager::ddClearFlag(BDDExpr *f) const
155 {
156  if (!Cudd_IsComplement(f->next))
157  return;
158  /* Clear visited flag. */
159  f->next = Cudd_Regular(f->next);
160  if (cuddIsConstant(f))
161  return;
162  ddClearFlag(cuddT(f));
163  ddClearFlag(Cudd_Regular(cuddE(f)));
164  return;

167 void BDDExprManager::BddSupportStep(BDDExpr *f, NodeBS &support) const
168 {
169  if (cuddIsConstant(f) || Cudd_IsComplement(f->next))
170  return;
171 
172  support.set(f->index);
173 
174  BddSupportStep(cuddT(f), support);
175  BddSupportStep(Cudd_Regular(cuddE(f)), support);
176  /* Mark as visited. */
177  f->next = Cudd_Complement(f->next);
178 }
