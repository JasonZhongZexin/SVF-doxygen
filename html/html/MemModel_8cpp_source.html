71 bool ObjTypeInfo::isNonPtrFieldObj(const LocationSet& ls)
72 {
73  if (isHeap() || isStaticObj())
74  return false;
75 
76  const Type* ety = getType();
77  while (const ArrayType *AT= SVFUtil::dyn_cast<ArrayType>(ety))
78  {
79  ety = AT->getElementType();
80  }
81 
82  if (SVFUtil::isa<StructType>(ety) || SVFUtil::isa<ArrayType>(ety))
83  {
84  bool hasIntersection = false;
85  const vector<FieldInfo> &infovec = SymbolTableInfo::SymbolInfo()->getFlattenFieldInfoVec(ety);
86  vector<FieldInfo>::const_iterator it = infovec.begin();
87  vector<FieldInfo>::const_iterator eit = infovec.end();
88  for (; it != eit; ++it)
89  {
90  const FieldInfo& fieldLS = *it;
91  if (ls.intersects(LocationSet(fieldLS)))
92  {
93  hasIntersection = true;
94  if (fieldLS.getFlattenElemTy()->isPointerTy())
95  return false;
96  }
97  }
98  assert(hasIntersection && "cannot find field of specified offset");
99  return true;
100  }
101  else
102  {
103  if (isStaticObj() || isHeap())
104  {
105  // TODO: Objects which cannot find proper field for a certain offset including
106  // arguments in main(), static objects allocated before main and heap
107  // objects. Right now they're considered to have infinite fields and we
108  // treat each field as pointers conservatively.
109  // Try to model static and heap objects more accurately in the future.
110  return false;
111  }
112  else
113  {
114  // TODO: Using new memory model (locMM) may create objects with spurious offset
115  // as we simply return new offset by mod operation without checking its
116  // correctness in LocSymTableInfo::getModulusOffset(). So the following
117  // assertion may fail. Try to refine the new memory model.
118  //assert(ls.getOffset() == 0 && "cannot get a field from a non-struct type");
119  return (hasPtrObj() == false);
120  }
121  }
122 }
