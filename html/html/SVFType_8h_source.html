67 template <class S, class T> struct Hash<std::pair<S, T>>
68 {
69  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
70  static size_t szudzik(size_t a, size_t b)
71  {
72  return a > b ? b * b + a : a * a + a + b;
73  }
74 
75  size_t operator()(const std::pair<S, T> &t) const
76  {
77  Hash<decltype(t.first)> first;
78  Hash<decltype(t.second)> second;
79  return szudzik(first(t.first), second(t.second));
80  }
81 };
