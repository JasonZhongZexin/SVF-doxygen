25 BVDataPTAImpl::BVDataPTAImpl(PAG* p, PointerAnalysis::PTATY type, bool alias_check) :
26  PointerAnalysis(p, type, alias_check), ptCache()
27 {
28  if (type == Andersen_BASE || type == Andersen_WPA || type == AndersenWaveDiff_WPA || type == AndersenHCD_WPA || type == AndersenHLCD_WPA
29  || type == AndersenLCD_WPA || type == TypeCPP_WPA || type == FlowS_DDA || type == AndersenWaveDiffWithType_WPA
30  || type == AndersenSCD_WPA || type == AndersenSFR_WPA)
31  {
32  // Only maintain reverse points-to when the analysis is field-sensitive, as objects turning
33  // field-insensitive is all it is used for.
34  bool maintainRevPts = Options::MaxFieldLimit != 0;
35  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(maintainRevPts);
36  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), maintainRevPts);
37  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
38  }
39  else if (type == Steensgaard_WPA)
40  {
41  // Steensgaard is only field-insensitive (for now?), so no reverse points-to.
42  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(false);
43  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), false);
44  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
45  }
46  else if (type == FSSPARSE_WPA || type == FSTBHC_WPA)
47  {
48  if (Options::INCDFPTData)
49  {
50  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutIncDFPTDataTy(false);
51  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersIncDFPTDataTy(getPtCache(), false);
52  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
53  }
54  else
55  {
56  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDFPTDataTy(false);
57  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDFPTDataTy(getPtCache(), false);
58  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
59  }
60  }
61  else if (type == VFS_WPA)
62  {
63  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutVersionedPTDataTy(false);
64  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersVersionedPTDataTy(getPtCache(), false);
65  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");

183 bool BVDataPTAImpl::readFromFile(const string& filename)
184 {
185  // If the module annotations are available, read from there instead
186  auto mainModule = SVF::LLVMModuleSet::getLLVMModuleSet()->getMainLLVMModule();
187  if (mainModule->getNamedMetadata("PAG-Annotated") != nullptr)
188  {
189  return readFromModule();
190  }
191 
192  outs() << "Loading pointer analysis results from '" << filename << "'...";
193 
194  ifstream F(filename.c_str());
195  if (!F.is_open())
196  {
197  outs() << " error opening file for reading!\n";
198  return false;
199  }
200 
201  // Read analysis results from file
202  PTDataTy *ptD = getPTDataTy();
203  string line;
204 
205  // Read points-to sets
206  string delimiter1 = " -> { ";
207  string delimiter2 = " }";
208  map<NodeID, string> nodePtsMap;
209  map<string, PointsTo> strPtsMap;
210 
211  while (F.good())
212  {
213  // Parse a single line in the form of "var -> { obj1 obj2 obj3 }"
214  getline(F, line);
215  size_t pos = line.find(delimiter1);
216  if (pos == string::npos) break;
217  if (line.back() != '}') break;
218 
219  // var
220  NodeID var = atoi(line.substr(0, pos).c_str());
221 
222  // objs
223  pos = pos + delimiter1.length();
224  size_t len = line.length() - pos - delimiter2.length();
225  string objs = line.substr(pos, len);
226  PointsTo dstPts;
227 
228  if (!objs.empty())
229  {
230  // map the variable ID to its unique string pointer set
231  nodePtsMap[var] = objs;
232  if (strPtsMap.count(objs)) continue;
233 
234  istringstream ss(objs);
235  NodeID obj;
236  while (ss.good())
237  {
238  ss >> obj;
239  dstPts.set(obj);
240  }
241  // map the string pointer set to the parsed PointsTo set
242  strPtsMap[objs] = dstPts;
243  }
244  }
245 
246  // map the variable ID to its pointer set
247  for (auto t: nodePtsMap)
248  ptD->unionPts(t.first, strPtsMap[t.second]);
249 
250  // Read PAG offset nodes
251  while (F.good())
252  {
253  if (line == "------") break;
254  // Parse a single line in the form of "ID baseNodeID offset"
255  istringstream ss(line);
256  NodeID id;
257  NodeID base;
258  size_t offset;
259  ss >> id >> base >> offset;
260 
261  NodeID n = pag->getGepObjNode(pag->getObject(base), LocationSet(offset));
262  assert(id == n && "Error adding GepObjNode into PAG!");
263 
264  getline(F, line);
265  }
266 
267  // Read BaseNode insensitivity
268  while (F.good())
269  {
270  getline(F, line);
271  // Parse a single line in the form of "baseNodeID insensitive"
272  istringstream ss(line);
273  NodeID base;
274  bool insensitive;
275  ss >> base >> insensitive;
276 
277  if (insensitive)
278  setObjFieldInsensitive(base);
279  }
280 
281  // Update callgraph
282  updateCallGraph(pag->getIndirectCallsites());
283 
284  F.close();
285  outs() << "\n";
286 
287  return true;
288 }

395 void BVDataPTAImpl::normalizePointsTo() {
396  PAG::MemObjToFieldsMap &memToFieldsMap = pag->getMemToFieldsMap();
397  PAG::NodeLocationSetMap &GepObjNodeMap = pag->getGepObjNodeMap();
398 
399  // collect each gep node whose base node has been set as field-insensitive
400  NodeBS dropNodes;
401  for (auto t: memToFieldsMap){
402  NodeID base = t.first;
403  const MemObj* memObj = pag->getObject(base);
404  assert(memObj && "Invalid memobj in memToFieldsMap");
405  if (memObj->isFieldInsensitive()) {
406  for (NodeID id : t.second) {
407  if (SVFUtil::isa<GepObjPN>(pag->getPAGNode(id))) {
408  dropNodes.set(id);
409  } else
410  assert(id == base && "Not a GepObj Node or a baseObj Node?");
411  }
412  }
413  }
414 
415  // remove the collected redundant gep nodes in each pointers's pts
416  for (PAG::iterator nIter = pag->begin(); nIter != pag->end(); ++nIter) {
417  NodeID n = nIter->first;
418 
419  const PointsTo &tmpPts = getPts(n);
420  for (NodeID obj : tmpPts) {
421  if (!dropNodes.test(obj))
422  continue;
423  NodeID baseObj = pag->getBaseObjNode(obj);
424  clearPts(n, obj);
425  addPts(n, baseObj);
426  }
427  }
428 
429  // clear GepObjNodeMap and memToFieldsMap for redundant gepnodes
430  // and remove those nodes from pag
431  for (NodeID n: dropNodes) {
432  NodeID base = pag->getBaseObjNode(n);
433  GepObjPN *gepNode = SVFUtil::dyn_cast<GepObjPN>(pag->getPAGNode(n));
434  const LocationSet ls = gepNode->getLocationSet();
435  GepObjNodeMap.erase(std::make_pair(base, ls));
436  memToFieldsMap[base].reset(n);
437 
438  pag->removeGNode(gepNode);
439  }
440 }
