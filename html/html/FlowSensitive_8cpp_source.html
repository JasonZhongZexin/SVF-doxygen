162 bool FlowSensitive::processSVFGNode(SVFGNode* node)
163 {
164  double start = stat->getClk();
165  bool changed = false;
166  if(AddrSVFGNode* addr = SVFUtil::dyn_cast<AddrSVFGNode>(node))
167  {
168  numOfProcessedAddr++;
169  if(processAddr(addr))
170  changed = true;
171  }
172  else if(CopySVFGNode* copy = SVFUtil::dyn_cast<CopySVFGNode>(node))
173  {
174  numOfProcessedCopy++;
175  if(processCopy(copy))
176  changed = true;
177  }
178  else if(GepSVFGNode* gep = SVFUtil::dyn_cast<GepSVFGNode>(node))
179  {
180  numOfProcessedGep++;
181  if(processGep(gep))
182  changed = true;
183  }
184  else if(LoadSVFGNode* load = SVFUtil::dyn_cast<LoadSVFGNode>(node))
185  {
186  numOfProcessedLoad++;
187  if(processLoad(load))
188  changed = true;
189  }
190  else if(StoreSVFGNode* store = SVFUtil::dyn_cast<StoreSVFGNode>(node))
191  {
192  numOfProcessedStore++;
193  if(processStore(store))
194  changed = true;
195  }
196  else if(PHISVFGNode* phi = SVFUtil::dyn_cast<PHISVFGNode>(node))
197  {
198  numOfProcessedPhi++;
199  if (processPhi(phi))
200  changed = true;
201  }
202  else if(SVFUtil::isa<MSSAPHISVFGNode>(node) || SVFUtil::isa<FormalINSVFGNode>(node)
203  || SVFUtil::isa<FormalOUTSVFGNode>(node) || SVFUtil::isa<ActualINSVFGNode>(node)
204  || SVFUtil::isa<ActualOUTSVFGNode>(node))
205  {
206  numOfProcessedMSSANode++;
207  changed = true;
208  }
209  else if(SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalParmSVFGNode>(node)
210  || SVFUtil::isa<ActualRetSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node)
211  || SVFUtil::isa<NullPtrSVFGNode>(node))
212  {
213  changed = true;
214  }
215  else if(SVFUtil::isa<CmpVFGNode>(node) || SVFUtil::isa<BinaryOPVFGNode>(node) || SVFUtil::dyn_cast<UnaryOPVFGNode>(node))
216  {
217 
218  }
219  else
220  assert(false && "unexpected kind of SVFG nodes");
221 
222  double end = stat->getClk();
223  processTime += (end - start) / TIMEINTERVAL;
224 
225  return changed;

256 bool FlowSensitive::propAlongDirectEdge(const DirectSVFGEdge* edge)
257 {
258  double start = stat->getClk();
259  bool changed = false;
260 
261  SVFGNode* src = edge->getSrcNode();
262  SVFGNode* dst = edge->getDstNode();
263  // If this is an actual-param or formal-ret, top-level pointer's pts must be
264  // propagated from src to dst.
265  if (ActualParmSVFGNode* ap = SVFUtil::dyn_cast<ActualParmSVFGNode>(src))
266  changed = propagateFromAPToFP(ap, dst);
267  else if (FormalRetSVFGNode* fp = SVFUtil::dyn_cast<FormalRetSVFGNode>(src))
268  changed = propagateFromFRToAR(fp, dst);
269  else
270  {
271  // Direct SVFG edge links between def and use of a top-level pointer.
272  // There's no points-to information propagated along direct edge.
273  // Since the top-level pointer's value has been changed at src node,
274  // return TRUE to put dst node into the work list.
275  changed = true;
276  }
277 
278  double end = stat->getClk();
279  directPropaTime += (end - start) / TIMEINTERVAL;
280  return changed;
281 }

318 bool FlowSensitive::propAlongIndirectEdge(const IndirectSVFGEdge* edge)
319 {
320  double start = stat->getClk();
321 
322  SVFGNode* src = edge->getSrcNode();
323  SVFGNode* dst = edge->getDstNode();
324 
325  bool changed = false;
326 
327  // Get points-to targets may be used by next SVFG node.
328  // Propagate points-to set for node used in dst.
329  const PointsTo& pts = edge->getPointsTo();
330  for (PointsTo::iterator ptdIt = pts.begin(), ptdEit = pts.end(); ptdIt != ptdEit; ++ptdIt)
331  {
332  NodeID ptd = *ptdIt;
333 
334  if (propVarPtsFromSrcToDst(ptd, src, dst))
335  changed = true;
336 
337  if (isFieldInsensitive(ptd))
338  {
340  const NodeBS& allFields = getAllFieldsObjNode(ptd);
341  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end();
342  fieldIt != fieldEit; ++fieldIt)
343  {
344  if (propVarPtsFromSrcToDst(*fieldIt, src, dst))
345  changed = true;
346  }
347  }
348  }
349 
350  double end = stat->getClk();
351  indirectPropaTime += (end - start) / TIMEINTERVAL;
352  return changed;

427 bool FlowSensitive::processGep(const GepSVFGNode* edge)
428 {
429  double start = stat->getClk();
430  bool changed = false;
431  const PointsTo& srcPts = getPts(edge->getPAGSrcNodeID());
432 
433  PointsTo tmpDstPts;
434  if (SVFUtil::isa<VariantGepPE>(edge->getPAGEdge()))
435  {
436  for (NodeID o : srcPts)
437  {

651 void FlowSensitive::updateConnectedNodes(const SVFGEdgeSetTy& edges)
652 {
653  for (SVFGEdgeSetTy::const_iterator it = edges.begin(), eit = edges.end();
654  it != eit; ++it)
655  {
656  const SVFGEdge* edge = *it;
657  SVFGNode* dstNode = edge->getDstNode();
658  if (SVFUtil::isa<PHISVFGNode>(dstNode))
659  {
662  pushIntoWorklist(dstNode->getId());
663  }
664  else if (SVFUtil::isa<FormalINSVFGNode>(dstNode) || SVFUtil::isa<ActualOUTSVFGNode>(dstNode))
665  {
668  bool changed = false;
669 
670  SVFGNode* srcNode = edge->getSrcNode();
671 
672  const PointsTo& pts = SVFUtil::cast<IndirectSVFGEdge>(edge)->getPointsTo();
673  for (PointsTo::iterator ptdIt = pts.begin(), ptdEit = pts.end(); ptdIt != ptdEit; ++ptdIt)
674  {
675  NodeID ptd = *ptdIt;
676 
677  if (propVarPtsAfterCGUpdated(ptd, srcNode, dstNode))
678  changed = true;
679 
680  if (isFieldInsensitive(ptd))
681  {
683  const NodeBS& allFields = getAllFieldsObjNode(ptd);
684  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end();
685  fieldIt != fieldEit; ++fieldIt)
686  {
687  if (propVarPtsAfterCGUpdated(*fieldIt, srcNode, dstNode))
688  changed = true;
689  }
690  }
691  }
692 
693  if (changed)
694  pushIntoWorklist(dstNode->getId());
695  }
696  }
697 }

718 void FlowSensitive::printCTirAliasStats(void)
719 {
720  DCHGraph *dchg = SVFUtil::dyn_cast<DCHGraph>(chgraph);
721  assert(dchg && "eval-ctir-aliases needs DCHG.");
722 
723  // < SVFG node ID (loc), PAG node of interest (top-level pointer) >.
724  Set<std::pair<NodeID, NodeID>> cmpLocs;
725  for (SVFG::iterator npair = svfg->begin(); npair != svfg->end(); ++npair)
726  {
727  NodeID loc = npair->first;
728  SVFGNode *node = npair->second;
729 
730  // Only care about loads, stores, and GEPs.
731  if (StmtSVFGNode *stmt = SVFUtil::dyn_cast<StmtSVFGNode>(node))
732  {
733  if (!SVFUtil::isa<LoadSVFGNode>(stmt) && !SVFUtil::isa<StoreSVFGNode>(stmt)
734  && !SVFUtil::isa<GepSVFGNode>(stmt))
735  {
736  continue;
737  }
738 
739  if (!TypeBasedHeapCloning::getRawCTirMetadata(stmt->getInst() ? stmt->getInst() : stmt->getPAGEdge()->getValue()))
740  {
741  continue;
742  }
743 
744  NodeID p = 0;
745  if (SVFUtil::isa<LoadSVFGNode>(stmt))
746  {
747  p = stmt->getPAGSrcNodeID();
748  }
749  else if (SVFUtil::isa<StoreSVFGNode>(stmt))
750  {
751  p = stmt->getPAGDstNodeID();
752  }
753  else if (SVFUtil::isa<GepSVFGNode>(stmt))
754  {
755  p = stmt->getPAGSrcNodeID();
756  }
757  else
758  {
759  // Not interested.
760  continue;
761  }
762 
763  cmpLocs.insert(std::make_pair(loc, p));
764  }
765  }
766 
767  unsigned mayAliases = 0, noAliases = 0;
768  countAliases(cmpLocs, &mayAliases, &noAliases);
769 
770  unsigned total = mayAliases + noAliases;
771  llvm::outs() << "eval-ctir-aliases "
772  << total << " "
773  << mayAliases << " "
774  << noAliases << " "
775  << "\n";
776  llvm::outs() << " " << "TOTAL : " << total << "\n"
777  << " " << "MAY : " << mayAliases << "\n"
778  << " " << "MAY % : " << 100 * ((double)mayAliases/(double)(total)) << "\n"
779  << " " << "NO : " << noAliases << "\n"
780  << " " << "NO % : " << 100 * ((double)noAliases/(double)(total)) << "\n";
781 }

783 void FlowSensitive::countAliases(Set<std::pair<NodeID, NodeID>> cmp, unsigned *mayAliases, unsigned *noAliases)
784 {
785  for (std::pair<NodeID, NodeID> locPA : cmp)
786  {
787  // loc doesn't make a difference for FSPTA.
788  NodeID p = locPA.second;
