276  inline void addBBSuccessorPos(const BasicBlock *bb, const BasicBlock* succ,const u32_t pos)
277  {
278  Map<const BasicBlock*, const Map<const BasicBlock*, const u32_t>>::iterator bbSuccessorPosMapIter = bbSuccessorPosMap.find(bb);
279  if(bbSuccessorPosMapIter != bbSuccessorPosMap.end())
280  {
281  Map<const BasicBlock*, const u32_t> foundValue = bbSuccessorPosMapIter->second;
282  foundValue.insert({succ,pos});
283  bbSuccessorPosMap.insert({bb,foundValue});
284  }
285  else
286  {
287  Map<const BasicBlock*, const u32_t> valueMap;
288  valueMap.insert({succ,pos});
289  bbSuccessorPosMap.insert({bb,valueMap});
290  }
291  }
292 
293  inline void addBBPredecessorPos(const BasicBlock *bb, const BasicBlock* Pred,const u32_t pos)
294  {
295  Map<const BasicBlock*, const Map<const BasicBlock*, const u32_t>>::iterator bbPredecessorPosMapIter = bbPredecessorPosMap.find(bb);
296  if(bbPredecessorPosMapIter != bbPredecessorPosMap.end())
297  {
298  Map<const BasicBlock*, const u32_t> foundValue = bbPredecessorPosMapIter->second;
299  foundValue.insert({Pred,pos});
300  bbPredecessorPosMap.insert({bb,foundValue});
301  }
302  else
303  {
304  Map<const BasicBlock*, const u32_t> valueMap;
305  valueMap.insert({Pred,pos});
306  bbPredecessorPosMap.insert({bb,valueMap});
307  }
308  }
