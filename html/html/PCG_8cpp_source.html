42 bool PCG::mayHappenInParallelBetweenFunctions(const Function* fun1, const Function* fun2) const
43 {
44  // if neither of functions are spawnees, then they won't happen in parallel
45  if (isSpawneeFun(fun1) == false && isSpawneeFun(fun2) == false)
46  return false;
47  // if there exit one of the function are not spawner, spawnee or follower, then they won't happen in parallel
48  if (isSpawnerFun(fun1) == false && isSpawneeFun(fun1) == false && isFollowerFun(fun1) == false)
49  return false;
50  if (isSpawnerFun(fun2) == false && isSpawneeFun(fun2) == false && isFollowerFun(fun2) == false)
51  return false;
52 
53  return true;
54 }

118 void PCG::collectSpawners()
119 {
120 
122  FunWorkList worklist;
123  for (FunSet::iterator it = spawners.begin(), eit = spawners.end(); it != eit; ++it)
124  {
125  worklist.push(*it);
126  }
127  while (!worklist.empty())
128  {
129  const Function* fun = worklist.pop();
130  const SVFFunction* svffun = getSVFFun(fun);
131  PTACallGraphNode* funNode = callgraph->getCallGraphNode(svffun);
132  for (PTACallGraphNode::const_iterator it = funNode->InEdgeBegin(), eit = funNode->InEdgeEnd(); it != eit;
133  ++it)
134  {
135  PTACallGraphEdge* callEdge = (*it);
136  const Function* caller = callEdge->getSrcNode()->getFunction()->getLLVMFun();
137  if (isSpawnerFun(caller) == false)
138  {
139  worklist.push(caller);
140  addSpawnerFun(caller);
141  }
143  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->directCallsBegin(), deit =
144  callEdge->directCallsEnd(); dit != deit; ++dit)
145  {
146  addSpawnsite((*dit)->getCallSite());
147  }
148  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->indirectCallsBegin(), deit =
149  callEdge->indirectCallsEnd(); dit != deit; ++dit)

191 void PCG::identifyFollowers()
192 {
193 
194  for (CallInstSet::const_iterator sit = spawnSitesBegin(), esit = spawnSitesEnd(); sit != esit; ++sit)
195  {
196  const Instruction* inst = *sit;
197  BBWorkList bb_worklist;
198  Set<const BasicBlock*> visitedBBs;
199  bb_worklist.push(inst->getParent());
200  while (!bb_worklist.empty())
201  {
202  const BasicBlock* bb = bb_worklist.pop();
203  for (BasicBlock::const_iterator it = bb->begin(), eit = bb->end(); it != eit; ++it)
204  {
205  const Instruction* inst = &*it;
206  // mark the callee of this callsite as follower
207  // if this is an call/invoke instruction but not a spawn site
208  if ((SVFUtil::isCallSite(inst)) && !isSpawnsite(inst))
209  {
210  CallICFGNode* cbn = getCallICFGNode(inst);
211  if (callgraph->hasCallGraphEdge(cbn))
212  {
213  for (PTACallGraph::CallGraphEdgeSet::const_iterator cgIt = callgraph->getCallEdgeBegin(cbn),
214  ecgIt = callgraph->getCallEdgeEnd(cbn); cgIt != ecgIt; ++cgIt)
215  {
216  const PTACallGraphEdge* edge = *cgIt;
217  addFollowerFun(edge->getDstNode()->getFunction()->getLLVMFun());
218  }
219  }
220  }
221  }
222  for (succ_const_iterator succ_it = succ_begin(bb); succ_it != succ_end(bb); succ_it++)
223  {
224  const BasicBlock* succ_bb = *succ_it;
225  if (visitedBBs.count(succ_bb) == 0)
226  {
227  visitedBBs.insert(succ_bb);
228  bb_worklist.push(succ_bb);
229  }
230  }
231  }
232  }
