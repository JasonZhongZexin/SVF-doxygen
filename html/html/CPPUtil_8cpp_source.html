56 static bool isOperOverload(const string name)
57 {
58  u32_t leftnum = 0, rightnum = 0;
59  string subname = name;
60  size_t leftpos, rightpos;
61  leftpos = subname.find("<");
62  while (leftpos != string::npos)
63  {
64  subname = subname.substr(leftpos+1);
65  leftpos = subname.find("<");
66  leftnum++;
67  }
68  subname = name;
69  rightpos = subname.find(">");
70  while (rightpos != string::npos)
71  {
72  subname = subname.substr(rightpos+1);
73  rightpos = subname.find(">");
74  rightnum++;
75  }
76  if (leftnum != rightnum)
77  {
78  return true;
79  }
80  else
81  {
82  return false;
83  }
84 }

86 static string getBeforeParenthesis(const string &name)
87 {
88  size_t lastRightParen = name.rfind(")");
89  assert(lastRightParen > 0);
90 
91  s32_t paren_num = 1, pos;
92  for (pos = lastRightParen - 1; pos >= 0; pos--)
93  {
94  if (name[pos] == ')')
95  paren_num++;
96  if (name[pos] == '(')
97  paren_num--;
98  if (paren_num == 0)
99  break;
100  }
101  return name.substr(0, pos);
102 }

105 {
106  if (name.size() == 0 || name[name.size() - 1] != '>')
107  {

111  for (pos = name.size() - 2; pos >= 0; pos--)
112  {
113  if (name[pos] == '>')
114  bracket_num++;
115  if (name[pos] == '<')

135 static void handleThunkFunction(cppUtil::DemangledName &dname)
136 {
137  // when handling multi-inheritance,
138  // the compiler may generate thunk functions
139  // to perform `this` pointer adjustment
140  // they are indicated with `virtual thunk to `
141  // and `nun-virtual thunk to`.
142  // if the classname starts with part of a
143  // demangled name starts with
144  // these prefixes, we need to remove the prefix
145  // to get the real class name
146 
147  static vector<string> thunkPrefixes = {VThunkFuncLabel, NVThunkFunLabel};
148  for (unsigned i = 0; i < thunkPrefixes.size(); i++)
149  {
150  auto prefix = thunkPrefixes[i];
151  if (dname.className.size() > prefix.size() &&
152  dname.className.compare(0, prefix.size(), prefix) == 0)
153  {
154  dname.className = dname.className.substr(prefix.size());
155  dname.isThunkFunc = true;
156  return;
157 
158  }
159  }
160 }

177  * ^

183 struct cppUtil::DemangledName cppUtil::demangle(const string &name)
184 {
185  struct cppUtil::DemangledName dname;
186  dname.isThunkFunc = false;
187 
188  s32_t status;
189  char *realname = abi::__cxa_demangle(name.c_str(), 0, 0, &status);
190  if (realname == nullptr)
191  {
192  dname.className = "";
193  dname.funcName = "";
194  }
195  else
196  {
197  string realnameStr = string(realname);
198  string beforeParenthesis = getBeforeParenthesis(realnameStr);
199  if (beforeParenthesis.find("::") == string::npos ||
200  isOperOverload(beforeParenthesis))
201  {
202  dname.className = "";
203  dname.funcName = "";
204  }
205  else
206  {
207  string beforeBracket = getBeforeBrackets(beforeParenthesis);
208  size_t colon = beforeBracket.rfind("::");
209  if (colon == string::npos)
210  {
211  dname.className = "";
212  dname.funcName = "";
213  }
214  else
215  {
216  dname.className = beforeParenthesis.substr(0, colon);
217  dname.funcName = beforeParenthesis.substr(colon + 2);
218  }
219  }
220  std::free(realname);
221  }
222 
223  handleThunkFunction(dname);
224 
225  return dname;
226 }

260 {

262  if (cs.getCalledFunction() != nullptr || cs.arg_empty())

266  // virtual annotations.

268  {

270  }
271 
272  const Value *vfunc = cs.getCalledValue();
273  if (const LoadInst *vfuncloadinst = SVFUtil::dyn_cast<LoadInst>(vfunc))
274  {
275  const Value *vfuncptr = vfuncloadinst->getPointerOperand();
276  if (const GetElementPtrInst *vfuncptrgepinst =
277  SVFUtil::dyn_cast<GetElementPtrInst>(vfuncptr))

283  {
284  return true;
285  }
286  }
287  }

484  }

487  return true;
