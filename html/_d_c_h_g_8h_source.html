291  const DIType *getFieldType(const DIType *base, unsigned idx)
292  {
293  base = getCanonicalType(base);
294  if (base == nullptr)
295  {
296  // Conservative; the base object is untyped, sadly.
297  return nullptr;
298  }
299 
300  // For TBHC this is conservative because the union type is lower in the DCHG
301  // than its fields. TODO: make more precise.
302  if (base->getTag() == dwarf::DW_TAG_union_type)
303  {
304  return base;
305  }
306 
307  if (base->getTag() == dwarf::DW_TAG_array_type)
308  {
309  const DICompositeType *cbase = SVFUtil::dyn_cast<DICompositeType>(base);
310  assert(cbase && "DCHG: bad DIComposite case");
311  return cbase->getBaseType();
312  }
313 
314  if (!(base->getTag() == dwarf::DW_TAG_class_type
315  || base->getTag() == dwarf::DW_TAG_structure_type))
316  {
317  return nullptr;
318  }
319 
320  assert(fieldTypes.find(base) != fieldTypes.end() && "DCHG: base not flattened!");
321  std::vector<const DIType *> &fields = fieldTypes[base];
322  assert(fields.size() > idx && "DCHG: idx into struct larger than # fields!");
323  return getCanonicalType(fields[idx]);
324  }
