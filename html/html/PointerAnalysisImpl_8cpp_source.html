28 BVDataPTAImpl::BVDataPTAImpl(PAG* p, PointerAnalysis::PTATY type, bool alias_check) :
29  PointerAnalysis(p, type, alias_check)
30 {
31  if (type == Andersen_BASE || type == Andersen_WPA || type == AndersenWaveDiff_WPA || type == AndersenHCD_WPA || type == AndersenHLCD_WPA
32  || type == AndersenLCD_WPA || type == TypeCPP_WPA || type == FlowS_DDA || type == AndersenWaveDiffWithType_WPA
33  || type == AndersenSCD_WPA || type == AndersenSFR_WPA)
34  {
35  // Only maintain reverse points-to when the analysis is field-sensitive, as objects turning
36  // field-insensitive is all it is used for.
37  bool maintainRevPts = Options::MaxFieldLimit != 0;
38  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(maintainRevPts);
39  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), maintainRevPts);
40  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
41  }
42  else if (type == Steensgaard_WPA)
43  {
44  // Steensgaard is only field-insensitive (for now?), so no reverse points-to.
45  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDiffPTDataTy(false);
46  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDiffPTDataTy(getPtCache(), false);
47  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
48  }
49  else if (type == FSSPARSE_WPA || type == FSTBHC_WPA)
50  {
51  if (Options::INCDFPTData)
52  {
53  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutIncDFPTDataTy(false);
54  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersIncDFPTDataTy(getPtCache(), false);
55  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
56  }
57  else
58  {
59  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutDFPTDataTy(false);
60  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersDFPTDataTy(getPtCache(), false);
61  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
62  }
63  }
64  else if (type == VFS_WPA)
65  {
66  if (Options::ptDataBacking == PTBackingType::Mutable) ptD = new MutVersionedPTDataTy(false);
67  else if (Options::ptDataBacking == PTBackingType::Persistent) ptD = new PersVersionedPTDataTy(getPtCache(), false);
68  else assert(false && "BVDataPTAImpl::BVDataPTAImpl: unexpected points-to backing type!");
69  }
70  else assert(false && "no points-to data available");
71 
72  ptaImplTy = BVDataImpl;

169 bool BVDataPTAImpl::readFromFile(const string& filename)
170 {
171  // If the module annotations are available, read from there instead
172  auto mainModule = SVF::LLVMModuleSet::getLLVMModuleSet()->getMainLLVMModule();
173  if (mainModule->getNamedMetadata("PAG-Annotated") != nullptr)
174  {
175  return readFromModule();
176  }
177 
178  outs() << "Loading pointer analysis results from '" << filename << "'...";
179 
180  ifstream F(filename.c_str());
181  if (!F.is_open())
182  {
183  outs() << " error opening file for reading!\n";
184  return false;
185  }
186 
187  // Read analysis results from file
188  PTDataTy *ptD = getPTDataTy();
189  string line;
190 
191  // Read points-to sets
192  string delimiter1 = " -> { ";
193  string delimiter2 = " }";
194  map<NodeID, string> nodePtsMap;
195  map<string, PointsTo> strPtsMap;
196 
197  while (F.good())
198  {
199  // Parse a single line in the form of "var -> { obj1 obj2 obj3 }"
200  getline(F, line);
201  size_t pos = line.find(delimiter1);
202  if (pos == string::npos) break;
203  if (line.back() != '}') break;
204 
205  // var
206  NodeID var = atoi(line.substr(0, pos).c_str());
207 
208  // objs
209  pos = pos + delimiter1.length();
210  size_t len = line.length() - pos - delimiter2.length();
211  string objs = line.substr(pos, len);
212  PointsTo dstPts;
213 
214  if (!objs.empty())
215  {
216  // map the variable ID to its unique string pointer set
217  nodePtsMap[var] = objs;
218  if (strPtsMap.count(objs)) continue;
219 
220  istringstream ss(objs);
221  NodeID obj;
222  while (ss.good())
223  {
224  ss >> obj;
225  dstPts.set(obj);
226  }
227  // map the string pointer set to the parsed PointsTo set
228  strPtsMap[objs] = dstPts;
229  }
230  }
231 
232  // map the variable ID to its pointer set
233  for (auto t: nodePtsMap)
234  ptD->unionPts(t.first, strPtsMap[t.second]);
235 
236  // Read PAG offset nodes
237  while (F.good())
238  {
239  if (line == "------") break;
240  // Parse a single line in the form of "ID baseNodeID offset"
241  istringstream ss(line);
242  NodeID id;
243  NodeID base;
244  size_t offset;
245  ss >> id >> base >> offset;
246 
247  NodeID n = pag->getGepObjNode(pag->getObject(base), LocationSet(offset));
248  assert(id == n && "Error adding GepObjNode into PAG!");
249 
250  getline(F, line);
251  }
252 
253  // Read BaseNode insensitivity
254  while (F.good())
255  {
256  getline(F, line);
257  // Parse a single line in the form of "baseNodeID insensitive"
258  istringstream ss(line);
259  NodeID base;
260  bool insensitive;
261  ss >> base >> insensitive;
262 
263  if (insensitive)
264  setObjFieldInsensitive(base);
265  }
266 
267  // Update callgraph
268  updateCallGraph(pag->getIndirectCallsites());
269 
270  F.close();
271  outs() << "\n";
272 
273  return true;
274 }

381 void BVDataPTAImpl::normalizePointsTo() {
382  PAG::MemObjToFieldsMap &memToFieldsMap = pag->getMemToFieldsMap();
383  PAG::NodeLocationSetMap &GepObjNodeMap = pag->getGepObjNodeMap();
384 
385  // collect each gep node whose base node has been set as field-insensitive
386  NodeBS dropNodes;
387  for (auto t: memToFieldsMap){
388  NodeID base = t.first;
389  const MemObj* memObj = pag->getObject(base);
390  assert(memObj && "Invalid memobj in memToFieldsMap");
391  if (memObj->isFieldInsensitive()) {
392  for (NodeID id : t.second) {
393  if (SVFUtil::isa<GepObjPN>(pag->getPAGNode(id))) {
394  dropNodes.set(id);
395  } else
396  assert(id == base && "Not a GepObj Node or a baseObj Node?");
397  }
398  }
399  }
400 
401  // remove the collected redundant gep nodes in each pointers's pts
402  for (PAG::iterator nIter = pag->begin(); nIter != pag->end(); ++nIter) {
403  NodeID n = nIter->first;
404 
405  const PointsTo &tmpPts = getPts(n);
406  for (NodeID obj : tmpPts) {
407  if (!dropNodes.test(obj))
408  continue;
409  NodeID baseObj = pag->getBaseObjNode(obj);
410  clearPts(n, obj);
411  addPts(n, baseObj);
412  }
413  }
414 
415  // clear GepObjNodeMap and memToFieldsMap for redundant gepnodes
416  // and remove those nodes from pag
417  for (NodeID n: dropNodes) {
418  NodeID base = pag->getBaseObjNode(n);
419  GepObjPN *gepNode = SVFUtil::dyn_cast<GepObjPN>(pag->getPAGNode(n));
420  const LocationSet ls = gepNode->getLocationSet();
421  GepObjNodeMap.erase(std::make_pair(base, ls));
422  memToFieldsMap[base].reset(n);
423 
424  pag->removeGNode(gepNode);
425  }
426 }
