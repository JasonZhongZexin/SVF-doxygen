229  SymID getValSym(const Value *val)
230  {
231 
232  if(isNullPtrSym(val))
233  return nullPtrSymID();
234  else if (SVFUtil::isBlackholeSym(val))
235  return blkPtrSymID();
236  else
237  {
238  ValueToIDMapTy::const_iterator iter = valSymMap.find(val);
239  assert(iter!=valSymMap.end() &&"value sym not found");
240  return iter->second;
241  }
242  }

244  inline bool hasValSym(const Value* val)
245  {
246  if (isNullPtrSym(val) || SVFUtil::isBlackholeSym(val))
247  return true;
248  else
