59 
60  for (const auto &func: *M)

84 void SaberCondAllocator::allocateForBB(const BasicBlock &bb)
85 {
86 
87  u32_t succ_number = SymbolTableInfo::getBBSuccessorNum(&bb);
88 
89  // if successor number greater than 1, allocate new decision variable for successors
90  if (succ_number > 1)
91  {
92 
93  //allocate log2(num_succ) decision variables
94  double num = log(succ_number) / log(2);
95  u32_t bit_num = (u32_t) ceil(num);
96  u32_t succ_index = 0;
97  std::vector<Condition> condVec;
98  for (u32_t i = 0; i < bit_num; i++)
99  {
100  condVec.push_back(newCond(bb.getTerminator()));
101  }
102 
103  // iterate each successor
104  for (succ_const_iterator succ_it = succ_begin(&bb);
105  succ_it != succ_end(&bb);
106  succ_it++, succ_index++)
107  {
108 
109  const BasicBlock *succ = *succ_it;
110 
111  Condition path_cond = getTrueCond();
112 
114 
115  // for each successor decide its bit representation
116  // decide whether each bit of succ_index is 1 or 0, if (three successor) succ_index is 000 then use C1^C2^C3
117  // if 001 use C1^C2^negC3
118  for (u32_t j = 0; j < bit_num; j++)
119  {
120  //test each bit of this successor's index (binary representation)
121  u32_t tool = 0x01 << j;
122  if (tool & succ_index)
123  {
124  path_cond = condAnd(path_cond, (condNeg(condVec.at(j))));
125  }
126  else
127  {
128  path_cond = condAnd(path_cond, condVec.at(j));
129  }
130  }
131  setBranchCond(&bb, succ, path_cond);
132  }
133 
134  }
135 }

184 SaberCondAllocator::evaluateTestNullLikeExpr(const BranchStmt *branchStmt, const BasicBlock *succ)
185 {
186 
187  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
188 
189  if (isTestNullExpr(branchStmt->getCondition()->getValue()))
190  {
191  // succ is then branch
192  if (succ1 == succ)
193  return getFalseCond();
194  // succ is else branch
195  else
196  return getTrueCond();
197  }
198  if (isTestNotNullExpr(branchStmt->getCondition()->getValue()))
199  {
200  // succ is then branch

204  else
205  return getFalseCond();
206  }
207  return Condition::nullExpr();

213 SaberCondAllocator::Condition SaberCondAllocator::evaluateProgExit(const BranchStmt *branchStmt, const BasicBlock *succ)
214 {
215  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
216  const BasicBlock *succ2 = branchStmt->getSuccessor(1)->getBB();
217 
218  bool branch1 = isBBCallsProgExit(succ1);
219  bool branch2 = isBBCallsProgExit(succ2);
220 
222  if (branch1 && !branch2)
223  {
224  // succ is then branch
225  if (succ1 == succ)
226  return getFalseCond();
227  // succ is else branch
228  else
229  return getTrueCond();
230  }
232  else if (!branch1 && branch2)
233  {
234  // succ is else branch
235  if (succ2 == succ)
236  return getFalseCond();
237  // succ is then branch
238  else
239  return getTrueCond();
240  }
241  // two branches both call program exit
242  else if (branch1 && branch2)
243  {
244  return getFalseCond();
245  }
247  else
248  return Condition::nullExpr();
249

257 SaberCondAllocator::Condition SaberCondAllocator::evaluateLoopExitBranch(const BasicBlock *bb, const BasicBlock *dst)
258 {
259  const Function *fun = bb->getParent();
260  assert(fun == dst->getParent() && "two basic blocks should be in the same function");
261 
262  const SVFFunction * svffun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
263  if (svffun->isLoopHeader(bb))
264  {
265  Set<const BasicBlock *> filteredbbs;
266  Set<const BasicBlock*> exitbbs;
267  svffun->getExitBlocksOfLoop(bb,exitbbs);
269  for(const BasicBlock *eb : exitbbs)
270  {
271  if(!isBBCallsProgExit(eb))
272  filteredbbs.insert(eb);
273  }
274 
276  bool allPDT = true;
277  for (const auto &filteredbb: filteredbbs)
278  {
279  if (!postDominate(dst, filteredbb))
280  allPDT = false;
281  }
282 
283  if (allPDT)
284  return getTrueCond();
285  }
286  return Condition::nullExpr();

294 SaberCondAllocator::Condition SaberCondAllocator::evaluateBranchCond(const BasicBlock *bb, const BasicBlock *succ)
295 {
296  if(SymbolTableInfo::getBBSuccessorNum(bb) == 1)
297  {
298  assert(bb->getTerminator()->getSuccessor(0) == succ && "not the unique successor?");
299  return getTrueCond();
300  }
301 
302  if (ICFGNode *icfgNode = getICFG()->getICFGNode(bb->getTerminator()))
303  {
304  for (const auto &svfStmt: icfgNode->getSVFStmts())
305  {
306  if (const BranchStmt *branchStmt = SVFUtil::dyn_cast<BranchStmt>(svfStmt))
307  {
308  if (branchStmt->getNumSuccessors() == 2)
309  {
310  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
311  const BasicBlock *succ2 = branchStmt->getSuccessor(1)->getBB();
312  assert((succ1 == succ || succ2 == succ) && "not a successor??");
313 
314  Condition evalLoopExit = evaluateLoopExitBranch(bb, succ);
315  if (!eq(evalLoopExit, Condition::nullExpr()))
316  return evalLoopExit;
317 
318  Condition evalProgExit = evaluateProgExit(branchStmt, succ);
319  if (!eq(evalProgExit, Condition::nullExpr()))
320  return evalProgExit;
321 
322  Condition evalTestNullLike = evaluateTestNullLikeExpr(branchStmt, succ);
323  if (!eq(evalTestNullLike, Condition::nullExpr()))
324  return evalTestNullLike;
325  break;
326  }
327  }
328  }
329  }
330 
331  return getBranchCond(bb, succ);
332 }

500 SaberCondAllocator::Condition SaberCondAllocator::ComputeIntraVFGGuard(const BasicBlock *srcBB, const BasicBlock *dstBB)
501 {
502 
503  assert(srcBB->getParent() == dstBB->getParent() && "two basic blocks are not in the same function??");
504 
505  if (postDominate(dstBB, srcBB))
506  return getTrueCond();
507 
508  CFWorkList worklist;
509  worklist.push(srcBB);
510  setCFCond(srcBB, getTrueCond());
511 
512  while (!worklist.empty())
513  {
514  const BasicBlock *bb = worklist.pop();
515  Condition cond = getCFCond(bb);
516 
519  Condition loopExitCond = evaluateLoopExitBranch(bb, dstBB);
520  if (!eq(loopExitCond, Condition::nullExpr()))
521  return condAnd(cond, loopExitCond);
522 
523 
524  for (succ_const_iterator succ_it = succ_begin(bb);
525  succ_it != succ_end(bb); succ_it++)
526  {
527  const BasicBlock *succ = *succ_it;
532  Condition brCond;
533  if (postDominate(succ, bb))
534  brCond = getTrueCond();
535  else
536  brCond = getEvalBrCond(bb, succ);
537 
538  DBOUT(DSaber, outs() << " bb (" << bb->getName().str() <<
539  ") --> " << "succ_bb (" << succ->getName().str() << ") condition: " << brCond << "\n");
540  Condition succPathCond = condAnd(cond, brCond);
541  if (setCFCond(succ, condOr(getCFCond(succ), succPathCond)))
542  worklist.push(succ);
543  }
544  }
545 
546  DBOUT(DSaber, outs() << " src_bb (" << srcBB->getName().str() <<
547  ") --> " << "dst_bb (" << dstBB->getName().str() << ") condition: " << getCFCond(dstBB)
548  << "\n");
549 
550  return getCFCond(dstBB);
551 }
