66 void SaberSVFGBuilder::collectGlobals(BVDataPTAImpl* pta)
67 {
68  SVFIR* pag = svfg->getPAG();
69  NodeVector worklist;
70  for(SVFIR::iterator it = pag->begin(), eit = pag->end(); it!=eit; it++)
71  {
72  PAGNode* pagNode = it->second;
73  if(SVFUtil::isa<DummyValVar>(pagNode) || SVFUtil::isa<DummyObjVar>(pagNode))
74  continue;
75 
76  if(GepObjVar* gepobj = SVFUtil::dyn_cast<GepObjVar>(pagNode))
77  {
78  if(SVFUtil::isa<DummyObjVar>(pag->getGNode(gepobj->getBaseNode())))
79  continue;
80  }
81  if(const Value* val = pagNode->getValue())
82  {
83  if(SVFUtil::isa<GlobalVariable>(val))
84  worklist.push_back(it->first);
85  }
86  }
87 
88  NodeToPTSSMap cachedPtsMap;
89  while(!worklist.empty())
90  {
91  NodeID id = worklist.back();
92  worklist.pop_back();
93  globs.set(id);
94  const PointsTo& pts = pta->getPts(id);
95  for(PointsTo::iterator it = pts.begin(), eit = pts.end(); it!=eit; ++it)
96  {
97  globs |= CollectPtsChain(pta,*it,cachedPtsMap);
98  }
99  }
100 }

102 PointsTo& SaberSVFGBuilder::CollectPtsChain(BVDataPTAImpl* pta,NodeID id, NodeToPTSSMap& cachedPtsMap)
103 {
104  SVFIR* pag = svfg->getPAG();
105 
106  NodeID baseId = pag->getBaseObjVar(id);
107  NodeToPTSSMap::iterator it = cachedPtsMap.find(baseId);
108  if(it!=cachedPtsMap.end())
109  return it->second;
110  else
111  {
112  PointsTo& pts = cachedPtsMap[baseId];
113  pts |= pag->getFieldsAfterCollapse(baseId);
114 
115  WorkList worklist;
116  for(PointsTo::iterator it = pts.begin(), eit = pts.end(); it!=eit; ++it)
117  worklist.push(*it);
118 
119  while(!worklist.empty())
120  {
121  NodeID nodeId = worklist.pop();
122  const PointsTo& tmp = pta->getPts(nodeId);
123  for(PointsTo::iterator it = tmp.begin(), eit = tmp.end(); it!=eit; ++it)
124  {
125  pts |= CollectPtsChain(pta,*it,cachedPtsMap);
126  }
127  }
128  return pts;
129  }

146 void SaberSVFGBuilder::rmDerefDirSVFGEdges(BVDataPTAImpl* pta)
147 {
148 
149  for(SVFG::iterator it = svfg->begin(), eit = svfg->end(); it!=eit; ++it)
150  {
151  const SVFGNode* node = it->second;
152 
153  if(const StmtSVFGNode* stmtNode = SVFUtil::dyn_cast<StmtSVFGNode>(node))
154  {
156  if(SVFUtil::isa<StoreSVFGNode>(stmtNode))
157  {
158  const SVFGNode* def = svfg->getDefSVFGNode(stmtNode->getPAGDstNode());
159  if(SVFGEdge* edge = svfg->getIntraVFGEdge(def,stmtNode,SVFGEdge::IntraDirectVF))
160  svfg->removeSVFGEdge(edge);
161  else
162  assert((svfg->getKind()==VFG::FULLSVFG_OPT || svfg->getKind()==VFG::PTRONLYSVFG_OPT) && "Edge not found!");
163 
164  if(accessGlobal(pta,stmtNode->getPAGDstNode()))
165  {
166  globSVFGNodes.insert(stmtNode);
167  }
168  }
169  else if(SVFUtil::isa<LoadSVFGNode>(stmtNode))
170  {
171  const SVFGNode* def = svfg->getDefSVFGNode(stmtNode->getPAGSrcNode());
172  if(SVFGEdge* edge = svfg->getIntraVFGEdge(def,stmtNode,SVFGEdge::IntraDirectVF))
173  svfg->removeSVFGEdge(edge);
174  else
175  assert((svfg->getKind()==VFG::FULLSVFG_OPT || svfg->getKind()==VFG::PTRONLYSVFG_OPT) && "Edge not found!");
176 
177  if(accessGlobal(pta,stmtNode->getPAGSrcNode()))
178  {
179  globSVFGNodes.insert(stmtNode);
180  }
181  }

183  }

190 bool SaberSVFGBuilder::isStrongUpdate(const SVFGNode* node, NodeID& singleton, BVDataPTAImpl* pta)
191 {
192  bool isSU = false;
193  if (const StoreSVFGNode* store = SVFUtil::dyn_cast<StoreSVFGNode>(node))
194  {
195  const PointsTo& dstCPSet = pta->getPts(store->getPAGDstNodeID());
196  if (dstCPSet.count() == 1)
197  {
199  PointsTo::iterator it = dstCPSet.begin();
200  singleton = *it;
201 
202  // Strong update can be made if this points-to target is not heap, array or field-insensitive.
203  if (!pta->isHeapMemObj(singleton) && !pta->isArrayMemObj(singleton)
204  && SVFIR::getPAG()->getBaseObj(singleton)->isFieldInsensitive() == false
205  && !pta->isLocalVarInRecursiveFun(singleton))
206  {
207  isSU = true;
208  }
209  }
210  }
211  return isSU;
212 }

260 {

262  for(SVFIR::CSToArgsListMap::iterator it = pag->getCallSiteArgsMap().begin(),
263  eit = pag->getCallSiteArgsMap().end(); it!=eit; ++it)
264  {
265  PTACallGraph::FunctionSet callees;

268  ecit = callees.end(); cit != ecit; cit++)
269  {
270 
271  const SVFFunction* fun = *cit;
