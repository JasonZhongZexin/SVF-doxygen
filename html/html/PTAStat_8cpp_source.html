114 double PTAStat::getClk(bool mark) {
115  if (Options::MarkedClocksOnly && !mark) return 0.0;
116 
117  if (Options::ClockType == ClockType::Wall)
118  {
119  struct timespec time;
120  clock_gettime(CLOCK_MONOTONIC, &time);
121  return (double)(time.tv_nsec + time.tv_sec * 1000000000) / 1000000.0;
122  }
123  else if (Options::ClockType == ClockType::CPU)
124  {
125  return CLOCK_IN_MS();
126  }
127 
128  assert(false && "PTAStat::getClk: unknown clock type");
129  abort();
130 }

132 void PTAStat::performStat()
133 {
134 
135  callgraphStat();
136 
137  SVFIR* pag = SVFIR::getPAG();
138  u32_t numOfFunction = 0;
139  u32_t numOfGlobal = 0;
140  u32_t numOfStack = 0;
141  u32_t numOfHeap = 0;
142  u32_t numOfHasVarArray = 0;
143  u32_t numOfHasVarStruct = 0;
144  u32_t numOfHasConstArray = 0;
145  u32_t numOfHasConstStruct = 0;
146  u32_t numOfScalar = 0;
147  u32_t numOfConstant = 0;
148  u32_t fiObjNumber = 0;
149  u32_t fsObjNumber = 0;
150  Set<SymID> memObjSet;
151  for(SVFIR::iterator it = pag->begin(), eit = pag->end(); it!=eit; ++it)
152  {
153  PAGNode* node = it->second;
154  if(ObjVar* obj = SVFUtil::dyn_cast<ObjVar>(node))
155  {
156  const MemObj* mem = obj->getMemObj();
157  if (memObjSet.insert(mem->getId()).second == false)
158  continue;
159  if(mem->isBlackHoleObj())
160  continue;
161  if(mem->isFunction())
162  numOfFunction++;
163  if(mem->isGlobalObj())
164  numOfGlobal++;
165  if(mem->isStack())
166  numOfStack++;
167  if(mem->isHeap())
168  numOfHeap++;
169  if(mem->isVarArray())
170  numOfHasVarArray++;
171  if(mem->isVarStruct())
172  numOfHasVarStruct++;
173  if(mem->isConstantArray())
174  numOfHasConstArray++;
175  if(mem->isConstantStruct())
176  numOfHasConstStruct++;
177  if(mem->hasPtrObj() == false)
178  numOfScalar++;
179  if(mem->isConstDataOrConstGlobal())
180  numOfConstant++;
181 
182  if (mem->isFieldInsensitive())
183  fiObjNumber++;
184  else
185  fsObjNumber++;
186 
187  if(pta->isLocalVarInRecursiveFun(node->getId()))
188  {
189  localVarInRecursion.set(node->getId());
190  }
191  }
192  }
193 
194 
195 
196  generalNumMap[TotalNumOfPointers] = pag->getValueNodeNum() + pag->getFieldValNodeNum();
197  generalNumMap[TotalNumOfObjects] = pag->getObjectNodeNum();
198  generalNumMap[TotalNumOfFieldObjects] = pag->getFieldObjNodeNum();
199  generalNumMap[MaxStructSize] = SymbolTableInfo::SymbolInfo()->getMaxStructSize();
200  generalNumMap[TotalNumOfEdges] = pag->getPAGEdgeNum();
201  generalNumMap["TotalPTASVFStmts"] = pag->getPTAPAGEdgeNum();
202  generalNumMap[NumberOfFieldInsensitiveObj] = fiObjNumber;
203  generalNumMap[NumberOfFieldSensitiveObj] = fsObjNumber;
204 
205  generalNumMap[NumOfAddrs] = pag->getSVFStmtSet(SVFStmt::Addr).size();
206  generalNumMap[NumOfLoads] = pag->getSVFStmtSet(SVFStmt::Load).size();
207  generalNumMap[NumOfStores] = pag->getSVFStmtSet(SVFStmt::Store).size();
208  generalNumMap[NumOfCopys] = pag->getSVFStmtSet(SVFStmt::Copy).size();
209  generalNumMap[NumOfGeps] = pag->getSVFStmtSet(SVFStmt::Gep).size();
210  generalNumMap[NumOfCalls] = pag->getSVFStmtSet(SVFStmt::Call).size();
211  generalNumMap[NumOfReturns] = pag->getSVFStmtSet(SVFStmt::Ret).size();
212 
213  generalNumMap[NumOfFunctionObjs] = numOfFunction;
214  generalNumMap[NumOfGlobalObjs] = numOfGlobal;
215  generalNumMap[NumOfHeapObjs] = numOfHeap;
216  generalNumMap[NumOfStackObjs] = numOfStack;
217 
218  generalNumMap[NumOfObjsHasVarStruct] = numOfHasVarStruct;
219  generalNumMap[NumOfObjsHasVarArray] = numOfHasVarArray;
220  generalNumMap[NumOfObjsHasConstStruct] = numOfHasConstStruct;
221  generalNumMap[NumOfObjsHasConstArray] = numOfHasConstArray;
222  generalNumMap[NumOfNonPtrObjs] = numOfScalar;
223 
224  generalNumMap[NumOfIndirectCallSites] = pag->getIndirectCallsites().size();
225  generalNumMap["TotalCallSite"] = pag->getCallSiteSet().size();
226  generalNumMap["LocalVarInRecur"] = localVarInRecursion.count();
227  bitcastInstStat();
228  branchStat();
229 
230  printStat("General Stats");
231 
232 }

234 void PTAStat::callgraphStat()
235 {
236 
237  PTACallGraph* graph = pta->getPTACallGraph();
238  PointerAnalysis::CallGraphSCC* callgraphSCC = new PointerAnalysis::CallGraphSCC(graph);
239  callgraphSCC->find();
240 
241  unsigned totalNode = 0;
242  unsigned totalCycle = 0;
243  unsigned nodeInCycle = 0;
244  unsigned maxNodeInCycle = 0;
245  unsigned totalEdge = 0;
246  unsigned edgeInCycle = 0;
247 
248  NodeSet sccRepNodeSet;
249  PTACallGraph::iterator it = graph->begin();
250  PTACallGraph::iterator eit = graph->end();
251  for (; it != eit; ++it)
252  {
253  totalNode++;
254  if(callgraphSCC->isInCycle(it->first))
255  {
256  sccRepNodeSet.insert(callgraphSCC->repNode(it->first));
257  nodeInCycle++;
258  const NodeBS& subNodes = callgraphSCC->subNodes(it->first);
259  if(subNodes.count() > maxNodeInCycle)
260  maxNodeInCycle = subNodes.count();
261  }
262 
263  PTACallGraphNode::const_iterator edgeIt = it->second->InEdgeBegin();
264  PTACallGraphNode::const_iterator edgeEit = it->second->InEdgeEnd();
265  for (; edgeIt != edgeEit; ++edgeIt)
266  {
267  PTACallGraphEdge *edge = *edgeIt;
268  totalEdge+= edge->getDirectCalls().size() + edge->getIndirectCalls().size();
269  if(callgraphSCC->repNode(edge->getSrcID()) == callgraphSCC->repNode(edge->getDstID()))
270  {
271  edgeInCycle+=edge->getDirectCalls().size() + edge->getIndirectCalls().size();
272  }
273  }
274  }
275 
276  totalCycle = sccRepNodeSet.size();
277 
278  PTNumStatMap["TotalNode"] = totalNode;
279  PTNumStatMap["TotalCycle"] = totalCycle;
280  PTNumStatMap["NodeInCycle"] = nodeInCycle;
281  PTNumStatMap["MaxNodeInCycle"] = maxNodeInCycle;
282  PTNumStatMap["TotalEdge"] = totalEdge;
283  PTNumStatMap["CalRetPairInCycle"] = edgeInCycle;
284 
285  PTAStat::printStat("CallGraph Stats");
286 
287  delete callgraphSCC;
288 }

326 void PTAStat::bitcastInstStat()
327 {
328  SVFModule* module = pta->getModule();
329  u32_t numberOfBitCast = 0;
330  for (SVFModule::llvm_const_iterator funIter = module->llvmFunBegin(), funEiter = module->llvmFunEnd();
331  funIter != funEiter; ++funIter)
332  {
333  const Function* func = *funIter;
334  for (Function::const_iterator bbIt = func->begin(), bbEit = func->end();
335  bbIt != bbEit; ++bbIt)
336  {
337  const BasicBlock& bb = *bbIt;
338  for (BasicBlock::const_iterator instIt = bb.begin(), instEit = bb.end();
339  instIt != instEit; ++instIt)
340  {
341  const Instruction& inst = *instIt;
342  if (const BitCastInst* bitcast = SVFUtil::dyn_cast<BitCastInst>(&inst))
343  {
344  if (SVFUtil::isa<PointerType>(bitcast->getSrcTy()))
345  numberOfBitCast++;
346  }
347  }
348  }
349  }
350 
351  generalNumMap["BitCastNumber"] = numberOfBitCast;
352 }

354 void PTAStat::branchStat()
355 {
356  SVFModule* module = pta->getModule();
357  u32_t numOfBB_2Succ = 0;
358  u32_t numOfBB_3Succ = 0;
359  for (SVFModule::llvm_const_iterator funIter = module->llvmFunBegin(), funEiter = module->llvmFunEnd();
360  funIter != funEiter; ++funIter)
361  {
362  const Function* func = *funIter;
363  for (Function::const_iterator bbIt = func->begin(), bbEit = func->end();
364  bbIt != bbEit; ++bbIt)
365  {
366  const BasicBlock& bb = *bbIt;
367  u32_t numOfSucc = bb.getTerminator()->getNumSuccessors();
368  if (numOfSucc == 2)
369  numOfBB_2Succ++;
370  else if (numOfSucc > 2)
371  numOfBB_3Succ++;
372  }
373  }
374 
375  generalNumMap["BBWith2Succ"] = numOfBB_2Succ;
376  generalNumMap["BBWith3Succ"] = numOfBB_3Succ;
377 }
