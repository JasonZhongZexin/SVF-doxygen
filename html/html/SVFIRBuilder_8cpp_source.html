46 SVFIR* SVFIRBuilder::build(SVFModule* svfModule)
47 {
48 
49  // We read SVFIR from a user-defined txt instead of parsing SVFIR from LLVM IR
50  if (SVFModule::pagReadFromTXT())
51  {
52  PAGBuilderFromFile fileBuilder(SVFModule::pagFileName());
53  return fileBuilder.build();
54  }
55 
56  // If the SVFIR has been built before, then we return the unique SVFIR of the program
57  if(pag->getNodeNumAfterPAGBuild() > 1)
58  return pag;
59 
60  svfMod = svfModule;
61 
64  initialiseNodes();
67  visitGlobal(svfModule);
69 
71  for (SVFModule::iterator fit = svfModule->begin(), efit = svfModule->end();
72  fit != efit; ++fit)
73  {
74  const SVFFunction& fun = **fit;
76  if(!SVFUtil::isExtCall(&fun))
77  {
83  if(fun.getLLVMFun()->doesNotReturn() == false && fun.getLLVMFun()->getReturnType()->isVoidTy() == false)
84  pag->addFunRet(&fun,pag->getGNode(pag->getReturnNode(&fun)));
85 
88  for (Function::arg_iterator I = fun.getLLVMFun()->arg_begin(), E = fun.getLLVMFun()->arg_end();
89  I != E; ++I) {
90  setCurrentLocation(&*I,&fun.getLLVMFun()->getEntryBlock());
91  NodeID argValNodeId = pag->getValueNode(&*I);
92  // if this is the function does not have caller (e.g. main)
93  // or a dead function, shall we create a black hole address edge for it?
94  // it is (1) too conservative, and (2) make FormalParmVFGNode defined at blackhole address PAGEdge.
95  // if(SVFUtil::ArgInNoCallerFunction(&*I)) {
96  // if(I->getType()->isPointerTy())
97  // addBlackHoleAddrEdge(argValNodeId);
98  //}
99  pag->addFunArgs(&fun,pag->getGNode(argValNodeId));
100  }
101  }
102  for (Function::iterator bit = fun.getLLVMFun()->begin(), ebit = fun.getLLVMFun()->end();
103  bit != ebit; ++bit)
104  {
105  BasicBlock& bb = *bit;
106  for (BasicBlock::iterator it = bb.begin(), eit = bb.end();
107  it != eit; ++it)
108  {
109  Instruction& inst = *it;
110  setCurrentLocation(&inst,&bb);
111  visit(inst);
112  }
113  }
114  }
115 
116  sanityCheck();
117 
118  pag->initialiseCandidatePointers();
119 
120  pag->setNodeNumAfterPAGBuild(pag->getTotalNodeNum());
121 
122  return pag;
123 }

128 void SVFIRBuilder::initialiseNodes()
129 {
130  DBOUT(DPAGBuild, outs() << "Initialise SVFIR Nodes ...\n");
131 
132  SymbolTableInfo* symTable = SymbolTableInfo::SymbolInfo();
133 
134  pag->addBlackholeObjNode();
135  pag->addConstantObjNode();
136  pag->addBlackholePtrNode();
137  addNullPtrNode();
138 
139  for (SymbolTableInfo::ValueToIDMapTy::iterator iter =
140  symTable->valSyms().begin(); iter != symTable->valSyms().end();
141  ++iter)
142  {
143  DBOUT(DPAGBuild, outs() << "add val node " << iter->second << "\n");
144  if(iter->second == symTable->blkPtrSymID() || iter->second == symTable->nullPtrSymID())
145  continue;
146  pag->addValNode(iter->first, iter->second);
147  }
148 
149  for (SymbolTableInfo::ValueToIDMapTy::iterator iter =
150  symTable->objSyms().begin(); iter != symTable->objSyms().end();
151  ++iter)
152  {
153  DBOUT(DPAGBuild, outs() << "add obj node " << iter->second << "\n");
154  if(iter->second == symTable->blackholeSymID() || iter->second == symTable->constantSymID())
155  continue;
156  pag->addObjNode(iter->first, iter->second);
157  }
158 
159  for (SymbolTableInfo::FunToIDMapTy::iterator iter =
160  symTable->retSyms().begin(); iter != symTable->retSyms().end();
161  ++iter)
162  {
163  DBOUT(DPAGBuild, outs() << "add ret node " << iter->second << "\n");
164  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(iter->first);
165  pag->addRetNode(fun, iter->second);
166  }
167 
168  for (SymbolTableInfo::FunToIDMapTy::iterator iter =
169  symTable->varargSyms().begin();
170  iter != symTable->varargSyms().end(); ++iter)
171  {
172  DBOUT(DPAGBuild, outs() << "add vararg node " << iter->second << "\n");
173  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(iter->first);
174  pag->addVarargNode(fun, iter->second);
175  }
176 
178  for (SymbolTableInfo::ValueToIDMapTy::iterator iter =
179  symTable->objSyms().begin(); iter != symTable->objSyms().end(); ++iter)
180  {
181  DBOUT(DPAGBuild, outs() << "add address edges for constant node " << iter->second << "\n");
182  const Value* val = iter->first;
183  if (symTable->isConstantObjSym(val))
184  {
185  NodeID ptr = pag->getValueNode(val);
186  if(ptr!= pag->getBlkPtr() && ptr!= pag->getNullPtr())
187  {
188  setCurrentLocation(val, nullptr);
189  addAddrEdge(iter->second, ptr);
190  }
191  }
192  }
193 
194  assert(pag->getTotalNodeNum() >= symTable->getTotalSymNum()
195  && "not all node been inititalize!!!");
196 
197 }

213 void SVFIRBuilder::processCE(const Value *val)
214 {
215  if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
216  {
217  if (const ConstantExpr* gepce = isGepConstantExpr(ref))
218  {
219  DBOUT(DPAGBuild,
220  outs() << "handle gep constant expression " << *ref << "\n");
221  const Constant* opnd = gepce->getOperand(0);
222  // handle recursive constant express case (gep (bitcast (gep X 1)) 1)
223  processCE(opnd);
224  LocationSet ls;
225  bool constGep = computeGepOffset(gepce, ls);
226  // must invoke pag methods here, otherwise it will be a dead recursion cycle
227  const Value* cval = getCurrentValue();
228  const BasicBlock* cbb = getCurrentBB();
229  setCurrentLocation(gepce, nullptr);
230  /*
231  * The gep edge created are like constexpr (same edge may appear at multiple callsites)
232  * so bb/inst of this edge may be rewritten several times, we treat it as global here.
233  */
234  addGepEdge(pag->getValueNode(opnd), pag->getValueNode(gepce), ls, constGep);
235  setCurrentLocation(cval, cbb);
236  }
237  else if (const ConstantExpr* castce = isCastConstantExpr(ref))
238  {
239  DBOUT(DPAGBuild,
240  outs() << "handle cast constant expression " << *ref << "\n");
241  const Constant* opnd = castce->getOperand(0);
242  processCE(opnd);
243  const Value* cval = getCurrentValue();
244  const BasicBlock* cbb = getCurrentBB();
245  setCurrentLocation(castce, nullptr);
246  addCopyEdge(pag->getValueNode(opnd), pag->getValueNode(castce));
247  setCurrentLocation(cval, cbb);
248  }
249  else if (const ConstantExpr* selectce = isSelectConstantExpr(ref))
250  {
251  DBOUT(DPAGBuild,
252  outs() << "handle select constant expression " << *ref << "\n");
253  const Constant* src1 = selectce->getOperand(1);
254  const Constant* src2 = selectce->getOperand(2);
255  processCE(src1);
256  processCE(src2);
257  const Value* cval = getCurrentValue();
258  const BasicBlock* cbb = getCurrentBB();
259  setCurrentLocation(selectce, nullptr);
260  NodeID nsrc1 = pag->getValueNode(src1);
261  NodeID nsrc2 = pag->getValueNode(src2);
262  NodeID nres = pag->getValueNode(selectce);
263  addPhiStmt(nres,nsrc1);
264  addPhiStmt(nres,nsrc2);
265  setCurrentLocation(cval, cbb);
266  }
267  // if we meet a int2ptr, then it points-to black hole
268  else if (const ConstantExpr* int2Ptrce = isInt2PtrConstantExpr(ref))
269  {
270  addGlobalBlackHoleAddrEdge(pag->getValueNode(int2Ptrce), int2Ptrce);
271  }
272  else if (const ConstantExpr* ptr2Intce = isPtr2IntConstantExpr(ref))
273  {
274  const Constant* opnd = ptr2Intce->getOperand(0);
275  processCE(opnd);
276  const BasicBlock* cbb = getCurrentBB();
277  const Value* cval = getCurrentValue();
278  setCurrentLocation(ptr2Intce, nullptr);
279  addCopyEdge(pag->getValueNode(opnd), pag->getValueNode(ptr2Intce));
280  setCurrentLocation(cval, cbb);
281  }
282  else if(isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
283  {
284  // we don't handle trunc and cmp instruction for now
285  const Value* cval = getCurrentValue();
286  const BasicBlock* cbb = getCurrentBB();
287  setCurrentLocation(ref, nullptr);
288  NodeID dst = pag->getValueNode(ref);
289  addBlackHoleAddrEdge(dst);
290  setCurrentLocation(cval, cbb);
291  }
292  else if (isBinaryConstantExpr(ref))
293  {
294  // we don't handle binary constant expression like add(x,y) now
295  const Value* cval = getCurrentValue();
296  const BasicBlock* cbb = getCurrentBB();
297  setCurrentLocation(ref, nullptr);
298  NodeID dst = pag->getValueNode(ref);
299  addBlackHoleAddrEdge(dst);
300  setCurrentLocation(cval, cbb);
301  }
302  else if (isUnaryConstantExpr(ref))
303  {
304  // we don't handle unary constant expression like fneg(x) now
305  const Value* cval = getCurrentValue();
306  const BasicBlock* cbb = getCurrentBB();
307  setCurrentLocation(ref, nullptr);
308  NodeID dst = pag->getValueNode(ref);
309  addBlackHoleAddrEdge(dst);
310  setCurrentLocation(cval, cbb);
311  }
312  else if (SVFUtil::isa<ConstantAggregate>(ref))
313  {
314  // we don't handle constant agrgregate like constant vectors
315  }
316  else if (SVFUtil::isa<BlockAddress>(ref))
317  {
318  // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
319  // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
320  const Value *cval = getCurrentValue();
321  const BasicBlock *cbb = getCurrentBB();
322  setCurrentLocation(ref, nullptr);
323  NodeID dst = pag->getValueNode(ref);
324  addAddrEdge(pag->getConstantNode(), dst);
325  setCurrentLocation(cval, cbb);
326  }
327  else
328  {
329  if(SVFUtil::isa<ConstantExpr>(val))
330  assert(false && "we don't handle all other constant expression for now!");
331  }
332  }

358 /*For global variable initialization
359  * Give a simple global variable
360  * int x = 10; // store 10 x (constant, non pointer) |
361  * int *y = &x; // store x y (pointer type)
362  * Given a struct
363  * struct Z { int s; int *t;};
364  * Global initialization:
365  * struct Z z = {10,&x}; // store x z.t (struct type)
366  * struct Z *m = &z; // store z m (pointer type)
367  * struct Z n = {10,&z.s}; // store z.s n , &z.s constant expression (constant expression)
368  */
369 void SVFIRBuilder::InitialGlobal(const GlobalVariable *gvar, Constant *C,
370  u32_t offset)
371 {
372  DBOUT(DPAGBuild,
373  outs() << "global " << *gvar << " constant initializer: " << *C
374  << "\n");
375 
376  if (C->getType()->isSingleValueType())
377  {
378  NodeID src = getValueNode(C);
379  // get the field value if it is avaiable, otherwise we create a dummy field node.
380  setCurrentLocation(gvar, nullptr);
381  NodeID field = getGlobalVarField(gvar, offset);
382 
383  if (SVFUtil::isa<GlobalVariable>(C) || SVFUtil::isa<Function>(C))
384  {
385  setCurrentLocation(C, nullptr);
386  addStoreEdge(src, field);
387  }
388  else if (SVFUtil::isa<ConstantExpr>(C))
389  {
390  // add gep edge of C1 itself is a constant expression
391  processCE(C);
392  setCurrentLocation(C, nullptr);
393  addStoreEdge(src, field);
394  }
395  else if (SVFUtil::isa<BlockAddress>(C))
396  {
397  // blockaddress instruction (e.g. i8* blockaddress(@run_vm, %182))
398  // is treated as constant data object for now, see LLVMUtil.h:397, SymbolTableInfo.cpp:674 and SVFIRBuilder.cpp:183-194
399  processCE(C);
400  setCurrentLocation(C, nullptr);
401  addAddrEdge(pag->getConstantNode(), src);
402  }
403  else
404  {
405  setCurrentLocation(C, nullptr);
406  addStoreEdge(src, field);
408  if (C->getType()->isPtrOrPtrVectorTy() && src != pag->getNullPtr())
409  addCopyEdge(pag->getNullPtr(), src);
410  }
411  }
412  else if (SVFUtil::isa<ConstantArray>(C))
413  {
414  if (cppUtil::isValVtbl(gvar) == false)
415  for (u32_t i = 0, e = C->getNumOperands(); i != e; i++)
416  InitialGlobal(gvar, SVFUtil::cast<Constant>(C->getOperand(i)), offset);
417 
418  }
419  else if (SVFUtil::isa<ConstantStruct>(C))
420  {
421  const StructType *sty = SVFUtil::cast<StructType>(C->getType());
422  const std::vector<u32_t>& offsetvect =
423  SymbolTableInfo::SymbolInfo()->getFattenFieldIdxVec(sty);
424  for (u32_t i = 0, e = C->getNumOperands(); i != e; i++)
425  {
426  u32_t off = offsetvect[i];
427  InitialGlobal(gvar, SVFUtil::cast<Constant>(C->getOperand(i)), offset + off);
428  }
429 
430  }
431  else
432  {
433  //TODO:assert(false,"what else do we have");
434  }
435 }

440 void SVFIRBuilder::visitGlobal(SVFModule* svfModule)
441 {
442 
444  for (SVFModule::global_iterator I = svfModule->global_begin(), E =
445  svfModule->global_end(); I != E; ++I)
446  {
447  GlobalVariable *gvar = *I;
448  NodeID idx = getValueNode(gvar);
449  NodeID obj = getObjectNode(gvar);
450 
451  setCurrentLocation(gvar, nullptr);
452  addAddrEdge(obj, idx);
453 
454  if (gvar->hasInitializer())
455  {
456  Constant *C = gvar->getInitializer();
457  DBOUT(DPAGBuild, outs() << "add global var node " << *gvar << "\n");
458  InitialGlobal(gvar, C, 0);
459  }
460  }
461 
463  for (SVFModule::llvm_const_iterator I = svfModule->llvmFunBegin(), E =
464  svfModule->llvmFunEnd(); I != E; ++I)
465  {
466  const Function *fun = *I;
467  NodeID idx = getValueNode(fun);
468  NodeID obj = getObjectNode(fun);
469 
470  DBOUT(DPAGBuild, outs() << "add global function node " << fun->getName() << "\n");
471  setCurrentLocation(fun, nullptr);
472  addAddrEdge(obj, idx);
473  }
474 
475  // Handle global aliases (due to linkage of multiple bc files), e.g., @x = internal alias @y. We need to add a copy from y to x.
476  for (SVFModule::alias_iterator I = svfModule->alias_begin(), E = svfModule->alias_end(); I != E; I++)
477  {
478  NodeID dst = pag->getValueNode(*I);
479  NodeID src = pag->getValueNode((*I)->getAliasee());
480  processCE((*I)->getAliasee());
481  setCurrentLocation(*I, nullptr);
482  addCopyEdge(src,dst);
483  }
484 }

669 /*
670  * Visit callsites
671  */
672 void SVFIRBuilder::visitCallSite(CallSite cs)
673 {
674 
675  // skip llvm intrinsics
676  if(isIntrinsicInst(cs.getInstruction()))
677  return;
678 
679  DBOUT(DPAGBuild,
680  outs() << "process callsite " << *cs.getInstruction() << "\n");
681 
682  CallBlockNode* callBlockNode = pag->getICFG()->getCallBlockNode(cs.getInstruction());
683  RetBlockNode* retBlockNode = pag->getICFG()->getRetBlockNode(cs.getInstruction());
684 
685  pag->addCallSite(callBlockNode);
686 
688  for(CallSite::arg_iterator itA = cs.arg_begin(), ieA = cs.arg_end(); itA!=ieA; ++itA)
689  pag->addCallSiteArgs(callBlockNode,pag->getGNode(getValueNode(*itA)));
690 
691  if(!cs.getType()->isVoidTy())
692  pag->addCallSiteRets(retBlockNode,pag->getGNode(getValueNode(cs.getInstruction())));
693 
694  const SVFFunction *callee = getCallee(cs);
695 
696  if (callee)
697  {
698  if (isExtCall(callee))
699  {
700  // There is no extpag for the function, use the old method.
701  handleExtCall(cs, callee);
702  }
703  else
704  {
705  handleDirectCall(cs, callee);
706  }
707  }
708  else

957  {
958  // case 1: ret = new obj
959  if (isHeapAllocExtCallViaRet(cs) || isStaticExtCall(cs))
960  {
961  NodeID val = getValueNode(inst);
962  NodeID obj = getObjectNode(inst);
963  addAddrEdge(obj, val);
964  }
965  // case 2: *arg = new obj
966  else
967  {
968  assert(isHeapAllocExtCallViaArg(cs) && "Must be heap alloc call via arg.");
969  int arg_pos = getHeapAllocHoldingArgPosition(callee);
970  const Value *arg = cs.getArgument(arg_pos);
971  if (arg->getType()->isPointerTy())
972  {
973  NodeID vnArg = getValueNode(arg);
974  NodeID dummy = pag->addDummyValNode();
975  NodeID obj = pag->addDummyObjNode();
976  if (vnArg && dummy && obj)
977  {
978  addAddrEdge(obj, dummy);
979  addStoreEdge(dummy, vnArg);
980  }
981  }
982  else
983  {
984  writeWrnMsg("Arg receiving new object must be pointer type");
985  }
986  }
987  }
