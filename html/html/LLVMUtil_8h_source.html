133 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallBase* cs)
134 {
135  const PointerType *refType = nullptr;
136  const SVFInstruction* svfcall = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(cs);
137  CallSite svfcs = SVFUtil::getSVFCallSite(svfcall);
138  // Case 1: heap object held by *argument, we should get its element type.
139  if (SVFUtil::isHeapAllocExtCallViaArg(svfcs))
140  {
141  int argPos = SVFUtil::getHeapAllocHoldingArgPosition(svfcs);
142  const Value* arg = cs->getArgOperand(argPos);
143  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
144  refType = SVFUtil::dyn_cast<PointerType>(getPtrElementType(argType));
145  }
146  // Case 2: heap/static object held by return value.
147  else
148  {
149  assert((SVFUtil::isStaticExtCall(svfcs) || SVFUtil::isHeapAllocExtCallViaRet(svfcs))
150  && "Must be heap alloc via ret, or static allocation site");
151  refType = SVFUtil::dyn_cast<PointerType>(cs->getType());
152  }
153  assert(refType && "Allocated object must be held by a pointer-typed value.");
154  return refType;
155 }
