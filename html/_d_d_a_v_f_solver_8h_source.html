147  virtual void handleSingleStatement(const DPIm& dpm, CPtSet& pts)
148  {
150  resolveFunPtr(dpm);
151 
152  const SVFGNode* node = dpm.getLoc();
153  if(SVFUtil::isa<AddrSVFGNode>(node))
154  {
155  handleAddr(pts,dpm,SVFUtil::cast<AddrSVFGNode>(node));
156  }
157  else if(SVFUtil::isa<CopySVFGNode>(node) || SVFUtil::isa<PHISVFGNode>(node)
158  || SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalParmSVFGNode>(node)
159  || SVFUtil::isa<ActualRetSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node)
160  || SVFUtil::isa<NullPtrSVFGNode>(node))
161  {
162  backtraceAlongDirectVF(pts,dpm);
163  }
164  else if(SVFUtil::isa<GepSVFGNode>(node))
165  {
166  CPtSet gepPts;
167  backtraceAlongDirectVF(gepPts,dpm);
168  unionDDAPts(pts, processGepPts(SVFUtil::cast<GepSVFGNode>(node),gepPts));
169  }
170  else if(SVFUtil::isa<LoadSVFGNode>(node))
171  {
172  CPtSet loadpts;
173  startNewPTCompFromLoadSrc(loadpts,dpm);
174  for(typename CPtSet::iterator it = loadpts.begin(), eit = loadpts.end(); it!=eit; ++it)
175  {
176  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
177  }
178  }
179  else if(SVFUtil::isa<StoreSVFGNode>(node))
180  {
181  if(isMustAlias(getLoadDpm(dpm),dpm))
182  {
183  DBOUT(DDDA, SVFUtil::outs() << "+++must alias for load and store:");
184  DBOUT(DDDA, getLoadDpm(dpm).dump());
185  DBOUT(DDDA, dpm.dump());
186  DBOUT(DDDA, SVFUtil::outs() << "+++\n");
187  DOSTAT(ddaStat->_NumOfMustAliases++);
188  backtraceToStoreSrc(pts,dpm);
189  }
190  else
191  {
192  CPtSet storepts;
193  startNewPTCompFromStoreDst(storepts,dpm);
194  for(typename CPtSet::iterator it = storepts.begin(), eit = storepts.end(); it!=eit; ++it)
195  {
196  if(propagateViaObj(*it,getLoadCVar(dpm)))
197  {
198  backtraceToStoreSrc(pts,getDPImWithOldCond(dpm,*it,node));
199 
200  if(isStrongUpdate(storepts,SVFUtil::cast<StoreSVFGNode>(node)))
201  {
202  DBOUT(DDDA, SVFUtil::outs() << "backward strong update for obj " << dpm.getCurNodeID() << "\n");
203  DOSTAT(addSUStat(dpm,node);)
204  }
205  else
206  {
207  DOSTAT(rmSUStat(dpm,node);)
208  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
209  }
210  }
211  else
212  {
213  backtraceAlongIndirectVF(pts,dpm);
214  }
215  }
216  }
217  }
218  else if(SVFUtil::isa<MRSVFGNode>(node))
219  {
220  backtraceAlongIndirectVF(pts,dpm);
221  }
222  else
223  assert(false && "unexpected kind of SVFG nodes");
224  }

227  void reCompute(const DPIm& dpm)
228  {
230  SVFGEdgeSet newIndirectEdges;
231  if(_pag->isFunPtr(dpm.getCurNodeID()))
232  {
233  const CallSiteSet& csSet = _pag->getIndCallSites(dpm.getCurNodeID());
234  for(CallSiteSet::const_iterator it = csSet.begin(), eit = csSet.end(); it!=eit; ++it)
235  updateCallGraphAndSVFG(dpm, (*it),newIndirectEdges);
236  }
238  if(!newIndirectEdges.empty())
239  _callGraphSCC->find();
240  reComputeForEdges(dpm,newIndirectEdges,true);
241 
243  SVFGEdgeSet edgeSet(dpm.getLoc()->getOutEdges());
244  reComputeForEdges(dpm,edgeSet,false);
245  }
246 
248  void reComputeForEdges(const DPIm& dpm, const SVFGEdgeSet& edgeSet, bool indirectCall = false)
249  {
250  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
251  {
252  const SVFGEdge* edge = *it;
253  const SVFGNode* dst = edge->getDstNode();
254  typename LocToDPMVecMap::const_iterator locIt = getLocToDPMVecMap().find(dst->getId());
256  if (locIt == getLocToDPMVecMap().end())
257  continue;
258  DPTItemSet dpmSet(locIt->second.begin(), locIt->second.end());
259  for(typename DPTItemSet::const_iterator it = dpmSet.begin(),eit = dpmSet.end(); it!=eit; ++it)
260  {
261  const DPIm& dstDpm = *it;
262  if(!indirectCall && SVFUtil::isa<IndirectSVFGEdge>(edge) && !SVFUtil::isa<LoadSVFGNode>(edge->getDstNode()))
263  {
264  if(dstDpm.getCurNodeID() == dpm.getCurNodeID())
265  {
266  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute, forward from :");
267  DBOUT(DDDA, dpm.dump());
268  DOSTAT(ddaStat->_NumOfStepInCycle++);
269  clearbkVisited(dstDpm);
270  findPT(dstDpm);
271  }
272  }
273  else
274  {
275  if(indirectCall)
276  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute for indirect call from :");
277  else
278  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute forward from :");
279  DBOUT(DDDA, dpm.dump());
280  DOSTAT(ddaStat->_NumOfStepInCycle++);
281  clearbkVisited(dstDpm);
282  findPT(dstDpm);
283  }
284  }
285  }

324  void backtraceAlongIndirectVF(CPtSet& pts, const DPIm& oldDpm)
325  {
326  const SVFGNode* node = oldDpm.getLoc();
327  NodeID obj = oldDpm.getCurNodeID();
328  if (_pag->isConstantObj(obj) || _pag->isNonPointerObj(obj))
329  return;
330  const SVFGEdgeSet edgeSet(node->getInEdges());
331  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
332  {
333  if(const IndirectSVFGEdge* indirEdge = SVFUtil::dyn_cast<IndirectSVFGEdge>(*it))
334  {
335  PointsTo& guard = const_cast<PointsTo&>(indirEdge->getPointsTo());
336  if(guard.test(obj))
337  {
338  DBOUT(DDDA, SVFUtil::outs() << "\t\t==backtrace indirectVF svfgNode " <<
339  indirEdge->getDstID() << " --> " << indirEdge->getSrcID() << "\n");
340  backwardPropDpm(pts,oldDpm.getCurNodeID(),oldDpm,indirEdge);
341  }
342  }
343  }
344  }
346  void backtraceAlongDirectVF(CPtSet& pts, const DPIm& oldDpm)
347  {
348  const SVFGNode* node = oldDpm.getLoc();
349  const SVFGEdgeSet edgeSet(node->getInEdges());
350  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
351  {
352  if(const DirectSVFGEdge* dirEdge = SVFUtil::dyn_cast<DirectSVFGEdge>(*it))
353  {
354  DBOUT(DDDA, SVFUtil::outs() << "\t\t==backtrace directVF svfgNode " <<
355  dirEdge->getDstID() << " --> " << dirEdge->getSrcID() << "\n");
356  const SVFGNode* srcNode = dirEdge->getSrcNode();
357  backwardPropDpm(pts,getSVFG()->getLHSTopLevPtr(srcNode)->getId(),oldDpm,dirEdge);
358  }
359  }
360  }

398  virtual void backwardPropDpm(CPtSet& pts, NodeID ptr,const DPIm& oldDpm,const SVFGEdge* edge)
399  {
400  DPIm dpm(oldDpm);
401  dpm.setLocVar(edge->getSrcNode(),ptr);
402  DOTIMESTAT(double start = DDAStat::getClk());
404  if(handleBKCondition(dpm,edge)==false)
405  {
406  DOTIMESTAT(ddaStat->_TotalTimeOfBKCondition += DDAStat::getClk() - start);
407  DBOUT(DDDA, SVFUtil::outs() << "\t!!! infeasible path svfgNode: " << edge->getDstID() << " --| " << edge->getSrcID() << "\n");
408  DOSTAT(ddaStat->_NumOfInfeasiblePath++);
409  return;
410  }
411 
413  if(SVFUtil::isa<IndirectSVFGEdge>(edge))
414  addLoadDpmAndCVar(dpm,getLoadDpm(oldDpm),getLoadCVar(oldDpm));
415 
416  DOSTAT(ddaStat->_NumOfDPM++);
418  unionDDAPts(pts,findPT(dpm));
419  }

426  virtual bool isStrongUpdate(const CPtSet& dstCPSet, const StoreSVFGNode* store)
427  {
428  if (dstCPSet.count() == 1)
429  {
431  typename CPtSet::iterator it = dstCPSet.begin();
432  const CVar& var = *it;
433  // Strong update can be made if this points-to target is not heap, array or field-insensitive.
434  if (!isHeapCondMemObj(var,store) && !isArrayCondMemObj(var)
435  && !isFieldInsenCondMemObj(var) && !isLocalCVarInRecursion(var))
436  {
437  return true;
438  }
439  }
440  return false;
441  }
443  virtual inline bool isLocalCVarInRecursion(const CVar& var) const
444  {
445  NodeID id = getPtrNodeID(var);
446  const MemObj* obj = _pag->getObject(id);
447  assert(obj && "object not found!!");
448  if(obj->isStack())
449  {
450  if(const AllocaInst* local = SVFUtil::dyn_cast<AllocaInst>(obj->getRefVal()))
451  {
452  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(local->getFunction());
453  return _callGraphSCC->isInCycle(_callGraph->getCallGraphNode(fun)->getId());
454  }
455  }
456  return false;
457  }

467  void resolveFunPtr(const DPIm& dpm)
468  {
469  if(const CallBlockNode* cbn= getSVFG()->isCallSiteRetSVFGNode(dpm.getLoc()))
470  {
471  if(_pag->isIndirectCallSites(cbn))
472  {
473  NodeID funPtr = _pag->getFunPtr(cbn);
474  DPIm funPtrDpm(dpm);
475  funPtrDpm.setLocVar(getSVFG()->getDefSVFGNode(_pag->getPAGNode(funPtr)),funPtr);
476  findPT(funPtrDpm);
477  }
478  }
479  else if(const SVFFunction* fun = getSVFG()->isFunEntrySVFGNode(dpm.getLoc()))
480  {
481  CallInstSet csSet;
483  _ander->getPTACallGraph()->getIndCallSitesInvokingCallee(fun,csSet);
484  for(CallInstSet::const_iterator it = csSet.begin(), eit = csSet.end(); it!=eit; ++it)
485  {
486  NodeID funPtr = _pag->getFunPtr(*it);
487  DPIm funPtrDpm(dpm);

490  }
