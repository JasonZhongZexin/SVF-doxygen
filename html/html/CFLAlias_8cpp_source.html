61 void CFLAlias::onTheFlyCallGraphSolve(const CallSiteToFunPtrMap& callsites, CallEdgeMap& newEdges)
62 {
63  for(CallSiteToFunPtrMap::const_iterator iter = callsites.begin(), eiter = callsites.end(); iter!=eiter; ++iter)
64  {
65  const CallICFGNode* cs = iter->first;
66 
67  if (isVirtualCallSite(SVFUtil::getLLVMCallSite(cs->getCallSite())))
68  {
69  const Value *vtbl = getVCallVtblPtr(SVFUtil::getLLVMCallSite(cs->getCallSite()));
70  assert(pag->hasValueNode(vtbl));
71  NodeID vtblId = pag->getValueNode(vtbl);
72  resolveCPPIndCalls(cs, getCFLPts(vtblId), newEdges);
73  }
74  else
75  resolveIndCalls(iter->first,getCFLPts(iter->second),newEdges);
76  }
77 }

83 void CFLAlias::connectCaller2CalleeParams(CallSite cs, const SVFFunction* F)
84 {
85  assert(F);
86 
87  DBOUT(DAndersen, outs() << "connect parameters from indirect callsite " << SVFUtil::value2String(cs.getInstruction()) << " to callee " << *F << "\n");
88 
89  CallICFGNode* callBlockNode = svfir->getICFG()->getCallICFGNode(cs.getInstruction());
90  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
91 
92  if(SVFUtil::isHeapAllocExtFunViaRet(F) && svfir->callsiteHasRet(retBlockNode))
93  {
94  heapAllocatorViaIndCall(cs);
95  }
96 
97  if (svfir->funHasRet(F) && svfir->callsiteHasRet(retBlockNode))
98  {
99  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
100  const PAGNode* fun_return = svfir->getFunRet(F);
101  if (cs_return->isPointer() && fun_return->isPointer())
102  {
103  NodeID dstrec = cs_return->getId();
104  NodeID srcret = fun_return->getId();
105  addCopyEdge(srcret, dstrec);
106  }
107  else
108  {
109  DBOUT(DAndersen, outs() << "not a pointer ignored\n");
110  }
111  }
112 
113  if (svfir->hasCallSiteArgsMap(callBlockNode) && svfir->hasFunArgsList(F))
114  {
115 
116  // connect actual and formal param
117  const SVFIR::SVFVarList& csArgList = svfir->getCallSiteArgsList(callBlockNode);
118  const SVFIR::SVFVarList& funArgList = svfir->getFunArgsList(F);
119  //Go through the fixed parameters.
120  DBOUT(DPAGBuild, outs() << " args:");
121  SVFIR::SVFVarList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
122  SVFIR::SVFVarList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
123  for (; funArgIt != funArgEit; ++csArgIt, ++funArgIt)
124  {
125  //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
126  if (csArgIt == csArgEit)
127  {
128  DBOUT(DAndersen, outs() << " !! not enough args\n");
129  break;
130  }
131  const PAGNode *cs_arg = *csArgIt ;
132  const PAGNode *fun_arg = *funArgIt;
133 
134  if (cs_arg->isPointer() && fun_arg->isPointer())
135  {
136  DBOUT(DAndersen, outs() << "process actual parm " << cs_arg->toString() << " \n");
137  NodeID srcAA = cs_arg->getId();
138  NodeID dstFA = fun_arg->getId();
139  addCopyEdge(srcAA, dstFA);
140  }
141  }
142 
143  //Any remaining actual args must be varargs.
144  if (F->isVarArg())
145  {
146  NodeID vaF = svfir->getVarargNode(F);
147  DBOUT(DPAGBuild, outs() << "\n varargs:");
148  for (; csArgIt != csArgEit; ++csArgIt)
149  {
150  const PAGNode *cs_arg = *csArgIt;
151  if (cs_arg->isPointer())
152  {
153  NodeID vnAA = cs_arg->getId();
154  addCopyEdge(vnAA,vaF);
155  }
156  }
157  }
158  if(csArgIt != csArgEit)
159  {
160  writeWrnMsg("too many args to non-vararg func.");
161  writeWrnMsg("(" + getSourceLoc(cs.getInstruction()) + ")");
162  }
163  }
164 }

166 void CFLAlias::heapAllocatorViaIndCall(CallSite cs)
167 {
168  assert(SVFUtil::getCallee(cs) == nullptr && "not an indirect callsite?");
169  RetICFGNode* retBlockNode = svfir->getICFG()->getRetICFGNode(cs.getInstruction());
170  const PAGNode* cs_return = svfir->getCallSiteRet(retBlockNode);
171  NodeID srcret;
172  CallSite2DummyValPN::const_iterator it = callsite2DummyValPN.find(cs);
173  if(it != callsite2DummyValPN.end())
174  {
175  srcret = it->second;
176  }
177  else
178  {
179  NodeID valNode = svfir->addDummyValNode();
180  NodeID objNode = svfir->addDummyObjNode(cs.getType());
181  callsite2DummyValPN.insert(std::make_pair(cs,valNode));
182  graph->addCFLNode(valNode, new CFLNode(valNode));
183  graph->addCFLNode(objNode, new CFLNode(objNode));
184  srcret = valNode;
185  }
186 
187  NodeID dstrec = cs_return->getId();
188  addCopyEdge(srcret, dstrec);
189 }

217 void CFLAlias::analyze()
218 {
219  stat = new CFLStat(this);
220  GrammarBuilder grammarBuilder = GrammarBuilder(Options::GrammarFilename);
221  CFGNormalizer normalizer = CFGNormalizer();
222  AliasCFLGraphBuilder cflGraphBuilder = AliasCFLGraphBuilder();
223  CFLGramGraphChecker cflChecker = CFLGramGraphChecker();
224  if (Options::CFLGraph.empty())
225  {
226  PointerAnalysis::initialize();
227  GrammarBase *grammarBase = grammarBuilder.build();
228  ConstraintGraph *consCG = new ConstraintGraph(svfir);
229  if (Options::PEGTransfer)
230  {
231  graph = cflGraphBuilder.buildBiPEGgraph(consCG, grammarBase->getStartKind(), grammarBase, svfir);
232  }
233  else
234  {
235  graph = cflGraphBuilder.buildBigraph(consCG, grammarBase->getStartKind(), grammarBase);
236  }
237 
238  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
239  grammar = normalizer.normalize(grammarBase);
240  cflChecker.check(grammar, &cflGraphBuilder, graph);
241  delete consCG;
242  delete grammarBase;
243  }
244  else
245  {
246  GrammarBase *grammarBase = grammarBuilder.build();
247  graph = cflGraphBuilder.buildFromDot(Options::CFLGraph, grammarBase);
248  cflChecker.check(grammarBase, &cflGraphBuilder, graph);
249  grammar = normalizer.normalize(grammarBase);
250  cflChecker.check(grammar, &cflGraphBuilder, graph);
251  delete grammarBase;
252  }
253  solver = new CFLSolver(graph, grammar);
254  solver->solve();
255  while (updateCallGraph(svfir->getIndirectCallsites()))
256  solver->solve();
257  if(Options::PrintCFL == true)
258  {
259  svfir->dump("IR");
260  grammar->dump("Grammar");
261  graph->dump("CFLGraph");
262  }
263  if (Options::CFLGraph.empty())
264  {
265  PointerAnalysis::finalize();
266  }
267 }
