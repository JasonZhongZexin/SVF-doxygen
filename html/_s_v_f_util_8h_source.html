101 inline bool cmpPts (const PointsTo& lpts,const PointsTo& rpts)
102 {
103  if (lpts.count() != rpts.count())
104  return (lpts.count() < rpts.count());
105  else
106  {
107  PointsTo::iterator bit = lpts.begin(), eit = lpts.end();
108  PointsTo::iterator rbit = rpts.begin(), reit = rpts.end();
109  for (; bit != eit && rbit != reit; bit++, rbit++)
110  {
111  if (*bit != *rbit)
112  return (*bit < *rbit);
113  }
114 
115  return false;
116  }
117 }

121 inline bool isIntrinsicInst(const Instruction* inst)
122 {
123  if (const llvm::CallBase* call = llvm::dyn_cast<llvm::CallBase>(inst)) {
124  const Function* func = call->getCalledFunction();
125  if (func && (func->getIntrinsicID() == llvm::Intrinsic::donothing ||
126  func->getIntrinsicID() == llvm::Intrinsic::dbg_addr ||
127  func->getIntrinsicID() == llvm::Intrinsic::dbg_declare ||
128  func->getIntrinsicID() == llvm::Intrinsic::dbg_label ||
129  func->getIntrinsicID() == llvm::Intrinsic::dbg_value)) {
130  return true;
131  }
132  }
133  return false;
134 }

172 inline const SVFFunction* getFunction(StringRef name)
173 {
174  Function* fun = NULL;
175  LLVMModuleSet* llvmModuleset = LLVMModuleSet::getLLVMModuleSet();
176 
177  for (u32_t i = 0; i < llvmModuleset->getModuleNum(); ++i)
178  {
179  Module *mod = llvmModuleset->getModule(i);
180  fun = mod->getFunction(name);
181  if(fun && !fun->isDeclaration())

183  return llvmModuleset->getSVFFunction(fun);
184  }
185  }
