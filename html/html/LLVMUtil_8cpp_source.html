62 void LLVMUtil::getFunReachableBBs (const Function* fun, std::vector<const SVFBasicBlock*> &reachableBBs)
63 {
64  assert(!SVFUtil::isExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun)) && "The calling function cannot be an external function.");
65  //initial DominatorTree
66  DominatorTree dt;
67  dt.recalculate(const_cast<Function&>(*fun));
68 
69  Set<const BasicBlock*> visited;
70  std::vector<const BasicBlock*> bbVec;
71  bbVec.push_back(&fun->getEntryBlock());
72  while(!bbVec.empty())
73  {
74  const BasicBlock* bb = bbVec.back();
75  bbVec.pop_back();
76  const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
77  reachableBBs.push_back(svfbb);
78  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
79  {
80  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
81  DI != DE; ++DI)
82  {
83  const BasicBlock* succbb = (*DI)->getBlock();
84  if(visited.find(succbb)==visited.end())
85  visited.insert(succbb);
86  else
87  continue;
88  bbVec.push_back(succbb);
89  }
90  }
91  }
92 }

138 bool LLVMUtil::isUncalledFunction (const Function* fun)
139 {
140  if(fun->hasAddressTaken())
141  return false;
142  if(SVFUtil::isProgEntryFunction(fun))
143  return false;
144  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
145  {
146  if (LLVMUtil::isCallSite(*i))
147  return false;
148  }
149  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
150  {
151  const LLVMModuleSet::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
152  for (LLVMModuleSet::FunctionSetType::const_iterator it = decls.begin(),
153  eit = decls.end(); it != eit; ++it)
154  {
155  const Function* decl = *it;
156  if(decl->hasAddressTaken())
157  return false;
158  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
159  {
160  if (LLVMUtil::isCallSite(*i))
161  return false;
162  }
163  }
164  }
165  return true;
166 }

465 const std::string LLVMUtil::getSourceLoc(const Value* val )
466 {
467  if(val==nullptr) return "{ empty val }";
468 
469  std::string str;
470  std::stringstream rawstr(str);
471  rawstr << "{ ";
472 
473  if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
474  {
475  if (SVFUtil::isa<AllocaInst>(inst))
476  {
477  for (llvm::DbgInfoIntrinsic *DII : FindDbgAddrUses(const_cast<Instruction*>(inst)))
478  {
479  if (llvm::DbgDeclareInst *DDI = SVFUtil::dyn_cast<llvm::DbgDeclareInst>(DII))
480  {
481  llvm::DIVariable *DIVar = SVFUtil::cast<llvm::DIVariable>(DDI->getVariable());
482  rawstr << "ln: " << DIVar->getLine() << " fl: " << DIVar->getFilename().str();
483  break;
484  }
485  }
486  }
487  else if (MDNode *N = inst->getMetadata("dbg")) // Here I is an LLVM instruction
488  {
489  llvm::DILocation* Loc = SVFUtil::cast<llvm::DILocation>(N); // DILocation is in DebugInfo.h
490  unsigned Line = Loc->getLine();
491  unsigned Column = Loc->getColumn();
492  std::string File = Loc->getFilename().str();
493  //StringRef Dir = Loc.getDirectory();
494  if(File.empty() || Line == 0)
495  {
496  auto inlineLoc = Loc->getInlinedAt();
497  if(inlineLoc)
498  {
499  Line = inlineLoc->getLine();
500  Column = inlineLoc->getColumn();
501  File = inlineLoc->getFilename().str();
502  }
503  }
504  rawstr << "ln: " << Line << " cl: " << Column << " fl: " << File;
505  }
506  }
507  else if (const Argument* argument = SVFUtil::dyn_cast<Argument>(val))
508  {
509  if (argument->getArgNo()%10 == 1)
510  rawstr << argument->getArgNo() << "st";
511  else if (argument->getArgNo()%10 == 2)
512  rawstr << argument->getArgNo() << "nd";
513  else if (argument->getArgNo()%10 == 3)
514  rawstr << argument->getArgNo() << "rd";
515  else
516  rawstr << argument->getArgNo() << "th";
517  rawstr << " arg " << argument->getParent()->getName().str() << " "
518  << getSourceLocOfFunction(argument->getParent());
519  }
520  else if (const GlobalVariable* gvar = SVFUtil::dyn_cast<GlobalVariable>(val))
521  {
522  rawstr << "Glob ";
523  NamedMDNode* CU_Nodes = gvar->getParent()->getNamedMetadata("llvm.dbg.cu");
524  if(CU_Nodes)
525  {
526  for (unsigned i = 0, e = CU_Nodes->getNumOperands(); i != e; ++i)
527  {
528  llvm::DICompileUnit *CUNode = SVFUtil::cast<llvm::DICompileUnit>(CU_Nodes->getOperand(i));
529  for (llvm::DIGlobalVariableExpression *GV : CUNode->getGlobalVariables())
530  {
531  llvm::DIGlobalVariable * DGV = GV->getVariable();
532 
533  if(DGV->getName() == gvar->getName())
534  {
535  rawstr << "ln: " << DGV->getLine() << " fl: " << DGV->getFilename().str();
536  }
537 
538  }
539  }
540  }
541  }
542  else if (const Function* func = SVFUtil::dyn_cast<Function>(val))
543  {
544  rawstr << getSourceLocOfFunction(func);
545  }
546  else if (const BasicBlock* bb = SVFUtil::dyn_cast<BasicBlock>(val))
547  {
548  rawstr << "basic block: " << bb->getName().str() << " " << getSourceLoc(bb->getFirstNonPHI());
549  }
550  else if(LLVMUtil::isConstDataOrAggData(val))
551  {
552  rawstr << "constant data";
553  }
554  else
555  {
556  rawstr << "Can only get source location for instruction, argument, global var, function or constant data.";
557  }
558  rawstr << " }";
559 
560  if(rawstr.str()=="{ }")
561  return "";
562  return rawstr.str();
563 }

647 bool LLVMUtil::isConstantObjSym(const Value* val)
648 {
649  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
650  {
651  if (cppUtil::isValVtbl(v))
652  return false;
653  else if (!v->hasInitializer())
654  {
655  if(v->isExternalLinkage(v->getLinkage()))
656  return false;
657  else
658  return true;
659  }
660  else
661  {
662  StInfo *stInfo = LLVMModuleSet::getLLVMModuleSet()->getSVFType(v->getInitializer()->getType())->getTypeInfo();
663  const std::vector<const SVFType*> &fields = stInfo->getFlattenFieldTypes();
664  for (std::vector<const SVFType*>::const_iterator it = fields.begin(), eit = fields.end(); it != eit; ++it)
665  {
666  const SVFType* elemTy = *it;
667  assert(!SVFUtil::isa<SVFFunctionType>(elemTy) && "Initializer of a global is a function?");
668  if (SVFUtil::isa<SVFPointerType>(elemTy))
669  return false;
670  }
671 
672  return v->isConstant();
673  }
674  }
675  return LLVMUtil::isConstDataOrAggData(val);
676 }
