179  const std::vector<u32_t>& so = getTypeInfo(T)->getFlattenedFieldIdxVec();

238  outs() <<" {Type: ";

240  outs() << "}\n";

242  int field_idx = 0;

244  it!=eit; ++it, field_idx++)
245  {
246  outs() << " \tField_idx = " << field_idx;
247  outs() << ", field type: ";
248  outs() << (*it)->toString();
249  outs() << "\n";
250  }
251  outs() << "\n";
252  }
253 
254  else if (const SVFPointerType* pt= SVFUtil::dyn_cast<SVFPointerType> (type))
255  {
256  u32_t eSize = getNumOfFlattenElements(pt->getPtrElementType());
257  outs() << " {Type: ";
258  outs() << pt->toString();
259  outs() << "}\n";
260  outs() <<"\t [target size = " << eSize << "]\n";
261  outs() << "\n";
262  }
263 
264  else if ( const SVFFunctionType* fu= SVFUtil::dyn_cast<SVFFunctionType> (type))
265  {
266  outs() << " {Type: ";
267  outs() << fu->getReturnType()->toString();
268  outs() << "(Function)}\n\n";
269  }
270 
271  else
272  {
273  assert(type->isSingleValueType() && "not a single value type, then what else!!");
275  u32_t eSize = getNumOfFlattenElements(type);
276  outs() <<" {Type: ";
277  outs() << type->toString();
278  outs() << "}\n";
279  outs() <<"\t [object size = " << eSize << "]\n";
280  outs() << "\n";
281  }

369 bool ObjTypeInfo::isNonPtrFieldObj(const LocationSet& ls)
370 {
371  if (hasPtrObj() == false)
372  return true;
373 
374  const SVFType* ety = getType();
375 
376  if (SVFUtil::isa<SVFStructType>(ety) || SVFUtil::isa<SVFArrayType>(ety))
377  {
378  u32_t sz = 0;
379  if(Options::ModelArrays)
380  sz = SymbolTableInfo::SymbolInfo()->getTypeInfo(ety)->getFlattenElementTypes().size();
381  else
382  sz = SymbolTableInfo::SymbolInfo()->getTypeInfo(ety)->getFlattenFieldTypes().size();
383 
384  if(sz <= (u32_t)ls.accumulateConstantFieldIdx())
385  {
386  writeWrnMsg("out of bound error when accessing the struct/array");
387  return false;
388  }
389 
390  const SVFType* elemTy = SymbolTableInfo::SymbolInfo()->getFlatternedElemType(ety, ls.accumulateConstantFieldIdx());
391  return (elemTy->isPointerTy() == false);
392  }
393  else
394  {
395  return (hasPtrObj() == false);
396  }
397 }
