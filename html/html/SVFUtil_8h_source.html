105 inline bool cmpPts (const PointsTo& lpts,const PointsTo& rpts)
106 {
107  if (lpts.count() != rpts.count())
108  return (lpts.count() < rpts.count());
109  else
110  {
111  PointsTo::iterator bit = lpts.begin(), eit = lpts.end();
112  PointsTo::iterator rbit = rpts.begin(), reit = rpts.end();
113  for (; bit != eit && rbit != reit; bit++, rbit++)
114  {
115  if (*bit != *rbit)
116  return (*bit < *rbit);
117  }
118 
119  return false;
120  }
121 }

211 inline std::vector<std::string> split(const std::string& s, char seperator)
212 {
213  std::vector<std::string> output;
214  std::string::size_type prev_pos = 0, pos = 0;
215  while((pos = s.find(seperator, pos)) != std::string::npos)
216  {
217  std::string substring( s.substr(prev_pos, pos-prev_pos) );
218  output.push_back(substring);
219  prev_pos = ++pos;
220  }
221  output.push_back(s.substr(prev_pos, pos-prev_pos));
222  return output;
223 }
