387 void TCT::build()
388 {
389 
390  markRelProcs();
391 
392  collectLoopInfoForJoin();
393 
394  // the fork site of main function is initialized with nullptr.
395  // the context of main is initialized with empty
396  // start routine is empty
397 
398  collectEntryFunInCallGraph();
399  for (FunSet::iterator it=entryFuncSet.begin(), eit=entryFuncSet.end(); it!=eit; ++it)
400  {
401  if (!isCandidateFun(*it))
402  continue;
403  CallStrCxt cxt;
404  TCTNode* mainTCTNode = getOrCreateTCTNode(cxt, nullptr, cxt, *it);
405  CxtThreadProc t(mainTCTNode->getId(), cxt, getSVFFun(*it));
406  pushToCTPWorkList(t);
407  }
408 
409  while(!ctpList.empty())
410  {
411  CxtThreadProc ctp = popFromCTPWorkList();
412  PTACallGraphNode* cgNode = tcg->getCallGraphNode(ctp.getProc());
413  if(isCandidateFun(cgNode->getFunction()->getLLVMFun()) == false)
414  continue;
415 
416  for(PTACallGraphNode::const_iterator nit = cgNode->OutEdgeBegin(), neit = cgNode->OutEdgeEnd(); nit!=neit; nit++)
417  {
418  const PTACallGraphEdge* cgEdge = (*nit);
419 
420  for(PTACallGraphEdge::CallInstSet::const_iterator cit = cgEdge->directCallsBegin(),
421  ecit = cgEdge->directCallsEnd(); cit!=ecit; ++cit)
422  {
423  DBOUT(DMTA,outs() << "\nTCT handling direct call:" << **cit << "\t" << cgEdge->getSrcNode()->getFunction()->getName() << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
424  handleCallRelation(ctp,cgEdge,getLLVMCallSite((*cit)->getCallSite()));
425  }
426  for(PTACallGraphEdge::CallInstSet::const_iterator ind = cgEdge->indirectCallsBegin(),
427  eind = cgEdge->indirectCallsEnd(); ind!=eind; ++ind)
428  {
429  DBOUT(DMTA,outs() << "\nTCT handling indirect call:" << **ind << "\t" << cgEdge->getSrcNode()->getFunction()->getName() << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
430  handleCallRelation(ctp,cgEdge,getLLVMCallSite((*ind)->getCallSite()));
431  }
432  }
433  }
434 
435  collectMultiForkedThreads();
436 
437  if (Options::TCTDotGraph)
438  {
439  print();
440  dump("tct");
441  }
