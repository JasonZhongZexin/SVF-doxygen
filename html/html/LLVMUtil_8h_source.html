228 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallSite cs)
229 {
230  const PointerType *refType = nullptr;
231  // Case 1: heap object held by *argument, we should get its element type.
232  if (isHeapAllocExtCallViaArg(cs))
233  {
234  int argPos = getHeapAllocHoldingArgPosition(cs);
235  const Value *arg = cs.getArgument(argPos);
236  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
237  refType = SVFUtil::dyn_cast<PointerType>(argType->getElementType());
238  }
239  // Case 2: heap/static object held by return value.
240  else
241  {
242  assert((isStaticExtCall(cs) || isHeapAllocExtCallViaRet(cs))
243  && "Must be heap alloc via ret, or static allocation site");
244  refType = SVFUtil::dyn_cast<PointerType>(cs.getType());
245  }
246  assert(refType && "Allocated object must be held by a pointer-typed value.");
247  return refType;
248 }
