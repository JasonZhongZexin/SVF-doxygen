242  for(JoinEdgeSet::const_iterator jit = it->second.begin(), ejit = it->second.end(); jit!=ejit; ++jit)

395  inline ThreadJoinEdge* hasThreadJoinEdge(const CallBlockNode* call, PTACallGraphNode* joinFunNode, PTACallGraphNode* threadRoutineFunNode, CallSiteID csId) const
396  {
397  ThreadJoinEdge joinEdge(joinFunNode,threadRoutineFunNode, csId);
398  CallInstToJoinEdgesMap::const_iterator it = callinstToThreadJoinEdgesMap.find(call);
399  if(it != callinstToThreadJoinEdgesMap.end())
400  {
401  JoinEdgeSet::const_iterator jit = it->second.find(&joinEdge);
402  if(jit!=it->second.end())
403  return *jit;
404  }
405  return NULL;
406  }
