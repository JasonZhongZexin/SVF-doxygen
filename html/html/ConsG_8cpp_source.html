42 void ConstraintGraph::buildCG()
43 {
44 
45  // initialize nodes
46  for(PAG::iterator it = pag->begin(), eit = pag->end(); it!=eit; ++it)
47  {
48  addConstraintNode(new ConstraintNode(it->first), it->first);
49  }
50 
51  // initialize edges
52  PAGEdge::PAGEdgeSetTy& addrs = getPAGEdgeSet(PAGEdge::Addr);
53  for (PAGEdge::PAGEdgeSetTy::iterator iter = addrs.begin(), eiter =
54  addrs.end(); iter != eiter; ++iter)
55  {
56  PAGEdge* edge = *iter;
57  addAddrCGEdge(edge->getSrcID(),edge->getDstID());
58  }
59

61  for (PAGEdge::PAGEdgeSetTy::iterator iter = copys.begin(), eiter =
62  copys.end(); iter != eiter; ++iter)
63  {
64  PAGEdge* edge = *iter;
65  addCopyCGEdge(edge->getSrcID(),edge->getDstID());
66  }
67 
68  PAGEdge::PAGEdgeSetTy& calls = getPAGEdgeSet(PAGEdge::Call);
69  for (PAGEdge::PAGEdgeSetTy::iterator iter = calls.begin(), eiter =
70  calls.end(); iter != eiter; ++iter)
71  {
72  PAGEdge* edge = *iter;
73  addCopyCGEdge(edge->getSrcID(),edge->getDstID());

78  rets.end(); iter != eiter; ++iter)

157 {

271 void ConstraintGraph::reTargetDstOfEdge(ConstraintEdge* edge, ConstraintNode* newDstNode)
272 {
273  NodeID newDstNodeID = newDstNode->getId();
274  NodeID srcId = edge->getSrcID();
275  if(LoadCGEdge* load = SVFUtil::dyn_cast<LoadCGEdge>(edge))
276  {
277  removeLoadEdge(load);
278  addLoadCGEdge(srcId,newDstNodeID);
279  }
280  else if(StoreCGEdge* store = SVFUtil::dyn_cast<StoreCGEdge>(edge))
281  {
282  removeStoreEdge(store);
283  addStoreCGEdge(srcId,newDstNodeID);
284  }
285  else if(CopyCGEdge* copy = SVFUtil::dyn_cast<CopyCGEdge>(edge))
286  {
287  removeDirectEdge(copy);
288  addCopyCGEdge(srcId,newDstNodeID);
289  }
290  else if(NormalGepCGEdge* gep = SVFUtil::dyn_cast<NormalGepCGEdge>(edge))
291  {
292  const LocationSet ls = gep->getLocationSet();
293  removeDirectEdge(gep);
294  addNormalGepCGEdge(srcId,newDstNodeID,ls);
295  }
296  else if(VariantGepCGEdge* gep = SVFUtil::dyn_cast<VariantGepCGEdge>(edge))
297  {
298  removeDirectEdge(gep);
299  addVariantGepCGEdge(srcId,newDstNodeID);
300  }
301  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
302  {
303  removeAddrEdge(addr);
304  }
305  else
306  assert(false && "no other edge type!!");
307 }

407 bool ConstraintGraph::moveInEdgesToRepNode(ConstraintNode* node, ConstraintNode* rep )
408 {
409  std::vector<ConstraintEdge*> sccEdges;
410  std::vector<ConstraintEdge*> nonSccEdges;
411  for (ConstraintNode::const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit;
412  ++it)
413  {
414  ConstraintEdge* subInEdge = *it;
415  if(sccRepNode(subInEdge->getSrcID()) != rep->getId())
416  nonSccEdges.push_back(subInEdge);
417  else
418  {
419  sccEdges.push_back(subInEdge);
420  }
421  }
422  // if this edge is outside scc, then re-target edge dst to rep
423  while(!nonSccEdges.empty())
424  {
425  ConstraintEdge* edge = nonSccEdges.back();
426  nonSccEdges.pop_back();
427  reTargetDstOfEdge(edge,rep);
428  }
429 
430  bool criticalGepInsideSCC = false;
431  // if this edge is inside scc, then remove this edge and two end nodes
432  while(!sccEdges.empty())
433  {
434  ConstraintEdge* edge = sccEdges.back();
435  sccEdges.pop_back();
437  if(SVFUtil::isa<CopyCGEdge>(edge))
438  removeDirectEdge(edge);
439  else if (SVFUtil::isa<GepCGEdge>(edge))
440  {
441  // If the GEP is critical (i.e. may have a non-zero offset),
442  // then it brings impact on field-sensitivity.
443  if (!isZeroOffsettedGepCGEdge(edge))
444  {
445  criticalGepInsideSCC = true;
446  }
447  removeDirectEdge(edge);
448  }
449  else if(SVFUtil::isa<LoadCGEdge>(edge) || SVFUtil::isa<StoreCGEdge>(edge))
450  reTargetDstOfEdge(edge,rep);
451  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
452  {
453  removeAddrEdge(addr);
454  }
455  else
456  assert(false && "no such edge");
457  }
458  return criticalGepInsideSCC;
459 }
