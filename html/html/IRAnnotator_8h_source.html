31  void processAndersenResults(SVFIR *pag, BVDataPTAImpl *ptsTo, bool writeFlag)
32  {
33  this->ptsTo = ptsTo;
34  mainModule = LLVMModuleSet::getLLVMModuleSet()->getMainLLVMModule();
35 
36  // Add a named metadata node used to check whether or not
37  // this IR has been annotated with Andersen information
38  if (writeFlag)
39  mainModule->getOrInsertNamedMetadata("SVFIR-Annotated");
40 
41  for (auto it = pag->begin(); it != pag->end(); ++it)
42  {
43  auto nodeId = it->first;
44  auto pagNode = it->second;
45  auto gepNode = SVFUtil::dyn_cast<GepObjVar>(pagNode);
46 
47  if (gepNode && writeFlag)
48  {
49  writePAGgepNode(nodeId, gepNode);
50  }
51  else if (pagNode->hasValue())
52  {
53  processPAGNode(pagNode->getValue(), nodeId, writeFlag);
54  }
55  }
56 
57  if (!writeFlag)
58  readPAGgepNodes(pag);
59  }

62  // Write the PAGgepNode to the IR such that metadata name is the SVFIR node id and the operands
63  // are its base node's id and location offset
64  void writePAGgepNode(SVF::NodeID nodeId, GepObjVar* gepNode)
65  {
66  auto baseNodeId = gepNode->getBaseNode();
67  auto locationSetOffset = gepNode->getOffset();
68 
69  LLVMContext &context = mainModule->getContext();
70  llvm::SmallVector<llvm::Metadata *, 32> operands;
71  operands.push_back(llvm::MDString::get(context, std::to_string(baseNodeId)));
72  operands.push_back(llvm::MDString::get(context, std::to_string(locationSetOffset)));
73 
74  llvm::MDTuple *metadata = llvm::MDTuple::get(context, operands);
75  std::string label = "gepnode-" + std::to_string(nodeId);
76 
77  mainModule->getOrInsertNamedMetadata(label)->addOperand(metadata);
78  }

80  // Reads the PAGGepNodes in the annotated IR and creates a new SVFIR Node based on the
81  // data contained in the operands of the metadata node
82  void readPAGgepNodes(SVFIR *pag)
83  {
84  for (auto it = mainModule->named_metadata_begin(); it != mainModule->named_metadata_end(); ++it)
85  {
86  std::string label = it->getName().str();
87  std::string toErase = "gepnode-";
88  SVF::Size_t pos = label.find(toErase);
89  if (pos == std::string::npos)
90  {
91  continue;
92  }
93  label.erase(pos, toErase.length());
94 
95  SVF::NodeID nodeId = std::stoi(label);
96  auto mdNode = it->getOperand(0);
97 
98  SVF::NodeID baseNodeId = std::stoi(llvm::cast<llvm::MDString>(mdNode->getOperand(0))->getString().str());
99  SVF::Size_t locationSetOffset = std::stoi(llvm::cast<llvm::MDString>(mdNode->getOperand(1))->getString().str());
100 
101  LocationSet locationSet = LocationSet(locationSetOffset);
102  SVF::NodeID gepnodeId = pag->getGepObjVar(baseNodeId, locationSet);
103 
104  assert(nodeId == gepnodeId && "nodeId != gepnodeId");
105  }
106  }

108  // Deduce the LLVM value's type, and process the metadata accordingly
109  void processPAGNode(const Value *value, SVF::NodeID nodeId, bool writeFlag)
110  {
111  if (auto instruction = const_cast<Instruction *>(SVFUtil::dyn_cast<Instruction>(value)))
112  {
113  processInstructionNode(instruction, nodeId, writeFlag);
114  }
115  else if (auto argument = const_cast<Argument *>(SVFUtil::dyn_cast<Argument>(value)))
116  {
117  processArgumentNode(argument, nodeId, writeFlag);
118  }
119  else if (auto function = const_cast<Function *>(SVFUtil::dyn_cast<Function>(value)))
120  {
121  processFunctionNode(function, nodeId, writeFlag);
122  }
123  else if (auto globalVar = const_cast<GlobalVariable *>(SVFUtil::dyn_cast<GlobalVariable>(value)))
124  {
125  processGlobalVarNode(globalVar, nodeId, writeFlag);
126  }
127  else if (auto basicBlock = const_cast<BasicBlock *>(SVFUtil::dyn_cast<BasicBlock>(value)))
128  {
129  processBasicBlockNode(basicBlock, nodeId, writeFlag);
130  }
131  else if (auto constant = const_cast<Constant *>(SVFUtil::dyn_cast<Constant>(value)))
132  {
133  processConstantNode(nodeId, writeFlag);
134  }
135  else if (auto inlineAsm = const_cast<llvm::InlineAsm *>(SVFUtil::dyn_cast<llvm::InlineAsm>(value)))
136  {
137  processInlineAsmNode(inlineAsm, nodeId, writeFlag);
138  }
139  else
140  {
141  std::cout << "Value is NOT a Instruction, Argument, Function, GlobalVariable, BasicBlock, Constant or InlineAsm" << std::endl;
142  SVFUtil::outs() << *value << "\n";
143  }
144  }

147  {
