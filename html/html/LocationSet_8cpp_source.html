116 s32_t LocationSet::accumulateConstantOffset() const
117 {
118 
119  assert(isConstantOffset() && "not a constant offset");
120 
121  if(offsetValues.empty())
122  return accumulateConstantFieldIdx();
123 
124  s32_t totalConstOffset = 0;
125  for(int i = offsetValues.size() - 1; i >= 0; i--)
126  {
127  const Value* value = offsetValues[i].first;
128  const Type* type = offsetValues[i].second;
129  const ConstantInt *op = SVFUtil::dyn_cast<ConstantInt>(value);
130  assert(op && "not a constant offset?");
131  if(type==nullptr)
132  {
133  totalConstOffset += op->getSExtValue();
134  continue;
135  }
136 
137  if(const PointerType* pty = SVFUtil::dyn_cast<PointerType>(type))
138  totalConstOffset += op->getSExtValue() * getElementNum(getPtrElementType(pty));
139  else
140  {
141  s32_t offset = op->getSExtValue();
142  u32_t flattenOffset = SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(type, offset);
143  totalConstOffset += flattenOffset;
144  }
145  }
146  return totalConstOffset;
147 }

171 bool LocationSet::operator< (const LocationSet& rhs) const
172 {
173  if (fldIdx != rhs.fldIdx)
174  return (fldIdx < rhs.fldIdx);
175  else
176  {
177  const OffsetValueVec& pairVec = getOffsetValueVec();
178  const OffsetValueVec& rhsPairVec = rhs.getOffsetValueVec();
179  if (pairVec.size() != rhsPairVec.size())
180  return (pairVec.size() < rhsPairVec.size());
181  else
182  {
183  OffsetValueVec::const_iterator it = pairVec.begin();
184  OffsetValueVec::const_iterator rhsIt = rhsPairVec.begin();
185  for (; it != pairVec.end() && rhsIt != rhsPairVec.end(); ++it, ++rhsIt)
186  {
187  return (*it) < (*rhsIt);
188  }
189 
190  return false;
191  }
192  }
193 }

195 SVF::LocationSet::LSRelation LocationSet::checkRelation(const LocationSet& LHS, const LocationSet& RHS)
196 {
197  NodeBS lhsLocations = LHS.computeAllLocations();
198  NodeBS rhsLocations = RHS.computeAllLocations();
199  if (lhsLocations.intersects(rhsLocations))
200  {
201  if (lhsLocations == rhsLocations)
202  return Same;
203  else if (lhsLocations.contains(rhsLocations))
204  return Superset;
205  else if (rhsLocations.contains(lhsLocations))
206  return Subset;
207  else
208  return Overlap;
209  }
210  else
211  {
212  return NonOverlap;
213  }
214 }
