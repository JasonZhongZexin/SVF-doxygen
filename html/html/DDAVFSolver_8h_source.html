148  virtual void handleSingleStatement(const DPIm& dpm, CPtSet& pts)
149  {
151  resolveFunPtr(dpm);
152 
153  const SVFGNode* node = dpm.getLoc();
154  if(SVFUtil::isa<AddrSVFGNode>(node))
155  {
156  handleAddr(pts,dpm,SVFUtil::cast<AddrSVFGNode>(node));
157  }
158  else if(SVFUtil::isa<CopySVFGNode>(node) || SVFUtil::isa<PHISVFGNode>(node)
159  || SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalParmSVFGNode>(node)
160  || SVFUtil::isa<ActualRetSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node)
161  || SVFUtil::isa<NullPtrSVFGNode>(node))
162  {
163  backtraceAlongDirectVF(pts,dpm);
164  }
165  else if(SVFUtil::isa<GepSVFGNode>(node))
166  {
167  CPtSet gepPts;
168  backtraceAlongDirectVF(gepPts,dpm);
169  unionDDAPts(pts, processGepPts(SVFUtil::cast<GepSVFGNode>(node),gepPts));
170  }
171  else if(SVFUtil::isa<LoadSVFGNode>(node))
172  {
173  CPtSet loadpts;
174  startNewPTCompFromLoadSrc(loadpts,dpm);
175  for(typename CPtSet::iterator it = loadpts.begin(), eit = loadpts.end(); it!=eit; ++it)
176  {
177  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
178  }
179  }
180  else if(SVFUtil::isa<StoreSVFGNode>(node))
181  {
182  if(isMustAlias(getLoadDpm(dpm),dpm))
183  {
184  DBOUT(DDDA, SVFUtil::outs() << "+++must alias for load and store:");
185  DBOUT(DDDA, getLoadDpm(dpm).dump());
186  DBOUT(DDDA, dpm.dump());
187  DBOUT(DDDA, SVFUtil::outs() << "+++\n");
188  DOSTAT(ddaStat->_NumOfMustAliases++);
189  backtraceToStoreSrc(pts,dpm);
190  }
191  else
192  {
193  CPtSet storepts;
194  startNewPTCompFromStoreDst(storepts,dpm);
195  for(typename CPtSet::iterator it = storepts.begin(), eit = storepts.end(); it!=eit; ++it)
196  {
197  if(propagateViaObj(*it,getLoadCVar(dpm)))
198  {
199  backtraceToStoreSrc(pts,getDPImWithOldCond(dpm,*it,node));
200 
201  if(isStrongUpdate(storepts,SVFUtil::cast<StoreSVFGNode>(node)))
202  {
203  DBOUT(DDDA, SVFUtil::outs() << "backward strong update for obj " << dpm.getCurNodeID() << "\n");
204  DOSTAT(addSUStat(dpm,node);)
205  }
206  else
207  {
208  DOSTAT(rmSUStat(dpm,node);)
209  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
210  }
211  }
212  else
213  {
214  backtraceAlongIndirectVF(pts,dpm);
215  }
216  }
217  }
218  }
219  else if(SVFUtil::isa<MRSVFGNode>(node))
220  {
221  backtraceAlongIndirectVF(pts,dpm);
222  }
223  else
224  assert(false && "unexpected kind of SVFG nodes");
225  }

228  void reCompute(const DPIm& dpm)
229  {
231  SVFGEdgeSet newIndirectEdges;
232  if(_pag->isFunPtr(dpm.getCurNodeID()))
233  {
234  const CallSiteSet& csSet = _pag->getIndCallSites(dpm.getCurNodeID());
235  for(CallSiteSet::const_iterator it = csSet.begin(), eit = csSet.end(); it!=eit; ++it)
236  updateCallGraphAndSVFG(dpm, (*it),newIndirectEdges);
237  }
239  if(!newIndirectEdges.empty())
240  _callGraphSCC->find();
241  reComputeForEdges(dpm,newIndirectEdges,true);
242 
244  SVFGEdgeSet edgeSet(dpm.getLoc()->getOutEdges());
245  reComputeForEdges(dpm,edgeSet,false);
246  }

249  void reComputeForEdges(const DPIm& dpm, const SVFGEdgeSet& edgeSet, bool indirectCall = false)
250  {
251  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
252  {
253  const SVFGEdge* edge = *it;
254  const SVFGNode* dst = edge->getDstNode();
255  typename LocToDPMVecMap::const_iterator locIt = getLocToDPMVecMap().find(dst->getId());
257  if (locIt == getLocToDPMVecMap().end())
258  continue;
259  DPTItemSet dpmSet(locIt->second.begin(), locIt->second.end());
260  for(typename DPTItemSet::const_iterator it = dpmSet.begin(),eit = dpmSet.end(); it!=eit; ++it)
261  {
262  const DPIm& dstDpm = *it;
263  if(!indirectCall && SVFUtil::isa<IndirectSVFGEdge>(edge) && !SVFUtil::isa<LoadSVFGNode>(edge->getDstNode()))
264  {
265  if(dstDpm.getCurNodeID() == dpm.getCurNodeID())
266  {
267  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute, forward from :");
268  DBOUT(DDDA, dpm.dump());
269  DOSTAT(ddaStat->_NumOfStepInCycle++);
270  clearbkVisited(dstDpm);
271  findPT(dstDpm);
272  }
273  }
274  else
275  {
276  if(indirectCall)
277  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute for indirect call from :");
278  else
279  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute forward from :");
280  DBOUT(DDDA, dpm.dump());
281  DOSTAT(ddaStat->_NumOfStepInCycle++);
282  clearbkVisited(dstDpm);
283  findPT(dstDpm);
284  }
285  }
286  }

326  {

328  NodeID obj = oldDpm.getCurNodeID();
329  if (_pag->isConstantObj(obj) || _pag->isNonPointerObj(obj))
330  return;

332  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
