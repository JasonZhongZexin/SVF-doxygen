93  inline const Value* getValue() const
94  {
95  assert((this->getNodeKind() != DummyValNode && this->getNodeKind() != DummyObjNode) && "dummy node do not have value!");
96  assert((this->getId()!=SYMTYPE::BlackHole && this->getId() != SYMTYPE::ConstantObj) && "blackhole and constant obj do not have value");

109  inline bool hasValue() const

114  virtual inline bool isPointer() const

139  virtual const std::string getValueName() const = 0;
140 
142  virtual inline const Function* getFunction() const
143  {
144  if(value)
145  {
146  if(const Instruction* inst = SVFUtil::dyn_cast<Instruction>(value))
147  return inst->getParent()->getParent();
148  else if (const Argument* arg = SVFUtil::dyn_cast<Argument>(value))
149  return arg->getParent();
150  else if (const Function* fun = SVFUtil::dyn_cast<Function>(value))
151  return fun;
152  }
153  return NULL;
154  }

157  inline PAGEdge::PAGEdgeSetTy& getIncomingEdges(PAGEdge::PEDGEK kind)
158  {
159  return InEdgeKindToSetMap[kind];
160  }
161 
163  inline PAGEdge::PAGEdgeSetTy& getOutgoingEdges(PAGEdge::PEDGEK kind)
164  {
165  return OutEdgeKindToSetMap[kind];

169  inline bool hasIncomingEdges(PAGEdge::PEDGEK kind) const
170  {
171  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
172  if (it != InEdgeKindToSetMap.end())
173  return (!it->second.empty());
174  else
175  return false;
176  }
177 
179  inline bool hasIncomingVariantGepEdge() const
180  {
181  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = InEdgeKindToSetMap.find(PAGEdge::VariantGep);
182  if (it != InEdgeKindToSetMap.end())
183  {
184  return (!it->second.empty());
185  }
186  return false;
187  }
188 
190  inline PAGEdge::PAGEdgeSetTy::iterator getIncomingEdgesBegin(PAGEdge::PEDGEK kind) const
191  {
192  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
193  assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
194  return it->second.begin();
195  }
196 
198  inline PAGEdge::PAGEdgeSetTy::iterator getIncomingEdgesEnd(PAGEdge::PEDGEK kind) const
199  {
200  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = InEdgeKindToSetMap.find(kind);
201  assert(it!=InEdgeKindToSetMap.end() && "The node does not have such kind of edge");
202  return it->second.end();
203  }
204 
206  inline bool hasOutgoingEdges(PAGEdge::PEDGEK kind) const
207  {
208  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
209  if (it != OutEdgeKindToSetMap.end())
210  return (!it->second.empty());
211  else
212  return false;
213  }
214 
216  inline PAGEdge::PAGEdgeSetTy::iterator getOutgoingEdgesBegin(PAGEdge::PEDGEK kind) const
217  {
218  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
219  assert(it!=OutEdgeKindToSetMap.end() && "The node does not have such kind of edge");
220  return it->second.begin();
221  }
222 
224  inline PAGEdge::PAGEdgeSetTy::iterator getOutgoingEdgesEnd(PAGEdge::PEDGEK kind) const
225  {
226  PAGEdge::PAGKindToEdgeSetMapTy::const_iterator it = OutEdgeKindToSetMap.find(kind);
227  assert(it!=OutEdgeKindToSetMap.end() && "The node does not have such kind of edge");
228  return it->second.end();
229  }
231 
233 
234  inline void addInEdge(PAGEdge* inEdge)
235  {
236  GNodeK kind = inEdge->getEdgeKind();
237  InEdgeKindToSetMap[kind].insert(inEdge);
238  addIncomingEdge(inEdge);
239  }
240 
241  inline void addOutEdge(PAGEdge* outEdge)
242  {
243  GNodeK kind = outEdge->getEdgeKind();
244  OutEdgeKindToSetMap[kind].insert(outEdge);
245  addOutgoingEdge(outEdge);
246  }
247 
248  virtual const std::string toString() const;
249

255  o << node.toString();
256  return o;
257  }
