49 void SymbolTableBuilder::buildMemModel(SVFModule* svfModule)
50 {
51  SVFUtil::increaseStackSize();
52 
53  symInfo->setModule(svfModule);
54 
55  // Pointer #0 always represents the null pointer.
56  assert(symInfo->totalSymNum++ == SymbolTableInfo::NullPtr && "Something changed!");
57 
58  // Pointer #1 always represents the pointer points-to black hole.
59  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlkPtr && "Something changed!");
60 
61  // Object #2 is black hole the object that may point to any object
62  assert(symInfo->totalSymNum++ == SymbolTableInfo::BlackHole && "Something changed!");
63  symInfo->createBlkObj(SymbolTableInfo::BlackHole);
64 
65  // Object #3 always represents the unique constant of a program (merging all constants if Options::ModelConsts is disabled)
66  assert(symInfo->totalSymNum++ == SymbolTableInfo::ConstantObj && "Something changed!");
67  symInfo->createConstantObj(SymbolTableInfo::ConstantObj);
68 
69  // Add symbols for all the globals .
70  for (SVFModule::global_iterator I = svfModule->global_begin(), E =
71  svfModule->global_end(); I != E; ++I)
72  {
73  collectSym(*I);
74  }
75 
76  // Add symbols for all the global aliases
77  for (SVFModule::alias_iterator I = svfModule->alias_begin(), E =
78  svfModule->alias_end(); I != E; I++)
79  {
80  collectSym(*I);
81  collectSym((*I)->getAliasee());
82  }
83 
84  // Add symbols for all of the functions and the instructions in them.
85  for (SVFModule::llvm_iterator F = svfModule->llvmFunBegin(), E = svfModule->llvmFunEnd(); F != E; ++F)
86  {
87  Function *fun = *F;
88  collectSym(fun);
89  collectRet(fun);
90  if (fun->getFunctionType()->isVarArg())
91  collectVararg(fun);
92 
93  // Add symbols for all formal parameters.
94  for (Function::arg_iterator I = fun->arg_begin(), E = fun->arg_end();
95  I != E; ++I)
96  {
97  collectSym(&*I);
98  }
99 
100  // collect and create symbols inside the function body
101  for (inst_iterator II = inst_begin(*fun), E = inst_end(*fun); II != E; ++II)
102  {
103  const Instruction *inst = &*II;
104  collectSym(inst);
105 
106  // initialization for some special instructions
107  //{@
108  if (const StoreInst *st = SVFUtil::dyn_cast<StoreInst>(inst))
109  {
110  collectSym(st->getPointerOperand());
111  collectSym(st->getValueOperand());
112  }
113  else if (const LoadInst *ld = SVFUtil::dyn_cast<LoadInst>(inst))
114  {
115  collectSym(ld->getPointerOperand());
116  }
117  else if (const AllocaInst *alloc = SVFUtil::dyn_cast<AllocaInst>(inst))
118  {
119  collectSym(alloc->getArraySize());
120  }
121  else if (const PHINode *phi = SVFUtil::dyn_cast<PHINode>(inst))
122  {
123  for (u32_t i = 0; i < phi->getNumIncomingValues(); ++i)
124  {
125  collectSym(phi->getIncomingValue(i));
126  }
127  }
128  else if (const GetElementPtrInst *gep = SVFUtil::dyn_cast<GetElementPtrInst>(
129  inst))
130  {
131  collectSym(gep->getPointerOperand());
132  }
133  else if (const SelectInst *sel = SVFUtil::dyn_cast<SelectInst>(inst))
134  {
135  collectSym(sel->getTrueValue());
136  collectSym(sel->getFalseValue());
137  collectSym(sel->getCondition());
138  }
139  else if (const BinaryOperator *binary = SVFUtil::dyn_cast<BinaryOperator>(inst))
140  {
141  for (u32_t i = 0; i < binary->getNumOperands(); i++)
142  collectSym(binary->getOperand(i));
143  }
144  else if (const UnaryOperator *unary = SVFUtil::dyn_cast<UnaryOperator>(inst))
145  {
146  for (u32_t i = 0; i < unary->getNumOperands(); i++)
147  collectSym(unary->getOperand(i));
148  }
149  else if (const CmpInst *cmp = SVFUtil::dyn_cast<CmpInst>(inst))
150  {
151  for (u32_t i = 0; i < cmp->getNumOperands(); i++)
152  collectSym(cmp->getOperand(i));
153  }
154  else if (const CastInst *cast = SVFUtil::dyn_cast<CastInst>(inst))
155  {
156  collectSym(cast->getOperand(0));
157  }
158  else if (const ReturnInst *ret = SVFUtil::dyn_cast<ReturnInst>(inst))
159  {
160  if(ret->getReturnValue())
161  collectSym(ret->getReturnValue());
162  }
163  else if (const BranchInst *br = SVFUtil::dyn_cast<BranchInst>(inst))
164  {
165  Value* opnd = br->isConditional() ? br->getCondition() : br->getOperand(0);
166  collectSym(opnd);
167  }
168  else if (const SwitchInst *sw = SVFUtil::dyn_cast<SwitchInst>(inst))
169  {
170  collectSym(sw->getCondition());
171  }
172  else if (isNonInstricCallSite(inst))
173  {
174 
175  CallSite cs = SVFUtil::getLLVMCallSite(inst);
176  symInfo->callSiteSet.insert(cs);
177  for (User::op_iterator it = cs.getInstruction()->arg_begin();
178  it != cs.getInstruction()->arg_end(); ++it)
179  {
180  collectSym(*it);
181  }
182  // Calls to inline asm need to be added as well because the callee isn't
183  // referenced anywhere else.
184  const Value *Callee = cs.getCalledValue();
185  collectSym(Callee);
186 
187  //TODO handle inlineAsm
189 
190  }
192  }
193  }
194 
195  symInfo->totalSymNum = NodeIDAllocator::get()->endSymbolAllocation();
196  if (Options::SymTabPrint)
197  {
198  SymbolTableInfo::SymbolInfo()->dump();
199  }
200 }

214 void SymbolTableBuilder::collectSpecialSym(const Value* val)
215 {
216  if (LLVMUtil::isPtrInUncalledFunction(val))
217  {
218  symInfo->getModule()->addPtrInUncalledFunction(val);
219  }
220 
221  if (const Function *fun = SVFUtil::dyn_cast<Function>(val))
222  {
223  const SVFFunction *svfFun = symInfo->getModule()->getSVFFunction(fun);
224 
225  if (!isExtCall(svfFun))
226  {
227  for (Function::const_iterator bit = fun->begin(), ebit = fun->end(); bit != ebit; ++bit)
228  {
229  const BasicBlock *bb = &*bit;
230  const u32_t num = LLVMUtil::getBBSuccessorNum(bb);
231  symInfo->getModule()->addBBSuccessorNum(bb,num);
232  if (num >1)
233  {
234  for (succ_const_iterator succIt = succ_begin(bb); succIt != succ_end(bb); succIt++)
235  {
236  const BasicBlock* succ = *succIt;
237  const u32_t successorPos = LLVMUtil::getBBSuccessorPos(bb,succ);
238  if (successorPos != 0)
239  {
240  symInfo->getModule()->addBBSuccessorPos(bb,succ,successorPos);
241  }
242  }
243  }
244  for (succ_const_iterator succIt = succ_begin(bb); succIt != succ_end(bb); succIt++)
245  {
246  const BasicBlock* succ = *succIt;
247  const u32_t predecessorPos = LLVMUtil::getBBPredecessorPos(bb,succ);
248  if (predecessorPos != 0)
249  {
250  symInfo->getModule()->addBBPredecessorPos(bb,succ,predecessorPos);
251  }
252  }
253  }
254  }
255  }
256 
257  if (const Instruction* inst = SVFUtil::dyn_cast<Instruction>(val))
258  {
259  if (LLVMUtil::isReturn(inst))
260  symInfo->getModule()->addReturn(inst);
261  }
262 
263  if (const PointerType * ptrType = SVFUtil::dyn_cast<PointerType>(val->getType()))
264  {
265  const Type* type = LLVMUtil::getPtrElementType(ptrType);
266  symInfo->getModule()->addptrElementType(ptrType, type);
267  }
268 
269  if (LLVMUtil::isArgOfUncalledFunction(val))
270  {
271  symInfo->getModule()->addArgsOfUncalledFunction(val);
272  }
273 }

303 void SymbolTableBuilder::collectVal(const Value *val)
304 {
305  // collect and record special sym here
306  if (LLVMUtil::isNullPtrSym(val) || LLVMUtil::isBlackholeSym(val))
307  {
308  collectNullPtrBlackholeSyms(val);
309  return;
310  }
311  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->valSymMap.find(val);
312  if (iter == symInfo->valSymMap.end())
313  {
314  // create val sym and sym type
315  SymID id = NodeIDAllocator::get()->allocateValueId();
316  symInfo->valSymMap.insert(std::make_pair(val, id));
317  DBOUT(DMemModel,
318  outs() << "create a new value sym " << id << "\n");
320  if (const GlobalVariable* globalVar = SVFUtil::dyn_cast<GlobalVariable>(val))
321  handleGlobalCE(globalVar);
322  }
323 
324  if (LLVMUtil::isConstantObjSym(val))
325  collectObj(val);
326 }

331 void SymbolTableBuilder::collectObj(const Value *val)
332 {
333  val = getGlobalRep(val);
334  SymbolTableInfo::ValueToIDMapTy::iterator iter = symInfo->objSymMap.find(val);
335  if (iter == symInfo->objSymMap.end())
336  {
337  // if the object pointed by the pointer is a constant data (e.g., i32 0) or a global constant object (e.g. string)
338  // then we treat them as one ConstantObj
339  if((LLVMUtil::isConstantObjSym(val) && !symInfo->getModelConstants()))
340  {
341  symInfo->objSymMap.insert(std::make_pair(val, symInfo->constantSymID()));
342  }
343  // otherwise, we will create an object for each abstract memory location
344  else
345  {
346  // create obj sym and sym type
347  SymID id = NodeIDAllocator::get()->allocateObjectId();
348  symInfo->objSymMap.insert(std::make_pair(val, id));
349  DBOUT(DMemModel,
350  outs() << "create a new obj sym " << id << "\n");
351 
352  // create a memory object
353  MemObj* mem = new MemObj(id, createObjTypeInfo(val), val);
354  assert(symInfo->objMap.find(id) == symInfo->objMap.end());
355  symInfo->objMap[id] = mem;
356  }
357  }

394 void SymbolTableBuilder::handleCE(const Value *val)
395 {
396  if (const Constant* ref = SVFUtil::dyn_cast<Constant>(val))
397  {
398  if (const ConstantExpr* ce = isGepConstantExpr(ref))
399  {
400  DBOUT(DMemModelCE,
401  outs() << "handle constant expression " << SVFUtil::value2String(ref) << "\n");
402  collectVal(ce);
403  collectVal(ce->getOperand(0));
404  // handle the recursive constant express case
405  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
406  handleCE(ce->getOperand(0));
407  }
408  else if (const ConstantExpr* ce = isCastConstantExpr(ref))
409  {
410  DBOUT(DMemModelCE,
411  outs() << "handle constant expression " << SVFUtil::value2String(ref) << "\n");
412  collectVal(ce);
413  collectVal(ce->getOperand(0));
414  // handle the recursive constant express case
415  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
416  handleCE(ce->getOperand(0));
417  }
418  else if (const ConstantExpr* ce = isSelectConstantExpr(ref))
419  {
420  DBOUT(DMemModelCE,
421  outs() << "handle constant expression " << SVFUtil::value2String(ref) << "\n");
422  collectVal(ce);
423  collectVal(ce->getOperand(0));
424  collectVal(ce->getOperand(1));
425  collectVal(ce->getOperand(2));
426  // handle the recursive constant express case
427  // like (gep (bitcast (gep X 1)) 1); the inner gep is ce->getOperand(0)
428  handleCE(ce->getOperand(0));
429  handleCE(ce->getOperand(1));
430  handleCE(ce->getOperand(2));
431  }
432  // if we meet a int2ptr, then it points-to black hole
433  else if (const ConstantExpr *int2Ptrce = isInt2PtrConstantExpr(ref))
434  {
435  collectVal(int2Ptrce);
436  }
437  else if (const ConstantExpr *ptr2Intce = isPtr2IntConstantExpr(ref))
438  {
439  collectVal(ptr2Intce);
440  const Constant *opnd = ptr2Intce->getOperand(0);
441  handleCE(opnd);
442  }
443  else if (isTruncConstantExpr(ref) || isCmpConstantExpr(ref))
444  {
445  collectVal(ref);
446  }
447  else if (isBinaryConstantExpr(ref))
448  {
449  collectVal(ref);
450  }
451  else if (isUnaryConstantExpr(ref))
452  {
453  // we don't handle unary constant expression like fneg(x) now
454  collectVal(ref);
455  }
456  else if (SVFUtil::isa<ConstantAggregate>(ref))
457  {
458  // we don't handle constant agrgregate like constant vectors
459  collectVal(ref);
460  }
461  else
462  {
463  if (SVFUtil::isa<ConstantExpr>(val))
464  assert(false && "we don't handle all other constant expression for now!");
465  collectVal(ref);
466  }
467  }
468 }

473 void SymbolTableBuilder::handleGlobalCE(const GlobalVariable *G)
474 {
475  assert(G);
476 
477  //The type this global points to
478  const Type *T = G->getType()->getContainedType(0);
479  bool is_array = 0;
480  //An array is considered a single variable of its type.
481  while (const ArrayType *AT = SVFUtil::dyn_cast<ArrayType>(T))
482  {
483  T = AT->getElementType();
484  is_array = 1;
485  }
486 
487  if (SVFUtil::isa<StructType>(T))
488  {
489  //A struct may be used in constant GEP expr.
490  for (Value::const_user_iterator it = G->user_begin(), ie = G->user_end();
491  it != ie; ++it)
492  {
493  handleCE(*it);
494  }
495  }
496  else
497  {
498  if (is_array)
499  {
500  for (Value::const_user_iterator it = G->user_begin(), ie =
501  G->user_end(); it != ie; ++it)
502  {
503  handleCE(*it);
504  }
505  }
506  }
507 
508  if (G->hasInitializer())
509  {
510  handleGlobalInitializerCE(G->getInitializer());
511  }

565  //TODO:assert(SVFUtil::isa<ConstantVector>(C),"what else do we have");

572 ObjTypeInfo* SymbolTableBuilder::createObjTypeInfo(const Value *val)
573 {
574  const PointerType *refTy = nullptr;
575 
576  const Instruction *I = SVFUtil::dyn_cast<Instruction>(val);
577 
578  // We consider two types of objects:
579  // (1) A heap/static object from a callsite
580  if (I && isNonInstricCallSite(I))
581  refTy = getRefTypeOfHeapAllocOrStatic(I);
582  // (2) Other objects (e.g., alloca, global, etc.)
583  else
584  refTy = SVFUtil::dyn_cast<PointerType>(val->getType());
585 
586  if (refTy)
587  {
588  Type *objTy = getPtrElementType(refTy);
589  ObjTypeInfo* typeInfo = new ObjTypeInfo(objTy, Options::MaxFieldLimit);
590  initTypeInfo(typeInfo,val);
591  return typeInfo;
592  }
593  else
594  {
595  writeWrnMsg("try to create an object with a non-pointer type.");
596  writeWrnMsg(val->getName().str());
597  writeWrnMsg("(" + getSourceLoc(val) + ")");
598  if(LLVMUtil::isConstantObjSym(val))
599  {
600  ObjTypeInfo* typeInfo = new ObjTypeInfo(val->getType(), 0);
601  initTypeInfo(typeInfo,val);
602  return typeInfo;
603  }
604  else
605  {
606  assert(false && "Memory object must be either (1) held by a pointer-typed ref value or (2) a constant value (e.g., 10).");
607  abort();
608  }
609  }

615 void SymbolTableBuilder::analyzeObjType(ObjTypeInfo* typeinfo, const Value* val)
616 {
617 
618  const PointerType * refty = SVFUtil::dyn_cast<PointerType>(val->getType());
619  assert(refty && "this value should be a pointer type!");
620  Type* elemTy = getPtrElementType(refty);
621  bool isPtrObj = false;
622  // Find the inter nested array element
623  while (const ArrayType *AT= SVFUtil::dyn_cast<ArrayType>(elemTy))
624  {
625  elemTy = AT->getElementType();
626  if(elemTy->isPointerTy())
627  isPtrObj = true;
628  if(SVFUtil::isa<GlobalVariable>(val) && SVFUtil::cast<GlobalVariable>(val)->hasInitializer()
629  && SVFUtil::isa<ConstantArray>(SVFUtil::cast<GlobalVariable>(val)->getInitializer()))
630  typeinfo->setFlag(ObjTypeInfo::CONST_ARRAY_OBJ);
631  else
632  typeinfo->setFlag(ObjTypeInfo::VAR_ARRAY_OBJ);
633  }
634  if (const StructType *ST= SVFUtil::dyn_cast<StructType>(elemTy))
635  {
636  const std::vector<const Type*>& flattenFields = SymbolTableInfo::SymbolInfo()->getFlattenFieldTypes(ST);
637  for(std::vector<const Type*>::const_iterator it = flattenFields.begin(), eit = flattenFields.end();
638  it!=eit; ++it)
639  {
640  if((*it)->isPointerTy())
641  isPtrObj = true;
642  }
643  if(SVFUtil::isa<GlobalVariable>(val) && SVFUtil::cast<GlobalVariable>(val)->hasInitializer()
644  && SVFUtil::isa<ConstantStruct>(SVFUtil::cast<GlobalVariable>(val)->getInitializer()))
645  typeinfo->setFlag(ObjTypeInfo::CONST_STRUCT_OBJ);
646  else
647  typeinfo->setFlag(ObjTypeInfo::VAR_STRUCT_OBJ);
648  }
649  else if (elemTy->isPointerTy())
650  {
651  isPtrObj = true;
652  }
653 
654  if(isPtrObj)
655  typeinfo->setFlag(ObjTypeInfo::HASPTR_OBJ);
656 }

697 void SymbolTableBuilder::initTypeInfo(ObjTypeInfo* typeinfo, const Value* val)
698 {
699 
700  u32_t objSize = 1;
701  // Global variable
702  if (SVFUtil::isa<Function>(val))
703  {
704  typeinfo->setFlag(ObjTypeInfo::FUNCTION_OBJ);
705  analyzeObjType(typeinfo,val);
706  objSize = getObjSize(typeinfo->getType());
707  }
708  else if(const AllocaInst* allocaInst = SVFUtil::dyn_cast<AllocaInst>(val))
709  {
710  typeinfo->setFlag(ObjTypeInfo::STACK_OBJ);
711  analyzeObjType(typeinfo,val);
714  if(const ConstantInt *sz = SVFUtil::dyn_cast<ConstantInt>(allocaInst->getArraySize()))
715  objSize = sz->getZExtValue() * getObjSize(typeinfo->getType());
716  else
717  objSize = getObjSize(typeinfo->getType());
718  }
719  else if(SVFUtil::isa<GlobalVariable>(val))
720  {
721  typeinfo->setFlag(ObjTypeInfo::GLOBVAR_OBJ);
722  if(LLVMUtil::isConstantObjSym(val))
723  typeinfo->setFlag(ObjTypeInfo::CONST_GLOBAL_OBJ);
724  analyzeObjType(typeinfo,val);
725  objSize = getObjSize(typeinfo->getType());
726  }
727  else if (SVFUtil::isa<Instruction>(val) && isHeapAllocExtCall(SVFUtil::cast<Instruction>(val)))
728  {
729  analyzeHeapObjType(typeinfo,val);
730  // Heap object, label its field as infinite here
731  objSize = typeinfo->getMaxFieldOffsetLimit();
732  }
733  else if (SVFUtil::isa<Instruction>(val) && isStaticExtCall(SVFUtil::cast<Instruction>(val)))
734  {
735  analyzeStaticObjType(typeinfo,val);
736  // static object allocated before main, label its field as infinite here
737  objSize = typeinfo->getMaxFieldOffsetLimit();
738  }
739  else if(ArgInProgEntryFunction(val))
740  {
741  analyzeStaticObjType(typeinfo,val);
742  // user input data, label its field as infinite here
743  objSize = typeinfo->getMaxFieldOffsetLimit();
744  }
745  else if(SVFUtil::isConstantData(val))
746  {
747  typeinfo->setFlag(ObjTypeInfo::CONST_DATA);
748  objSize = SymbolTableInfo::SymbolInfo()->getNumOfFlattenElements(val->getType());
749  }
750  else
751  {
752  assert("what other object do we have??");
753  abort();
754  }
755 
756  // Reset maxOffsetLimit if it is over the total fieldNum of this object
757  if(typeinfo->getMaxFieldOffsetLimit() > objSize)
758  typeinfo->setNumOfElements(objSize);
759 }

766  assert(ety && "type is null?");
767  u32_t numOfFields = 1;
768  if (SVFUtil::isa<StructType>(ety) || SVFUtil::isa<ArrayType>(ety))
769  {
770  numOfFields = SymbolTableInfo::SymbolInfo()->getNumOfFlattenElements(ety);
771  }
772  return numOfFields;
773 }
