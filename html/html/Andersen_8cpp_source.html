362 bool Andersen::processGepPts(const PointsTo& pts, const GepCGEdge* edge)
363 {
364  numOfProcessedGep++;
365 
366  PointsTo tmpDstPts;
367  if (SVFUtil::isa<VariantGepCGEdge>(edge))
368  {
369  // If a pointer is connected by a variant gep edge,
370  // then set this memory object to be field insensitive,
371  // unless the object is a black hole/constant.
372  for (NodeID o : pts)
373  {
374  if (consCG->isBlkObjOrConstantObj(o))
375  {
376  tmpDstPts.set(o);
377  continue;
378  }
379 
380  if (!isFieldInsensitive(o))
381  {
382  setObjFieldInsensitive(o);
383  consCG->addNodeToBeCollapsed(consCG->getBaseObjNode(o));
384  }
385 
386  // Add the field-insensitive node into pts.
387  NodeID baseId = consCG->getFIObjNode(o);
388  tmpDstPts.set(baseId);
389  }
390  }
391  else if (const NormalGepCGEdge* normalGepEdge = SVFUtil::dyn_cast<NormalGepCGEdge>(edge))
392  {
393  // TODO: after the node is set to field insensitive, handling invariant
394  // gep edge may lose precision because offsets here are ignored, and the
395  // base object is always returned.
396  for (NodeID o : pts)
397  {
398  if (consCG->isBlkObjOrConstantObj(o))
399  {
400  tmpDstPts.set(o);
401  continue;
402  }
403 
404  if (!matchType(edge->getSrcID(), o, normalGepEdge)) continue;
405 
406  NodeID fieldSrcPtdNode = consCG->getGepObjNode(o, normalGepEdge->getLocationSet());
407  tmpDstPts.set(fieldSrcPtdNode);
408  addTypeForGepObjNode(fieldSrcPtdNode, normalGepEdge);
409  }
410  }
411  else
412  {
413  assert(false && "Andersen::processGepPts: New type GEP edge type?");
414  }
415 
416  NodeID dstId = edge->getDstID();
417  if (unionPts(dstId, tmpDstPts))
418  {
419  pushIntoWorklist(dstId);
420  return true;
421  }
422 
423  return false;
424 }

453 void Andersen::mergeSccCycle()
454 {
455  NodeStack revTopoOrder;
456  NodeStack & topoOrder = getSCCDetector()->topoNodeStack();
457  while (!topoOrder.empty())
458  {
459  NodeID repNodeId = topoOrder.top();
460  topoOrder.pop();
461  revTopoOrder.push(repNodeId);
462  const NodeBS& subNodes = getSCCDetector()->subNodes(repNodeId);
463  // merge sub nodes to rep node
464  mergeSccNodes(repNodeId, subNodes);
465  }
466 
467  // restore the topological order for later solving.
468  while (!revTopoOrder.empty())
469  {
470  NodeID nodeId = revTopoOrder.top();
471  revTopoOrder.pop();
472  topoOrder.push(nodeId);
473  }
474 }

516 bool Andersen::collapseField(NodeID nodeId)
517 {
522  if (consCG->isBlkObjOrConstantObj(nodeId))
523  return false;
524 
525  bool changed = false;
526 
527  double start = stat->getClk();
528 
529  // set base node field-insensitive.
530  setObjFieldInsensitive(nodeId);
531 
532  // replace all occurrences of each field with the field-insensitive node
533  NodeID baseId = consCG->getFIObjNode(nodeId);
534  NodeID baseRepNodeId = consCG->sccRepNode(baseId);
535  NodeBS & allFields = consCG->getAllFieldsObjNode(baseId);
536  for (NodeBS::iterator fieldIt = allFields.begin(), fieldEit = allFields.end(); fieldIt != fieldEit; fieldIt++)
537  {
538  NodeID fieldId = *fieldIt;
539  if (fieldId != baseId)
540  {
541  // use the reverse pts of this field node to find all pointers point to it
542  const NodeSet revPts = getRevPts(fieldId);
543  for (const NodeID o : revPts)
544  {
545  // change the points-to target from field to base node
546  clearPts(o, fieldId);
547  addPts(o, baseId);
548  pushIntoWorklist(o);
549 
550  changed = true;
551  }
552  // merge field node into base node, including edges and pts.
553  NodeID fieldRepNodeId = consCG->sccRepNode(fieldId);
554  if (fieldRepNodeId != baseRepNodeId)
555  mergeNodeToRep(fieldRepNodeId, baseRepNodeId);
556 
557  // collect each gep node whose base node has been set as field-insensitive
558  redundantGepNodes.set(fieldId);
559  }
560  }
561 
562  if (consCG->isPWCNode(baseRepNodeId))
563  if (collapseNodePts(baseRepNodeId))
564  changed = true;
565 
566  double end = stat->getClk();
567  timeOfCollapse += (end - start) / TIMEINTERVAL;
568 
569  return changed;
570 }

656 void Andersen::connectCaller2CalleeParams(CallSite cs, const SVFFunction* F, NodePairSet &cpySrcNodes)
657 {
658  assert(F);
659 
660  DBOUT(DAndersen, outs() << "connect parameters from indirect callsite " << *cs.getInstruction() << " to callee " << *F << "\n");
661 
662  CallBlockNode* callBlockNode = pag->getICFG()->getCallBlockNode(cs.getInstruction());
663  RetBlockNode* retBlockNode = pag->getICFG()->getRetBlockNode(cs.getInstruction());
664 
665  if(SVFUtil::isHeapAllocExtFunViaRet(F) && pag->callsiteHasRet(retBlockNode))
666  {
667  heapAllocatorViaIndCall(cs,cpySrcNodes);
668  }
669 
670  if (pag->funHasRet(F) && pag->callsiteHasRet(retBlockNode))
671  {
672  const PAGNode* cs_return = pag->getCallSiteRet(retBlockNode);
673  const PAGNode* fun_return = pag->getFunRet(F);
674  if (cs_return->isPointer() && fun_return->isPointer())
675  {
676  NodeID dstrec = sccRepNode(cs_return->getId());
677  NodeID srcret = sccRepNode(fun_return->getId());
678  if(addCopyEdge(srcret, dstrec))
679  {
680  cpySrcNodes.insert(std::make_pair(srcret,dstrec));
681  }
682  }
683  else
684  {
685  DBOUT(DAndersen, outs() << "not a pointer ignored\n");
686  }
687  }
688 
689  if (pag->hasCallSiteArgsMap(callBlockNode) && pag->hasFunArgsList(F))
690  {
691 
692  // connect actual and formal param
693  const PAG::PAGNodeList& csArgList = pag->getCallSiteArgsList(callBlockNode);
694  const PAG::PAGNodeList& funArgList = pag->getFunArgsList(F);
695  //Go through the fixed parameters.
696  DBOUT(DPAGBuild, outs() << " args:");
697  PAG::PAGNodeList::const_iterator funArgIt = funArgList.begin(), funArgEit = funArgList.end();
698  PAG::PAGNodeList::const_iterator csArgIt = csArgList.begin(), csArgEit = csArgList.end();
699  for (; funArgIt != funArgEit; ++csArgIt, ++funArgIt)
700  {
701  //Some programs (e.g. Linux kernel) leave unneeded parameters empty.
702  if (csArgIt == csArgEit)
703  {
704  DBOUT(DAndersen, outs() << " !! not enough args\n");
705  break;
706  }
707  const PAGNode *cs_arg = *csArgIt ;
708  const PAGNode *fun_arg = *funArgIt;
709 
710  if (cs_arg->isPointer() && fun_arg->isPointer())
711  {
712  DBOUT(DAndersen, outs() << "process actual parm " << cs_arg->toString() << " \n");
713  NodeID srcAA = sccRepNode(cs_arg->getId());
714  NodeID dstFA = sccRepNode(fun_arg->getId());
715  if(addCopyEdge(srcAA, dstFA))
716  {
717  cpySrcNodes.insert(std::make_pair(srcAA,dstFA));
718  }
719  }
720  }
721 
722  //Any remaining actual args must be varargs.
723  if (F->isVarArg())
724  {
725  NodeID vaF = sccRepNode(pag->getVarargNode(F));
726  DBOUT(DPAGBuild, outs() << "\n varargs:");
727  for (; csArgIt != csArgEit; ++csArgIt)
728  {
729  const PAGNode *cs_arg = *csArgIt;
730  if (cs_arg->isPointer())
731  {
732  NodeID vnAA = sccRepNode(cs_arg->getId());
733  if (addCopyEdge(vnAA,vaF))
734  {
735  cpySrcNodes.insert(std::make_pair(vnAA,vaF));
736  }
737  }
738  }
739  }
740  if(csArgIt != csArgEit)
741  {
742  writeWrnMsg("too many args to non-vararg func.");
743  writeWrnMsg("(" + getSourceLoc(cs.getInstruction()) + ")");
744  }
745  }
746 }

775 void Andersen::mergeNodeToRep(NodeID nodeId,NodeID newRepId)
776 {
777 
778  ConstraintNode* node = consCG->getConstraintNode(nodeId);
