126 void MemSSA::createMUCHI(const SVFFunction& fun)
127 {
128 
129  SVFIR* pag = pta->getPAG();
130 
131  DBOUT(DMSSA,
132  outs() << "\t creating mu chi for function " << fun.getName()
133  << "\n");
134  // 1. create mu/chi
135  // insert a set of mus for memory regions at each load
136  // inset a set of chis for memory regions at each store
137 
138  // 2. find global names (region name before renaming) of each memory region,
139  // collect used mrs in usedRegs, and collect its def basic block in reg2BBMap
140  // in the form of mu(r) and r = chi (r)
141  // a) mu(r):
142  // if(r \not\in varKills) global = global \cup r
143  // b) r = chi(r):
144  // if(r \not\in varKills) global = global \cup r
145  // varKills = varKills \cup r
146  // block(r) = block(r) \cup bb_{chi}
147 
150  BBList reachableBBs;
151  getFunReachableBBs(fun.getLLVMFun(),getDT(fun),reachableBBs);
152 
153  for (BBList::const_iterator iter = reachableBBs.begin(), eiter = reachableBBs.end();
154  iter != eiter; ++iter)
155  {
156  const BasicBlock* bb = *iter;
157  varKills.clear();
158  for (BasicBlock::const_iterator it = bb->begin(), eit = bb->end();
159  it != eit; ++it)
160  {
161  const Instruction* inst = &*it;
162  if(mrGen->hasSVFStmtList(inst))
163  {
164  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(inst);
165  for (SVFStmtList::const_iterator bit = pagEdgeList.begin(),
166  ebit = pagEdgeList.end(); bit != ebit; ++bit)
167  {
168  const PAGEdge* inst = *bit;
169  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(inst))
170  AddLoadMU(bb, load, mrGen->getLoadMRSet(load));
171  else if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(inst))
172  AddStoreCHI(bb, store, mrGen->getStoreMRSet(store));
173  }
174  }
175  if (isNonInstricCallSite(inst))
176  {
177  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(inst);
178  if(mrGen->hasRefMRSet(cs))
179  AddCallSiteMU(cs,mrGen->getCallSiteRefMRSet(cs));
180 
181  if(mrGen->hasModMRSet(cs))
182  AddCallSiteCHI(cs,mrGen->getCallSiteModMRSet(cs));
183  }
184  }
185  }
186 
187  // create entry chi for this function including all memory regions
188  // initialize them with version 0 and 1 r_1 = chi (r_0)
189  for (MRSet::iterator iter = usedRegs.begin(), eiter = usedRegs.end();
190  iter != eiter; ++iter)
191  {
192  const MemRegion* mr = *iter;
193  // initialize mem region version and stack for renaming phase
194  mr2CounterMap[mr] = 0;
195  mr2VerStackMap[mr].clear();
196  ENTRYCHI* chi = new ENTRYCHI(&fun, mr);
197  chi->setOpVer(newSSAName(mr,chi));
198  chi->setResVer(newSSAName(mr,chi));
199  funToEntryChiSetMap[&fun].insert(chi);
200 
203  if(functionDoesNotRet(fun.getLLVMFun()) == false)
204  {
205  RETMU* mu = new RETMU(&fun, mr);
206  funToReturnMuSetMap[&fun].insert(mu);
207  }
208 
209  }

216 void MemSSA::insertPHI(const SVFFunction& fun)
217 {
218 
219  DBOUT(DMSSA,
220  outs() << "\t insert phi for function " << fun.getName() << "\n");
221 
222  const DominanceFrontier* df = getDF(fun);
223  // record whether a phi of mr has already been inserted into the bb.
224  BBToMRSetMap bb2MRSetMap;
225 
226  // start inserting phi node
227  for (MRSet::iterator iter = usedRegs.begin(), eiter = usedRegs.end();
228  iter != eiter; ++iter)
229  {
230  const MemRegion* mr = *iter;
231 
232  BBList bbs = reg2BBMap[mr];
233  while (!bbs.empty())
234  {
235  const BasicBlock* bb = bbs.back();
236  bbs.pop_back();
237  DominanceFrontierBase::const_iterator it = df->find(const_cast<BasicBlock*>(bb));
238  if(it == df->end())
239  {
240  writeWrnMsg("bb not in the dominance frontier map??");
241  continue;
242  }
243  const DominanceFrontierBase::DomSetType& domSet = it->second;
244  for (DominanceFrontierBase::DomSetType::const_iterator bit =
245  domSet.begin(); bit != domSet.end(); ++bit)
246  {
247  const BasicBlock* pbb = *bit;
248  // if we never insert this phi node before
249  if (0 == bb2MRSetMap[pbb].count(mr))
250  {
251  bb2MRSetMap[pbb].insert(mr);
252  // insert phi node
253  AddMSSAPHI(pbb,mr);
254  // continue to insert phi in its iterative dominate frontiers
255  bbs.push_back(pbb);
256  }
257  }
258  }
259  }

279 void MemSSA::SSARenameBB(const BasicBlock& bb)
280 {
281 
282  SVFIR* pag = pta->getPAG();
283  // record which mem region needs to pop stack
284  MRVector memRegs;
285 
286  // rename phi result op
287  // for each r = phi (...)
288  // rewrite r as new name
289  if (hasPHISet(&bb))
290  RenamePhiRes(getPHISet(&bb),memRegs);
291 
292 
293  // process mu and chi
294  // for each mu(r)
295  // rewrite r with top mrver of stack(r)
296  // for each r = chi(r')
297  // rewrite r' with top mrver of stack(r)
298  // rewrite r with new name
299 
300  for (BasicBlock::const_iterator it = bb.begin(), eit = bb.end();
301  it != eit; ++it)
302  {
303  const Instruction* inst = &*it;
304  if(mrGen->hasSVFStmtList(inst))
305  {
306  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(inst);
307  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
308  bit!=ebit; ++bit)
309  {
310  const PAGEdge* inst = *bit;
311  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(inst))
312  RenameMuSet(getMUSet(load));
313 
314  else if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(inst))
315  RenameChiSet(getCHISet(store),memRegs);
316 
317  }
318  }
319  if (isNonInstricCallSite(inst))
320  {
321  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(inst);
322  if(mrGen->hasRefMRSet(cs))
323  RenameMuSet(getMUSet(cs));
324 
325  if(mrGen->hasModMRSet(cs))
326  RenameChiSet(getCHISet(cs),memRegs);
327  }
328  else if(isReturn(inst))
329  {
330  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(bb.getParent());
331  RenameMuSet(getReturnMuSet(fun));
332  }
333  }
334 
335 
336  // fill phi operands of succ basic blocks
337  for (succ_const_iterator sit = succ_begin(&bb), esit = succ_end(&bb);
338  sit != esit; ++sit)
339  {
340  const BasicBlock* succ = *sit;
341  u32_t pos = getBBPredecessorPos(&bb, succ);
342  if (hasPHISet(succ))
343  RenamePhiOps(getPHISet(succ),pos,memRegs);
344  }
345 
346  // for succ basic block in dominator tree
347  const SVFFunction* fun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(bb.getParent());
348  DominatorTree* dt = getDT(*fun);
349  if(DomTreeNode *dtNode = dt->getNode(const_cast<BasicBlock*>(&bb)))
350  {
351  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
352  DI != DE; ++DI)
353  {
354  SSARenameBB(*((*DI)->getBlock()));
355  }
356  }
357  // for each r = chi(..), and r = phi(..)
358  // pop ver stack(r)
359  while (!memRegs.empty())
360  {
361  const MemRegion* mr = memRegs.back();
362  memRegs.pop_back();
363  mr2VerStackMap[mr].pop_back();
364  }

595 void MemSSA::dumpMSSA(OutStream& Out)
596 {
597  SVFIR* pag = pta->getPAG();
598 
599  for (SVFModule::iterator fit = pta->getModule()->begin(), efit = pta->getModule()->end();
600  fit != efit; ++fit)
601  {
602  const SVFFunction* fun = *fit;
603  if(Options::MSSAFun!="" && Options::MSSAFun!=fun->getName())
604  continue;
605 
606  Out << "==========FUNCTION: " << fun->getName() << "==========\n";
607  // dump function entry chi nodes
608  if (hasFuncEntryChi(fun))
609  {
610  CHISet & entry_chis = getFuncEntryChiSet(fun);
611  for (CHISet::iterator chi_it = entry_chis.begin(); chi_it != entry_chis.end(); chi_it++)
612  {
613  (*chi_it)->dump();
614  }
615  }
616 
617  for (Function::iterator bit = fun->getLLVMFun()->begin(), ebit = fun->getLLVMFun()->end();
618  bit != ebit; ++bit)
619  {
620  BasicBlock& bb = *bit;
621  if (bb.hasName())
622  Out << bb.getName().str() << "\n";
623  PHISet& phiSet = getPHISet(&bb);
624  for(PHISet::iterator pi = phiSet.begin(), epi = phiSet.end(); pi !=epi; ++pi)
625  {
626  (*pi)->dump();
627  }
628 
629  bool last_is_chi = false;
630  for (BasicBlock::iterator it = bb.begin(), eit = bb.end();
631  it != eit; ++it)
632  {
633  Instruction& inst = *it;
634  bool isAppCall = isNonInstricCallSite(&inst) && !isExtCall(&inst);
635  if (isAppCall || isHeapAllocExtCall(&inst))
636  {
637  const CallICFGNode* cs = pag->getICFG()->getCallICFGNode(&inst);
638  if(hasMU(cs))
639  {
640  if (!last_is_chi)
641  {
642  Out << "\n";
643  }
644  for (MUSet::iterator mit = getMUSet(cs).begin(), emit = getMUSet(cs).end();
645  mit != emit; ++mit)
646  {
647  (*mit)->dump();
648  }
649  }
650 
651  Out << SVFUtil::value2String(&inst) << "\n";
652 
653  if(hasCHI(cs))
654  {
655  for (CHISet::iterator cit = getCHISet(cs).begin(), ecit = getCHISet(cs).end();
656  cit != ecit; ++cit)
657  {
658  (*cit)->dump();
659  }
660  Out << "\n";
661  last_is_chi = true;
662  }
663  else
664  last_is_chi = false;
665  }
666  else
667  {
668  bool dump_preamble = false;
669  SVFStmtList& pagEdgeList = mrGen->getPAGEdgesFromInst(&inst);
670  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
671  bit!=ebit; ++bit)
672  {
673  const PAGEdge* edge = *bit;
674  if (const LoadStmt* load = SVFUtil::dyn_cast<LoadStmt>(edge))
675  {
676  MUSet& muSet = getMUSet(load);
677  for(MUSet::iterator it = muSet.begin(), eit = muSet.end(); it!=eit; ++it)
678  {
679  if (!dump_preamble && !last_is_chi)
680  {
681  Out << "\n";
682  dump_preamble = true;
683  }
684  (*it)->dump();
685  }
686  }
687  }
688 
689  Out << SVFUtil::value2String(&inst) << "\n";
690 
691  bool has_chi = false;
692  for(SVFStmtList::const_iterator bit = pagEdgeList.begin(), ebit= pagEdgeList.end();
693  bit!=ebit; ++bit)
694  {
695  const PAGEdge* edge = *bit;
696  if (const StoreStmt* store = SVFUtil::dyn_cast<StoreStmt>(edge))
697  {
698  CHISet& chiSet = getCHISet(store);
699  for(CHISet::iterator it = chiSet.begin(), eit = chiSet.end(); it!=eit; ++it)
700  {
701  has_chi = true;
702  (*it)->dump();
703  }
704  }
705  }
706  if (has_chi)
707  {
708  Out << "\n";
709  last_is_chi = true;
710  }
711  else
712  last_is_chi = false;
713  }
714  }
715  }
716 
717  // dump return mu nodes
718  if (hasReturnMu(fun))
719  {
720  MUSet & return_mus = getReturnMuSet(fun);
721  for (MUSet::iterator mu_it = return_mus.begin(); mu_it != return_mus.end(); mu_it++)
722  {
723  (*mu_it)->dump();
724  }
725  }
726  }
