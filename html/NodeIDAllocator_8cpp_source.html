71  NodeID NodeIDAllocator::allocateGepObjectId(NodeID base, u32_t offset, u32_t maxFieldLimit)
72  {
73  NodeID id = 0;
74  if (strategy == Strategy::DENSE)
75  {
76  // Nothing different to the other case.
77  id = numObjects;
78  }
79  else if (strategy == Strategy::SEQ)
80  {
81  // Everything is sequential and intermixed.
82  id = numNodes;
83  }
84  else if (strategy == Strategy::DEBUG)
85  {
86  // For a gep id, base id is set at lower bits, and offset is set at higher bits
87  // e.g., 1100050 denotes base=50 and offset=10
88  // The offset is 10, not 11, because we add 1 to the offset to ensure that the
89  // high bits are never 0. For example, we do not want the gep id to be 50 when
90  // the base is 50 and the offset is 0.

92  numSymbols > maxFieldLimit ?
93  numSymbols : maxFieldLimit
94  )));

96  assert(id > numSymbols && "NodeIDAllocator::allocateGepObjectId: GEP allocation clashing with other nodes");
