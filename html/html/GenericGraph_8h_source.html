106  {

108  return lhs->edgeFlag < rhs->edgeFlag;
109  else if (lhs->getSrcID() != rhs->getSrcID())

457 // mapped_iter - This is a simple iterator adapter that causes a function to
458 // be applied whenever operator* is invoked on the iterator.
459 
460 template <typename ItTy, typename FuncTy,
461  typename FuncReturnTy =
462  decltype(std::declval<FuncTy>()(*std::declval<ItTy>()))>
463 class mapped_iter
464  : public iter_adaptor_base<
465  mapped_iter<ItTy, FuncTy>, ItTy,
466  typename std::iterator_traits<ItTy>::iterator_category,
467  typename std::remove_reference<FuncReturnTy>::type>
468 {
469 public:
470  mapped_iter(ItTy U, FuncTy F)
471  : mapped_iter::iter_adaptor_base(std::move(U)), F(std::move(F)) {}
472 
473  ItTy getCurrent()
474  {
475  return this->I;
476  }
477 
478  FuncReturnTy operator*() const
479  {
480  return F(*this->I);
481  }
482 
483 private:
484  FuncTy F;
485 };

488 // make_pair is useful for creating pairs...

490 inline mapped_iter<ItTy, FuncTy> map_iter(ItTy I, FuncTy F)
491 {
492  return mapped_iter<ItTy, FuncTy>(std::move(I), std::move(F));
