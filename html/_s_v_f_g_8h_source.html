294  virtual inline void getInterVFEdgeAtIndCSFromAPToFP(const PAGNode* cs_arg, const PAGNode* fun_arg, const CallBlockNode*, CallSiteID csId, SVFGEdgeSetTy& edges)
295  {
296  SVFGNode* actualParam = getSVFGNode(getDef(cs_arg));
297  SVFGNode* formalParam = getSVFGNode(getDef(fun_arg));
298  SVFGEdge* edge = hasInterVFGEdge(actualParam, formalParam, SVFGEdge::CallDirVF, csId);
299  assert(edge != NULL && "Can not find inter value flow edge from aparam to fparam");
300  edges.insert(edge);

312  virtual inline void getInterVFEdgeAtIndCSFromAInToFIn(ActualINSVFGNode* actualIn, const SVFFunction* callee, SVFGEdgeSetTy& edges)
313  {

315  {
316  SVFGEdge* edge = *outIt;

352  inline void setDef(const MRVer* mvar, const SVFGNode* node)
353  {
354  MSSAVarToDefMapTy::iterator it = MSSAVarToDefMap.find(mvar);
355  if(it==MSSAVarToDefMap.end())
356  {
357  MSSAVarToDefMap[mvar] = node->getId();
358  assert(hasSVFGNode(node->getId()) && "not in the map!!");
359  }
360  else
361  {
362  assert((it->second == node->getId()) && "a PAG node can only have unique definition ");
363  }
364  }
365  inline NodeID getDef(const MRVer* mvar) const
366  {
367  MSSAVarToDefMapTy::const_iterator it = MSSAVarToDefMap.find(mvar);
368  assert(it!=MSSAVarToDefMap.end() && "memory SSA does not have a definition??");
369  return it->second;
370  }

419  IntraMSSAPHISVFGNode* sNode = new IntraMSSAPHISVFGNode(totalVFGNode++,phi);
420  addSVFGNode(sNode, pag->getICFG()->getBlockICFGNode(&(phi->getBasicBlock()->front())));
421  for(MemSSA::PHI::OPVers::const_iterator it = phi->opVerBegin(), eit=phi->opVerEnd(); it!=eit; ++it)
