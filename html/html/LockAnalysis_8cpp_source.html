165 void LockAnalysis::analyzeIntraProcedualLock()
166 {
167 
168  // Identify the protected Instructions.
169  for (InstSet::const_iterator it = locksites.begin(), ie = locksites.end(); it != ie; ++it)
170  {
171  const CallInst *lockSite = SVFUtil::dyn_cast<CallInst>(*it);
172  assert(lockSite && "Lock acquire instruction must be CallInst");
173 
174  // Perform forward traversal
175  InstSet forwardInsts;
176  InstSet backwardInsts;
177  InstSet unlockSet;
178 
179  bool forward = intraForwardTraverse(lockSite,unlockSet,forwardInsts);
180  bool backward = intraBackwardTraverse(unlockSet,backwardInsts);
181 
183  if(forward && backward)
184  addIntraLock(lockSite,forwardInsts);
185  else if(forward && !backward)
186  addCondIntraLock(lockSite,forwardInsts);
187  }

193 bool LockAnalysis::intraForwardTraverse(const Instruction* lockSite, InstSet& unlockSet, InstSet& forwardInsts)
194 {
195 
196  const Function* fun = lockSite->getParent()->getParent();
197  const SVFFunction* svfFun = LLVMModuleSet::getLLVMModuleSet()->getSVFFunction(fun);
198 
199  InstVec worklist;
200  worklist.push_back(lockSite);
201  while (!worklist.empty())
202  {
203  const Instruction *I = worklist.back();
204  worklist.pop_back();
205 
206  if(&(getFunExitBB(svfFun)->back()) == I)
207  return false;
208 
209  // Skip the visited Instructions.
210  if (forwardInsts.find(I)!=forwardInsts.end())
211  continue;
212  forwardInsts.insert(I);
213 
214  if (isTDRelease(I) && isAliasedLocks(lockSite, I))
215  {
216  unlockSet.insert(I);
217  DBOUT(DMTA, outs() << "LockAnalysis ci lock -- " << SVFUtil::getSourceLoc(lockSite)<<"\n");
218  DBOUT(DMTA, outs() << "LockAnalysis ci unlock -- " << SVFUtil::getSourceLoc(I)<<"\n");
219  continue;
220  }
221 
222  InstVec nextInsts;
223  getNextInsts(I, nextInsts);
224  for (InstVec::const_iterator nit = nextInsts.begin(), enit = nextInsts.end(); nit != enit; ++nit)
225  {
226  worklist.push_back(*nit);
227  }
228  }
229 
230  return true;
231 }

237 bool LockAnalysis::intraBackwardTraverse(const InstSet& unlockSet, InstSet& backwardInsts)
238 {
239 
240  InstVec worklist;
241  for(InstSet::const_iterator it = unlockSet.begin(), eit = unlockSet.end(); it!=eit; ++it)
242  {
243  const Instruction* unlockSite = *it;
244  const Function* fun = unlockSite->getParent()->getParent();
245  const Instruction* entryInst = &(fun->getEntryBlock().back());
246  worklist.push_back(*it);
247 
248  while (!worklist.empty())
249  {
250  const Instruction *I = worklist.back();
251  worklist.pop_back();
252 
253  if(entryInst == I)
254  return false;
255 
256  // Skip the visited Instructions.
257  if (backwardInsts.find(I)!=backwardInsts.end())
258  continue;
259  backwardInsts.insert(I);
260 
261  if (isTDAcquire(I) && isAliasedLocks(unlockSite, I))
262  {
263  DBOUT(DMTA, outs() << "LockAnalysis ci lock -- " << SVFUtil::getSourceLoc(I)<<"\n");
264  DBOUT(DMTA, outs() << "LockAnalysis ci unlock -- " << SVFUtil::getSourceLoc(unlockSite)<<"\n");
265  continue;
266  }
267 
268  InstVec nextInsts;
269  getPrevInsts(I, nextInsts);
270  for (InstVec::const_iterator nit = nextInsts.begin(), enit = nextInsts.end(); nit != enit; ++nit)
271  {
272  worklist.push_back(*nit);
273  }
274  }
275  }
276 
277  return true;
278 }

281 void LockAnalysis::collectCxtLock()
282 {
283  FunSet entryFuncSet = tct->getEntryProcs();
284  for (FunSet::const_iterator it = entryFuncSet.begin(), eit = entryFuncSet.end(); it != eit; ++it)
285  {
286  if (!isLockCandidateFun(*it))
287  continue;
288  CallStrCxt cxt;
289  CxtLockProc t(cxt, tct->getSVFFun(*it));
290  pushToCTPWorkList(t);
291  }
292 
293  while (!clpList.empty())
294  {
295  CxtLockProc clp = popFromCTPWorkList();
296  PTACallGraphNode* cgNode = getTCG()->getCallGraphNode(clp.getProc());
297  // lzh TODO.
298  if (!isLockCandidateFun(cgNode->getFunction()->getLLVMFun()))
299  continue;
300 
301  for (PTACallGraphNode::const_iterator nit = cgNode->OutEdgeBegin(), neit = cgNode->OutEdgeEnd(); nit != neit; nit++)
302  {
303  const PTACallGraphEdge* cgEdge = (*nit);
304 
305  for (PTACallGraphEdge::CallInstSet::const_iterator cit = cgEdge->directCallsBegin(), ecit = cgEdge->directCallsEnd();
306  cit != ecit; ++cit)
307  {
308  DBOUT(DMTA,
309  outs() << "\nCollecting CxtLocks: handling direct call:" << **cit << "\t" << cgEdge->getSrcNode()->getFunction()->getName()
310  << "-->" << cgEdge->getDstNode()->getFunction()->getName() << "\n");
311  handleCallRelation(clp, cgEdge, getLLVMCallSite((*cit)->getCallSite()));
312  }
313  for (PTACallGraphEdge::CallInstSet::const_iterator ind = cgEdge->indirectCallsBegin(), eind = cgEdge->indirectCallsEnd();
314  ind != eind; ++ind)
315  {
316  DBOUT(DMTA,
317  outs() << "\nCollecting CxtLocks: handling indirect call:" << **ind << "\t"
318  << cgEdge->getSrcNode()->getFunction()->getName() << "-->" << cgEdge->getDstNode()->getFunction()->getName()
319  << "\n");
320  handleCallRelation(clp, cgEdge, getLLVMCallSite((*ind)->getCallSite()));
321  }
322  }

353 void LockAnalysis::analyzeLockSpanCxtStmt()
354 {
355 
356  FunSet entryFuncSet = tct->getEntryProcs();
357  for (FunSet::const_iterator it = entryFuncSet.begin(), eit = entryFuncSet.end(); it != eit; ++it)
358  {
359  if (!isLockCandidateFun(*it))
360  continue;
361  CallStrCxt cxt;
362  CxtStmt cxtstmt(cxt, &((*it)->front().front()));
363  pushToCTSWorkList(cxtstmt);
364  }
365 
366  while (!cxtStmtList.empty())
367  {
368  CxtStmt cts = popFromCTSWorkList();
369 
370  touchCxtStmt(cts);
371  const Instruction* curInst = cts.getStmt();
372  instToCxtStmtSet[curInst].insert(cts);
373 
374  DBOUT(DMTA, outs() << "\nVisit cxtStmt: ");
375  DBOUT(DMTA, cts.dump());
376 
377  DBOUT(DMTA, outs() << "\nIts cxt lock sets: ");
378  DBOUT(DMTA, printLocks(cts));
379 
380  if (isTDFork(curInst))
381  {
382  handleFork(cts);
383  }
384  else if (isTDAcquire(curInst))
385  {
386  assert(hasCxtLock(cts) && "context-sensitive lock not found!!");
387  if(addCxtStmtToSpan(cts,cts))
388  handleIntra(cts);
389  }
390  else if (isTDRelease(curInst))
391  {
392  if(removeCxtStmtToSpan(cts,cts))
393  handleIntra(cts);
394  }
395  else if (SVFUtil::isa<CallInst>(curInst) && !isExtCall(curInst))
396  {
397  handleCall(cts);
398  }
399  else if (SVFUtil::isa<ReturnInst>(curInst))
400  {
401  handleRet(cts);
402  }
403  else
404  {
405  handleIntra(cts);
406  }
407 
408  }
