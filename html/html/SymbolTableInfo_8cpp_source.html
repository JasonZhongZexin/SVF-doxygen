136 void SymbolTableInfo::collectArrayInfo(const ArrayType* ty)
137 {
138  u64_t totalElemNum = ty->getNumElements();
139  const Type* elemTy = ty->getElementType();
140  while (const ArrayType* aty = SVFUtil::dyn_cast<ArrayType>(elemTy))
141  {
142  totalElemNum *= aty->getNumElements();
143  elemTy = aty->getElementType();
144  }
145 
146  StInfo* stinfo = new StInfo(totalElemNum);
147  typeToFieldInfo[ty] = stinfo;
148 
150  if(totalElemNum==0)
151  {
152  stinfo->addFldWithType(0, elemTy, 0);
153  stinfo->setNumOfFieldsAndElems(1, 1);
154  stinfo->getFlattenFieldTypes().push_back(elemTy);
155  stinfo->getFlattenElementTypes().push_back(elemTy);
156  return;
157  }
158 
161  StInfo* elemStInfo = getStructInfo(elemTy);
162  u32_t nfE = elemStInfo->getNumOfFlattenFields();
163  for (u32_t j = 0; j < nfE; j++)
164  {
165  const Type* fieldTy = elemStInfo->getFlattenFieldTypes()[j];
166  stinfo->getFlattenFieldTypes().push_back(fieldTy);
167  }
168 
171  u32_t outArrayElemNum = ty->getNumElements();
172  for(u32_t i = 0; i < outArrayElemNum; i++)
173  stinfo->addFldWithType(0, elemTy, (i * nfE * totalElemNum)/outArrayElemNum);
174 
175  for(u32_t i = 0; i < totalElemNum; i++)
176  {
177  for(u32_t j = 0; j < nfE; j++)
178  {
179  stinfo->getFlattenElementTypes().push_back(elemStInfo->getFlattenFieldTypes()[j]);
180  }
181  }
182 
183  assert(stinfo->getFlattenElementTypes().size() == nfE * totalElemNum && "typeForArray size incorrect!!!");
184  stinfo->setNumOfFieldsAndElems(nfE, nfE * totalElemNum);
185 }

192 void SymbolTableInfo::collectStructInfo(const StructType *sty)
193 {
195  StInfo* stinfo = new StInfo(1);
196  typeToFieldInfo[sty] = stinfo;
197 
198  // Number of fields after flattening the struct
199  u32_t nf = 0;
200  // The offset when considering array stride info
201  u32_t strideOffset = 0;
202  for (StructType::element_iterator it = sty->element_begin(), ie =
203  sty->element_end(); it != ie; ++it)
204  {
205  const Type *et = *it;
207  stinfo->addFldWithType(nf, et, strideOffset);
208 
209  if (SVFUtil::isa<StructType>(et) || SVFUtil::isa<ArrayType>(et))
210  {
211  StInfo * subStinfo = getStructInfo(et);
212  u32_t nfE = subStinfo->getNumOfFlattenFields();
213  //Copy ST's info, whose element 0 is the size of ST itself.
214  for (u32_t j = 0; j < nfE; j++)
215  {
216  const Type* elemTy = subStinfo->getFlattenFieldTypes()[j];
217  stinfo->getFlattenFieldTypes().push_back(elemTy);
218  }
219  nf += nfE;
220  strideOffset += nfE * subStinfo->getStride();
221  for(u32_t tpi = 0; tpi < subStinfo->getStride(); tpi++)
222  {
223  for(u32_t tpj = 0; tpj < nfE; tpj++)
224  {
225  stinfo->getFlattenElementTypes().push_back(subStinfo->getFlattenFieldTypes()[tpj]);
226  }
227  }
228  }
229  else //simple type
230  {
231  nf += 1;
232  strideOffset += 1;
233  stinfo->getFlattenFieldTypes().push_back(et);
234  stinfo->getFlattenElementTypes().push_back(et);
235  }
236  }
237 
238  assert(stinfo->getFlattenElementTypes().size() == strideOffset && "typeForStruct size incorrect!");
239  stinfo->setNumOfFieldsAndElems(nf,strideOffset);
240 
241  //Record the size of the complete struct and update max_struct.
242  if (nf > maxStSize)
243  {
244  maxStruct = sty;
245  maxStSize = nf;
246  }
247 }

336 {
337  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
338  {

356  if (SVFUtil::isa<PointerType>(elemTy))
