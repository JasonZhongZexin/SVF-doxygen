64 bool PCG::mayHappenInParallelBetweenFunctions(const Function* fun1, const Function* fun2) const
65 {
66  // if neither of functions are spawnees, then they won't happen in parallel
67  if (isSpawneeFun(fun1) == false && isSpawneeFun(fun2) == false)
68  return false;
69  // if there exit one of the function are not spawner, spawnee or follower, then they won't happen in parallel
70  if (isSpawnerFun(fun1) == false && isSpawneeFun(fun1) == false && isFollowerFun(fun1) == false)
71  return false;
72  if (isSpawnerFun(fun2) == false && isSpawneeFun(fun2) == false && isFollowerFun(fun2) == false)
73  return false;
74 
75  return true;
76 }

140 void PCG::collectSpawners()
141 {
142 
144  FunWorkList worklist;
145  for (FunSet::iterator it = spawners.begin(), eit = spawners.end(); it != eit; ++it)
146  {
147  worklist.push(*it);
148  }
149  while (!worklist.empty())
150  {
151  const Function* fun = worklist.pop();
152  const SVFFunction* svffun = getSVFFun(fun);
153  PTACallGraphNode* funNode = callgraph->getCallGraphNode(svffun);
154  for (PTACallGraphNode::const_iterator it = funNode->InEdgeBegin(), eit = funNode->InEdgeEnd(); it != eit;
155  ++it)
156  {
157  PTACallGraphEdge* callEdge = (*it);
158  const Function* caller = callEdge->getSrcNode()->getFunction()->getLLVMFun();
159  if (isSpawnerFun(caller) == false)
160  {
161  worklist.push(caller);
162  addSpawnerFun(caller);
163  }
165  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->directCallsBegin(), deit =
166  callEdge->directCallsEnd(); dit != deit; ++dit)
167  {
168  addSpawnsite((*dit)->getCallSite());
169  }
170  for (PTACallGraphEdge::CallInstSet::const_iterator dit = callEdge->indirectCallsBegin(), deit =
171  callEdge->indirectCallsEnd(); dit != deit; ++dit)
172  {
173  addSpawnsite((*dit)->getCallSite());
174  }
175  }
176  }

213 void PCG::identifyFollowers()
214 {
215 
216  for (CallInstSet::const_iterator sit = spawnSitesBegin(), esit = spawnSitesEnd(); sit != esit; ++sit)
217  {
218  const Instruction* inst = *sit;
219  BBWorkList bb_worklist;
220  Set<const BasicBlock*> visitedBBs;
221  bb_worklist.push(inst->getParent());
222  while (!bb_worklist.empty())
223  {
224  const BasicBlock* bb = bb_worklist.pop();
225  for (BasicBlock::const_iterator it = bb->begin(), eit = bb->end(); it != eit; ++it)
226  {
227  const Instruction* inst = &*it;
228  // mark the callee of this callsite as follower
229  // if this is an call/invoke instruction but not a spawn site
230  if ((SVFUtil::isCallSite(inst)) && !isSpawnsite(inst))
231  {
232  CallICFGNode* cbn = getCallICFGNode(inst);
233  if (callgraph->hasCallGraphEdge(cbn))
234  {
235  for (PTACallGraph::CallGraphEdgeSet::const_iterator cgIt = callgraph->getCallEdgeBegin(cbn),
236  ecgIt = callgraph->getCallEdgeEnd(cbn); cgIt != ecgIt; ++cgIt)
237  {
238  const PTACallGraphEdge* edge = *cgIt;
239  addFollowerFun(edge->getDstNode()->getFunction()->getLLVMFun());
240  }
241  }
242  }
243  }
244  for (succ_const_iterator succ_it = succ_begin(bb); succ_it != succ_end(bb); succ_it++)
245  {
246  const BasicBlock* succ_bb = *succ_it;
247  if (visitedBBs.count(succ_bb) == 0)
248  {
249  visitedBBs.insert(succ_bb);
250  bb_worklist.push(succ_bb);
251  }
252  }
253  }
254  }

302 
303 // DBOUT(DMTA, outs() << pasMsg("Starting Race Detection\n"));
304 
305  PCG::FunVec worklist;
306  for (SVFModule::const_iterator F = mod->begin(), E = mod->end(); F != E; ++F)

308  const SVFFunction* fun = *F;
309  if (isExtCall(fun))

314  while (!worklist.empty())

321  {

323  if (mayHappenInParallelBetweenFunctions(fun1, fun2))
324  {
325  ismhpfun = true;
