212  bool allCTir(void) const
213  {
214  // Iterate over all modules. If a single module does not have the correct ctir module flag,
215  // short-circuit and return false.
216  for (u32_t i = 0; i < getModuleNum(); ++i)
217  {
218  llvm::Metadata *ctirModuleFlag = llvmModuleSet->getModule(i)->getModuleFlag(cppUtil::ctir::derefMDName);
219  if (ctirModuleFlag == nullptr)
220  {
221  return false;
222  }
223 
224  llvm::ConstantAsMetadata *flagConstMetadata = SVFUtil::dyn_cast<llvm::ConstantAsMetadata>(ctirModuleFlag);
225  ConstantInt *flagConstInt = SVFUtil::dyn_cast<ConstantInt>(flagConstMetadata->getValue());
226  if (flagConstInt->getZExtValue() != cppUtil::ctir::moduleFlagValue)
227  {
228  return false;
229  }
230  }
231 
232  return true;
233  }
