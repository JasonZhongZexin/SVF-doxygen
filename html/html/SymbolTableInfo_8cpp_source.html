88 void SymbolTableInfo::collectArrayInfo(const ArrayType* ty)
89 {
90  StInfo* stinfo = new StInfo();
91  typeToFieldInfo[ty] = stinfo;
92 
93  u64_t out_num = ty->getNumElements();
94  const llvm::Type* elemTy = ty->getElementType();
95  u32_t out_stride = getTypeSizeInBytes(elemTy);
96  while (const ArrayType* aty = SVFUtil::dyn_cast<ArrayType>(elemTy))
97  {
98  out_num *= aty->getNumElements();
99  elemTy = aty->getElementType();
100  out_stride = getTypeSizeInBytes(elemTy);
101  }
102 
104  stinfo->addFldWithType(0,elemTy);
105 
108  StInfo* elemStInfo = getStructInfo(elemTy);
109  u32_t nfE = elemStInfo->getFlattenedFieldInfoVec().size();
110  for (u32_t j = 0; j < nfE; j++)
111  {
112  u32_t idx = elemStInfo->getFlattenedFieldInfoVec()[j].getFlattenFldIdx();
113  const Type* fieldTy = elemStInfo->getFlattenedFieldInfoVec()[j].getFlattenElemTy();
114  FlattenedFieldInfo field(idx, fieldTy);
115  stinfo->getFlattenedFieldInfoVec().push_back(field);
116  }
117 }

124 void SymbolTableInfo::collectStructInfo(const StructType *sty)
125 {
127  StInfo* stinfo = new StInfo();
128  typeToFieldInfo[sty] = stinfo;
129 
130  // Number of fields after flattening the struct
131  u32_t nf = 0;
132  // field of the current struct
133  u32_t field_idx = 0;
134  for (StructType::element_iterator it = sty->element_begin(), ie =
135  sty->element_end(); it != ie; ++it, ++field_idx)
136  {
137  const Type *et = *it;
139  stinfo->addFldWithType(nf, et);
140 
141  if (SVFUtil::isa<StructType>(et) || SVFUtil::isa<ArrayType>(et))
142  {
143  StInfo * subStinfo = getStructInfo(et);
144  u32_t nfE = subStinfo->getFlattenedFieldInfoVec().size();
145  //Copy ST's info, whose element 0 is the size of ST itself.
146  for (u32_t j = 0; j < nfE; j++)
147  {
148  u32_t fldIdx = nf + subStinfo->getFlattenedFieldInfoVec()[j].getFlattenFldIdx();
149  const Type* elemTy = subStinfo->getFlattenedFieldInfoVec()[j].getFlattenElemTy();
150  FlattenedFieldInfo field(fldIdx, elemTy);
151  stinfo->getFlattenedFieldInfoVec().push_back(field);
152  }
153  nf += nfE;
154  }
155  else //simple type
156  {
157  FlattenedFieldInfo field(nf, et);
158  stinfo->getFlattenedFieldInfoVec().push_back(field);
159  ++nf;
160  }
161  }
162 
163  //Record the size of the complete struct and update max_struct.
164  if (nf > maxStSize)
165  {
166  maxStruct = sty;
167  maxStSize = nf;
168  }
169 }

191 LocationSet SymbolTableInfo::getModulusOffset(const MemObj* obj, const LocationSet& ls)
192 {
193 
197 
198  s64_t offset = ls.accumulateConstantFieldIdx();
199  if(offset < 0)

267 bool SymbolTableInfo::isConstantObjSym(const Value *val)
268 {
269  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
270  {
271  if (cppUtil::isValVtbl(v))
272  return false;
273  else if (!v->hasInitializer()){
274  if(v->isExternalLinkage(v->getLinkage()))
275  return false;
276  else
277  return true;
278  }
279  else
280  {
281  StInfo *stInfo = getStructInfo(v->getInitializer()->getType());
282  const std::vector<FlattenedFieldInfo> &fields = stInfo->getFlattenedFieldInfoVec();
283  for (std::vector<FlattenedFieldInfo>::const_iterator it = fields.begin(), eit = fields.end(); it != eit; ++it)
284  {
285  const FlattenedFieldInfo &field = *it;
286  const Type *elemTy = field.getFlattenElemTy();
287  assert(!SVFUtil::isa<FunctionType>(elemTy) && "Initializer of a global is a function?");
288  if (SVFUtil::isa<PointerType>(elemTy))
289  return false;
290  }
291 
292  return v->isConstant();
293  }
294  }
295  return SVFUtil::isConstantData(val);
296 }

538 bool ObjTypeInfo::isNonPtrFieldObj(const LocationSet& ls)
539 {
540  if (isHeap() || isStaticObj())
541  return false;
542 
543  const Type* ety = getType();
544  while (const ArrayType *AT= SVFUtil::dyn_cast<ArrayType>(ety))
545  {
546  ety = AT->getElementType();
547  }
548 
549  if (SVFUtil::isa<StructType>(ety) || SVFUtil::isa<ArrayType>(ety))
550  {
551  bool hasIntersection = false;
552  const vector<FlattenedFieldInfo> &infovec = SymbolTableInfo::SymbolInfo()->getFlattenedFieldInfoVec(ety);
553  vector<FlattenedFieldInfo>::const_iterator it = infovec.begin();
554  vector<FlattenedFieldInfo>::const_iterator eit = infovec.end();
555  for (; it != eit; ++it)
556  {
557  const FlattenedFieldInfo& fieldLS = *it;
558  if (ls.intersects(LocationSet(fieldLS)))
559  {
560  hasIntersection = true;
561  if (fieldLS.getFlattenElemTy()->isPointerTy())
562  return false;
563  }
564  }
565  assert(hasIntersection && "cannot find field of specified offset");
566  return true;
567  }
568  else
569  {
570  if (isStaticObj() || isHeap())
571  {
572  // TODO: Objects which cannot find proper field for a certain offset including
573  // arguments in main(), static objects allocated before main and heap
574  // objects. Right now they're considered to have infinite fields and we
575  // treat each field as pointers conservatively.
576  // Try to model static and heap objects more accurately in the future.
577  return false;
578  }
579  else
580  {
581  // TODO: Using new memory model (locMM) may create objects with spurious offset
582  // as we simply return new offset by mod operation without checking its
583  // correctness in LocSymTableInfo::getModulusOffset(). So the following
584  // assertion may fail. Try to refine the new memory model.
585  //assert(ls.getFieldOffset() == 0 && "cannot get a field from a non-struct type");
586  return (hasPtrObj() == false);
587  }
588  }
589 }
