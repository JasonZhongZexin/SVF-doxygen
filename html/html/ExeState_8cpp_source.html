43 void ExeState::initValVar(const ValVar *valVar, Z3Expr &e)
44 {
45  std::string str;
46  raw_string_ostream rawstr(str);
47  SVFIR *svfir = PAG::getPAG();
48 
49  rawstr << "ValVar" << valVar->getId();
50  if (const Type *type = valVar->getType())
51  {
52  if (type->isIntegerTy() || type->isFloatingPointTy() || type->isPointerTy() || type->isFunctionTy()
53  || type->isStructTy() || type->isArrayTy() || type->isVoidTy() || type->isLabelTy() || type->isMetadataTy())
54  e = getContext().int_const(rawstr.str().c_str());
55  else
56  {
57  SVFUtil::errs() << value2String(valVar->getValue()) << "\n" << " type: " << type2String(type) << "\n";
58  assert(false && "what other types we have");
59  }
60  }
61  else
62  {
63  if (svfir->getNullPtr() == valVar->getId())
64  e = getContext().int_val(0);
65  else
66  e = getContext().int_const(rawstr.str().c_str());
67  assert(SVFUtil::isa<DummyValVar>(valVar) && "not a DummValVar if it has no type?");
68  }
69 }

76 void ExeState::initObjVar(const ObjVar *objVar, Z3Expr &e)
77 {
78  std::string str;
79  raw_string_ostream rawstr(str);
80  rawstr << "ObjVar" << objVar->getId();
81 
82  if (objVar->hasValue())
83  {
84  const MemObj *obj = objVar->getMemObj();
86  if (obj->isConstantData() || obj->isConstantArray() || obj->isConstantStruct())
87  {
88  if (const ConstantInt *consInt = SVFUtil::dyn_cast<ConstantInt>(obj->getValue()))
89  {
90  if (consInt == llvm::ConstantInt::getTrue(LLVMModuleSet::getLLVMModuleSet()->getContext()))
91  {
92  e = getContext().int_val(1);
93  }
94  else if (consInt == llvm::ConstantInt::getFalse(LLVMModuleSet::getLLVMModuleSet()->getContext()))
95  e = getContext().int_val(0);
96  else
97  {
98  e = getContext().int_val(consInt->getSExtValue());
99  }
100  }
101  else if (const llvm::ConstantFP *consFP = SVFUtil::dyn_cast<llvm::ConstantFP>(obj->getValue()))
102  e = getContext().int_val(static_cast<u32_t>(consFP->getValueAPF().convertToFloat()));
103  else if (SVFUtil::isa<ConstantPointerNull>(obj->getValue()))
104  e = getContext().int_val(0);
105  else if (SVFUtil::isa<GlobalVariable>(obj->getValue()))
106  e = getContext().int_val(getVirtualMemAddress(objVar->getId()));
107  else if (SVFUtil::isa<ConstantAggregate>(obj->getValue()))
108  assert(false && "implement this part");
109  else
110  {
111  std::cerr << value2String(obj->getValue()) << "\n";
112  assert(false && "what other types of values we have?");
113  }
114  }
116  else
117  {
118  e = getContext().int_val(getVirtualMemAddress(objVar->getId()));
119  }
120  }
121  else
122  {
123  assert(SVFUtil::isa<DummyObjVar>(objVar) &&
124  "it should either be a blackhole or constant dummy if this obj has no value?");
125  e = getContext().int_val(getVirtualMemAddress(objVar->getId()));
126  }
127 }

134 Z3Expr &ExeState::getZ3Expr(u32_t varId)
135 {
136  assert(getInternalID(varId) == varId && "SVFVar idx overflow > 0x7f000000?");
137  Z3Expr &e = varToVal[varId];
138  if (eq(e, Z3Expr::nullExpr()))
139  {
140  std::string str;
141  raw_string_ostream rawstr(str);
142  SVFIR *svfir = PAG::getPAG();
143  SVFVar *svfVar = svfir->getGNode(varId);
144  if (const ValVar *valVar = dyn_cast<ValVar>(svfVar))
145  {
146  initValVar(valVar, e);
147  }
148  else if (const ObjVar *objVar = dyn_cast<ObjVar>(svfVar))
149  {
150  initObjVar(objVar, e);
151  }
152  else
153  {
154  assert(false && "var type not supported");
155  }
156  }
157  return e;
158 }

187 bool ExeState::operator<(const ExeState &rhs) const
188 {
189  // judge from path constraint
190  if (!eq(pathConstraint, rhs.getPathConstraint()))
191  return pathConstraint.id() < rhs.getPathConstraint().id();
192  if (lessThanVarToValMap(varToVal, rhs.getVarToVal()) || lessThanVarToValMap(locToVal, rhs.getLocToVal()))
193  return true;
194  return false;

197 bool ExeState::eqVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs) const
198 {
199  if(lhs.size() != rhs.size()) return false;
200  for (const auto &item: lhs)
201  {
202  auto it = rhs.find(item.first);
203  // return false if SVFVar not exists in rhs or z3Expr not equal
204  if (it == rhs.end() || !eq(item.second, it->second))
205  return false;
206  }
207  return true;
208 }

210 bool ExeState::lessThanVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs) const
211 {
212  if(lhs.size() != rhs.size()) return lhs.size() < rhs.size();
213  for (const auto &item: lhs)
214  {
215  auto it = rhs.find(item.first);
216  // lhs > rhs if SVFVar not exists in rhs
217  if (it == rhs.end())
218  return false;
219  // judge from expr id
220  if (!eq(item.second, it->second))
221  return item.second.id() < it->second.id();
222  }
223  return false;
224 }
