142  if (lhs == getFalseCond() || rhs == getFalseCond())

144  else if (lhs == getTrueCond())

208 void BddCondManager::ddClearFlag(DdNode * f) const
209 {
210  if (!Cudd_IsComplement(f->next))
211  return;
212  /* Clear visited flag. */
213  f->next = Cudd_Regular(f->next);
214  if (cuddIsConstant(f))
215  return;
216  ddClearFlag(cuddT(f));
217  ddClearFlag(Cudd_Regular(cuddE(f)));
218  return;
219 }

221 void BddCondManager::BddSupportStep(DdNode * f, NodeBS &support) const
222 {
223  if (cuddIsConstant(f) || Cudd_IsComplement(f->next))
224  return;
225 
226  support.set(f->index);
227 
228  BddSupportStep(cuddT(f), support);
229  BddSupportStep(Cudd_Regular(cuddE(f)), support);
230  /* Mark as visited. */
231  f->next = Cudd_Complement(f->next);
232 }
