78 void TypeAnalysis::callGraphSolveBasedOnCHA(const CallSiteToFunPtrMap& callsites, CallEdgeMap& newEdges)
79 {
80  for(CallSiteToFunPtrMap::const_iterator iter = callsites.begin(), eiter = callsites.end(); iter!=eiter; ++iter)
81  {
82  const CallBlockNode* cbn = iter->first;
83  CallSite cs = SVFUtil::getLLVMCallSite(cbn->getCallSite());
84  if (isVirtualCallSite(cs))
85  {
86  virtualCallSites.insert(cs);
87  const Value *vtbl = getVCallVtblPtr(cs);
88  assert(pag->hasValueNode(vtbl));
89  VFunSet vfns;
90  getVFnsFromCHA(cbn, vfns);
91  connectVCallToVFns(cbn, vfns, newEdges);
92  }

97 void TypeAnalysis::dumpCHAStats()
98 {
99 
100  const CHGraph *chgraph = SVFUtil::dyn_cast<CHGraph>(getCHGraph());
101  if (chgraph == nullptr)
102  {
103  SVFUtil::errs() << "dumpCHAStats only implemented for standard CHGraph.\n";
104  return;
105  }
106 
107  s32_t pure_abstract_class_num = 0,
108  multi_inheritance_class_num = 0;
109  for (CHGraph::const_iterator it = chgraph->begin(), eit = chgraph->end();
110  it != eit; ++it)
111  {
112  CHNode *node = it->second;
113  outs() << "class " << node->getName() << "\n";
114  if (node->isPureAbstract())
115  pure_abstract_class_num++;
116  if (node->isMultiInheritance())
117  multi_inheritance_class_num++;
118  }
119  outs() << "class_num:\t" << chgraph->getTotalNodeNum() << '\n';
120  outs() << "pure_abstract_class_num:\t" << pure_abstract_class_num << '\n';
121  outs() << "multi_inheritance_class_num:\t" << multi_inheritance_class_num << '\n';
122 
123  /*
124  * count the following info:
125  * vtblnum
126  * total vfunction
127  * vtbl max vfunction
128  * pure abstract class
129  */
130  s32_t vtblnum = 0,
131  vfunc_total = 0,
132  vtbl_max = 0,
133  pure_abstract = 0;
134  set<const SVFFunction*> allVirtualFunctions;
135  for (CHGraph::const_iterator it = chgraph->begin(), eit = chgraph->end();
136  it != eit; ++it)
137  {
138  CHNode *node = it->second;
139  if (node->isPureAbstract())
140  pure_abstract++;
141 
142  s32_t vfuncs_size = 0;
143  const vector<CHNode::FuncVector>& vecs = node->getVirtualFunctionVectors();
144  for (vector<CHNode::FuncVector>::const_iterator vit = vecs.begin(),
145  veit = vecs.end(); vit != veit; ++vit)
146  {
147  vfuncs_size += (*vit).size();
148  for (vector<const SVFFunction*>::const_iterator fit = (*vit).begin(),
149  feit = (*vit).end(); fit != feit; ++fit)
150  {
151  const SVFFunction* func = *fit;
152  allVirtualFunctions.insert(func);
153  }
154  }
155  if (vfuncs_size > 0)
156  {
157  vtblnum++;
158  if (vfuncs_size > vtbl_max)
159  {
160  vtbl_max = vfuncs_size;
161  }
162  }
163  }
164  vfunc_total = allVirtualFunctions.size();
165 
166  outs() << "vtblnum:\t" << vtblnum << '\n';
167  outs() << "vtbl_average:\t" << (double)(vfunc_total)/vtblnum << '\n';
168  outs() << "vtbl_max:\t" << vtbl_max << '\n';
169 }
