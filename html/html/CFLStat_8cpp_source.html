68 void CFLStat::collectCycleInfo(ConstraintGraph* consCG)
69 {
70  _NumOfCycles = 0;
71  _NumOfPWCCycles = 0;
72  _NumOfNodesInCycles = 0;
73  NodeSet repNodes;
74  repNodes.clear();
75  for(ConstraintGraph::iterator it = consCG->begin(), eit = consCG->end(); it!=eit; ++it)
76  {
77  // sub nodes have been removed from the constraint graph, only rep nodes are left.
78  NodeID repNode = consCG->sccRepNode(it->first);
79  NodeBS& subNodes = consCG->sccSubNodes(repNode);
80  NodeBS clone = subNodes;
81  for (NodeBS::iterator it = subNodes.begin(), eit = subNodes.end(); it != eit; ++it)
82  {
83  NodeID nodeId = *it;
84  PAGNode* pagNode = pta->getPAG()->getGNode(nodeId);
85  if (SVFUtil::isa<ObjVar>(pagNode) && pta->isFieldInsensitive(nodeId))
86  {
87  NodeID baseId = consCG->getBaseObjVar(nodeId);
88  clone.reset(nodeId);
89  clone.set(baseId);
90  }
91  }
92  u32_t num = clone.count();
93  if (num > 1)
94  {
95  if(repNodes.insert(repNode).second)
96  {
97  _NumOfNodesInCycles += num;
98  if(consCG->isPWCNode(repNode))
99  _NumOfPWCCycles ++;
100  }
101  if( num > _MaxNumOfNodesInSCC)
102  _MaxNumOfNodesInSCC = num;
103  }
104  }
105  _NumOfCycles += repNodes.size();
106 }

162 void CFLStat::statNullPtr()
163 {
164 
165  _NumOfNullPtr = 0;
166  for (SVFIR::iterator iter = pta->getPAG()->begin(), eiter = pta->getPAG()->end();
167  iter != eiter; ++iter)
168  {
169  NodeID pagNodeId = iter->first;
170  PAGNode* pagNode = iter->second;
171  if (SVFUtil::isa<ValVar>(pagNode) == false)
172  continue;
173  SVFStmt::SVFStmtSetTy& inComingStore = pagNode->getIncomingEdges(SVFStmt::Store);
174  SVFStmt::SVFStmtSetTy& outGoingLoad = pagNode->getOutgoingEdges(SVFStmt::Load);
175  if (inComingStore.empty()==false || outGoingLoad.empty()==false)
176  {
178  const PointsTo& pts = pta->getPts(pagNodeId);
179  if (pta->containBlackHoleNode(pts))
180  _NumOfBlackholePtr++;
181 
182  if (pta->containConstantNode(pts))
183  _NumOfConstantPtr++;
184 
185  if(pts.empty())
186  {
187  std::string str;
188  raw_string_ostream rawstr(str);
189  if (!SVFUtil::isa<DummyValVar>(pagNode) && !SVFUtil::isa<DummyObjVar>(pagNode) )
190  {
191  // if a pointer is in dead function, we do not care
192  if(SymbolTableInfo::isPtrInUncalledFunction(pagNode->getValue()) == false)
193  {
194  _NumOfNullPtr++;
195  rawstr << "##Null Pointer : (NodeID " << pagNode->getId()
196  << ") PtrName:" << pagNode->getValue()->getName();
197  writeWrnMsg(rawstr.str());
198  //pagNode->getValue()->dump();
199  }
200  }
201  else
202  {
203  _NumOfNullPtr++;
204  rawstr << "##Null Pointer : (NodeID " << pagNode->getId() << ")";
205  writeWrnMsg(rawstr.str());
206  }
207  }
208  }
209  }

226  PTAStat::printStat("CFL-reachability analysis Stats");
227 
228  u32_t totalPointers = 0;
229  u32_t totalTopLevPointers = 0;
230  u32_t totalPtsSize = 0;
231  u32_t totalTopLevPtsSize = 0;
232  for (SVFIR::iterator iter = pta->getPAG()->begin(), eiter = pta->getPAG()->end();
233  iter != eiter; ++iter)
234  {
235  NodeID node = iter->first;
236  const PointsTo& pts = pta->getPts(node);
237  u32_t size = pts.count();
238  totalPointers++;
239  totalPtsSize+=size;
240 
241  if(pta->getPAG()->isValidTopLevelPtr(pta->getPAG()->getGNode(node)))
242  {
243  totalTopLevPointers++;
244  totalTopLevPtsSize+=size;
245  }
246 
247  if(size > _MaxPtsSize )
248  _MaxPtsSize = size;
249  }
250 
251 
252  PTAStat::performStat();
253 
254  constraintGraphStat();
255 
256  PTNumStatMap["PointsToConstPtr"] = _NumOfConstantPtr;
257  PTNumStatMap["PointsToBlkPtr"] = _NumOfBlackholePtr;
258 
259  PTAStat::printStat("CFL Alias Analysis Stats");
260 }
