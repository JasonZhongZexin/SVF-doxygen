82 SVFIR2ItvExeState::VAddrs SVFIR2ItvExeState::getGepObjAddress(u32_t pointer, u32_t offset)
83 {
84  assert(!getVAddrs(pointer).empty());

86  VAddrs ret;
87  for (const auto &addr: addrs)
88  {
89  int64_t baseObj = getInternalID(addr);

91  {
92  ret.insert(getVirtualMemAddress(0));

97  initSVFVar(gepObj);
98  ret.insert(getVirtualMemAddress(gepObj));
99  }

104 std::pair<s32_t, s32_t> SVFIR2ItvExeState::getGepOffset(const GepStmt *gep)
105 {
106  if (gep->getOffsetValueVec().empty())
107  return std::make_pair(gep->getConstantFieldIdx(), gep->getConstantFieldIdx());
108 
109  s32_t totalOffsetLb = 0;
110  s32_t totalOffsetUb = 0;
111  for (int i = gep->getOffsetValueVec().size() - 1; i >= 0; i--)
112  {
113  const SVFValue *value = gep->getOffsetValueVec()[i].first;
114  const SVFType *type = gep->getOffsetValueVec()[i].second;
115  const SVFConstantInt *op = SVFUtil::dyn_cast<SVFConstantInt>(value);
116  s32_t offsetLb = 0;
117  s32_t offsetUb = 0;
119  if (op)
120  {
121  offsetLb = offsetUb = op->getSExtValue();
122  }
124  else
125  {
126  u32_t idx = _svfir->getValueNode(value);
127  if (!inVarToIValTable(idx)) return std::make_pair(-1, -1);
128  IntervalValue &idxVal = _es[idx];
129  if(idxVal.isBottom() || idxVal.isTop()) return std::make_pair(-1, -1);
130  if (idxVal.is_numeral())
131  {
132  offsetLb = offsetUb = idxVal.lb().getNumeral();
133  }
134  else
135  {
136  offsetLb = idxVal.lb().getNumeral() < 0 ? 0 : idxVal.lb().getNumeral();
137  offsetUb = idxVal.ub().getNumeral() < 0 ? 0 : idxVal.ub().getNumeral();
138  }
139  }
140  if (type == nullptr)
141  {
142  totalOffsetLb += offsetLb;
143  totalOffsetUb += offsetUb;
144  continue;
145  }
147  if (const SVFPointerType *pty = SVFUtil::dyn_cast<SVFPointerType>(type))
148  {
149  totalOffsetLb += offsetLb * gep->getLocationSet().getElementNum(pty->getPtrElementType());
150  totalOffsetUb += offsetUb * gep->getLocationSet().getElementNum(pty->getPtrElementType());
151  }
152  else
153  {
154  const std::vector<u32_t> &so = SymbolTableInfo::SymbolInfo()->getTypeInfo(type)->getFlattenedElemIdxVec();
155  if(so.empty() || (u32_t)offsetUb >= so.size() || (u32_t)offsetLb >= so.size()) return std::make_pair(-1, -1);
156  totalOffsetLb += SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(type, offsetLb);
157  totalOffsetUb += SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(type, offsetUb);
158  }
159  }
160  std::pair<s32_t, s32_t> offSetPair;
161  offSetPair.first = totalOffsetLb;
162  offSetPair.second = totalOffsetUb;
163  return offSetPair;
164 }

205 void SVFIR2ItvExeState::initObjVar(const ObjVar *objVar, u32_t varId)
206 {
207 
208  if (objVar->hasValue())
209  {
210  const MemObj *obj = objVar->getMemObj();
212  if (obj->isConstDataOrConstGlobal() || obj->isConstantArray() || obj->isConstantStruct())
213  {
214  if (const SVFConstantInt *consInt = SVFUtil::dyn_cast<SVFConstantInt>(obj->getValue()))
215  {
216  if (LLVMModuleSet::getLLVMModuleSet()->getLLVMValue(consInt) ==
217  llvm::ConstantInt::getTrue(LLVMModuleSet::getLLVMModuleSet()->getContext()))
218  {
219  IntervalExeState::globalES[varId] = IntervalValue(1, 1);
220  }
221  else if (LLVMModuleSet::getLLVMModuleSet()->getLLVMValue(consInt) ==
222  llvm::ConstantInt::getFalse(LLVMModuleSet::getLLVMModuleSet()->getContext()))
223  IntervalExeState::globalES[varId] = IntervalValue(0, 0);
224  else
225  {
226  double numeral = (double)consInt->getSExtValue();
227  IntervalExeState::globalES[varId] = IntervalValue(numeral, numeral);
228  }
229  }
230  else if (const SVFConstantFP* consFP = SVFUtil::dyn_cast<SVFConstantFP>(obj->getValue()))
231  {
232  IntervalExeState::globalES[varId] = IntervalValue(consFP->getFPValue(), consFP->getFPValue());
233  }
234  else if (SVFUtil::isa<SVFConstantNullPtr>(obj->getValue()))
235  {
236  IntervalExeState::globalES[varId] = IntervalValue(0, 0);
237  }
238  else if (SVFUtil::isa<SVFGlobalValue>(obj->getValue()))
239  {
240  IntervalExeState::globalES.getVAddrs(varId).insert(getVirtualMemAddress(varId));
241  }
242  else if (obj->isConstantArray() || obj->isConstantStruct())
243  {
244  IntervalExeState::globalES[varId] = IntervalValue::top();
245  }
246  else
247  {
248  IntervalExeState::globalES[varId] = IntervalValue::top();
249  }
250  }
251  else
252  {
253  IntervalExeState::globalES.getVAddrs(varId).insert(getVirtualMemAddress(varId));
254  }
255  }
256  else
257  {
258  IntervalExeState::globalES.getVAddrs(varId).insert(getVirtualMemAddress(varId));
259  }

305 void SVFIR2ItvExeState::translateBinary(const BinaryOPStmt *binary)
306 {
307  u32_t op0 = binary->getOpVarID(0);
308  u32_t op1 = binary->getOpVarID(1);
309  u32_t res = binary->getResID();
310  if (!inVarToIValTable(op0)) _es[op0] = IntervalValue::top();
311  if (!inVarToIValTable(op1)) _es[op1] = IntervalValue::top();
312  if (inVarToIValTable(op0) && inVarToIValTable(op1))
313  {
314  IntervalValue &lhs = _es[op0], &rhs = _es[op1];
315  IntervalValue resVal;
316  switch (binary->getOpcode())
317  {
318  case BinaryOperator::Add:
319  case BinaryOperator::FAdd:
320  resVal = (lhs + rhs);
321  break;
322  case BinaryOperator::Sub:
323  case BinaryOperator::FSub:
324  resVal = (lhs - rhs);
325  break;
326  case BinaryOperator::Mul:
327  case BinaryOperator::FMul:
328  resVal = (lhs * rhs);
329  break;
330  case BinaryOperator::SDiv:
331  case BinaryOperator::FDiv:
332  case BinaryOperator::UDiv:
333  resVal = (lhs / rhs);
334  break;
335  case BinaryOperator::SRem:
336  case BinaryOperator::FRem:
337  case BinaryOperator::URem:
338  resVal = (lhs % rhs);
339  break;
340  case BinaryOperator::Xor:
341  resVal = (lhs ^ rhs);
342  break;
343  case BinaryOperator::And:
344  resVal = (lhs & rhs);
345  break;
346  case BinaryOperator::Or:
347  resVal = (lhs | rhs);
348  break;
349  case BinaryOperator::AShr:
350  resVal = (lhs >> rhs);
351  break;
352  case BinaryOperator::Shl:
353  resVal = (lhs << rhs);
354  break;
355  case BinaryOperator::LShr:
356  resVal = (lhs >> rhs);
357  break;
358  default:
359  {
360  assert(false && "undefined binary: ");
361  }
362  }
363  _es[res] = resVal;
364  }

367 void SVFIR2ItvExeState::translateCmp(const CmpStmt *cmp)
368 {
369  u32_t op0 = cmp->getOpVarID(0);
370  u32_t op1 = cmp->getOpVarID(1);
371  u32_t res = cmp->getResID();
372  if (inVarToIValTable(op0) && inVarToIValTable(op1))
373  {
374  IntervalValue resVal;
375  IntervalValue &lhs = _es[op0], &rhs = _es[op1];
376  auto predicate = cmp->getPredicate();
377  switch (predicate)
378  {
379  case CmpInst::ICMP_EQ:
380  case CmpInst::FCMP_OEQ:
381  case CmpInst::FCMP_UEQ:
382  resVal = (lhs == rhs);
383  break;
384  case CmpInst::ICMP_NE:
385  case CmpInst::FCMP_ONE:
386  case CmpInst::FCMP_UNE:
387  resVal = (lhs != rhs);
388  break;
389  case CmpInst::ICMP_UGT:
390  case CmpInst::ICMP_SGT:
391  case CmpInst::FCMP_OGT:
392  case CmpInst::FCMP_UGT:
393  resVal = (lhs > rhs);
394  break;
395  case CmpInst::ICMP_UGE:
396  case CmpInst::ICMP_SGE:
397  case CmpInst::FCMP_OGE:
398  case CmpInst::FCMP_UGE:
399  resVal = (lhs >= rhs);
400  break;
401  case CmpInst::ICMP_ULT:
402  case CmpInst::ICMP_SLT:
403  case CmpInst::FCMP_OLT:
404  case CmpInst::FCMP_ULT:
405  resVal = (lhs < rhs);
406  break;
407  case CmpInst::ICMP_ULE:
408  case CmpInst::ICMP_SLE:
409  case CmpInst::FCMP_OLE:
410  case CmpInst::FCMP_ULE:
411  resVal = (lhs <= rhs);
412  break;
413  case CmpInst::FCMP_FALSE:
414  resVal = IntervalValue(0, 0);
415  break;
416  case CmpInst::FCMP_TRUE:
417  resVal = IntervalValue(1, 1);
418  break;
419  default:
420  {
421  assert(false && "undefined compare: ");
422  }
423  }
424  _es[res] = resVal;
425  }
426  else if (inVarToAddrsTable(op0) && inVarToAddrsTable(op1))
427  {
428  IntervalValue resVal;
429  VAddrs &lhs = getVAddrs(op0), &rhs = getVAddrs(op1);
430 // if (lhs.empty() || rhs.empty()) {
431 // outs() << "empty address cmp?\n";
432 // return;
433 // }
434  assert(!lhs.empty() && !rhs.empty() && "empty address?");
435  auto predicate = cmp->getPredicate();
436  switch (predicate)
437  {
438  case CmpInst::ICMP_EQ:
439  case CmpInst::FCMP_OEQ:
440  case CmpInst::FCMP_UEQ:
441  {
442  if (lhs.size() == 1 && rhs.size() == 1)
443  {
444  resVal = IntervalValue(lhs.equals(rhs));
445  }
446  else
447  {
448  if (lhs.hasIntersect(rhs))
449  {
450  resVal = IntervalValue::top();
451  }
452  else
453  {
454  resVal = IntervalValue(0);
455  }
456  }
457  break;
458  }
459  case CmpInst::ICMP_NE:
460  case CmpInst::FCMP_ONE:
461  case CmpInst::FCMP_UNE:
462  {
463  if (lhs.size() == 1 && rhs.size() == 1)
464  {
465  resVal = IntervalValue(!lhs.equals(rhs));
466  }
467  else
468  {
469  if (lhs.hasIntersect(rhs))
470  {
471  resVal = IntervalValue::top();
472  }
473  else
474  {
475  resVal = IntervalValue(1);
476  }
477  }
478  break;
479  }
480  case CmpInst::ICMP_UGT:
481  case CmpInst::ICMP_SGT:
482  case CmpInst::FCMP_OGT:
483  case CmpInst::FCMP_UGT:
484  {
485  if (lhs.size() == 1 && rhs.size() == 1)
486  {
487  resVal = IntervalValue(*lhs.begin() > *rhs.begin());
488  }
489  else
490  {
491  resVal = IntervalValue::top();
492  }
493  break;
494  }
495  case CmpInst::ICMP_UGE:
496  case CmpInst::ICMP_SGE:
497  case CmpInst::FCMP_OGE:
498  case CmpInst::FCMP_UGE:
499  {
500  if (lhs.size() == 1 && rhs.size() == 1)
501  {
502  resVal = IntervalValue(*lhs.begin() >= *rhs.begin());
503  }
504  else
505  {
506  resVal = IntervalValue::top();
507  }
508  break;
509  }
510  case CmpInst::ICMP_ULT:
511  case CmpInst::ICMP_SLT:
512  case CmpInst::FCMP_OLT:
513  case CmpInst::FCMP_ULT:
514  {
515  if (lhs.size() == 1 && rhs.size() == 1)
516  {
517  resVal = IntervalValue(*lhs.begin() < *rhs.begin());
518  }
519  else
520  {
521  resVal = IntervalValue::top();
522  }
523  break;
524  }
525  case CmpInst::ICMP_ULE:
526  case CmpInst::ICMP_SLE:
527  case CmpInst::FCMP_OLE:
528  case CmpInst::FCMP_ULE:
529  {
530  if (lhs.size() == 1 && rhs.size() == 1)
531  {
532  resVal = IntervalValue(*lhs.begin() <= *rhs.begin());
533  }
534  else
535  {
536  resVal = IntervalValue::top();
537  }
538  break;
539  }
540  case CmpInst::FCMP_FALSE:
541  resVal = IntervalValue(0, 0);
542  break;
543  case CmpInst::FCMP_TRUE:
544  resVal = IntervalValue(1, 1);
545  break;
546  default:
547  {
548  assert(false && "undefined compare: ");
549  }
550  }
551  _es[res] = resVal;
552  }
553 }

555 void SVFIR2ItvExeState::translateLoad(const LoadStmt *load)
556 {
557  u32_t rhs = load->getRHSVarID();
558  u32_t lhs = load->getLHSVarID();

560  {
561  VAddrs &addrs = getVAddrs(rhs);
562  assert(!getVAddrs(rhs).empty());
563  for (const auto &addr: addrs)

575  }

578  {
579  if (!inVarToAddrsTable(lhs))
580  {
581  getVAddrs(lhs) = _es.loadVAddrs(addr);
582  }

643 void SVFIR2ItvExeState::translateGep(const GepStmt *gep)
644 {
645  u32_t rhs = gep->getRHSVarID();
646  u32_t lhs = gep->getLHSVarID();
647  if (!inVarToAddrsTable(rhs)) return;
648  assert(!getVAddrs(rhs).empty());
649  VAddrs &rhsVal = getVAddrs(rhs);
650  if (rhsVal.empty()) return;
651  std::pair<s32_t, s32_t> offsetPair = getGepOffset(gep);
652  if(offsetPair.first == -1 && offsetPair.second == -1) return;
653  if (!isVirtualMemAddress(*rhsVal.begin()))
654  {
655  return;
656  }
657  else
658  {
659  VAddrs gepAddrs;
660  for (s32_t i = offsetPair.first; i <= offsetPair.second; i++)
661  {
662  gepAddrs.join_with(getGepObjAddress(rhs, i));
663  }
664  getVAddrs(lhs) = gepAddrs;
665  return;
666  }

669 void SVFIR2ItvExeState::translateSelect(const SelectStmt *select)
670 {
671  u32_t res = select->getResID();
672  u32_t tval = select->getTrueValue()->getId();
