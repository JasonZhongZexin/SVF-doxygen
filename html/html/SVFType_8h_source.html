66 template <class S, class T> struct Hash<std::pair<S, T>>
67 {
68  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
69  static size_t szudzik(size_t a, size_t b)
70  {
71  return a > b ? b * b + a : a * a + a + b;
72  }
73 
74  size_t operator()(const std::pair<S, T>& t) const
75  {
76  Hash<decltype(t.first)> first;
77  Hash<decltype(t.second)> second;
78  return szudzik(first(t.first), second(t.second));
79  }
80 };
