123 void MTASVFGBuilder::performRemovingMHPEdges()
124 {
125  while (!recordedges.empty())
126  {
127  std::pair<NodeID, NodeID> edgepair = *recordedges.begin();
128  recordedges.erase(recordedges.begin());
129 
130  PointsTo remove_pts = edge2pts[edgepair];
131  const StmtSVFGNode* n1 = SVFUtil::cast<StmtSVFGNode>(svfg->getSVFGNode(edgepair.first));
132  const StmtSVFGNode* n2 = SVFUtil::cast<StmtSVFGNode>(svfg->getSVFGNode(edgepair.second));
133 
134  assert (n1&&n2 && "one node of removed pair is null");
135  assert (n1->hasOutgoingEdge() && "n1 doesn't have out edge");
136  assert (n2->hasIncomingEdge() && "n2 doesn't have in edge");
137 
138  Set<SVFGEdge*> removededges;
139  for (SVFGEdge::SVFGEdgeSetTy::iterator iter = n1->OutEdgeBegin(); iter != n1->OutEdgeEnd(); ++iter)
140  {
141  SVFGEdge* edge = *iter;
142  if (edge->isIndirectVFGEdge() && (edge->getDstNode()==n2))
143  {
144  IndirectSVFGEdge* e = SVFUtil::cast<IndirectSVFGEdge>(edge);
145  const NodeBS& pts = e->getPointsTo();
146  for (PointsTo::iterator o = remove_pts.begin(), eo = remove_pts.end(); o != eo; ++o)
147  {
148  if (const_cast<NodeBS&>(pts).test(*o))
149  {
150  const_cast<NodeBS&>(pts).reset(*o);
151  MTASVFGBuilder::numOfRemovedPTS ++;
152  }
153  }
154 
155  if (0 == e->getPointsTo().count())
156  {
157  removededges.insert(edge);
158  }
159  }
160  }
161 
162  while(!removededges.empty())
163  {
164  SVFGEdge* edge = *removededges.begin();
165  removededges.erase(removededges.begin());
166  svfg->removeSVFGEdge(edge);
167  DBOUT(DMTA,outs()<<"Read Precision remove: "<<edge->getSrcID()<<" -> "<<edge->getDstID()<<"\n");
168  MTASVFGBuilder::numOfRemovedSVFGEdges++;
169  }
170  }

321 MTASVFGBuilder::SVFGNodeIDSet MTASVFGBuilder::getPrevNodes(const StmtSVFGNode* n)
322 {
323  if (prevset.find(n)!=prevset.end())
324  return prevset[n];
325 
326  SVFGNodeIDSet prev;
327  SVFGNodeSet worklist;
328  SVFGNodeSet visited;
329 
330  for (SVFGEdge::SVFGEdgeSetTy::iterator iter = n->InEdgeBegin(); iter != n->InEdgeEnd(); ++iter)
331  {
332  SVFGEdge* edge = *iter;
333  if (edge->isIndirectVFGEdge())
334  {
335  worklist.insert(edge->getSrcNode());
336  }
337  }
338 
339  while (!worklist.empty())
340  {
341  const SVFGNode* node = *worklist.begin();
342  worklist.erase(worklist.begin());
343  visited.insert(node);
344  if (SVFUtil::isa<StoreSVFGNode>(node))
345  prev.set(node->getId());
346  else
347  {
348  for (SVFGEdge::SVFGEdgeSetTy::iterator iter = node->InEdgeBegin(); iter != node->InEdgeEnd(); ++iter)
349  {
350  SVFGEdge* edge = *iter;
351  if (edge->isIndirectVFGEdge() && visited.find(edge->getSrcNode())==visited.end())
352  worklist.insert(edge->getSrcNode());
353  }
354  }
355  }
356  prevset[n]=prev;
357  return prev;
358 }

426  {

429  {

635 void MTASVFGBuilder::readPrecision()
636 {
637 
638  recordedges.clear();
639  edge2pts.clear();
640 
641  for (SVFGNodeSet::iterator it1 = stnodeSet.begin(), eit1 = stnodeSet.end(); it1 != eit1; ++it1)
642  {
643  const StmtSVFGNode* n1 = SVFUtil::cast<StmtSVFGNode>(*it1);
644 
645  for (SVFGEdge::SVFGEdgeSetTy::iterator iter = n1->InEdgeBegin(); iter != n1->InEdgeEnd(); ++iter)
646  {
647  SVFGEdge* edge = *iter;
648  if (edge->isIndirectVFGEdge() && SVFUtil::isa<StoreSVFGNode>(edge->getSrcNode()))
649  {
650  const StmtSVFGNode* n2 = SVFUtil::cast<StmtSVFGNode>(edge->getSrcNode());
651 
652  IndirectSVFGEdge* e = SVFUtil::cast<IndirectSVFGEdge>(edge);
653  NodeBS pts = e->getPointsTo();
654  PointsTo remove_pts;
655 
656  for (NodeBS::iterator o = pts.begin(), eo = pts.end(); o != eo; ++o)
657  {
658  SVFGNodeIDSet succ1 = getSuccNodes(n1, *o);
659  SVFGNodeIDSet succ2 = getSuccNodes(n2, *o);
660 
661  bool remove = true;
662  for (SVFGNodeIDSet::iterator sn1 = succ1.begin(), esn1 = succ1.end(); sn1 != esn1; sn1++)
663  {
664  if (!succ2.test(*sn1))
665  {
666  remove = false;
667  break;
668  }
669  }
670  if (remove)
671  remove_pts.set(*o);
672  }
673 
674  if (remove_pts.count())
675  recordRemovingEdge(n2->getId(), n1->getId(), remove_pts);
676  }
677  }
678  }
679 
680  performRemovingMHPEdges();
681 }
