103  virtual inline void solve()
104  {
105  initWorklist();
106  do
107  {
108  numOfIteration++;
109  if (0 == numOfIteration % iterationForPrintStat)
110  printStat();
111 
112  reanalyze = false;
113 
114  solveWorklist();
115 
116  if (updateCallGraph())
117  reanalyze = true;
118 
119  }
120  while (reanalyze);
121  }

123  virtual inline void initWorklist()
124  {
125  NodeStack& nodeStack = SCCDetect();
126  while (!nodeStack.empty())
127  {
128  NodeID nodeId = nodeStack.top();
129  nodeStack.pop();
130  pushIntoWorklist(nodeId);
131  }
132  }
133 
134  virtual inline void solveWorklist()
135  {
136  while (!isWorklistEmpty())
137  {
138  NodeID nodeId = popFromWorklist();
139  collapsePWCNode(nodeId);
140  // Keep solving until workList is empty.
141  processNode(nodeId);
142  collapseFields();
143  }
144  }

161  virtual void propagate(GNODE* v)
162  {
163  child_iterator EI = GTraits::direct_child_begin(*v);
164  child_iterator EE = GTraits::direct_child_end(*v);
165  for (; EI != EE; ++EI)
166  {
167  if (propFromSrcToDst(*(EI.getCurrent())))
168  pushIntoWorklist(Node_Index(*EI));
169  }
