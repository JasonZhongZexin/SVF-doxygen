54 template <class S, class T> struct Hash<std::pair<S, T>> {
55  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
56  static size_t szudzik(size_t a, size_t b)
57  {
58  return a > b ? b * b + a : a * a + a + b;
59  }
60 
61  size_t operator()(const std::pair<S, T> &t) const {
62  Hash<decltype(t.first)> first;
63  Hash<decltype(t.second)> second;
64  return szudzik(first(t.first), second(t.second));
65  }
66 };

276 template <typename T, unsigned N>
277 struct std::hash<SVF::SmallVector<T, N>>
278 {
279  size_t operator()(const SVF::SmallVector<T, N> &sv) const {
280  if (sv.empty()) return 0;
281  if (sv.size() == 1) return sv[0];
282 
283  // Iterate and accumulate the hash.
284  size_t hash = 0;
285  SVF::Hash<std::pair<T, size_t>> hts;
286  std::hash<T> ht;
287  for (const T &t : sv)
288  {
289  hash = hts(std::make_pair(ht(t), hash));
290  }
291 
292  return hash;
293  }
294 };
