242  inline void RenameMuSet(const MUSet& muSet)
244  {
245  for (MUSet::const_iterator mit = muSet.begin(), emit = muSet.end();
246  mit != emit; ++mit)

254  inline void RenameChiSet(const CHISet& chiSet, MRVector& memRegs)
255  {
256  for (CHISet::const_iterator cit = chiSet.begin(), ecit = chiSet.end();
257  cit != ecit; ++cit)
258  {
259  CHI* chi = (*cit);
260  chi->setOpVer(getTopStackVer(chi->getMR()));
261  chi->setResVer(newSSAName(chi->getMR(),chi));
262  memRegs.push_back(chi->getMR());
263  }
264  }
