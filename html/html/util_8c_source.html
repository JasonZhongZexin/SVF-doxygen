79 /*
80  * These are interface routines to be placed between a program and the
81  * system memory allocator.
82  *
83  * It forces well-defined semantics for several 'borderline' cases:
84  *
85  * malloc() of a 0 size object is guaranteed to return something
86  * which is not 0, and can safely be freed (but not dereferenced)
