42 // Get environment variables $SVF_DIR and "npm root" through popen() method
43 static std::string GetStdoutFromCommand(const std::string &command)
44 {
45  char buffer[128];
46  std::string result = "";
47  // Open pipe to file
48  FILE *pipe = popen(command.c_str(), "r");
49  if (!pipe)
50  {
51  return "popen failed!";
52  }
53  // read till end of process:
54  while (!feof(pipe))
55  {
56  // use buffer to read and add to result
57  if (fgets(buffer, 128, pipe) != NULL)
58  result += buffer;
59  }
60  pclose(pipe);
61  // remove "\n"
62  result.erase(remove(result.begin(), result.end(), '\n'), result.end());
63  return result;
64 }

90 static cJSON *parseJson(const std::string &path, off_t fileSize)
91 {
92  FILE *file = fopen(path.c_str(), "r");
93  if (!file)
94  {
95  return nullptr;
96  }
97 
98  // allocate memory size matched with file size
99  char *jsonStr = (char *)calloc(fileSize + 1, sizeof(char));
100 
101  // read json string from file
102  u32_t size = fread(jsonStr, sizeof(char), fileSize, file);
103  if (size == 0)
104  {
105  SVFUtil::errs() << SVFUtil::errMsg("\t Wrong ExtAPI.json path!! ") << "The current ExtAPI.json path is: " << path << "\n";
106  assert(false && "Read ExtAPI.json file fails!");
107  return nullptr;
108  }
109  fclose(file);
110 
111  // convert json string to json pointer variable
112  cJSON *root = cJSON_Parse(jsonStr);
113  if (!root)
114  {
115  free(jsonStr);
116  return nullptr;
117  }
118  free(jsonStr);
119  return root;
120 }

344 std::vector<ExtAPI::Operation> ExtAPI::getAllOperations(std::string funName)
345 {
346  std::vector<ExtAPI::Operation> allOperations;
347  cJSON *item = get_FunJson(funName);
348  if (item != nullptr)
349  {
350  cJSON *obj = item->child;
351  // Get the first operation of the function
352  obj = obj -> next -> next -> next -> next;
353  std::vector<ExtAPI::Operation *> operations;
354  while (obj)
355  {
356  std::string op;
357  std::vector<std::string> operandsStr;
358  std::map<std::string, NodeID> opMap;
359  if (obj->type == cJSON_Object || obj->type == cJSON_Array)
360  {
361  op = get_opName(obj -> string);
362  cJSON *value = obj->child;
363  std::vector<std::string> args;
364  while (value)
365  {
366  if (value->type == cJSON_String)
367  {
368  operandsStr.push_back(value->valuestring);

370  value = value->next;
371  }
372  }

518 // Does (F) have the same return type(pointer or nonpointer) and same number of arguments
519 bool ExtAPI::is_sameSignature(const SVFFunction *F)
520 {
521  cJSON *item = get_FunJson(F->getName());
522  // If return type is pointer
523  bool isPointer = false;
524  // The number of arguments
525  u32_t argNum = 0;
526  if (item != nullptr)
527  {
528  // Get the "return" attribute
529  cJSON *obj = item->child;
530  if (strlen(obj->valuestring) == 0) // e.g. "return": ""
531  assert(false && "'return' should not be empty!");
532  // If "return": "..." includes "*", the return type of extern function is a pointer
533  if (strstr(obj->valuestring, "*") != NULL)
534  isPointer = true;
535  // Get the "arguments" attribute
536  obj = obj -> next;
537  if (strlen(obj->valuestring) == 0) // e.g. "arguments": "",
538  assert(false && "'arguments' should not be empty!");
539  // If "arguments": "()", the number of arguments is 0, otherwise, number >= 1;
540  if (strcmp(obj->valuestring, "()") != 0)
541  {
542  argNum++;
543  for (u32_t i = 0; i < strlen(obj->valuestring); i++)
544  if (obj->valuestring[i] == ',') // Calculate the number of arguments based on the number of ","
545  argNum++;
546  }
547  }
548  if (F->arg_size() != argNum) // The number of arguments is different
549  return false;
550  // Is the return type the same?
551  return F->getReturnType()->isPointerTy() == isPointer;
552 }

554 // Should (F) be considered "external" (either not defined in the program
555 // or a user-defined version of a known alloc or no-op)?
556 bool ExtAPI::is_ext(const SVFFunction *F)
557 {
558  assert(F);
559  bool res;
560  if (F->isDeclaration() || F->isIntrinsic())
561  {
562  res = 1;
563  }

574  else if (overwrittenAppFunction == 1)
575  res = 1;
576  // overwrittenAppFunction = 0: Execute user-defined functions
577  // F is not considered as external function
578  else
579  res = 0;
580  }
581  else
582  res = 0;
583  }
