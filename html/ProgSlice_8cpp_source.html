44 bool ProgSlice::AllPathReachableSolve()
45 {
46  const SVFGNode* source = getSource();
47  VFWorkList worklist;
48  worklist.push(source);
50  setVFCond(source,getTrueCond());
51 
52  while(!worklist.empty())
53  {
54  const SVFGNode* node = worklist.pop();
55  setCurSVFGNode(node);
56  Condition* cond = getVFCond(node);
57  for(SVFGNode::const_iterator it = node->OutEdgeBegin(), eit = node->OutEdgeEnd(); it!=eit; ++it)
58  {
59  const SVFGEdge* edge = (*it);
60  const SVFGNode* succ = edge->getDstNode();
61  if(inBackwardSlice(succ))
62  {
63  Condition* vfCond = nullptr;
64  const BasicBlock* nodeBB = getSVFGNodeBB(node);
65  const BasicBlock* succBB = getSVFGNodeBB(succ);
67  clearCFCond();
68 
69  if(edge->isCallVFGEdge())
70  {
71  vfCond = ComputeInterCallVFGGuard(nodeBB,succBB, getCallSite(edge)->getParent());
72  }
73  else if(edge->isRetVFGEdge())
74  {
75  vfCond = ComputeInterRetVFGGuard(nodeBB,succBB, getRetSite(edge)->getParent());
76  }
77  else
78  vfCond = ComputeIntraVFGGuard(nodeBB,succBB);
79 
80  Condition* succPathCond = condAnd(cond, vfCond);
81  if(setVFCond(succ, condOr(getVFCond(succ), succPathCond) ))
82  worklist.push(succ);
83  }
84 
85  DBOUT(DSaber, outs() << " node (" << node->getId() <<
86  ") --> " << "succ (" << succ->getId() << ") condition: " << getVFCond(succ) << "\n");
87  }
88  }
89 
90  return isSatisfiableForAll();
91 }

96 bool ProgSlice::isSatisfiableForAll()
97 {
98 
99  Condition* guard = getFalseCond();
100  for(SVFGNodeSetIter it = sinksBegin(), eit = sinksEnd(); it!=eit; ++it)
101  {
102  guard = condOr(guard,getVFCond(*it));
103  }
104  setFinalCond(guard);
105 
106  return guard == getTrueCond();

112 bool ProgSlice::isSatisfiableForPairs()
113 {
114 
115  for(SVFGNodeSetIter it = sinksBegin(), eit = sinksEnd(); it!=eit; ++it)
116  {
117  for(SVFGNodeSetIter sit = it, esit = sinksEnd(); sit!=esit; ++sit)
118  {
119  if(*it == *sit)
120  continue;
121  Condition* guard = condAnd(getVFCond(*sit),getVFCond(*it));
122  if(guard != getFalseCond())
123  {
124  setFinalCond(guard);
125  return false;
126  }
127  }
128  }
129 
130  return true;
131 }

154 const Value* ProgSlice::getLLVMValue(const SVFGNode* node) const
155 {
156  if(const StmtSVFGNode* stmt = SVFUtil::dyn_cast<StmtSVFGNode>(node))
157  {
158  if(SVFUtil::isa<StoreSVFGNode>(stmt) == false)
159  {
160  if(stmt->getPAGDstNode()->hasValue())
161  return stmt->getPAGDstNode()->getValue();
162  }
163  }
164  else if(const PHISVFGNode* phi = SVFUtil::dyn_cast<PHISVFGNode>(node))
165  {
166  return phi->getRes()->getValue();
167  }
168  else if(const ActualParmSVFGNode* ap = SVFUtil::dyn_cast<ActualParmSVFGNode>(node))
169  {
170  return ap->getParam()->getValue();
171  }
172  else if(const FormalParmSVFGNode* fp = SVFUtil::dyn_cast<FormalParmSVFGNode>(node))
173  {
174  return fp->getParam()->getValue();
175  }
176  else if(const ActualRetSVFGNode* ar = SVFUtil::dyn_cast<ActualRetSVFGNode>(node))
177  {
178  return ar->getRev()->getValue();
179  }
180  else if(const FormalRetSVFGNode* fr = SVFUtil::dyn_cast<FormalRetSVFGNode>(node))
181  {
182  return fr->getRet()->getValue();
183  }
184 
185  return nullptr;
186 }
