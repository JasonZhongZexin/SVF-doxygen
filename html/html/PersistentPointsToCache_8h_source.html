94  PointsToID unionPts(PointsToID lhs, PointsToID rhs)
95  {
96  static const DataOp unionOp = [](const Data &lhs, const Data &rhs) { return lhs | rhs; };
97 
98  ++totalUnions;
99 
100  // Order operands so we don't perform x U y and y U x separately.
101  std::pair<PointsToID, PointsToID> operands = std::minmax(lhs, rhs);
102 
103  // Property cases.
104  // EMPTY_SET U x
105  if (operands.first == emptyPointsToId())
106  {
107  ++propertyUnions;
108  return operands.second;
109  }
110 
111  // x U x
112  if (operands.first == operands.second)
113  {
114  ++propertyUnions;
115  return operands.first;
116  }
117 
118  bool opPerformed = false;
119  PointsToID result = opPts(lhs, rhs, unionOp, unionCache, true, opPerformed);
120 
121  if (opPerformed)
122  {
123  ++uniqueUnions;
124 
125  // We can use lhs/rhs here rather than our ordered operands,
126  // because the operation was commutative.
127 
128  // if x U y = z, then x U z = z,
129  if (lhs != result)
130  {
131  unionCache[std::minmax(lhs, result)] = result;
132  ++propertyUnions;
133  ++totalUnions;
134  }
135 
136  // and y U z = z.
137  if (rhs != result)
138  {
139  unionCache[std::minmax(rhs, result)] = result;
140  ++propertyUnions;
141  ++totalUnions;
142  }
143  } else ++lookupUnions;

321  inline PointsToID opPts(PointsToID lhs, PointsToID rhs, const DataOp &dataOp, OpCache &opCache,
322  bool commutative, bool &opPerformed)
323  {
324  std::pair<PointsToID, PointsToID> operands;
325  // If we're commutative, we want to always perform the same operation: x op y.
326  // Performing x op y sometimes and y op x other times is a waste of time.
327  if (commutative) operands = std::minmax(lhs, rhs);
328  else operands = std::make_pair(lhs, rhs);
329 
330  // Check if we have performed this operation
331  OpCache::const_iterator foundResult = opCache.find(operands);
332  if (foundResult != opCache.end()) return foundResult->second;
333 
334  opPerformed = true;
335 
336  const Data &lhsPts = getActualPts(lhs);
337  const Data &rhsPts = getActualPts(rhs);
338 
339  Data result = dataOp(lhsPts, rhsPts);
340 
341  PointsToID resultId;
342  // Intern points-to set: check if result already exists.
343  typename PTSToIDMap::const_iterator foundId = ptsToId.find(result);
344  if (foundId != ptsToId.end()) resultId = foundId->second;
345  else
346  {
347  resultId = newPointsToId();
348  idToPts.push_back(new Data(result));
349  ptsToId[result] = resultId;
350  }
351 
352  // Cache the result, for hash-consing.
353  opCache[operands] = resultId;
354 
355  return resultId;
356  }
