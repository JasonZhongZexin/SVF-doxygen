62 void VersionedFlowSensitive::prelabel(void)
63 {
64  double start = stat->getClk(true);
65  for (SVFG::iterator it = svfg->begin(); it != svfg->end(); ++it)
66  {
67  NodeID l = it->first;
68  const SVFGNode *ln = it->second;
69 
70  if (const StoreSVFGNode *stn = SVFUtil::dyn_cast<StoreSVFGNode>(ln))
71  {
72  // l: *p = q.
73  // If p points to o (Andersen's), l yields a new version for o.
74  NodeID p = stn->getPAGDstNodeID();
75  ObjToMeldVersionMap &myl = meldYield[l];
76  for (NodeID o : ander->getPts(p))
77  {
78  myl[o] = newMeldVersion(o);
79  }
80 
81  vWorklist.push(l);
82 
83  if (ander->getPts(p).count() != 0) ++numPrelabeledNodes;
84  }
85  else if (delta(l))
86  {
87  // The outgoing edges are not only what will later be propagated. SVFGOPT may
88  // move around nodes such that there can be an MRSVFGNode with no incoming or
89  // outgoing edges which will be added at runtime. In essence, we can no
90  // longer rely on the outgoing edges of a delta node when SVFGOPT is enabled.
91  const MRSVFGNode *mr = SVFUtil::dyn_cast<MRSVFGNode>(ln);
92  if (mr != nullptr)
93  {
94  ObjToMeldVersionMap &mcl = meldConsume[l];
95  for (const NodeID o : mr->getPointsTo())
96  {
97  mcl[o] = newMeldVersion(o);
98  }
99 
100  // Push into worklist because its consume == its yield.
101  vWorklist.push(l);
102  if (mr->getPointsTo().count() != 0) ++numPrelabeledNodes;
103  }
104  }
105  }
106 
107  double end = stat->getClk(true);
108  prelabelingTime = (end - start) / TIMEINTERVAL;
109 }

111 void VersionedFlowSensitive::meldLabel(void) {
112  double start = stat->getClk(true);
113 
114  while (!vWorklist.empty()) {
115  NodeID l = vWorklist.pop();
116  const SVFGNode *ln = svfg->getSVFGNode(l);
117 
118  // Propagate l's y to lp's c for all l --o--> lp.
119  for (const SVFGEdge *e : ln->getOutEdges()) {
120  const IndirectSVFGEdge *ie = SVFUtil::dyn_cast<IndirectSVFGEdge>(e);
121  if (!ie) continue;
122 
123  NodeID lp = ie->getDstNode()->getId();
124  // Delta nodes had c set already and they are permanent.
125  if (delta(lp)) continue;
126 
127  bool lpIsStore = SVFUtil::isa<StoreSVFGNode>(svfg->getSVFGNode(lp));
128  // Consume and yield are the same at non-stores, so ignore any self-loop
129  // at a non-store.
130  if (l == lp && !lpIsStore) continue;
131 
132  // At stores yield != consume, otherwise they are the same (so just use meldConsume).
133  const ObjToMeldVersionMap &myl = SVFUtil::isa<StoreSVFGNode>(ln) ? meldYield[l] : meldConsume[l];
134  ObjToMeldVersionMap &mclp = meldConsume[lp];
135  bool yieldChanged = false;
136  for (NodeID o : ie->getPointsTo()) {
137  ObjToMeldVersionMap::const_iterator myloIt = myl.find(o);
138  if (myloIt == myl.end()) continue;
139 
140  // Yield == consume for non-stores, so when consume is updated, so is yield.
141  // For stores, yield was already set, and it's static.
142  yieldChanged = (meld(mclp[o], myloIt->second) && !lpIsStore) || yieldChanged;
143  }
144 
145  if (yieldChanged) vWorklist.push(lp);
146  }
147  }
148 
149  double end = stat->getClk(true);
150  meldLabelingTime = (end - start) / TIMEINTERVAL;
151 }

159 void VersionedFlowSensitive::mapMeldVersions(void)
160 {
161  double start = stat->getClk(true);
162 
163  // We want to uniquely map MeldVersions (SparseBitVectors) to a Version (unsigned integer).
164  // mvv keeps track, and curVersion is used to generate new Versions.
165  Map<MeldVersion, Version> mvv;
166  Version curVersion = 1;
167 
168  // meldConsume -> consume.
169  for (LocMeldVersionMap::value_type &lomv : meldConsume)
170  {
171  NodeID l = lomv.first;
172  bool lIsStore = SVFUtil::isa<StoreSVFGNode>(svfg->getSVFGNode(l));
173  for (ObjToMeldVersionMap::value_type &omv : lomv.second)
174  {
175  NodeID o = omv.first;
176  MeldVersion &mv = omv.second;
177 
178  Map<MeldVersion, Version>::const_iterator foundVersion = mvv.find(mv);
179  // If a mapping for foudnVersion exists, use it, otherwise create a new Version (++),
180  // keep track of it ([mv]), and use that.
181  Version v = foundVersion == mvv.end() ? mvv[mv] = ++curVersion : foundVersion->second;
182  setConsume(l, o, v);
183  // At non-stores, consume == yield.
184  // Unlike meldYield, we use yield for all yields, not just where consume != yield.
185  // This affords simplicity later. meldYield is expensive to explicitly represent
186  // always, unlike yield.
187  if (!lIsStore) setYield(l, o, v);
188  }
189  }
190 
191  // meldYield -> yield.
192  for (LocMeldVersionMap::value_type &lomv : meldYield)
193  {
194  NodeID l = lomv.first;
195  for (ObjToMeldVersionMap::value_type &omv : lomv.second)
196  {
197  NodeID o = omv.first;
198  MeldVersion &mv = omv.second;
199 
200  Map<MeldVersion, Version>::const_iterator foundVersion = mvv.find(mv);
201  Version v = foundVersion == mvv.end() ? mvv[mv] = ++curVersion : foundVersion->second;
202  setYield(l, o, v);
203  }
204  }
205 
206  // No longer necessary.
207  meldConsume.clear();
208  meldYield.clear();
209 
210  double end = stat->getClk(true);
211  meldMappingTime += (end - start) / TIMEINTERVAL;
212 }

214 bool VersionedFlowSensitive::delta(NodeID l) const
215 {
216  // Whether a node is a delta node or not. Decent boon to performance.
217  static Map<NodeID, bool> deltaCache;
218 
219  Map<NodeID, bool>::const_iterator isDeltaIt = deltaCache.find(l);
220  if (isDeltaIt != deltaCache.end()) return isDeltaIt->second;
221 
222  const SVFGNode *s = svfg->getSVFGNode(l);
223  // Cases:
224  // * Function entry: can get new incoming indirect edges through ind. callsites.
225  // * Callsite returns: can get new incoming indirect edges if the callsite is indirect.
226  // * Otherwise: static.
227  bool isDelta = false;
228  if (const SVFFunction *fn = svfg->isFunEntrySVFGNode(s))
229  {
230  PTACallGraphEdge::CallInstSet callsites;
232  ander->getPTACallGraph()->getIndCallSitesInvokingCallee(fn, callsites);
233  isDelta = !callsites.empty();
234  }
235  else if (const CallBlockNode *cbn = svfg->isCallSiteRetSVFGNode(s))
236  {
237  isDelta = cbn->isIndirectCall();
238  }
239 
240  deltaCache[l] = isDelta;
241  return isDelta;
242 }

253 void VersionedFlowSensitive::determineReliance(void)
254 {
255  // Use a set-based version to build, then we'll move things to vectors.
256  Map<NodeID, Map<Version, Set<Version>>> setVersionReliance;
257 
258  double start = stat->getClk(true);
259  for (SVFG::iterator it = svfg->begin(); it != svfg->end(); ++it)
260  {
261  NodeID l = it->first;
262  const SVFGNode *ln = it->second;
263  for (const SVFGEdge *e : ln->getOutEdges())
264  {
265  const IndirectSVFGEdge *ie = SVFUtil::dyn_cast<IndirectSVFGEdge>(e);
266  if (!ie) continue;
267 
268  for (const NodeID o : ie->getPointsTo())
269  {
270  // Given l --o--> lp, c(o) at lp relies on y(o) at l.
271  const NodeID lp = ie->getDstNode()->getId();
272 
273  const Version y = getYield(l, o);
274  if (y == invalidVersion) continue;
275  const Version cp = getConsume(lp, o);
276  if (cp == invalidVersion) continue;
277 
278  if (cp != y) setVersionReliance[o][y].insert(cp);
279  }
280  }
281 
282  // When an object/version points-to set changes, these nodes need to know.
283  if (SVFUtil::isa<LoadSVFGNode>(ln) || SVFUtil::isa<StoreSVFGNode>(ln))
284  {
285  const LocVersionMap::const_iterator lovmIt = consume.find(l);
286  if (lovmIt != consume.end())
287  {
288  for (const ObjToVersionMap::value_type &ov : lovmIt->second)
289  {
290  const NodeID o = ov.first;
291  const Version v = ov.second;
292  stmtReliance[o][v].set(l);
293  }
294  }
295  }
296  }
297 
298  for (const std::pair<NodeID, Map<Version, Set<Version>>> &ovvs : setVersionReliance)
299  {
300  const NodeID o = ovvs.first;
301  Map<Version, std::vector<Version>> &osRelying = versionReliance[o];
302  for (const std::pair<Version, Set<Version>> &vvs : ovvs.second)
303  {
304  const Version v = vvs.first;
305  const Set<Version> &vs = vvs.second;
306  osRelying[v] = std::vector<Version>(vs.begin(), vs.end());
307  }
308  }
309 
310  double end = stat->getClk(true);
311  relianceTime = (end - start) / TIMEINTERVAL;
312 }

331 void VersionedFlowSensitive::propagateVersion(const NodeID o, const Version v, const Version vp, bool time/*=true*/)
332 {
333  double start = time ? stat->getClk() : 0.0;
334 
335  const VersionedVar srcVar = atKey(o, v);
336  const VersionedVar dstVar = atKey(o, vp);
337  if (vPtD->unionPts(dstVar, srcVar))
338  {
339  // o:vp has changed.
340  // Add the dummy propagation node to tell the solver to propagate it later.
341  const DummyVersionPropSVFGNode *dvp = nullptr;
342  VarToPropNodeMap::const_iterator dvpIt = versionedVarToPropNode.find(dstVar);
343  if (dvpIt == versionedVarToPropNode.end())
344  {
345  dvp = svfg->addDummyVersionPropSVFGNode(o, vp);
346  versionedVarToPropNode[dstVar] = dvp;
347  } else dvp = dvpIt->second;
348 
349  assert(dvp != nullptr && "VFS::propagateVersion: propagation dummy node not found?");
350  pushIntoWorklist(dvp->getId());
351 
352  // Notify nodes which rely on o:vp that it changed.
353  for (NodeID s : stmtReliance[o][vp]) pushIntoWorklist(s);
354  }
355 
356  double end = time ? stat->getClk() : 0.0;
357  if (time) versionPropTime += (end - start) / TIMEINTERVAL;
358 }

375 void VersionedFlowSensitive::updateConnectedNodes(const SVFGEdgeSetTy& newEdges)
376 {
377  for (const SVFGEdge *e : newEdges)
378  {
379  SVFGNode *dstNode = e->getDstNode();
380  NodeID src = e->getSrcNode()->getId();
381  NodeID dst = dstNode->getId();
382 
383  assert(delta(dst) && "VFS::updateConnectedNodes: new edges should be to delta nodes!");
384 
385  if (SVFUtil::isa<PHISVFGNode>(dstNode)
386  || SVFUtil::isa<FormalParmSVFGNode>(dstNode)
387  || SVFUtil::isa<ActualRetSVFGNode>(dstNode))
388  {
389  pushIntoWorklist(dst);
390  }
391  else
392  {
393  const IndirectSVFGEdge *ie = SVFUtil::dyn_cast<IndirectSVFGEdge>(e);
394  assert(ie != nullptr && "VFS::updateConnectedNodes: given direct edge?");
395 
396  const NodeBS &ept = ie->getPointsTo();
397  // For every o, such that src --o--> dst, we need to set up reliance (and propagate).
398  for (const NodeID o : ept)
399  {
400  Version srcY = getYield(src, o);
401  if (srcY == invalidVersion) continue;
402  Version dstC = getConsume(dst, o);
403  if (dstC == invalidVersion) continue;
404 
405  std::vector<Version> &versionsRelyingOnSrcY = versionReliance[o][srcY];
406  if (std::find(versionsRelyingOnSrcY.begin(), versionsRelyingOnSrcY.end(), dstC) == versionsRelyingOnSrcY.end())
407  {
408  versionsRelyingOnSrcY.push_back(dstC);
409  propagateVersion(o, srcY, dstC);
410  }
411  }
412  }
413  }

459 bool VersionedFlowSensitive::processStore(const StoreSVFGNode* store)
460 {
461  NodeID p = store->getPAGDstNodeID();
462  const PointsTo &ppt = getPts(p);
463 
464  if (ppt.empty()) return false;
465 
466  NodeID q = store->getPAGSrcNodeID();
467  const PointsTo &qpt = getPts(q);
468 
469  NodeID l = store->getId();
470  // l: *p = q
471 
472  double start = stat->getClk();
473  bool changed = false;
474  // The version for these objects would be y_l(o).
475  NodeBS changedObjects;
476 
477  if (!qpt.empty())
478  {
479  for (NodeID o : ppt)
480  {
481  if (pag->isConstantObj(o) || pag->isNonPointerObj(o)) continue;
482 
483  const Version y = getYield(l, o);
484  if (y != invalidVersion && vPtD->unionPts(atKey(o, y), q))
485  {
486  changed = true;
487  changedObjects.set(o);
488  }
489  }
490  }
491 
492  double end = stat->getClk();
493  storeTime += (end - start) / TIMEINTERVAL;
494 
495  double updateStart = stat->getClk();
496 
497  NodeID singleton = 0;
498  bool isSU = isStrongUpdate(store, singleton);
499  if (isSU) svfgHasSU.set(l);
500  else svfgHasSU.reset(l);
501 
502  // For all objects, perform pts(o:y) = pts(o:y) U pts(o:c) at loc,
503  // except when a strong update is taking place.
504  const LocVersionMap::const_iterator lovmIt = consume.find(l);
505  if (lovmIt != consume.end())
506  {
507  for (const ObjToVersionMap::value_type &oc : lovmIt->second)
508  {
509  const NodeID o = oc.first;
510  const Version c = oc.second;
511 
512  // Strong-updated; don't propagate.
513  if (isSU && o == singleton) continue;
514 
515  const Version y = getYield(l, o);
516  if (y != invalidVersion && vPtD->unionPts(atKey(o, y), atKey(o, c)))
517  {
518  changed = true;
519  changedObjects.set(o);
520  }
521  }
522  }
523 
524  double updateEnd = stat->getClk();
525  updateTime += (updateEnd - updateStart) / TIMEINTERVAL;
526 
527  // Changed objects need to be propagated. Time here should be inconsequential
528  // *except* for time taken for propagateVersion, which will time itself.
529  if (!changedObjects.empty())
530  {
531  for (const NodeID o : changedObjects)
532  {
533  // Definitely has a yielded version (came from prelabelling) as these are
534  // the changed objects which must've been pointed to in Andersen's too.
535  const Version y = getYield(l, o);
536  propagateVersion(o, y);
537 
538  // Some o/v pairs changed: statements need to know.
539  for (NodeID s : stmtReliance[o][y]) pushIntoWorklist(s);
540  }
541  }
542 
543  return changed;
544 }

567 {
