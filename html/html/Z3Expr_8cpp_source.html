78 Z3Expr Z3Expr::AND(const Z3Expr &lhs, const Z3Expr &rhs)
79 {
80  if (eq(lhs, Z3Expr::getFalseCond()) || eq(rhs, Z3Expr::getFalseCond()))
81  {
82  return Z3Expr::getFalseCond();
83  }
84  else if (eq(lhs, Z3Expr::getTrueCond()))
85  {
86  return rhs;
87  }
88  else if (eq(rhs, Z3Expr::getTrueCond()))
89  {
90  return lhs;
91  }
92  else
93  {
94  Z3Expr expr = lhs && rhs;
95  // check subexpression size and option limit
96  if (Z3Expr::getExprSize(expr) > Options::MaxZ3Size)
97  {
98  getSolver().push();
99  getSolver().add(expr.getExpr());
100  z3::check_result res = getSolver().check();
101  getSolver().pop();
102  if (res != z3::unsat)
103  {
104  return lhs;
105  }
106  else
107  {
108  return Z3Expr::getFalseCond();
109  }
110  }
111  return expr;
112  }
113 }
