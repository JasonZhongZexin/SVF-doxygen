673 NodeID PAG::getGepObjNode(const MemObj* obj, const LocationSet& ls)
674 {
675  NodeID base = getObjectNode(obj);
676 
678  if (obj->isFieldInsensitive())
679  return getFIObjNode(obj);
680 
681  LocationSet newLS = SymbolTableInfo::SymbolInfo()->getModulusOffset(obj,ls);
682 
683  // Base and first field are the same memory location.
684  if (Options::FirstFieldEqBase && newLS.getOffset() == 0) return base;
685 
686  NodeLocationSetMap::iterator iter = GepObjNodeMap.find(std::make_pair(base, newLS));
687  if (iter == GepObjNodeMap.end())
688  return addGepObjNode(obj, newLS);
689  else
690  return iter->second;
691
