40 SVFVar::SVFVar(const Value* val, NodeID i, PNODEK k) :
41  GenericPAGNodeTy(i,k), value(val)
42 {
43  assert( ValNode <= k && k <= CloneDummyObjNode && "new SVFIR node kind?");
44  switch (k)
45  {
46  case ValNode:
47  case GepValNode:
48  {
49  assert(val != nullptr && "value is nullptr for ValVar or GepValNode");
50  isPtr = val->getType()->isPointerTy();
51  break;
52  }
53  case RetNode:
54  {
55  assert(val != nullptr && "value is nullptr for RetNode");
56  isPtr = SVFUtil::cast<Function>(val)->getReturnType()->isPointerTy();
57  break;
58  }
59  case VarargNode:
60  case DummyValNode:
61  {
62  isPtr = true;
63  break;
64  }
65  case ObjNode:
66  case GepObjNode:
67  case FIObjNode:
68  case DummyObjNode:
69  case CloneGepObjNode:
70  case CloneFIObjNode:
71  case CloneDummyObjNode:
72  {
73  isPtr = true;
74  if(val)
75  isPtr = val->getType()->isPointerTy();
76  break;
77  }
78  }
79 }

81 bool SVFVar::isIsolatedNode() const{
82  if (getInEdges().empty() && getOutEdges().empty())
83  return true;
84  else if (isConstantData())
85  return true;
86  else if (value && SVFUtil::isa<Function>(value))
87  return SVFUtil::isIntrinsicFun(SVFUtil::cast<Function>(value));
88  else
89  return false;
90 }
