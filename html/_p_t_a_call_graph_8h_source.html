329  inline CallSiteID addCallSite(const CallBlockNode* cs, const SVFFunction* callee)
330  {
331  std::pair<const CallBlockNode*, const SVFFunction*> newCS(std::make_pair(cs, callee));
332  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
333  //assert(it == csToIdMap.end() && "cannot add a callsite twice");
334  if(it == csToIdMap.end())
335  {
336  CallSiteID id = totalCallSiteNum++;
337  csToIdMap.insert(std::make_pair(newCS, id));
338  idToCSMap.insert(std::make_pair(id, newCS));
339  return id;
340  }
341  return it->second;
342  }
343  inline CallSiteID getCallSiteID(const CallBlockNode* cs, const SVFFunction* callee) const
344  {
345  CallSitePair newCS(std::make_pair(cs, callee));
346  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
347  assert(it != csToIdMap.end() && "callsite id not found! This maybe a partially resolved callgraph, please check the indCallEdge limit");
348  return it->second;
349  }
350  inline bool hasCallSiteID(const CallBlockNode* cs, const SVFFunction* callee) const
351  {
352  CallSitePair newCS(std::make_pair(cs, callee));
353  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
