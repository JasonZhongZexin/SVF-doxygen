23 void MTAStat::performThreadCallGraphStat(ThreadCallGraph* tcg)
24 {
25  u32_t numOfForkEdge = 0;

28  u32_t numOfIndForkEdge = 0;

30  {
31  bool indirectfork = false;
32  const Function* spawnee = SVFUtil::dyn_cast<Function>(tcg->getThreadAPI()->getForkedFun(*it));
33  if(spawnee==nullptr)
34  {
35  numOfIndForksite++;
36  indirectfork = true;
37  }
38  for (ThreadCallGraph::ForkEdgeSet::const_iterator cgIt = tcg->getForkEdgeBegin(*it), ecgIt =
39  tcg->getForkEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
40  {
41  numOfForkEdge++;
42  if(indirectfork)
43  numOfIndForkEdge++;
44  }
45  }
46 
47  for (ThreadCallGraph::CallSiteSet::iterator it = tcg->joinsitesBegin(), eit = tcg->joinsitesEnd(); it != eit; ++it)
48  {
49  for (ThreadCallGraph::JoinEdgeSet::const_iterator cgIt = tcg->getJoinEdgeBegin(*it), ecgIt =
50  tcg->getJoinEdgeEnd(*it); cgIt != ecgIt; ++cgIt)
51  {
52  numOfJoinEdge++;
53  }
54  }
55 
56 
57  PTNumStatMap.clear();
58  PTNumStatMap["NumOfForkSite"] = tcg->getNumOfForksite();
59  PTNumStatMap["NumOfForkEdge"] = numOfForkEdge;
60  PTNumStatMap["NumOfJoinEdge"] = numOfJoinEdge;
61  PTNumStatMap["NumOfJoinSite"] = tcg->getNumOfJoinsite();
62  PTNumStatMap["NumOfIndForkSite"] = numOfIndForksite;
63  PTNumStatMap["NumOfIndForkEdge"] = numOfIndForkEdge;
64  PTNumStatMap["NumOfIndCallEdge"] = tcg->getNumOfResolvedIndCallEdge();

92 {
93 
94  if(Options::AllPairMHP)
95  {
96  InstSet instSet1;
97  InstSet instSet2;
98  SVFModule* mod = mhp->getThreadCallGraph()->getModule();
99  for (SVFModule::iterator F = mod->begin(), E = mod->end(); F != E; ++F)
100  {
101  const Function* fun = (*F);
102  if(SVFUtil::isExtCall(fun))
103  continue;
104  if(!mhp->isConnectedfromMain(fun))
105  continue;
106  for (const_inst_iterator II = inst_begin(fun), E = inst_end(fun); II != E; ++II)
107  {
108  const Instruction *inst = &*II;
109  if(SVFUtil::isa<LoadInst>(inst))
110  {
111  instSet1.insert(inst);
112  }
113  else if(SVFUtil::isa<StoreInst>(inst))
114  {
115  instSet1.insert(inst);
116  instSet2.insert(inst);
117  }
118  }
119  }
120 
121 
122  for(InstSet::const_iterator it1 = instSet1.begin(), eit1 = instSet1.end(); it1!=eit1; ++it1)
123  {
124  for(InstSet::const_iterator it2 = instSet2.begin(), eit2 = instSet2.end(); it2!=eit2; ++it2)
125  {
126  mhp->mayHappenInParallel(*it1,*it2);
127  }
128  }
129  }
