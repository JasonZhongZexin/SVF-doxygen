42 void ConstraintGraph::buildCG()
43 {
44 
45  // initialize nodes
46  for(SVFIR::iterator it = pag->begin(), eit = pag->end(); it!=eit; ++it)
47  {
48  addConstraintNode(new ConstraintNode(it->first), it->first);
49  }
50 
51  // initialize edges
52  SVFStmt::SVFStmtSetTy& addrs = getPAGEdgeSet(SVFStmt::Addr);
53  for (SVFStmt::SVFStmtSetTy::iterator iter = addrs.begin(), eiter =
54  addrs.end(); iter != eiter; ++iter)
55  {
56  const AddrStmt* edge = SVFUtil::cast<AddrStmt>(*iter);
57  addAddrCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
58  }
59 
60  SVFStmt::SVFStmtSetTy& copys = getPAGEdgeSet(SVFStmt::Copy);
61  for (SVFStmt::SVFStmtSetTy::iterator iter = copys.begin(), eiter =
62  copys.end(); iter != eiter; ++iter)
63  {
64  const CopyStmt* edge = SVFUtil::cast<CopyStmt>(*iter);
65  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
66  }
67 
68  SVFStmt::SVFStmtSetTy& phis = getPAGEdgeSet(SVFStmt::Phi);
69  for (SVFStmt::SVFStmtSetTy::iterator iter = phis.begin(), eiter =
70  phis.end(); iter != eiter; ++iter)
71  {
72  const PhiStmt* edge = SVFUtil::cast<PhiStmt>(*iter);
73  for(const auto opVar : edge->getOpndVars())
74  addCopyCGEdge(opVar->getId(),edge->getResID());
75  }
76 
77  SVFStmt::SVFStmtSetTy& selects = getPAGEdgeSet(SVFStmt::Select);
78  for (SVFStmt::SVFStmtSetTy::iterator iter = selects.begin(), eiter =
79  selects.end(); iter != eiter; ++iter)
80  {
81  const SelectStmt* edge = SVFUtil::cast<SelectStmt>(*iter);
82  for(const auto opVar : edge->getOpndVars())
83  addCopyCGEdge(opVar->getId(),edge->getResID());
84  }
85 
86  SVFStmt::SVFStmtSetTy& calls = getPAGEdgeSet(SVFStmt::Call);
87  for (SVFStmt::SVFStmtSetTy::iterator iter = calls.begin(), eiter =
88  calls.end(); iter != eiter; ++iter)
89  {
90  const CallPE* edge = SVFUtil::cast<CallPE>(*iter);
91  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
92  }
93 
94  SVFStmt::SVFStmtSetTy& rets = getPAGEdgeSet(SVFStmt::Ret);
95  for (SVFStmt::SVFStmtSetTy::iterator iter = rets.begin(), eiter =
96  rets.end(); iter != eiter; ++iter)
97  {
98  const RetPE* edge = SVFUtil::cast<RetPE>(*iter);
99  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
100  }
101 
102  SVFStmt::SVFStmtSetTy& tdfks = getPAGEdgeSet(SVFStmt::ThreadFork);
103  for (SVFStmt::SVFStmtSetTy::iterator iter = tdfks.begin(), eiter =
104  tdfks.end(); iter != eiter; ++iter)
105  {
106  const TDForkPE* edge = SVFUtil::cast<TDForkPE>(*iter);
107  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
108  }
109 
110  SVFStmt::SVFStmtSetTy& tdjns = getPAGEdgeSet(SVFStmt::ThreadJoin);
111  for (SVFStmt::SVFStmtSetTy::iterator iter = tdjns.begin(), eiter =
112  tdjns.end(); iter != eiter; ++iter)
113  {
114  const TDJoinPE* edge = SVFUtil::cast<TDJoinPE>(*iter);
115  addCopyCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
116  }
117 
118  SVFStmt::SVFStmtSetTy& ngeps = getPAGEdgeSet(SVFStmt::Gep);
119  for (SVFStmt::SVFStmtSetTy::iterator iter = ngeps.begin(), eiter =
120  ngeps.end(); iter != eiter; ++iter)
121  {
122  GepStmt* edge = SVFUtil::cast<GepStmt>(*iter);
123  if(edge->isVariantFieldGep())
124  addVariantGepCGEdge(edge->getRHSVarID(),edge->getLHSVarID());
125  else
126  addNormalGepCGEdge(edge->getRHSVarID(),edge->getLHSVarID(),edge->getLocationSet());
127  }
128 
129  SVFStmt::SVFStmtSetTy& loads = getPAGEdgeSet(SVFStmt::Load);
130  for (SVFStmt::SVFStmtSetTy::iterator iter = loads.begin(), eiter =
131  loads.end(); iter != eiter; ++iter)
132  {
133  LoadStmt* edge = SVFUtil::cast<LoadStmt>(*iter);

169 AddrCGEdge* ConstraintGraph::addAddrCGEdge(NodeID src, NodeID dst)
170 {
171  ConstraintNode* srcNode = getConstraintNode(src);
172  ConstraintNode* dstNode = getConstraintNode(dst);
173  if(hasEdge(srcNode,dstNode,ConstraintEdge::Addr))
174  return nullptr;
175  AddrCGEdge* edge = new AddrCGEdge(srcNode, dstNode, edgeIndex++);
176  bool added = AddrCGEdgeSet.insert(edge).second;
177  assert(added && "not added??");
178  srcNode->addOutgoingAddrEdge(edge);
179  dstNode->addIncomingAddrEdge(edge);
180  return edge;

284 void ConstraintGraph::reTargetDstOfEdge(ConstraintEdge* edge, ConstraintNode* newDstNode)
285 {
286  NodeID newDstNodeID = newDstNode->getId();
287  NodeID srcId = edge->getSrcID();
288  if(LoadCGEdge* load = SVFUtil::dyn_cast<LoadCGEdge>(edge))
289  {
290  removeLoadEdge(load);
291  addLoadCGEdge(srcId,newDstNodeID);
292  }
293  else if(StoreCGEdge* store = SVFUtil::dyn_cast<StoreCGEdge>(edge))
294  {
295  removeStoreEdge(store);
296  addStoreCGEdge(srcId,newDstNodeID);
297  }
298  else if(CopyCGEdge* copy = SVFUtil::dyn_cast<CopyCGEdge>(edge))
299  {
300  removeDirectEdge(copy);
301  addCopyCGEdge(srcId,newDstNodeID);
302  }
303  else if(NormalGepCGEdge* gep = SVFUtil::dyn_cast<NormalGepCGEdge>(edge))
304  {
305  const LocationSet ls = gep->getLocationSet();
306  removeDirectEdge(gep);
307  addNormalGepCGEdge(srcId,newDstNodeID,ls);
308  }
309  else if(VariantGepCGEdge* gep = SVFUtil::dyn_cast<VariantGepCGEdge>(edge))
310  {
311  removeDirectEdge(gep);
312  addVariantGepCGEdge(srcId,newDstNodeID);
313  }
314  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
315  {
316  removeAddrEdge(addr);
317  }
318  else
319  assert(false && "no other edge type!!");
320 }

420 bool ConstraintGraph::moveInEdgesToRepNode(ConstraintNode* node, ConstraintNode* rep )
421 {
422  std::vector<ConstraintEdge*> sccEdges;
423  std::vector<ConstraintEdge*> nonSccEdges;
424  for (ConstraintNode::const_iterator it = node->InEdgeBegin(), eit = node->InEdgeEnd(); it != eit;
425  ++it)
426  {
427  ConstraintEdge* subInEdge = *it;
428  if(sccRepNode(subInEdge->getSrcID()) != rep->getId())
429  nonSccEdges.push_back(subInEdge);
430  else
431  {
432  sccEdges.push_back(subInEdge);
433  }
434  }
435  // if this edge is outside scc, then re-target edge dst to rep
436  while(!nonSccEdges.empty())
437  {
438  ConstraintEdge* edge = nonSccEdges.back();
439  nonSccEdges.pop_back();
440  reTargetDstOfEdge(edge,rep);
441  }
442 
443  bool criticalGepInsideSCC = false;
444  // if this edge is inside scc, then remove this edge and two end nodes
445  while(!sccEdges.empty())
446  {
447  ConstraintEdge* edge = sccEdges.back();
448  sccEdges.pop_back();
450  if(SVFUtil::isa<CopyCGEdge>(edge))
451  removeDirectEdge(edge);
452  else if (SVFUtil::isa<GepCGEdge>(edge))
453  {
454  // If the GEP is critical (i.e. may have a non-zero offset),
455  // then it brings impact on field-sensitivity.
456  if (!isZeroOffsettedGepCGEdge(edge))
457  {
458  criticalGepInsideSCC = true;
459  }
460  removeDirectEdge(edge);
461  }
462  else if(SVFUtil::isa<LoadCGEdge>(edge) || SVFUtil::isa<StoreCGEdge>(edge))
463  reTargetDstOfEdge(edge,rep);
464  else if(AddrCGEdge* addr = SVFUtil::dyn_cast<AddrCGEdge>(edge))
465  {
466  removeAddrEdge(addr);
467  }
468  else
469  assert(false && "no such edge");
470  }
471  return criticalGepInsideSCC;
472 }

545 void ConstraintGraph::print()
546 {
547 
548  outs() << "-----------------ConstraintGraph--------------------------------------\n";
549 
550  ConstraintEdge::ConstraintEdgeSetTy& addrs = this->getAddrCGEdges();
551  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = addrs.begin(),
552  eiter = addrs.end(); iter != eiter; ++iter)
553  {
554  outs() << (*iter)->getSrcID() << " -- Addr --> " << (*iter)->getDstID()
555  << "\n";
556  }
557 
558  ConstraintEdge::ConstraintEdgeSetTy& directs = this->getDirectCGEdges();
559  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = directs.begin(),
560  eiter = directs.end(); iter != eiter; ++iter)
561  {
562  if (CopyCGEdge* copy = SVFUtil::dyn_cast<CopyCGEdge>(*iter))
563  {
564  outs() << copy->getSrcID() << " -- Copy --> " << copy->getDstID()
565  << "\n";
566  }
567  else if (NormalGepCGEdge* ngep = SVFUtil::dyn_cast<NormalGepCGEdge>(*iter))
568  {
569  outs() << ngep->getSrcID() << " -- NormalGep (" << ngep->getConstantFieldIdx()
570  << ") --> " << ngep->getDstID() << "\n";
571  }
572  else if (VariantGepCGEdge* vgep = SVFUtil::dyn_cast<VariantGepCGEdge>(*iter))
573  {
574  outs() << vgep->getSrcID() << " -- VarintGep --> "
575  << vgep->getDstID() << "\n";
576  }
577  else
578  assert(false && "wrong constraint edge kind!");
579  }
580 
581  ConstraintEdge::ConstraintEdgeSetTy& loads = this->getLoadCGEdges();
582  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = loads.begin(),
583  eiter = loads.end(); iter != eiter; ++iter)
584  {
585  outs() << (*iter)->getSrcID() << " -- Load --> " << (*iter)->getDstID()
586  << "\n";
587  }
588 
589  ConstraintEdge::ConstraintEdgeSetTy& stores = this->getStoreCGEdges();
590  for (ConstraintEdge::ConstraintEdgeSetTy::iterator iter = stores.begin(),
591  eiter = stores.end(); iter != eiter; ++iter)
592  {
593  outs() << (*iter)->getSrcID() << " -- Store --> " << (*iter)->getDstID()
594  << "\n";
595  }
596 
597  outs()
598  << "--------------------------------------------------------------\n";
599
