42 bool LLVMUtil::isObject(const Value* ref)
43 {
44  bool createobj = false;
45  if (SVFUtil::isa<Instruction>(ref) && SVFUtil::isStaticExtCall(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(SVFUtil::cast<Instruction>(ref))) )
46  createobj = true;
47  if (SVFUtil::isa<Instruction>(ref) && SVFUtil::isHeapAllocExtCallViaRet(LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(SVFUtil::cast<Instruction>(ref))))
48  createobj = true;
49  if (SVFUtil::isa<GlobalVariable>(ref))
50  createobj = true;
51  if (SVFUtil::isa<Function>(ref) || SVFUtil::isa<AllocaInst>(ref) )
52  createobj = true;
53 
54  return createobj;

60 bool LLVMUtil::isConstantObjSym(const Value* val)
61 {
62  if (const GlobalVariable* v = SVFUtil::dyn_cast<GlobalVariable>(val))
63  {
64  if (cppUtil::isValVtbl(v))
65  return false;
66  else if (!v->hasInitializer())
67  {
68  if(v->isExternalLinkage(v->getLinkage()))
69  return false;
70  else
71  return true;
72  }
73  else
74  {
75  StInfo *stInfo = SymbolTableInfo::SymbolInfo()->getStructInfo(v->getInitializer()->getType());
76  const std::vector<const Type*> &fields = stInfo->getFlattenFieldTypes();
77  for (std::vector<const Type*>::const_iterator it = fields.begin(), eit = fields.end(); it != eit; ++it)
78  {
79  const Type *elemTy = *it;
80  assert(!SVFUtil::isa<FunctionType>(elemTy) && "Initializer of a global is a function?");
81  if (SVFUtil::isa<PointerType>(elemTy))
82  return false;
83  }
84 
85  return v->isConstant();
86  }
87  }
88  return SVFUtil::isConstantData(val);
89 }

94 void LLVMUtil::getFunReachableBBs (const SVFFunction* svfFun, std::vector<const SVFBasicBlock*> &reachableBBs)
95 {
96  assert(!SVFUtil::isExtCall(svfFun) && "The calling function cannot be an external function.");
97  //initial DominatorTree
98  DominatorTree dt;
99  dt.recalculate(const_cast<Function&>(*svfFun->getLLVMFun()));
100 
101  Set<const BasicBlock*> visited;
102  std::vector<const BasicBlock*> bbVec;
103  bbVec.push_back(&svfFun->getLLVMFun()->getEntryBlock());
104  while(!bbVec.empty())
105  {
106  const BasicBlock* bb = bbVec.back();
107  bbVec.pop_back();
108  const SVFBasicBlock* svfbb = LLVMModuleSet::getLLVMModuleSet()->getSVFBasicBlock(bb);
109  reachableBBs.push_back(svfbb);
110  if(DomTreeNode *dtNode = dt.getNode(const_cast<BasicBlock*>(bb)))
111  {
112  for (DomTreeNode::iterator DI = dtNode->begin(), DE = dtNode->end();
113  DI != DE; ++DI)
114  {
115  const BasicBlock* succbb = (*DI)->getBlock();
116  if(visited.find(succbb)==visited.end())
117  visited.insert(succbb);
118  else
119  continue;
120  bbVec.push_back(succbb);
121  }
122  }
123  }
124 }

170 bool LLVMUtil::isUncalledFunction (const Function* fun)
171 {
172  if(fun->hasAddressTaken())
173  return false;
174  if(SVFUtil::isProgEntryFunction(fun))
175  return false;
176  for (Value::const_user_iterator i = fun->user_begin(), e = fun->user_end(); i != e; ++i)
177  {
178  if (SVFUtil::isCallSite(*i))
179  return false;
180  }
181  if (LLVMModuleSet::getLLVMModuleSet()->hasDeclaration(fun))
182  {
183  const LLVMModuleSet::FunctionSetType &decls = LLVMModuleSet::getLLVMModuleSet()->getDeclaration(fun);
184  for (LLVMModuleSet::FunctionSetType::const_iterator it = decls.begin(),
185  eit = decls.end(); it != eit; ++it)
186  {
187  const Function* decl = *it;
188  if(decl->hasAddressTaken())
189  return false;
190  for (Value::const_user_iterator i = decl->user_begin(), e = decl->user_end(); i != e; ++i)
191  {
192  if (SVFUtil::isCallSite(*i))
193  return false;
194  }
195  }
196  }
197  return true;
198 }

260 void LLVMUtil::getNextInsts(const SVFInstruction* curInst, std::vector<const SVFInstruction*>& instList)
261 {
262  if (!curInst->isTerminator())
263  {
264  const Instruction* nextInst = curInst->getLLVMInstruction()->getNextNode();
265  const SVFInstruction* svfNextInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(nextInst);
266  if (SVFUtil::isIntrinsicInst(svfNextInst))
267  getNextInsts(svfNextInst, instList);
268  else
269  instList.push_back(svfNextInst);
270  }
271  else
272  {
273  const BasicBlock* BB = curInst->getParent()->getLLVMBasicBlock();
274  // Visit all successors of BB in the CFG
275  for (succ_const_iterator it = succ_begin(BB), ie = succ_end(BB); it != ie; ++it)
276  {
277  const Instruction* nextInst = &((*it)->front());
278  const SVFInstruction* svfNextInst = LLVMModuleSet::getLLVMModuleSet()->getSVFInstruction(nextInst);
279  if (SVFUtil::isIntrinsicInst(svfNextInst))
280  getNextInsts(svfNextInst, instList);
281  else
282  instList.push_back(svfNextInst);
283  }
284  }
285 }

319  * Get the first dominated cast instruction for heap allocations since they typically come from void* (i8*)
320  * for example, %4 = call align 16 i8* @malloc(i64 10); %5 = bitcast i8* %4 to i32*
321  * return %5 whose type is i32* but not %4 whose type is i8*
322  */
323 const Value* LLVMUtil::getUniqueUseViaCastInst(const Value* val)
324 {
325  const PointerType * type = SVFUtil::dyn_cast<PointerType>(val->getType());
326  assert(type && "this value should be a pointer type!");
328  if (IntegerType *IT = SVFUtil::dyn_cast<IntegerType>(getPtrElementType(type)))
329  {
330  if (IT->getBitWidth() == 8 && val->getNumUses()==1)
331  {
332  const Use *u = &*val->use_begin();
333  return SVFUtil::dyn_cast<BitCastInst>(u->getUser());
334  }
335  }
336  return nullptr;

342 const Type* LLVMUtil::getTypeOfHeapAlloc(const SVFInstruction *inst)
343 {
344  const PointerType* type = SVFUtil::dyn_cast<PointerType>(inst->getType());
345 
346  if(SVFUtil::isHeapAllocExtCallViaRet(inst))
347  {
348  if(const Value* v = getUniqueUseViaCastInst(inst->getLLVMInstruction()))
349  {
350  if(const PointerType* newTy = SVFUtil::dyn_cast<PointerType>(v->getType()))
351  type = newTy;
352  }
353  }
354  else if(SVFUtil::isHeapAllocExtCallViaArg(inst))
355  {
356  CallSite cs = SVFUtil::getLLVMCallSite(inst);
357  int arg_pos = SVFUtil::getHeapAllocHoldingArgPosition(SVFUtil::getCallee(cs));
358  const Value* arg = cs.getArgument(arg_pos);
359  type = SVFUtil::dyn_cast<PointerType>(arg->getType());
360  }
361  else
362  {
363  assert( false && "not a heap allocation instruction?");
364  }
365 
366  assert(type && "not a pointer type?");
367  return getPtrElementType(type);
