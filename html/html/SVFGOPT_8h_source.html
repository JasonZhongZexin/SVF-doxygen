89  virtual inline void connectAParamAndFParam(const PAGNode* cs_arg, const PAGNode* fun_arg, const CallBlockNode*, CallSiteID csId, SVFGEdgeSetTy& edges)
90  {
91  NodeID phiId = getDef(fun_arg);
92  SVFGEdge* edge = addCallEdge(getDef(cs_arg), phiId, csId);
93  if (edge != nullptr)
94  {
95  PHISVFGNode* phi = SVFUtil::cast<PHISVFGNode>(getSVFGNode(phiId));
96  addInterPHIOperands(phi, cs_arg);
97  edges.insert(edge);
98  }
99  }
101  virtual inline void connectFRetAndARet(const PAGNode* fun_ret, const PAGNode* cs_ret, CallSiteID csId, SVFGEdgeSetTy& edges)
102  {
103  NodeID phiId = getDef(cs_ret);
104  SVFGEdge* edge = addRetEdge(getDef(fun_ret), phiId, csId);
105  if (edge != nullptr)
106  {
107  PHISVFGNode* phi = SVFUtil::cast<PHISVFGNode>(getSVFGNode(phiId));
108  addInterPHIOperands(phi, fun_ret);
109  edges.insert(edge);
110  }

113  virtual inline void connectAInAndFIn(const ActualINSVFGNode* actualIn, const FormalINSVFGNode* formalIn, CallSiteID csId, SVFGEdgeSetTy& edges)
114  {
115  PointsTo intersection = actualIn->getPointsTo();
116  intersection &= formalIn->getPointsTo();
117  if (intersection.empty() == false)
118  {
119  NodeID aiDef = getActualINDef(actualIn->getId());
120  SVFGEdge* edge = addCallIndirectSVFGEdge(aiDef,formalIn->getId(),csId,intersection);
121  if (edge != nullptr)
122  edges.insert(edge);
123  }
124  }
126  virtual inline void connectFOutAndAOut(const FormalOUTSVFGNode* formalOut, const ActualOUTSVFGNode* actualOut, CallSiteID csId, SVFGEdgeSetTy& edges)
127  {
128  PointsTo intersection = formalOut->getPointsTo();
129  intersection &= actualOut->getPointsTo();
130  if (intersection.empty() == false)
131  {
132  NodeID foDef = getFormalOUTDef(formalOut->getId());
133  SVFGEdge* edge = addRetIndirectSVFGEdge(foDef,actualOut->getId(),csId,intersection);
134  if (edge != nullptr)
135  edges.insert(edge);
136  }
137  }
