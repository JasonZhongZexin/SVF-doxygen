159 // * Check (1) write-write race
160 // * (2) write-read race
161 // * (3) read-read race
162 // * when two memory access may-happen in parallel and are not protected by the same lock
163 // * (excluding global constraints because they are initialized before running the main function)
164 // */
165 void MTA::detect(SVFModule* module)
166 {
167 
168  DBOUT(DGENERAL, outs() << pasMsg("Starting Race Detection\n"));
169 
170  LoadSet loads;
171  StoreSet stores;
172 
173  Set<const Instruction*> needcheckinst;
174  // Add symbols for all of the functions and the instructions in them.
175  for (SVFModule::iterator F = module->begin(), E = module->end(); F != E; ++F)
176  {
177  // collect and create symbols inside the function body
178  for (inst_iterator II = inst_begin((*F)->getLLVMFun()), E = inst_end((*F)->getLLVMFun()); II != E; ++II)
179  {
180  const Instruction *inst = &*II;
181  if (const LoadInst* load = SVFUtil::dyn_cast<LoadInst>(inst))
182  {
183  loads.insert(load);
184  }
185  else if (const StoreInst* store = SVFUtil::dyn_cast<StoreInst>(inst))
186  {
187  stores.insert(store);
188  }
189  }
190  }
191 
192  for (LoadSet::const_iterator lit = loads.begin(), elit = loads.end(); lit != elit; ++lit)
193  {
194  const LoadInst* load = *lit;
195  bool loadneedcheck = false;
196  for (StoreSet::const_iterator sit = stores.begin(), esit = stores.end(); sit != esit; ++sit)
197  {
198  const StoreInst* store = *sit;
199 
200  loadneedcheck = true;
201  needcheckinst.insert(store);
202  }
203  if (loadneedcheck)
204  needcheckinst.insert(load);
205  }
206 
207  outs() << "HP needcheck: " << needcheckinst.size() << "\n";
208 }
