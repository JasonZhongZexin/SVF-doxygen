58 template <class S, class T> struct Hash<std::pair<S, T>>
59 {
60  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
61  static size_t szudzik(size_t a, size_t b)
62  {
63  return a > b ? b * b + a : a * a + a + b;
64  }
65 
66  size_t operator()(const std::pair<S, T> &t) const
67  {
68  Hash<decltype(t.first)> first;
69  Hash<decltype(t.second)> second;
70  return szudzik(first(t.first), second(t.second));
71  }
72 };
