81 void CFLGraph::buildFromDot(std::string fileName)
82 {
83  std::cout << "Building CFL Graph from dot file: " << fileName << "..\n";
84  std::string lineString;
85  std::ifstream inputFile(fileName);
86 
87  std::regex reg("Node(\\w+)\\s*->\\s*Node(\\w+)\\s*\\[.*label=(.*)\\]");
88 
89  std::cout << std::boolalpha;
90  u32_t lineNum = 0 ;
91 
92  while (getline(inputFile, lineString))
93  {
94  lineNum += 1;
95  std::smatch matches;
96  if (std::regex_search(lineString, matches, reg))
97  {
98  CFLNode *src, *dst;
99  if (hasGNode(std::stoul(matches.str(1), nullptr, 16))==false)
100  {
101  src = new CFLNode(std::stoul(matches.str(1), nullptr, 16));
102  addCFLNode(src->getId(), src);
103  }
104  else
105  {
106  src = getGNode(std::stoul(matches.str(1), nullptr, 16));
107  }
108  if (hasGNode(std::stoul(matches.str(2), nullptr, 16))==false)
109  {
110  dst = new CFLNode(std::stoul(matches.str(2), nullptr, 16));
111  addCFLNode(dst->getId(), dst);
112  }
113  else
114  {
115  dst = getGNode(std::stoul(matches.str(2), nullptr, 16));
116  }
117  if (externMap == false)
118  {
119  if (label2SymMap.find(matches.str(3)) != label2SymMap.end())
120  {
121  addCFLEdge(src, dst, label2SymMap[matches.str(3)]);
122  }
123  else
124  {
125  label2SymMap.insert({matches.str(3), current++});
126  addCFLEdge(src, dst, label2SymMap[matches.str(3)]);
127  }
128  }
129  else
130  {
131  if (label2SymMap.find(matches.str(3)) != label2SymMap.end())
132  {
133  addCFLEdge(src, dst, label2SymMap[matches.str(3)]);
134  }
135  else
136  {
137  if(Options::FlexSymMap == true)
138  {
139  label2SymMap.insert({matches.str(3), current++});
140  addCFLEdge(src, dst, label2SymMap[matches.str(3)]);
141  }
142  else
143  {
144  std::string msg = "In line " + std::to_string(lineNum) + " sym can not find in grammar, please correct the input dot or set --flexsymmap.";
145  SVFUtil::errMsg(msg);
146  std::cout << msg;
147  abort();
148  }
149  }
150  }
151  }
152  }
153  inputFile.close();
154 }

156 void CFLGraph::setMap(Map<std::string, Symbol>* terminals, Map<std::string, Symbol>* nonterminals)
157 {
158  externMap = true;
159  for(auto pairV : *terminals)
160  {
161  if(label2SymMap.find(pairV.first) == label2SymMap.end())
162  {
163  label2SymMap.insert(pairV);
164  }
165  }
166 
167  for(auto pairV : *nonterminals)
168  {
169  if(label2SymMap.find(pairV.first) == label2SymMap.end())
170  {
171  label2SymMap.insert(pairV);
172  }
173  }
174  current = label2SymMap.size();
175 }
