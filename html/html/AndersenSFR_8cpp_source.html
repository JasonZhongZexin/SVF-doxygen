83 bool AndersenSFR::processGepPts(PointsTo& pts, const GepCGEdge* edge)
84 {
85  ConstraintNode* dst = edge->getDstNode();
86  NodeID dstId = dst->getId();
87 
88  if (!dst->strides.empty() && SVFUtil::isa<NormalGepCGEdge>(edge)) // dst is in pwc
89  {
90  PointsTo tmpDstPts;
91  PointsTo srcInits = pts - getPts(dstId);
92 
93  if (!srcInits.empty())
94  {
95  NodeSet sortSrcInits;
96  for (NodeID ptd : srcInits)
97  sortSrcInits.insert(ptd);
98 
99  s64_t offset = SVFUtil::dyn_cast<NormalGepCGEdge>(edge)->getConstantFieldIdx();
100  fieldExpand(sortSrcInits, offset, dst->strides, tmpDstPts);
101  }
102 
103  if (unionPts(dstId, tmpDstPts))
104  {
105  pushIntoWorklist(dstId);
106  return true;
107  }
108  else
109  return false;
110  }
111  else
112  return Andersen::processGepPts(pts, edge);
113 }

119 void AndersenSFR::fieldExpand(NodeSet& initials, s64_t offset, NodeBS& strides, PointsTo& expandPts)
120 {
121  numOfFieldExpand++;
122 
123  while (!initials.empty())
124  {
125  NodeID init = *initials.begin();
126  initials.erase(init);
127 
128  if (consCG->isBlkObjOrConstantObj(init))
129  expandPts.set(init);
130  else
131  {
132  PAGNode* initPN = pag->getGNode(init);
133  const MemObj* obj = pag->getBaseObj(init);
134  const u32_t maxLimit = obj->getMaxFieldOffsetLimit();
135  s64_t initOffset;
136  if (GepObjVar *gepNode = SVFUtil::dyn_cast<GepObjVar>(initPN))
137  initOffset = gepNode->getConstantFieldIdx();
138  else if (SVFUtil::isa<FIObjVar>(initPN) || SVFUtil::isa<DummyObjVar>(initPN))
139  initOffset = 0;
140  else
141  assert(false && "Not an object node!!");
142 
143  Set<s64_t> offsets;
144  offsets.insert(offset);
145 
146  // calculate offsets
147  bool loopFlag = true;
148  while (loopFlag)
149  {
150  loopFlag = false;
151  for (auto _f : offsets)
152  for (auto _s : strides)
153  {
154  s64_t _f1 = _f + _s;
155  loopFlag = (offsets.find(_f1) == offsets.end()) && (initOffset + _f1 < maxLimit);
156  if (loopFlag)
157  offsets.insert(_f1);
158  }
159  }
160 
161  // get gep objs
162  for (s64_t _f : offsets)
163  {
164  NodeID gepId = consCG->getGepObjVar(init, LocationSet(_f));
165  initials.erase(gepId); // gep id in initials should be removed to avoid redundant derivation
166  expandPts.set(gepId);
167  }
168  }
169  }
170 }
