57 void AndersenLCD::handleCopyGep(ConstraintNode* node)
58 {
59  double propStart = stat->getClk();
60 
61  NodeID nodeId = node->getId();
62  computeDiffPts(nodeId);
63 
64  for (ConstraintEdge* edge : node->getCopyOutEdges())
65  {
66  NodeID dstNodeId = edge->getDstID();
67  const PointsTo& srcPts = getPts(nodeId);
68  const PointsTo& dstPts = getPts(dstNodeId);
69  // In one edge, if the pts of src node equals to that of dst node, and the edge
70  // is never met, push it into 'metEdges' and push the dst node into 'lcdCandidates'
71  if (!srcPts.empty() && srcPts == dstPts && !isMetEdge(edge))
72  {
73  addMetEdge(edge);
74  addLCDCandidate((edge)->getDstID());
75  }
76  processCopy(nodeId, edge);
77  }
78  for (ConstraintEdge* edge : node->getGepOutEdges())
79  {
80  if (GepCGEdge* gepEdge = SVFUtil::dyn_cast<GepCGEdge>(edge))
81  processGep(nodeId, gepEdge);
82  }
83 
84  double propEnd = stat->getClk();
85  timeOfProcessCopyGep += (propEnd - propStart) / TIMEINTERVAL;
86 }
