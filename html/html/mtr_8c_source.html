445 MtrNode *
446 Mtr_MakeGroup(
447  MtrNode * root /* root of the group tree */,
448  unsigned int low /* lower bound of the group */,
449  unsigned int size /* size of the group */,
450  unsigned int flags /* flags for the new group */)
451 {
452  MtrNode *node,
453  *first,
454  *last,
455  *previous,
456  *newn;
457 
458  /* Sanity check. */
459  if (size == 0)
460  return(NULL);
461 
462  /* Check whether current group includes new group. This check is
463  ** necessary at the top-level call. In the subsequent calls it is
464  ** redundant. */
465  if (low < (unsigned int) root->low ||
466  low + size > (unsigned int) (root->low + root->size))
467  return(NULL);
468 
469  /* At this point we know that the new group is contained
470  ** in the group of root. We have two possible cases here:
471  ** - root is a terminal node;
472  ** - root has children. */
473 
474  /* Root has no children: create a new group. */
475  if (root->child == NULL) {
476  newn = Mtr_AllocNode();
477  if (newn == NULL) return(NULL); /* out of memory */
478  newn->low = low;
479  newn->size = size;
480  newn->flags = flags;
481  newn->parent = root;
482  newn->elder = newn->younger = newn->child = NULL;
483  root->child = newn;
484  return(newn);
485  }
486 
487  /* Root has children: Find all children of root that are included
488  ** in the new group. If the group of any child entirely contains
489  ** the new group, call Mtr_MakeGroup recursively. */
490  previous = NULL;
491  first = root->child; /* guaranteed to be non-NULL */
492  while (first != NULL && low >= (unsigned int) (first->low + first->size)) {
493  previous = first;
494  first = first->younger;
495  }
496  if (first == NULL) {
497  /* We have scanned the entire list and we need to append a new
498  ** child at the end of it. Previous points to the last child
499  ** of root. */
500  newn = Mtr_AllocNode();
501  if (newn == NULL) return(NULL); /* out of memory */
502  newn->low = low;
503  newn->size = size;
504  newn->flags = flags;
505  newn->parent = root;
506  newn->elder = previous;
507  previous->younger = newn;
508  newn->younger = newn->child = NULL;
509  return(newn);
510  }
511  /* Here first is non-NULL and low < first->low + first->size. */
512  if (low >= (unsigned int) first->low &&
513  low + size <= (unsigned int) (first->low + first->size)) {
514  /* The new group is contained in the group of first. */
515  newn = Mtr_MakeGroup(first, low, size, flags);
516  return(newn);
517  } else if (low + size <= first->low) {
518  /* The new group is entirely contained in the gap between
519  ** previous and first. */
520  newn = Mtr_AllocNode();
521  if (newn == NULL) return(NULL); /* out of memory */
522  newn->low = low;
523  newn->size = size;
524  newn->flags = flags;
525  newn->child = NULL;
526  newn->parent = root;
527  newn->elder = previous;
528  newn->younger = first;
529  first->elder = newn;
530  if (previous != NULL) {
531  previous->younger = newn;
532  } else {
533  root->child = newn;
534  }
535  return(newn);
536  } else if (low < (unsigned int) first->low &&
537  low + size < (unsigned int) (first->low + first->size)) {
538  /* Trying to cut an existing group: not allowed. */
539  return(NULL);
540  } else if (low > first->low) {
541  /* The new group neither is contained in the group of first
542  ** (this was tested above) nor contains it. It is therefore
543  ** trying to cut an existing group: not allowed. */
544  return(NULL);
545  }
546 
547  /* First holds the pointer to the first child contained in the new
548  ** group. Here low <= first->low and low + size >= first->low +
549  ** first->size. One of the two inequalities is strict. */
550  last = first->younger;
551  while (last != NULL &&
552  (unsigned int) (last->low + last->size) < low + size) {
553  last = last->younger;
554  }
555  if (last == NULL) {
556  /* All the chilren of root from first onward become children
557  ** of the new group. */
558  newn = Mtr_AllocNode();
559  if (newn == NULL) return(NULL); /* out of memory */
560  newn->low = low;
561  newn->size = size;
562  newn->flags = flags;
563  newn->child = first;
564  newn->parent = root;
565  newn->elder = previous;
566  newn->younger = NULL;
567  first->elder = NULL;
568  if (previous != NULL) {
569  previous->younger = newn;
570  } else {
571  root->child = newn;
572  }
573  last = first;
574  while (last != NULL) {
575  last->parent = newn;
576  last = last->younger;
577  }
578  return(newn);
579  }
580 
581  /* Here last != NULL and low + size <= last->low + last->size. */
582  if (low + size - 1 >= (unsigned int) last->low &&
583  low + size < (unsigned int) (last->low + last->size)) {
584  /* Trying to cut an existing group: not allowed. */
585  return(NULL);
586  }
587 
588  /* First and last point to the first and last of the children of
589  ** root that are included in the new group. Allocate a new node
590  ** and make all children of root between first and last chidren of
591  ** the new node. Previous points to the child of root immediately
592  ** preceeding first. If it is NULL, then first is the first child
593  ** of root. */
594  newn = Mtr_AllocNode();
595  if (newn == NULL) return(NULL); /* out of memory */
596  newn->low = low;
597  newn->size = size;
598  newn->flags = flags;
599  newn->child = first;
600  newn->parent = root;
601  if (previous == NULL) {
602  root->child = newn;
603  } else {
604  previous->younger = newn;
605  }
606  newn->elder = previous;
607  newn->younger = last->younger;
608  if (last->younger != NULL) {
609  last->younger->elder = newn;
610  }
611  last->younger = NULL;
612  first->elder = NULL;
613  for (node = first; node != NULL; node = node->younger) {
614  node->parent = newn;
615  }
616 
617  return(newn);
618 
619 } /* end of Mtr_MakeGroup */

702 Mtr_ReorderGroups(
703  MtrNode *treenode,
704  int *permutation)
705 {
706  MtrNode *auxnode;
707  /* Initialize sorted list to first element. */
708  MtrNode *sorted = treenode;
709  sorted->low = permutation[sorted->index];
710  if (sorted->child != NULL)
711  Mtr_ReorderGroups(sorted->child, permutation);
712 
713  auxnode = treenode->younger;
714  while (auxnode != NULL) {
715  MtrNode *rightplace;
716  MtrNode *moving = auxnode;
717  auxnode->low = permutation[auxnode->index];
718  if (auxnode->child != NULL)
719  Mtr_ReorderGroups(auxnode->child, permutation);
720  rightplace = auxnode->elder;
721  /* Find insertion point. */
722  while (rightplace != NULL && auxnode->low < rightplace->low)
723  rightplace = rightplace->elder;
724  auxnode = auxnode->younger;
725  if (auxnode != NULL) {
726  auxnode->elder = moving->elder;
727  auxnode->elder->younger = auxnode;
728  } else {
729  moving->elder->younger = NULL;
730  }
731  if (rightplace == NULL) { /* Move to head of sorted list. */
732  sorted->elder = moving;
733  moving->elder = NULL;
734  moving->younger = sorted;
735  sorted = moving;
736  } else { /* Splice. */
737  moving->elder = rightplace;
738  moving->younger = rightplace->younger;
739  if (rightplace->younger != NULL)
740  rightplace->younger->elder = moving;
741  rightplace->younger = moving;
742  }
743  }
744  /* Fix parent. */
745  if (sorted->parent != NULL)
746  sorted->parent->child = sorted;

748 } /* end of Mtr_ReorderGroups */
