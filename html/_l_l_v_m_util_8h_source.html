227 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallSite cs)
228 {
229  const PointerType *refType = NULL;
230  // Case 1: heap object held by *argument, we should get its element type.
231  if (isHeapAllocExtCallViaArg(cs))
232  {
233  int argPos = getHeapAllocHoldingArgPosition(cs);
234  const Value *arg = cs.getArgument(argPos);
235  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
236  refType = SVFUtil::dyn_cast<PointerType>(argType->getElementType());
237  }
238  // Case 2: heap/static object held by return value.
239  else
240  {
241  assert((isStaticExtCall(cs) || isHeapAllocExtCallViaRet(cs))
242  && "Must be heap alloc via ret, or static allocation site");
243  refType = SVFUtil::dyn_cast<PointerType>(cs.getType());
244  }
245  assert(refType && "Allocated object must be held by a pointer-typed value.");
246  return refType;
247 }
