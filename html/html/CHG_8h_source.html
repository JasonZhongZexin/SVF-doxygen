241  inline u32_t getVirtualFunctionID(const SVFFunction* vfn) const
242  {
243  Map<const SVFFunction*, u32_t>::const_iterator it =
244  virtualFunctionToIDMap.find(vfn);
245  if (it != virtualFunctionToIDMap.end())
246  return it->second;
247  else
248  return -1;

250  inline const SVFFunction* getVirtualFunctionBasedonID(u32_t id) const
251  {
252  Map<const SVFFunction*, u32_t>::const_iterator it, eit;
253  for (it = virtualFunctionToIDMap.begin(), eit =
254  virtualFunctionToIDMap.end(); it != eit; ++it)
255  {
256  if (it->second == id)
257  return it->first;
258  }
259  return nullptr;
260  }
