533 void PointerAnalysis::connectVCallToVFns(const CallBlockNode* cs, const VFunSet &vfns, CallEdgeMap& newEdges)
534 {
536  for (VFunSet::const_iterator fit = vfns.begin(),
537  feit = vfns.end(); fit != feit; ++fit)
538  {
539  const SVFFunction* callee = *fit;
540  callee = getDefFunForMultipleModule(callee->getLLVMFun());
541  if (getIndCallMap()[cs].count(callee) > 0)
542  continue;
543  if(SVFUtil::getLLVMCallSite(cs->getCallSite()).arg_size() == callee->arg_size() ||
544  (SVFUtil::getLLVMCallSite(cs->getCallSite()).getFunctionType()->isVarArg() && callee->isVarArg()))
545  {
546  newEdges[cs].insert(callee);
547  getIndCallMap()[cs].insert(callee);
548  const CallBlockNode* callBlockNode = pag->getICFG()->getCallBlockNode(cs->getCallSite());
549  ptaCallGraph->addIndirectCallGraphEdge(callBlockNode, cs->getCaller(),callee);

571 void PointerAnalysis::validateSuccessTests(std::string fun)
572 {
573 
574  // check for must alias cases, whether our alias analysis produce the correct results
575  if (const SVFFunction* checkFun = getFunction(fun))
576  {
577  if(!checkFun->getLLVMFun()->use_empty())
578  outs() << "[" << this->PTAName() << "] Checking " << fun << "\n";
579 
580  for (Value::user_iterator i = checkFun->getLLVMFun()->user_begin(), e =
581  checkFun->getLLVMFun()->user_end(); i != e; ++i)
582  if (SVFUtil::isCallSite(*i))
583  {
584 
585  CallSite cs(*i);
586  assert(cs.getNumArgOperands() == 2
587  && "arguments should be two pointers!!");
588  Value* V1 = cs.getArgOperand(0);
589  Value* V2 = cs.getArgOperand(1);
590  AliasResult aliasRes = alias(V1, V2);
591 
592  bool checkSuccessful = false;
593  if (fun == aliasTestMayAlias || fun == aliasTestMayAliasMangled)
594  {
595  if (aliasRes == llvm::MayAlias || aliasRes == llvm::MustAlias)
596  checkSuccessful = true;
597  }
598  else if (fun == aliasTestNoAlias || fun == aliasTestNoAliasMangled)
599  {
600  if (aliasRes == llvm::NoAlias)
601  checkSuccessful = true;
602  }
603  else if (fun == aliasTestMustAlias || fun == aliasTestMustAliasMangled)
604  {
605  // change to must alias when our analysis support it
606  if (aliasRes == llvm::MayAlias || aliasRes == llvm::MustAlias)
607  checkSuccessful = true;
608  }
609  else if (fun == aliasTestPartialAlias || fun == aliasTestPartialAliasMangled)
610  {
611  // change to partial alias when our analysis support it
612  if (aliasRes == llvm::MayAlias)
613  checkSuccessful = true;
614  }
615  else
616  assert(false && "not supported alias check!!");
617 
618  NodeID id1 = pag->getValueNode(V1);
619  NodeID id2 = pag->getValueNode(V2);
620 
621  if (checkSuccessful)
622  outs() << sucMsg("\t SUCCESS :") << fun << " check <id:" << id1 << ", id:" << id2 << "> at ("
623  << getSourceLoc(*i) << ")\n";
624  else
625  {
626  SVFUtil::errs() << errMsg("\t FAILURE :") << fun
627  << " check <id:" << id1 << ", id:" << id2
628  << "> at (" << getSourceLoc(*i) << ")\n";
629  assert(false && "test case failed!");
630  }
631  }
632  else
633  assert(false && "alias check functions not only used at callsite??");
634 
635  }
636 }

643

650  checkFun->getLLVMFun()->user_end(); i != e; ++i)
651  if (CallInst *call = SVFUtil::dyn_cast<CallInst>(*i))
652  {
653  assert(call->getNumArgOperands() == 2

655  Value* V1 = call->getArgOperand(0);
656  Value* V2 = call->getArgOperand(1);
657  AliasResult aliasRes = alias(V1, V2);
658

661  {

664  expectedFailure = true;
665  }
666  else if (fun == aliasTestFailNoAlias || fun == aliasTestFailNoAliasMangled)
667  {
668  // change to partial alias when our analysis support it
669  if (aliasRes == llvm::MayAlias || aliasRes == llvm::PartialAlias || aliasRes == llvm::MustAlias)
670  expectedFailure = true;

681  else

683  SVFUtil::errs() << errMsg("\t UNEXPECTED FAILURE :") << fun << " check <id:" << id1 << ", id:" << id2 << "> at ("
