118 s32_t LocationSet::accumulateConstantOffset() const
119 {
120 
121  assert(isConstantOffset() && "not a constant offset");
122 
123  if(offsetValues.empty())
124  return accumulateConstantFieldIdx();
125 
126  s32_t totalConstOffset = 0;
127  for(int i = offsetValues.size() - 1; i >= 0; i--)
128  {
129  const SVFValue* value = offsetValues[i].first;
130  const SVFType* type = offsetValues[i].second;
131  const SVFConstantInt* op = SVFUtil::dyn_cast<SVFConstantInt>(value);
132  assert(op && "not a constant offset?");
133  if(type==nullptr)
134  {
135  totalConstOffset += op->getSExtValue();
136  continue;
137  }
138 
139  if(const SVFPointerType* pty = SVFUtil::dyn_cast<SVFPointerType>(type))
140  totalConstOffset += op->getSExtValue() * getElementNum(pty->getPtrElementType());
141  else
142  {
143  s32_t offset = op->getSExtValue();
144  u32_t flattenOffset = SymbolTableInfo::SymbolInfo()->getFlattenedElemIdx(type, offset);
145  totalConstOffset += flattenOffset;
146  }
147  }
148  return totalConstOffset;
149 }

173 bool LocationSet::operator< (const LocationSet& rhs) const
174 {
175  if (fldIdx != rhs.fldIdx)
176  return (fldIdx < rhs.fldIdx);
177  else
178  {
179  const OffsetValueVec& pairVec = getOffsetValueVec();
180  const OffsetValueVec& rhsPairVec = rhs.getOffsetValueVec();
181  if (pairVec.size() != rhsPairVec.size())
182  return (pairVec.size() < rhsPairVec.size());
183  else
184  {
185  OffsetValueVec::const_iterator it = pairVec.begin();
186  OffsetValueVec::const_iterator rhsIt = rhsPairVec.begin();
187  for (; it != pairVec.end() && rhsIt != rhsPairVec.end(); ++it, ++rhsIt)
188  {
189  return (*it) < (*rhsIt);
190  }
191 
192  return false;
193  }
194  }

197 SVF::LocationSet::LSRelation LocationSet::checkRelation(const LocationSet& LHS, const LocationSet& RHS)
198 {
199  NodeBS lhsLocations = LHS.computeAllLocations();
200  NodeBS rhsLocations = RHS.computeAllLocations();
201  if (lhsLocations.intersects(rhsLocations))
202  {
203  if (lhsLocations == rhsLocations)
204  return Same;
205  else if (lhsLocations.contains(rhsLocations))
206  return Superset;
207  else if (rhsLocations.contains(lhsLocations))
208  return Subset;
209  else
210  return Overlap;
211  }
212  else
213  {
214  return NonOverlap;
215  }
216 }
