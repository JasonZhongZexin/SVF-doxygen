67 SVFG* SVFGBuilder::build(BVDataPTAImpl* pta, VFG::VFGK kind)
68 {
69 
70  MemSSA* mssa = buildMSSA(pta, (VFG::PTRONLYSVFG==kind || VFG::PTRONLYSVFG_OPT==kind));
71 
72  DBOUT(DGENERAL, outs() << pasMsg("Build Sparse Value-Flow Graph \n"));
73  if(Options::SingleVFG)
74  {
75  if(globalSvfg==nullptr)
76  {
78  if(kind == VFG::FULLSVFG_OPT || kind == VFG::PTRONLYSVFG_OPT)
79  svfg = globalSvfg = new SVFGOPT(mssa, kind);
80  else
81  svfg = globalSvfg = new SVFG(mssa, kind);
82  buildSVFG();
83  }
84  }
85  else
86  {
87  if(kind == VFG::FULLSVFG_OPT || kind == VFG::PTRONLYSVFG_OPT)
88  svfg = new SVFGOPT(mssa, kind);
89  else
90  svfg = new SVFG(mssa,kind);
91  buildSVFG();
92  }
93 
95  if(Options::SVFGWithIndirectCall || SVFGWithIndCall)
96  svfg->updateCallGraph(pta);
97 
98  if(mssa->getPTA()->printStat())
99  svfg->performStat();
100 
101  if(Options::DumpVFG)
102  svfg->dump("svfg_final");
103 
104  return svfg;
105 }

115 MemSSA* SVFGBuilder::buildMSSA(BVDataPTAImpl* pta, bool ptrOnlyMSSA)
116 {
117 
118  DBOUT(DGENERAL, outs() << pasMsg("Build Memory SSA \n"));
119 
120  MemSSA* mssa = new MemSSA(pta, ptrOnlyMSSA);
121 
122  DominatorTree dt;
123  MemSSADF df;
124 
125  SVFModule* svfModule = mssa->getPTA()->getModule();
126  for (SVFModule::const_iterator iter = svfModule->begin(), eiter = svfModule->end();
127  iter != eiter; ++iter)
128  {
129 
130  const SVFFunction *fun = *iter;
131  if (isExtCall(fun))
132  continue;
133 
134  dt.recalculate(*fun->getLLVMFun());
135  df.runOnDT(dt);
136 
137  mssa->buildMemSSA(*fun, &df, &dt);
138  }
139 
140  mssa->performStat();
141  if (Options::DumpMSSA)
142  {
143  mssa->dumpMSSA();
144  }
145 
146  return mssa;
147 }
