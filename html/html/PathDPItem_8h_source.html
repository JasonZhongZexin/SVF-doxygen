92  }

95  {
96  for(EdgeSet::const_iterator it = edges.begin(), eit =edges.end(); it!=eit; ++it)
97  {
98  if(it->second == node)
99  return true;
100  }
101  return false;
102  }
104  inline bool hasOutgoingEdge(NodeID node) const
105  {
106  for(EdgeSet::const_iterator it = edges.begin(), eit =edges.end(); it!=eit; ++it)
107  {
108  if(it->first == node)
109  return true;
110  }
111  return false;
112  }
113  inline bool addPath(CondManager* allocator, PathCond* c, NodeID from, NodeID to)
114  {
115  if(pathLen() < maximumPathLen)
116  {
117  if(!hasVFEdge(from,to))
118  {
120  if(hasOutgoingEdge(from))
121  c = allocator->getTrueCond();
122  }
123  addVFEdge(from,to);
124  return condAnd(allocator,c);
125  }
126  // DBOUT(DDDA, SVFUtil::outs() << "\t\t!!path length beyond limits \n");
127  return true;
128  }

131  inline bool condAnd(CondManager* allocator, PathCond* c)
132  {
133  path = allocator->AND(path,c);
134  return path != allocator->getFalseCond();
135  }
136  inline void condOr(CondManager* allocator, PathCond* c)
137  {
138  path = allocator->OR(path,c);
139  }
