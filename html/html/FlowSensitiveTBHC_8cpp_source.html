65 {

91 bool FlowSensitiveTBHC::propAlongIndirectEdge(const IndirectSVFGEdge* edge)
92 {
93  SVFGNode* src = edge->getSrcNode();
94  SVFGNode* dst = edge->getDstNode();
95 
96  bool changed = false;
97 
98  // Get points-to targets may be used by next SVFG node.
99  // Propagate points-to set for node used in dst.
100  const NodeBS & pts = edge->getPointsTo();
101 
102  // Since the base Andersen's analysis does NOT perform type-based heap cloning,
103  // it uses only the base objects; we want to account for clones too.
104  NodeBS edgePtsAndClones;
105 
106  // TODO: the conditional bool may be unnecessary.
107  // Adding all clones is redundant, and introduces too many calls to propVarPts...
108  // This introduces performance and precision penalties.
109  // We should filter out according to src.
110  bool isStore = false;
111  const DIType *tildet = nullptr;
112  PointsTo storePts;
113  if (const StoreSVFGNode *store = SVFUtil::dyn_cast<StoreSVFGNode>(src))
114  {
115  tildet = getTypeFromCTirMetadata(store);
116  isStore = true;
117  storePts = getPts(store->getPAGDstNodeID());
118  }
119 
120  const PointsTo &filterSet = getFilterSet(src->getId());
121  for (NodeID o : pts)
122  {
123  if (!filterSet.test(o))
124  {
125  edgePtsAndClones.set(o);
126  }
127 
128  for (NodeID c : getClones(o))
129  {
130  if (!isStore)
131  {
132  if (!filterSet.test(c))
133  {
134  edgePtsAndClones.set(c);
135  }
136  }
137  else
138  {
139  if (storePts.test(c) && !filterSet.test(c))
140  {
141  edgePtsAndClones.set(c);
142  }
143  }
144  }
145 
146  if (GepObjVar *gep = SVFUtil::dyn_cast<GepObjVar>(pag->getGNode(o)))
147  {
148  // Want the geps which are at the same "level" as this one (same mem obj, same offset).
149  const NodeBS &geps = getGepObjsFromMemObj(gep->getMemObj(), gep->getConstantFieldIdx());
150  for (NodeID g : geps)
151  {
152  const DIType *gepType = getType(g);
153  if (!isStore || gepType || isBase(tildet, gepType))
154  {
155  if (!filterSet.test(g))
156  {
157  edgePtsAndClones.set(g);
158  }
159  }
160  }
161  }
162  }
163 
164  for (NodeID o : edgePtsAndClones)
165  {
166  if (propVarPtsFromSrcToDst(o, src, dst))
167  changed = true;
168 
169  if (isFIObjNode(o))
170  {
172  const NodeBS &allFields = getAllFieldsObjVars(o);
173  for (NodeID f : allFields)
174  {
175  if (propVarPtsFromSrcToDst(f, src, dst))
176  changed = true;
177  }
178  }
179  }
180 
181  return changed;
182 }

185 {
186  double start = stat->getClk();

188

269 bool FlowSensitiveTBHC::processGep(const GepSVFGNode* gep)
270 {
271  // Copy of that in FlowSensitive.cpp + some changes.
272  double start = stat->getClk();
273  bool changed = false;
274 
275  NodeID q = gep->getPAGSrcNodeID();
276 
277  const DIType *tildet = getTypeFromCTirMetadata(gep);
278  if (tildet != undefType)
279  {
280  bool reuse = Options::TBHCAllReuse || (Options::TBHCStoreReuse && !gepIsLoad(gep->getId()));
281  changed = init(gep->getId(), q, tildet, reuse, true);
282  }
283 
284  if (!gep->getPAGEdge()->isPTAEdge())
285  {
286  return changed;
287  }
288 
289  const PointsTo& qPts = getPts(q);
290  PointsTo &filterSet = getFilterSet(gep->getId());
291  PointsTo tmpDstPts;
292  for (NodeID oq : qPts)
293  {
294  if (filterSet.test(oq)) continue;
295 
296  if (TypeBasedHeapCloning::isBlkObjOrConstantObj(oq))
297  {
298  tmpDstPts.set(oq);
299  }
300  else
301  {
302  const GepStmt* gepStmt = SVFUtil::cast<GepStmt>(gep->getPAGEdge());
303  if (gepStmt->isVariantFieldGep())
304  {
305  setObjFieldInsensitive(oq);
306  tmpDstPts.set(oq);
307  const DIType *t = getType(oq);
308  if (t && (t->getTag() == dwarf::DW_TAG_array_type || t->getTag() == dwarf::DW_TAG_pointer_type))
309  {
310  const NodeBS fieldClones = getGepObjClones(oq, 1);
311  for (NodeID fc : fieldClones)
312  {
313  gepToSVFGRetrievers[getOriginalObj(fc)].set(gep->getId());
314  tmpDstPts.set(fc);
315  }
316  }
317  }
318  else
319  {
320  const DIType *baseType = getType(oq);
321 
322  // Drop down to field insensitive.
323  if (baseType == nullptr)
324  {
325  setObjFieldInsensitive(oq);
326  NodeID fiObj = oq;
327  tmpDstPts.set(fiObj);
328  continue;
329  }
330 
331  if (DCHGraph::isAgg(baseType) && baseType->getTag() != dwarf::DW_TAG_array_type
332  && ((u32_t)gepStmt->getConstantFieldIdx() >= dchg->getNumFields(baseType)))
333  {
334  // If the field offset is too high for this object, it is killed. It seems that a
335  // clone was made on this GEP but this is not the base (e.g. base->f1->f2), and SVF
336  // expects to operate on the base (hence the large offset). The base will have been
337  // cloned at another GEP and back-propagated, thus it'll reach here safe and sound.
338  // We ignore arrays/pointers because those are array accesses/pointer arithmetic we
339  // assume are correct.
340  // Obviously, non-aggregates cannot have their fields taken so they are spurious.
341  filterSet.set(oq);
342  }
343  else
344  {
345  // Operate on the field and all its clones.
346  const NodeBS fieldClones = getGepObjClones(oq, gepStmt->getConstantFieldIdx());
347  for (NodeID fc : fieldClones)
348  {
349  gepToSVFGRetrievers[getOriginalObj(fc)].set(gep->getId());
350  tmpDstPts.set(fc);
351  }
352  }
353  }
354  }
355  }
356 
357  double end = stat->getClk();
358  gepTime += (end - start) / TIMEINTERVAL;
359 
360  changed = unionPts(gep->getPAGDstNodeID(), tmpDstPts) || changed;
361  return changed;
362 }

364 bool FlowSensitiveTBHC::processLoad(const LoadSVFGNode* load)
365 {
366  double start = stat->getClk();
367 
368  bool changed = false;
369  const DIType *tildet = getTypeFromCTirMetadata(load);
370  if (tildet != undefType)
371  {
372  changed = init(load->getId(), load->getPAGSrcNodeID(), tildet, Options::TBHCAllReuse);
373  }
374 
375  // We want to perform the initialisation for non-pointer nodes but not process the load.
376  if (!load->getPAGEdge()->isPTAEdge())
377  {
378  return changed;
379  }
380 
381  NodeID dstVar = load->getPAGDstNodeID();
382 
383  const PointsTo& srcPts = getPts(load->getPAGSrcNodeID());
384  const PointsTo &filterSet = getFilterSet(load->getId());
385  // unionPtsFromIn is going to call getOriginalObj on ptd anyway.
386  // This results in fewer loop iterations. o_t, o_s --> o.
387  PointsTo srcOriginalObjs;
388  for (NodeID s : srcPts)
389  {
390  if (filterSet.test(s)) continue;
391  if (pag->isConstantObj(s) || pag->isNonPointerObj(s)) continue;
392  srcOriginalObjs.set(getOriginalObj(s));
393  }
394 
395  for (NodeID ptd : srcOriginalObjs)
396  {
397  // filterSet tests happened while building srcOriginalObjs.
398  if (unionPtsFromIn(load, ptd, dstVar))
399  changed = true;
400 
401  if (isFIObjNode(ptd))
402  {
405  const NodeBS &allFields = getAllFieldsObjVars(ptd);
406  for (NodeID f : allFields)
407  {
408  if (unionPtsFromIn(load, f, dstVar))
409  changed = true;
410  }
411  }
412  }
413 
414  double end = stat->getClk();
415  loadTime += (end - start) / TIMEINTERVAL;
416  return changed;
417 }

419 bool FlowSensitiveTBHC::processStore(const StoreSVFGNode* store)
420 {
421  double start = stat->getClk();
422 
423  bool changed = false;
424  const DIType *tildet = getTypeFromCTirMetadata(store);
425  if (tildet != undefType)
426  {
427  changed = init(store->getId(), store->getPAGDstNodeID(), tildet, Options::TBHCAllReuse || Options::TBHCStoreReuse);
428  }
429 
430  // Like processLoad: we want to perform initialisation for non-pointers but not the store.
431  if (!store->getPAGEdge()->isPTAEdge())
432  {
433  // Pass through and return because there may be some pointer nodes
434  // relying on this node's parents.
435  changed = getDFPTDataTy()->updateAllDFOutFromIn(store->getId(), 0, false);
436  return changed;
437  }
438 
439  const PointsTo & dstPts = getPts(store->getPAGDstNodeID());
440 
447  if (dstPts.empty())
448  {
449  return changed;
450  }
451 
452  changed = false;
453  const PointsTo &filterSet = getFilterSet(store->getId());
454  if(getPts(store->getPAGSrcNodeID()).empty() == false)
455  {
456  for (NodeID ptd : dstPts)
457  {
458  if (filterSet.test(ptd)) continue;
459 
460  if (pag->isConstantObj(ptd) || pag->isNonPointerObj(ptd))
461  continue;
462 
463  if (unionPtsFromTop(store, store->getPAGSrcNodeID(), ptd))
464  changed = true;
465  }
466  }
467 
468  double end = stat->getClk();
469  storeTime += (end - start) / TIMEINTERVAL;
470 
471  double updateStart = stat->getClk();
472  // also merge the DFInSet to DFOutSet.
474  NodeID singleton;
475  bool isSU = isStrongUpdate(store, singleton);
476  if (isSU)
477  {
478  svfgHasSU.set(store->getId());
479  if (strongUpdateOutFromIn(store, singleton))
480  changed = true;
481  }
482  else
483  {
484  svfgHasSU.reset(store->getId());
485  if (weakUpdateOutFromIn(store))
486  changed = true;
487  }
488  double updateEnd = stat->getClk();
489  updateTime += (updateEnd - updateStart) / TIMEINTERVAL;
490 
491  return changed;
492 }

574 void FlowSensitiveTBHC::determineWhichGepsAreLoads(void)
575 {

577  {
578  SVFGNode *svfgNode = nI->second;
579  if (const StmtSVFGNode *gep = SVFUtil::dyn_cast<GepSVFGNode>(svfgNode))
580  {
581  // Only care about ctir nodes - they have the reuse problem.
582  if (getTypeFromCTirMetadata(gep))
583  {
584  bool isLoad = true;
585  for (const SVFGEdge *e : gep->getOutEdges())

589  // Loop on itself - don't care.

594  isLoad = false;
595  break;
596  }
597  }
598 
599  if (isLoad)
600  {
601  loadGeps.set(gep->getId());
602  }
603  }
604  }

656 void FlowSensitiveTBHC::countAliases(Set<std::pair<NodeID, NodeID>> cmp, unsigned *mayAliases, unsigned *noAliases)
657 {
658  Map<std::pair<NodeID, NodeID>, PointsTo> filteredPts;
659  for (std::pair<NodeID, NodeID> locP : cmp)
660  {
661  const PointsTo &filterSet = getFilterSet(locP.first);
662  const PointsTo &pts = getPts(locP.second);
663  PointsTo &ptsFiltered = filteredPts[locP];
664 
665  for (NodeID o : pts)
666  {
667  if (filterSet.test(o)) continue;
668  ptsFiltered.set(o);
669  }
670  }
671 
672  for (std::pair<NodeID, NodeID> locPA : cmp)
673  {
674  const PointsTo &aPts = filteredPts[locPA];
675  for (std::pair<NodeID, NodeID> locPB : cmp)
676  {
677  if (locPB == locPA) continue;
678  const PointsTo &bPts = filteredPts[locPB];
679 
680  switch (alias(aPts, bPts))
681  {
682  case AliasResult::NoAlias:
683  ++(*noAliases);
684  break;
