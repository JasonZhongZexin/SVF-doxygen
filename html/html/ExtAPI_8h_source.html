47 public:
48  //External Function types
49  //Assume a call in the form LHS= F(arg0, arg1, arg2, arg3).
50  enum extf_t
51  {
52  EFT_NOOP= 0, //no effect on pointers
53  EFT_ALLOC, //returns a ptr to a newly allocated object
54  EFT_REALLOC, //like L_A0 if arg0 is a non-null ptr, else ALLOC
55  EFT_FREE, //free memory arg0 and all pointers passing into free function
56  EFT_FREE_MULTILEVEL, //any argument with 2-level pointer passing too a free wrapper function e.g., XFree(void**) which frees memory for void* and void**
57  EFT_NOSTRUCT_ALLOC, //like ALLOC but only allocates non-struct data
58  EFT_STAT, //retval points to an unknown static var X
59  EFT_STAT2, //ret -> X -> Y (X, Y - external static vars)
60  EFT_L_A0, //copies arg0, arg1, or arg2 into LHS
61  EFT_L_A1,
62  EFT_L_A2,
63  EFT_L_A8,
64  EFT_L_A0__A0R_A1, //stores arg1 into *arg0 and returns arg0 (currently only for memset)
65  EFT_L_A0__A0R_A1R, //copies the data that arg1 points to into the location
66  EFT_L_A1__FunPtr, //obtain the address of a symbol based on the arg1 (char*) and parse a function to LHS (e.g., void *dlsym(void *handle, char *funname);)
67  // arg0 points to; note that several fields may be
68  // copied at once if both point to structs.
69  // Returns arg0.
70  EFT_A1R_A0R, //copies *arg0 into *arg1, with non-ptr return
71  EFT_A3R_A1R_NS, //copies *arg1 into *arg3 (non-struct copy only)
72  EFT_A1R_A0, //stores arg0 into *arg1
73  EFT_A2R_A1, //stores arg1 into *arg2
74  EFT_A4R_A1, //stores arg1 into *arg4
75  EFT_L_A0__A2R_A0, //stores arg0 into *arg2 and returns it
76  EFT_L_A0__A1_A0, //store arg1 into arg0's base and returns arg0
77  EFT_A0R_NEW, //stores a pointer to an allocated object in *arg0
78  EFT_A1R_NEW, //as above, into *arg1, etc.
79  EFT_A2R_NEW,
80  EFT_A4R_NEW,
81  EFT_A11R_NEW,
82  EFT_STD_RB_TREE_INSERT_AND_REBALANCE, // Some complex effects
83  EFT_STD_RB_TREE_INCREMENT, // Some complex effects
84  EFT_STD_LIST_HOOK, // Some complex effects
85 
86  CPP_EFT_A0R_A1, //stores arg1 into *arg0
87  CPP_EFT_A0R_A1R, //copies *arg1 into *arg0
88  CPP_EFT_A1R, //load arg1
89  EFT_CXA_BEGIN_CATCH, //__cxa_begin_catch
90  CPP_EFT_DYNAMIC_CAST, // dynamic_cast
91  EFT_OTHER //not found in the list
92  };

132  //Return the extf_t of (F).
133  extf_t get_type(const SVFFunction* F) const
134  {
135  assert(F);
136  std::string funName = F->getName();
137  if(F->isIntrinsic())
138  {
139  unsigned start = funName.find('.');
140  unsigned end = funName.substr(start + 1).find('.');
141  funName = "llvm." + funName.substr(start + 1, end);
142  }
143  Map<std::string, extf_t>::const_iterator it= info.find(funName);
144  if(it == info.end())
145  return EFT_OTHER;
146  else
147  return it->second;
148  }
149 
150  //Does (F) have a static var X (unavailable to us) that its return points to?
151  bool has_static(const SVFFunction* F) const
152  {
153  extf_t t= get_type(F);
154  return t==EFT_STAT || t==EFT_STAT2;
155  }
156  //Assuming hasStatic(F), does (F) have a second static Y where X -> Y?
157  bool has_static2(const SVFFunction* F) const
158  {
159  return get_type(F) == EFT_STAT2;
160  }
161  //Does (F) allocate a new object and return it?
162  bool is_alloc(const SVFFunction* F) const
163  {
164  extf_t t= get_type(F);
165  return t==EFT_ALLOC || t==EFT_NOSTRUCT_ALLOC;
166  }
167  //Does (F) allocate a new object and assign it to one of its arguments?
168  bool is_arg_alloc(const SVFFunction* F) const
169  {
170  extf_t t= get_type(F);
171  return t>=EFT_A0R_NEW && t<=EFT_A11R_NEW;
172  }
173  //Get the position of argument which holds the new object
174  int get_alloc_arg_pos(const SVFFunction* F) const
175  {
176  extf_t t= get_type(F);
177  switch(t)
178  {
179  case EFT_A0R_NEW:
180  return 0;
181  case EFT_A1R_NEW:
182  return 1;
183  case EFT_A2R_NEW:
184  return 2;
185  case EFT_A4R_NEW:
186  return 4;
187  case EFT_A11R_NEW:
188  return 11;
189  default:
190  assert(false && "Not an alloc call via argument.");
191  return -1;
192  }
193  }
194  //Does (F) allocate only non-struct objects?
195  bool no_struct_alloc(const SVFFunction* F) const
196  {
197  return get_type(F) == EFT_NOSTRUCT_ALLOC;
198  }
199  //Does (F) not free/release any memory?
200  bool is_dealloc(const SVFFunction* F) const
201  {
202  extf_t t= get_type(F);
203  return t == EFT_FREE;
204  }
205  //Does (F) not do anything with the known pointers?
206  bool is_noop(const SVFFunction* F) const
207  {
208  extf_t t= get_type(F);
209  return t == EFT_NOOP || t == EFT_FREE;
210  }
211  //Does (F) reallocate a new object?
212  bool is_realloc(const SVFFunction* F) const
213  {
214  extf_t t= get_type(F);
215  return t==EFT_REALLOC;
216  }
217  //Should (F) be considered "external" (either not defined in the program
218  // or a user-defined version of a known alloc or no-op)?
219  bool is_ext(const SVFFunction* F)
220  {
221  assert(F);
222  //Check the cache first; everything below is slower.
223  Map<const SVFFunction*, bool>::iterator i_iec= isext_cache.find(F);
224  if(i_iec != isext_cache.end())
225  return i_iec->second;
226 
227  bool res;
228  if(F->isDeclaration() || F->isIntrinsic())
229  {
230  res= 1;
231  }
232  else
233  {
234  extf_t t= get_type(F);
235  res= t==EFT_ALLOC || t==EFT_REALLOC || t==EFT_NOSTRUCT_ALLOC
236  || t==EFT_NOOP || t==EFT_FREE;
237  }
238  isext_cache[F]= res;
239  return res;
240  }
