61 
62  for (const auto &func: *M)

86 void SaberCondAllocator::allocateForBB(const BasicBlock &bb)
87 {
88 
89  u32_t succ_number = getBBSuccessorNum(&bb);
90 
91  // if successor number greater than 1, allocate new decision variable for successors
92  if (succ_number > 1)
93  {
94 
95  //allocate log2(num_succ) decision variables
96  double num = log(succ_number) / log(2);
97  u32_t bit_num = (u32_t) ceil(num);
98  u32_t succ_index = 0;
99  std::vector<Condition> condVec;
100  for (u32_t i = 0; i < bit_num; i++)
101  {
102  condVec.push_back(newCond(bb.getTerminator()));
103  }
104 
105  // iterate each successor
106  for (succ_const_iterator succ_it = succ_begin(&bb);
107  succ_it != succ_end(&bb);
108  succ_it++, succ_index++)
109  {
110 
111  const BasicBlock *succ = *succ_it;
112 
113  Condition path_cond = getTrueCond();
114 
116 
117  // for each successor decide its bit representation
118  // decide whether each bit of succ_index is 1 or 0, if (three successor) succ_index is 000 then use C1^C2^C3
119  // if 001 use C1^C2^negC3
120  for (u32_t j = 0; j < bit_num; j++)
121  {
122  //test each bit of this successor's index (binary representation)
123  u32_t tool = 0x01 << j;
124  if (tool & succ_index)
125  {
126  path_cond = condAnd(path_cond, (condNeg(condVec.at(j))));
127  }
128  else
129  {
130  path_cond = condAnd(path_cond, condVec.at(j));
131  }
132  }
133  setBranchCond(&bb, succ, path_cond);
134  }
135 
136  }

186 SaberCondAllocator::evaluateTestNullLikeExpr(const BranchStmt *branchStmt, const BasicBlock *succ)
187 {
188 
189  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();

191  if (isTestNullExpr(branchStmt->getCondition()->getValue()))
192  {

199  }

201  {

215 SaberCondAllocator::Condition SaberCondAllocator::evaluateProgExit(const BranchStmt *branchStmt, const BasicBlock *succ)
216 {
217  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
218  const BasicBlock *succ2 = branchStmt->getSuccessor(1)->getBB();
219 
220  bool branch1 = isBBCallsProgExit(succ1);
221  bool branch2 = isBBCallsProgExit(succ2);
222 
224  if (branch1 && !branch2)
225  {
226  // succ is then branch
227  if (succ1 == succ)
228  return getFalseCond();
229  // succ is else branch
230  else
231  return getTrueCond();
232  }
234  else if (!branch1 && branch2)
235  {
236  // succ is else branch
237  if (succ2 == succ)
238  return getFalseCond();
239  // succ is then branch
240  else
241  return getTrueCond();
242  }
243  // two branches both call program exit
244  else if (branch1 && branch2)
245  {
246  return getFalseCond();
247  }
249  else
250  return Condition::nullExpr();
251

259 SaberCondAllocator::Condition SaberCondAllocator::evaluateLoopExitBranch(const BasicBlock *bb, const BasicBlock *dst)
260 {
261  const Function *fun = bb->getParent();
262  assert(fun == dst->getParent() && "two basic blocks should be in the same function");
263 
264  const LoopInfo *loopInfo = getLoopInfo(fun);
265  if (loopInfo->isLoopHeader(const_cast<BasicBlock *>(bb)))
266  {
267  const Loop *loop = loopInfo->getLoopFor(bb);
268  SmallBBVector exitbbs;
269  Set<BasicBlock *> filteredbbs;
270  loop->getExitBlocks(exitbbs);
272  while (!exitbbs.empty())
273  {
274  BasicBlock *eb = exitbbs.pop_back_val();
275  if (!isBBCallsProgExit(eb))
276  filteredbbs.insert(eb);
277  }
278 
280  bool allPDT = true;
281  PostDominatorTree *pdt = getPostDT(fun);
282  for (const auto &filteredbb: filteredbbs)
283  {
284  if (!pdt->dominates(dst, filteredbb))
285  allPDT = false;
286  }
287 
288  if (allPDT)
289  return getTrueCond();
290  }
291  return Condition::nullExpr();
292 }

299 SaberCondAllocator::Condition SaberCondAllocator::evaluateBranchCond(const BasicBlock *bb, const BasicBlock *succ)
300 {
301  if (getBBSuccessorNum(bb) == 1)
302  {
303  assert(bb->getTerminator()->getSuccessor(0) == succ && "not the unique successor?");
304  return getTrueCond();
305  }
306 
307  if (ICFGNode *icfgNode = getICFG()->getICFGNode(bb->getTerminator()))
308  {
309  for (const auto &svfStmt: icfgNode->getSVFStmts())
310  {
311  if (const BranchStmt *branchStmt = SVFUtil::dyn_cast<BranchStmt>(svfStmt))
312  {
313  if (branchStmt->getNumSuccessors() == 2)
314  {
315  const BasicBlock *succ1 = branchStmt->getSuccessor(0)->getBB();
316  const BasicBlock *succ2 = branchStmt->getSuccessor(1)->getBB();
317  assert((succ1 == succ || succ2 == succ) && "not a successor??");
318 
319  Condition evalLoopExit = evaluateLoopExitBranch(bb, succ);
320  if (!eq(evalLoopExit, Condition::nullExpr()))
321  return evalLoopExit;
322 
323  Condition evalProgExit = evaluateProgExit(branchStmt, succ);
324  if (!eq(evalProgExit, Condition::nullExpr()))
325  return evalProgExit;
326 
327  Condition evalTestNullLike = evaluateTestNullLikeExpr(branchStmt, succ);
328  if (!eq(evalTestNullLike, Condition::nullExpr()))
329  return evalTestNullLike;
330  break;
331  }
332  }
333  }
334  }
335 
336  return getBranchCond(bb, succ);
337 }

506 SaberCondAllocator::Condition SaberCondAllocator::ComputeIntraVFGGuard(const BasicBlock *srcBB, const BasicBlock *dstBB)
507 {
508 
509  assert(srcBB->getParent() == dstBB->getParent() && "two basic blocks are not in the same function??");
510 
511  PostDominatorTree *postDT = getPostDT(srcBB->getParent());
512  if (postDT->dominates(dstBB, srcBB))
513  return getTrueCond();
514 
515  CFWorkList worklist;
516  worklist.push(srcBB);
517  setCFCond(srcBB, getTrueCond());
518 
519  while (!worklist.empty())
520  {
521  const BasicBlock *bb = worklist.pop();
522  Condition cond = getCFCond(bb);
523 
526  Condition loopExitCond = evaluateLoopExitBranch(bb, dstBB);
527  if (!eq(loopExitCond, Condition::nullExpr()))
528  return condAnd(cond, loopExitCond);
529 
530 
531  for (succ_const_iterator succ_it = succ_begin(bb);
532  succ_it != succ_end(bb); succ_it++)
533  {
534  const BasicBlock *succ = *succ_it;
539  Condition brCond;
540  if (postDT->dominates(succ, bb))
541  brCond = getTrueCond();
542  else
543  brCond = getEvalBrCond(bb, succ);
544 
545  DBOUT(DSaber, outs() << " bb (" << bb->getName().str() <<
546  ") --> " << "succ_bb (" << succ->getName().str() << ") condition: " << brCond << "\n");
547  Condition succPathCond = condAnd(cond, brCond);
548  if (setCFCond(succ, condOr(getCFCond(succ), succPathCond)))
549  worklist.push(succ);
550  }
551  }
552 
553  DBOUT(DSaber, outs() << " src_bb (" << srcBB->getName().str() <<
554  ") --> " << "dst_bb (" << dstBB->getName().str() << ") condition: " << getCFCond(dstBB)
555  << "\n");
556 
557  return getCFCond(dstBB);
558 }
