341  static bool eqVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
342  {
343  if (lhs.size() != rhs.size()) return false;
344  for (const auto &item: lhs)
345  {
346  auto it = rhs.find(item.first);
347  if (it == rhs.end())
348  return false;
349  if (!item.second.equals(it->second))
350  {
351  return false;
352  }
353  }
354  return true;
355  }
356 
357 
358  static bool lessThanVarToValMap(const VarToValMap &lhs, const VarToValMap &rhs)
359  {
360  if (lhs.size() != rhs.size()) return lhs.size() < rhs.size();
361  for (const auto &item: lhs)
362  {
363  auto it = rhs.find(item.first);
364  // lhs > rhs if SVFVar not exists in rhs
365  if (it == rhs.end())
366  return false;
367  // judge from expr id
368  if (!item.second.equals(it->second))
369  {
370  return !item.second.geq(it->second);
371  }
372  }
373  return false;
374  }
