57 template <class S, class T> struct Hash<std::pair<S, T>>
58 {
59  // Pairing function from: http://szudzik.com/ElegantPairing.pdf
60  static size_t szudzik(size_t a, size_t b)
61  {
62  return a > b ? b * b + a : a * a + a + b;
63  }
64 
65  size_t operator()(const std::pair<S, T> &t) const
66  {
67  Hash<decltype(t.first)> first;
68  Hash<decltype(t.second)> second;
69  return szudzik(first(t.first), second(t.second));
70  }
71 };
